# 第 8 章：类型提示（Type Hints）

> **Python 版本要求：3.11+**
> 本章使用 Python 3.11+ 现代语法，主要改进包括：
> - 使用内置泛型（`list[int]` 而非 `List[int]`）
> - 使用联合操作符（`int | str` 而非 `Union[int, str]`）
> - 无需从 typing 导入基础容器类型

Python 类型提示完全指南，从入门到精通。

---

## 1. 为什么要用类型提示？

### 1.1 对比示例

```python
# ❌ 没有类型提示 - 不知道参数和返回值是什么
def process(data):
    return data.get('value', 0) * 2

# ✅ 有类型提示 - 一目了然
def process(data: dict) -> int:
    return data.get('value', 0) * 2
```

### 1.2 类型提示的好处

```python
# IDE 会给出智能提示
def greet(name: str) -> str:
    return f"Hello, {name}"

# greet(123)  # IDE 会报错：int 不能赋值给 str
# greet()     # IDE 会报错：缺少必需参数 name
```

## 2. 基础类型注解

### 2.1 函数参数和返回值

```python
# 语法：参数名: 类型 = 默认值
#      -> 返回值类型

# 最简单的例子
def say_hello() -> None:
    """无参数无返回值"""
    print("Hello!")

# 带参数
def greet(name: str) -> None:
    """一个字符串参数，无返回值"""
    print(f"Hello, {name}!")

# 带返回值
def add(a: int, b: int) -> int:
    """两个整数参数，返回整数"""
    return a + b

# 多个不同类型参数
def print_info(name: str, age: int, height: float) -> None:
    print(f"{name}, {age}岁, {height}米")

# 调用示例
result = add(1, 2)        # result 是 int 类型
greet("张三")             # 正确
# greet(123)              # 错误！类型不匹配
```

### 2.2 变量类型注解

```python
# 语法：变量名: 类型 = 值

# 基本类型
name: str = "张三"
age: int = 25
height: float = 1.75
is_student: bool = True

# 容器类型
numbers: list = [1, 2, 3]
info: dict = {"name": "张三", "age": 25}
point: tuple = (10, 20)

# Python 3.9+ 可以直接用内置类型
numbers2: list[int] = [1, 2, 3]      # 推荐
info2: dict[str, int] = {"a": 1}    # 推荐
```

### 2.3 所有基础类型

```python
# 基础类型
a: int = 10          # 整数
b: float = 3.14     # 浮点数
c: str = "hello"    # 字符串
d: bool = True      # 布尔值
e: bytes = b"data" # 字节

# 特殊类型
f: None = None      # 空值
g: type = int       # 类型本身

# 复合类型（后面详细讲）
h: any = "任意类型"  # 任意类型
```

## 3. 容器类型详解

### 3.1 List（列表）

```python
# 定义整数列表
def get_numbers() -> list[int]:
    return [1, 2, 3, 4, 5]

# 字符串列表
def get_names() -> list[str]:
    return ["张三", "李四", "王五"]

# 混合类型列表（不推荐）
def get_mixed() -> list:
    return [1, "two", 3.0]

# 使用示例
numbers: list[int] = [1, 2, 3]
# numbers.append("four")  # 错误！只能添加 int
```

### 3.2 Dict（字典）

```python
# 字符串到整数的映射
def count_words(text: str) -> dict[str, int]:
    result: dict[str, int] = {}
    for word in text.split():
        result[word] = result.get(word, 0) + 1
    return result

# 使用示例
ages: dict[str, int] = {
    "张三": 25,
    "李四": 30,
    "王五": 35
}

# 访问
print(ages["张三"])      # 25
# print(ages["赵六"])   # KeyError
```

### 3.3 Tuple（元组）

```python
# 固定长度和类型
def get_point() -> tuple[float, float]:
    return (10.5, 20.3)

# 混合类型
def get_user() -> tuple[str, int, bool]:
    return ("张三", 25, True)

# 可变长度（任意数量的 int）
def get_scores() -> tuple[int, ...]:
    return (90, 85, 78, 92, 88)

# 使用示例
point: tuple[float, float] = (1.0, 2.0)
x, y = point  # 解包

user: tuple[str, int, bool] = ("张三", 25, True)
name, age, _ = user
```

### 3.4 Set（集合）

```python
# 字符串集合
def unique_items(items: list[str]) -> set[str]:
    return set(items)

# 使用示例
unique: set[str] = {"apple", "banana", "orange"}
unique.add("grape")
# unique.add(123)  # 错误！只能添加 str
```

## 4. Optional 和 Union

### 4.1 Optional（可选值）

```python
# 可能返回 None
def find_user(user_id: int) -> dict | None:
    """查找用户，不存在返回 None"""
    users = {
        1: {"name": "张三", "email": "zhangsan@example.com"},
        2: {"name": "李四", "email": "lisi@example.com"}
    }
    return users.get(user_id)

# 使用示例
user = find_user(1)
if user is not None:
    print(user["name"])  # 类型检查器知道 user 不是 None

user = find_user(999)
if user is None:
    print("用户不存在")
```

### 4.2 Union（联合类型）

```python
# 多种类型之一
def parse(value: str) -> int | float | bool:
    if value.isdigit():
        return int(value)
    try:
        return float(value)
    except ValueError:
        return value.lower() == "true"

# 使用示例
result = parse("123")     # int: 123
result = parse("3.14")   # float: 3.14
result = parse("true")   # bool: True
```

### 4.3 详细对比

```python
# 方式1：使用 | 语法（推荐，Python 3.10+）
def get_name(user_id: int) -> str | None:
    """返回用户名或 None"""
    if user_id > 0:
        return "张三"
    return None

# 旧语法对比（已不推荐）：
# from typing import Optional
# def get_name(user_id: int) -> Optional[str]: ...

# 或：
# from typing import Union
# def get_name(user_id: int) -> Union[str, None]: ...
```

## 5. 函数类型 Callable

### 5.1 基本用法

```python
from typing import Callable

# 定义一个接受两个 int，返回 int 的函数
def apply(func: Callable[[int, int], int], a: int, b: int) -> int:
    return func(a, b)

# 使用示例
def add(x: int, y: int) -> int:
    return x + y

def multiply(x: int, y: int) -> int:
    return x * y

result = apply(add, 5, 3)      # 8
result = apply(multiply, 5, 3) # 15
```

### 5.2 无返回的回调

```python
from typing import Callable

def on_click(callback: Callable[[], None]) -> None:
    """点击后调用回调"""
    print("点击了")
    callback()

def handle_click() -> None:
    print("处理点击事件")

on_click(handle_click)
```

### 5.3 匿名函数

```python
from typing import Callable

# 匿名函数的类型
add: Callable[[int, int], int] = lambda x, y: x + y

# 在函数参数中使用
def math_operation(
    a: int,
    b: int,
    operation: Callable[[int, int], int]
) -> int:
    return operation(a, b)

# 使用
result = math_operation(10, 5, lambda x, y: x + y)   # 15
result = math_operation(10, 5, lambda x, y: x - y)    # 5
result = math_operation(10, 5, lambda x, y: x * y)    # 50
```

## 6. 泛型 TypeVar

### 6.1 什么是泛型？

```python
from typing import TypeVar

# 定义一个类型变量（泛型）
T = TypeVar('T')

# 这个函数可以处理任意类型
def first(items: list[T]) -> T:
    """返回列表的第一个元素"""
    return items[0]

# 使用
numbers = [1, 2, 3]
first_number = first(numbers)   # 类型是 int

strings = ["a", "b", "c"]
first_string = first(strings)   # 类型是 str

# 混合类型列表
mixed = [1, "two", 3.0]
first_item = first(mixed)      # 类型是 int | str | float
```

### 6.2 泛型类

```python
from typing import TypeVar, Generic

T = TypeVar('T')

class Box(Generic[T]):
    """一个可以装任何东西的盒子"""
    
    def __init__(self, item: T) -> None:
        self._item = item
    
    def get(self) -> T:
        return self._item
    
    def set(self, item: T) -> None:
        self._item = item

# 使用
int_box: Box[int] = Box(42)
value: int = int_box.get()     # 42
int_box.set(100)

str_box: Box[str] = Box("hello")
text: str = str_box.get()      # "hello"
str_box.set("world")

# 错误示例
# int_box.set("hello")  # 错误！只能放 int
```

### 6.3 泛型函数

```python
from typing import TypeVar

T = TypeVar('T')

def reverse(items: list[T]) -> list[T]:
    """反转列表"""
    return items[::-1]

def get_middle(items: list[T]) -> T:
    """获取中间元素"""
    return items[len(items) // 2]

def pair(items: list[T]) -> tuple[T, T]:
    """取前两个元素"""
    return (items[0], items[1])

# 使用
reverse([1, 2, 3])        # [3, 2, 1]
reverse(["a", "b", "c"])  # ["c", "b", "a"]

get_middle([1, 2, 3])           # 2
get_middle(["a", "b", "c"])     # "b"
```

## 7. 协议 Protocol（结构化类型）

### 7.1 基本概念

```python
from typing import Protocol

# 定义一个协议（接口）
class Drawable(Protocol):
    def draw(self) -> None: ...

# 只要有 draw 方法的类都符合这个协议
class Circle:
    def draw(self) -> None:
        print("画圆")

class Square:
    def draw(self) -> None:
        print("画正方形")

# 任何有 draw 方法的对象都可以用
def render(shape: Drawable) -> None:
    shape.draw()

render(Circle())  # 画圆
render(Square())  # 画正方形
```

### 7.2 完整示例

```python
from typing import Protocol

class Comparable(Protocol):
    def compare_to(self, other: "Comparable") -> int: ...

class Number:
    def __init__(self, value: int) -> None:
        self.value = value
    
    def compare_to(self, other: "Number") -> int:
        if self.value < other.value:
            return -1
        elif self.value > other.value:
            return 1
        return 0

def find_max(items: list[Comparable]) -> Comparable:
    """找到最大的元素"""
    max_item = items[0]
    for item in items[1:]:
        if item.compare_to(max_item) > 0:
            max_item = item
    return max_item

# 使用
numbers = [Number(1), Number(5), Number(3)]
max_num = find_max(numbers)
print(max_num.value)  # 5
```

## 8. 类型别名

### 8.1 简单别名

```python
# 给复杂类型起个简单的名字
UserId = int
PostId = int
Timestamp = float

# 使用
def get_user(user_id: UserId) -> dict:
    return {"id": user_id}

def create_post(post_id: PostId) -> dict:
    return {"id": post_id}

user_id: UserId = 123
post_id: PostId = 456
```

### 8.2 复杂别名

```python
# 用户列表
UserList = list[dict]
# 用户字典
UserDict = dict[str, dict]
# 用户配置
UserConfig = dict[str, str | int | bool]

# 使用
def get_users() -> UserList:
    return [{"name": "张三"}]

def get_user_dict() -> UserDict:
    return {"user1": {"name": "张三"}}

def get_config() -> UserConfig:
    return {"name": "张三", "age": 25, "active": True}
```

## 9. Literal（字面量）

### 9.1 基本用法

```python
from typing import Literal

# 限制为特定值
def set_status(status: Literal["pending", "active", "completed"]) -> None:
    print(f"状态设置为: {status}")

# 使用
set_status("pending")    # ✓
set_status("active")     # ✓
set_status("completed")  # ✓
# set_status("draft")   # ✗ 错误！不在允许的值中
```

### 9.2 实际应用

```python
from typing import Literal

def http_request(
    method: Literal["GET", "POST", "PUT", "DELETE"],
    url: str
) -> dict:
    return {"method": method, "url": url}

def set_mode(mode: Literal["read", "write", "append"]) -> None:
    print(f"模式: {mode}")

# 使用
http_request("GET", "/api/users")
http_request("POST", "/api/users")
# http_request("PATCH", "/api/users")  # 错误！

set_mode("read")
# set_mode("execute")  # 错误！
```

## 10. 完整示例

### 10.1 学生管理系统

```python
# 类型别名
StudentId = int
Score = float

class Student:
    def __init__(
        self,
        id: StudentId,
        name: str,
        age: int,
        scores: list[Score]
    ) -> None:
        self.id = id
        self.name = name
        self.age = age
        self.scores = scores
    
    def average_score(self) -> float:
        if not self.scores:
            return 0.0
        return sum(self.scores) / len(self.scores)
    
    def to_dict(self) -> dict[str, any]:
        return {
            "id": self.id,
            "name": self.name,
            "age": self.age,
            "scores": self.scores,
            "average": self.average_score()
        }

# 使用
students: list[Student] = [
    Student(1, "张三", 18, [85, 90, 92]),
    Student(2, "李四", 19, [78, 82, 88]),
    Student(3, "王五", 17, [95, 98, 100])
]

# 查找学生
def find_student(students: list[Student], student_id: StudentId) -> Student | None:
    for student in students:
        if student.id == student_id:
            return student
    return None

# 测试
student = find_student(students, 1)
if student:
    print(f"{student.name}的平均分: {student.average_score()}")
```

## 本章总结

### 核心语法

| 语法 | 说明 | 示例 |
|------|------|------|
| `x: int` | 变量类型 | `name: str = "张三"` |
| `def f(x: int) -> int` | 函数类型 | `def add(a: int) -> int` |
| `list[int]` | 列表类型 | `numbers: list[int]` |
| `dict[str, int]` | 字典类型 | `ages: dict[str, int]` |
| `str \| None` | 可选类型 | `name: str \| None` |
| `int \| str` | 联合类型 | `value: int \| str` |
| `Callable[[int], int]` | 函数类型 | `func: Callable[[int], int]` |
| `TypeVar('T')` | 泛型 | `T = TypeVar('T')` |
| `Protocol` | 协议 | `class X(Protocol):` |

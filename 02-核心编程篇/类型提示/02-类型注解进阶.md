## 1. 运行时类型检查

### 1.1 get_type_hints 的使用

```python
from typing import get_type_hints, List

def greet(name: str, age: int) -> str:
    return f"Hello {name}, you are {age} years old"

# 获取函数的所有类型提示
hints = get_type_hints(greet)
print(hints)
# {'name': <class 'str'>, 'age': <class 'int'>, 'return': <class 'str'>}

# 逐个查看
print(f"参数 name: {hints['name']}")
print(f"参数 age: {hints['age']}")
print(f"返回值: {hints['return']}")
```

### 1.2 手动实现运行时检查

```python
from typing import get_type_hints, Any, get_origin, get_args
import functools

def validate_args(func):
    """运行时参数类型验证装饰器"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        hints = get_type_hints(func)
        
        # 获取参数名列表（排除返回值）
        param_names = list(hints.keys())
        
        # 检查位置参数
        for i, arg in enumerate(args):
            if i < len(param_names):
                param_name = param_names[i]
                expected_type = hints[param_name]
                if not _check_type(arg, expected_type):
                    raise TypeError(
                        f"参数 '{param_name}' 类型错误: "
                        f"期望 {expected_type}, 实际得到 {type(arg).__name__}"
                    )
        
        # 检查关键字参数
        for key, value in kwargs.items():
            if key in hints:
                expected_type = hints[key]
                if not _check_type(value, expected_type):
                    raise TypeError(
                        f"参数 '{key}' 类型错误: "
                        f"期望 {expected_type}, 实际得到 {type(value).__name__}"
                    )
        
        return func(*args, **kwargs)
    return wrapper

def _check_type(value: Any, expected: Any) -> bool:
    """检查值是否符合类型"""
    # 处理 Optional[X] 即 X | None
    if get_origin(expected) is type(None):
        return value is None
    
    # 处理 Union
    if get_origin(expected) is Union:
        args = get_args(expected)
        return isinstance(value, args) or value is None
    
    # 处理泛型容器
    if get_origin(expected):
        return isinstance(value, expected)
    
    # 基本类型
    return isinstance(value, expected)

# 使用
from typing import Union

@validate_args
def add(a: int, b: int) -> int:
    return a + b

# 测试
print(add(1, 2))        # 3 正常
# add("1", 2)          # TypeError: 参数 'a' 类型错误
# add(1, "2")          # TypeError: 参数 'b' 类型错误
```

### 1.3 完整示例：类型检查工具类

```python
from typing import get_type_hints, Any, get_origin, get_args
from dataclasses import dataclass

class TypeChecker:
    """运行时类型检查器"""
    
    @staticmethod
    def check(value: Any, expected: Any) -> bool:
        """检查单个值"""
        # None 检查
        if expected is type(None):
            return value is None
        
        # Union 类型
        if get_origin(expected) is Union:
            args = get_args(expected)
            return isinstance(value, args) or value is None
        
        # 泛型容器
        if get_origin(expected):
            return isinstance(value, expected)
        
        # 基本类型
        return isinstance(value, expected)
    
    @classmethod
    def validate(cls, obj: Any, hints: dict[str, Any]) -> dict[str, list[str]]:
        """验证对象的所有字段"""
        errors: dict[str, list[str]] = {}
        
        for field_name, expected_type in hints.items():
            if hasattr(obj, field_name):
                value = getattr(obj, field_name)
                if not cls.check(value, expected_type):
                    errors[field_name] = [
                        f"期望 {expected_type}, 得到 {type(value).__name__}"
                    ]
        
        return errors

# 使用示例
from typing import Optional

@dataclass
class User:
    name: str
    age: int
    email: Optional[str] = None

user = User(name="张三", age=25, email="test@example.com")
hints = {"name": str, "age": int, "email": Optional[str]}

errors = TypeChecker.validate(user, hints)
if errors:
    print("验证失败:", errors)
else:
    print("验证通过")
```

## 2. 泛型高级用法

### 2.1 泛型约束

```python
from typing import TypeVar, Generic

# 约束类型必须是数字
Number = TypeVar('Number', int, float)

def double(value: Number) -> Number:
    """翻倍"""
    return value * 2

# 使用
result1 = double(5)      # int: 10
result2 = double(5.0)    # float: 10.0
# double("hello")        # 错误！字符串不支持
```

### 2.2 泛型类 - 栈

```python
from typing import TypeVar, Generic, List, Optional

T = TypeVar('T')

class Stack(Generic[T]):
    """泛型栈"""
    
    def __init__(self) -> None:
        self._items: List[T] = []
    
    def push(self, item: T) -> None:
        """入栈"""
        self._items.append(item)
    
    def pop(self) -> T:
        """出栈"""
        if not self._items:
            raise IndexError("Stack is empty")
        return self._items.pop()
    
    def peek(self) -> T:
        """查看栈顶"""
        if not self._items:
            raise IndexError("Stack is empty")
        return self._items[-1]
    
    def is_empty(self) -> bool:
        return len(self._items) == 0
    
    def __len__(self) -> int:
        return len(self._items)

# 使用
int_stack: Stack[int] = Stack()
int_stack.push(1)
int_stack.push(2)
int_stack.push(3)

print(int_stack.pop())   # 3
print(int_stack.peek())  # 2
print(len(int_stack))    # 2

# 字符串栈
str_stack: Stack[str] = Stack()
str_stack.push("a")
str_stack.push("b")
print(str_stack.pop())   # "b"
```

### 2.3 泛型类 - 仓库

```python
from typing import TypeVar, Generic, List, Optional
from dataclasses import dataclass

T = TypeVar('T')

@dataclass
class Entity:
    id: int

class Repository(Generic[T]):
    """泛型仓储"""
    
    def __init__(self) -> None:
        self._storage: List[T] = []
    
    def add(self, item: T) -> int:
        self._storage.append(item)
        return len(self._storage) - 1
    
    def get(self, index: int) -> Optional[T]:
        if 0 <= index < len(self._storage):
            return self._storage[index]
        return None
    
    def get_all(self) -> List[T]:
        return self._storage.copy()
    
    def update(self, index: int, item: T) -> bool:
        if 0 <= index < len(self._storage):
            self._storage[index] = item
            return True
        return False
    
    def delete(self, index: int) -> bool:
        if 0 <= index < len(self._storage):
            self._storage.pop(index)
            return True
        return False

# 使用
class User(Entity):
    def __init__(self, id: int, name: str):
        super().__init__(id)
        self.name = name

user_repo: Repository[User] = Repository()

# 添加用户
user_repo.add(User(1, "张三"))
user_repo.add(User(2, "李四"))

# 获取用户
user = user_repo.get(0)
if user:
    print(f"用户: {user.name}")

# 更新用户
user_repo.update(0, User(1, "张三（更新）"))

# 删除用户
user_repo.delete(1)

# 获取所有
for user in user_repo.get_all():
    print(f"ID: {user.id}, 姓名: {user.name}")
```

## 3. 协议进阶

### 3.1 协议继承

```python
from typing import Protocol, runtime_checkable

# 定义基础协议
class Drawable(Protocol):
    def draw(self) -> None: ...

class Fillable(Protocol):
    def fill(self) -> None: ...

# 组合协议
class Shape(Drawable, Fillable, Protocol):
    def move(self, x: int, y: int) -> None: ...

# 实现类
class Circle:
    def draw(self) -> None:
        print("画圆")
    
    def fill(self) -> None:
        print("填充圆")
    
    def move(self, x: int, y: int) -> None:
        print(f"移动到 ({x}, {y})")

# 使用
def render_shape(shape: Shape) -> None:
    shape.draw()
    shape.fill()
    shape.move(0, 0)

render_shape(Circle())
```

### 3.2 运行时检查

```python
from typing import Protocol, runtime_checkable

@runtime_checkable
class Serializable(Protocol):
    def to_json(self) -> str: ...

class User:
    def to_json(self) -> str:
        return '{"name": "张三", "age": 25}'

class SimpleClass:
    pass

# 运行时检查
print(isinstance(User(), Serializable))  # True
print(isinstance(SimpleClass(), Serializable))  # False
```

### 3.3 完整示例：排序协议

```python
from typing import Protocol, List

class Comparable(Protocol):
    def compare_to(self, other: "Comparable") -> int:
        """返回负数、零、正数"""
        ...

class Person:
    def __init__(self, name: str, age: int):
        self.name = name
        self.age = age
    
    def compare_to(self, other: "Person") -> int:
        return self.age - other.age
    
    def __repr__(self) -> str:
        return f"Person({self.name}, {self.age})"

def sort_items(items: List[Comparable]) -> List[Comparable]:
    """排序"""
    return sorted(items, key=lambda x: x.compare_to(items[0]))

# 使用
people = [
    Person("张三", 25),
    Person("李四", 18),
    Person("王五", 30)
]

sorted_people = sort_items(people)
print(sorted_people)
# [Person(李四, 18), Person(张三, 25), Person(王五, 30)]
```

### 4.2 递归类型

```python
from typing import Optional

class TreeNode:
    def __init__(
        self,
        value: str,
        left: Optional["TreeNode"] = None,
        right: Optional["TreeNode"] = None
    ) -> None:
        self.value = value
        self.left = left
        self.right = right

# 使用
tree = TreeNode(
    value="root",
    left=TreeNode("left"),
    right=TreeNode(
        value="right",
        left=TreeNode("right-left")
    )
)

def traverse(node: Optional[TreeNode]) -> None:
    if node is None:
        return
    print(node.value)
    traverse(node.left)
    traverse(node.right)

traverse(tree)
```

### 4.3 条件类型（overload）

```python
from typing import overload, Union

@overload
def process(value: int) -> int: ...
@overload
def process(value: str) -> str: ...
@overload
def process(value: list[int]) -> int: ...

def process(value: int | str | list[int]) -> int | str | int:
    if isinstance(value, int):
        return value * 2
    elif isinstance(value, str):
        return value.upper()
    else:
        return sum(value)

# 类型检查器会自动推断返回类型
x: int = process(5)           # int: 10
y: str = process("hello")    # str: "HELLO"
z: int = process([1, 2, 3])   # int: 6
```

## 5. 类型守卫

### 5.1 TypeGuard

```python
from typing import TypeGuard, List

def is_string_list(val: list[object]) -> TypeGuard[list[str]]:
    """类型守卫：缩小类型范围"""
    return all(isinstance(x, str) for x in val)

def process(items: list[object]) -> None:
    if is_string_list(items):
        # 类型检查器知道 items 是 list[str]
        print(" ".join(items))  # 正确
    else:
        print("不是字符串列表")

# 测试
process(["a", "b", "c"])  # a b c
process([1, "b", 3])      # 不是字符串列表
```

### 5.2 isinstance 守卫

```python
from typing import Union

def process(value: int | str | list) -> str:
    if isinstance(value, int):
        # value 是 int
        return f"整数: {value * 2}"
    elif isinstance(value, str):
        # value 是 str
        return f"字符串: {value.upper()}"
    elif isinstance(value, list):
        # value 是 list
        return f"列表长度: {len(value)}"
    return ""

# 测试
print(process(5))           # 整数: 10
print(process("hello"))    # 字符串: HELLO
print(process([1, 2, 3]))  # 列表长度: 3
```

## 6. 特殊类型

### 6.1 NoReturn

```python
from typing import NoReturn

def error(message: str) -> NoReturn:
    """永远不返回的函数（抛出异常）"""

def process() -> None:
    result = input("继续? (y/n): ")
```

## 7. 完整示例：类型安全的 API

```python
from typing import TypeVar, Generic, Protocol, List, Optional
from dataclasses import dataclass

# 定义协议
class Serializable(Protocol):
    def to_dict(self) -> dict: ...

# 泛型
T = TypeVar('T', bound=Serializable)

@dataclass
class User:
    id: int
    name: str
    email: str
    
    def to_dict(self) -> dict:
        return {
            "id": self.id,
            "name": self.name,
            "email": self.email
        }

class Repository(Generic[T]):
    """泛型仓储"""
    
    def __init__(self) -> None:
        self._data: List[T] = []
    
    def add(self, item: T) -> T:
        self._data.append(item)
        return item
    
    def get(self, predicate: callable[[T], bool]) -> Optional[T]:
        for item in self._data:
            if predicate(item):
                return item
        return None
    
    def get_all(self) -> List[T]:
        return self._data.copy()
    
    def filter(self, predicate: callable[[T], bool]) -> List[T]:
        return [item for item in self._data if predicate(item)]

# 使用
user_repo: Repository[User] = Repository()

# 添加
user_repo.add(User(1, "张三", "zhangsan@example.com"))
user_repo.add(User(2, "李四", "lisi@example.com"))

# 查找
user = user_repo.get(lambda u: u.id == 1)
if user:
    print(f"找到: {user.name}")

# 过滤
adults = user_repo.filter(lambda u: u.id > 0)
for u in adults:
    print(u.to_dict())
```

## 为什么需要闭包？

### 实际场景引入

想象你正在开发一个计数器系统：

```python
# 问题场景：如何保持计数器的状态？
count = 0  # 全局变量，容易被修改

def increment() -> int:
    global count
    count += 1
    return count

print(increment())  # 1
print(increment())  # 2
count = 100  # 外部意外修改！
print(increment())  # 101（错误！）
```

**痛点**：
- 使用全局变量容易被意外修改
- 无法创建多个独立的计数器
- 代码耦合度高，难以维护

**解决思路**：如果能让函数"记住"自己的状态，问题就解决了！

## 第一部分：什么是闭包

### 概念动机

闭包（Closure）是一种特殊的嵌套函数，它能"记住"外层函数的变量，即使外层函数已经执行完毕。

```
┌─────────────────────────────────────────┐
│              闭包的概念                 │
├─────────────────────────────────────────┤
│                                         │
│  闭包（Closure）                        │
│  ─────────────────                      │
│  内层函数引用了外层函数的变量，         │
│  即使外层函数已执行完毕，               │
│  内层函数仍能访问这些变量。             │
│                                         │
│  ─────────────────────────────────      │
│                                         │
│  构成闭包的条件：                       │
│  1. 必须有嵌套函数                      │
│  2. 内层函数引用外层变量                │
│  3. 外层函数返回内层函数                │
│                                         │
└─────────────────────────────────────────┘
```

### 最简示例

```python
def make_multiplier(factor: int) -> callable:
    """创建乘法器工厂函数"""
    def multiply(number: int) -> int:
        # 内层函数引用了外层变量 factor
        return number * factor
    return multiply

# 使用闭包
double = make_multiplier(2)
triple = make_multiplier(3)

# factor 被"记住"了
print(double(5))   # 10（5 * 2）
print(triple(5))   # 15（5 * 3）
```

### 详细讲解：闭包工作原理

```
调用 make_multiplier(2)：

┌─────────────────────────────────┐
│  make_multiplier(factor=2)      │
│  ─────────────────────────      │
│  factor = 2                     │
│                                 │
│  def multiply(number):          │
│      return number * factor  ◄──┼── 引用 factor
│                                 │
│  return multiply ───────────────┼── 返回函数
└─────────────────────────────────│
                                  │
        返回的函数：               │
        double = multiply          │
        factor=2 被"封闭"在其中   │
                                  │
        double(5) → 5 * 2 = 10   │
```

```python
# 验证闭包确实记住了变量
def make_multiplier(factor: int) -> callable:
    def multiply(number: int) -> int:
        return number * factor
    return multiply

double = make_multiplier(2)
print(double.__code__.co_freevars)  # ('factor',) - factor 被保存在闭包中
```

### 渐进复杂化

回到计数器问题，使用闭包解决：

```python
from typing import Callable

def create_counter(start: int = 0) -> Callable[[], int]:
    """创建独立的计数器"""
    count = start  # 私有变量
    
    def increment() -> int:
        nonlocal count
        count += 1
        return count
    
    return increment

# 创建多个独立的计数器
counter_a = create_counter()
counter_b = create_counter(100)

print(counter_a())  # 1
print(counter_a())  # 2
print(counter_b())  # 101（互不影响）
print(counter_a())  # 3
```

### 实际应用：配置管理

```python
from typing import Callable, Any

def create_config_manager(defaults: dict[str, Any]) -> Callable[..., Any]:
    """创建配置管理器"""
    config = defaults.copy()  # 私有配置
    
    def get(key: str, default: Any = None) -> Any:
        """获取配置"""
        return config.get(key, default)
    
    def set_value(key: str, value: Any) -> None:
        """设置配置"""
        config[key] = value
    
    def get_all() -> dict[str, Any]:
        """获取所有配置"""
        return config.copy()
    
    # 返回一个包含多个方法的对象
    class ConfigManager:
        get = staticmethod(get)
        set = staticmethod(set_value)
        all = staticmethod(get_all)
    
    return ConfigManager

# 使用
manager = create_config_manager({"debug": False, "host": "localhost"})
print(manager.get("debug"))  # False
manager.set("debug", True)
print(manager.get("debug"))  # True
```

## 第二部分：nonlocal 关键字

### 概念动机

在闭包中，如果内层函数需要修改外层变量，必须使用 `nonlocal` 关键字声明。

### 最简示例

```python
# ❌ 不使用 nonlocal（会出错）
def counter_wrong() -> Callable[[], int]:
    count = 0
    
    def increment() -> int:
        count += 1  # UnboundLocalError!
        return count
    
    return increment

# ✅ 使用 nonlocal（正确）
def counter() -> Callable[[], int]:
    count = 0
    
    def increment() -> int:
        nonlocal count  # 声明使用外层变量
        count += 1
        return count
    
    return increment

my_counter = counter()
print(my_counter())  # 1
print(my_counter())  # 2
print(my_counter())  # 3
```

### 详细讲解

```python
def demonstrate_nonlocal() -> None:
    """演示 nonlocal 的作用"""
    x = 10
    
    def read_only() -> int:
        # 不修改，直接读取是可以的
        return x * 2
    
    def modify_with_nonlocal() -> int:
        nonlocal x
        x = 20
        return x
    
    def try_modify_without_nonlocal() -> int:
        # 这会创建一个新的局部变量 x
        x = 30  # 这是局部变量，不影响外层 x
        return x
    
    print(f"原始 x: {x}")  # 10
    print(f"read_only: {read_only()}")  # 20
    print(f"修改后 x: {x}")  # 10（未改变）
    
    print(f"modify_with_nonlocal: {modify_with_nonlocal()}")  # 20
    print(f"修改后 x: {x}")  # 20（已改变）
```

### 渐进复杂化：多层嵌套

```python
def outer() -> Callable[[], None]:
    x = "outer"
    
    def middle() -> Callable[[], None]:
        y = "middle"
        
        def inner() -> None:
            nonlocal x, y  # 可以声明多个外层变量
            x = "modified outer"
            y = "modified middle"
            print(f"inner: x={x}, y={y}")
        
        inner()
        print(f"middle: x={x}, y={y}")
    
    middle()
    print(f"outer: x={x}")

# outer() 输出:
# inner: x=modified outer, y=modified middle
# middle: x=modified outer, y=modified middle
# outer: x=modified outer
```

## 第三部分：闭包的应用

### 应用一：数据隐藏/封装

```python
from typing import Callable

def create_account(initial_balance: float) -> dict[str, Callable]:
    """创建银行账户（使用闭包隐藏余额）"""
    balance = initial_balance  # 私有变量，外部无法直接访问
    
    def get_balance() -> float:
        return balance
    
    def deposit(amount: float) -> bool:
        nonlocal balance
        if amount > 0:
            balance += amount
            return True
        return False
    
    def withdraw(amount: float) -> bool:
        nonlocal balance
        if 0 < amount <= balance:
            balance -= amount
            return True
        return False
    
    return {
        "balance": get_balance,
        "deposit": deposit,
        "withdraw": withdraw
    }

# 使用
account = create_account(1000)
print(account["balance"]())  # 1000
account["deposit"](500)
print(account["balance"]())  # 1500
account["withdraw"](200)
print(account["balance"]())  # 1300
# balance 变量无法从外部直接访问！
```

### 应用二：缓存（Memoization）

```python
from typing import Callable, TypeVar

T = TypeVar('T')

def memoize(func: Callable[..., T]) -> Callable[..., T]:
    """缓存装饰器：避免重复计算"""
    cache: dict[tuple, T] = {}
    
    def wrapper(*args: tuple) -> T:
        if args in cache:
            print(f"缓存命中: {args}")
            return cache[args]
        result = func(*args)
        cache[args] = result
        return result
    
    return wrapper

@memoize
def fibonacci(n: int) -> int:
    """计算斐波那契数"""
    if n < 2:
        return n
    return fibonacci(n - 1) + fibonacci(n - 2)

print(fibonacci(10))  # 55
print(fibonacci(10))  # 缓存命中: (10,) → 55
```

### 应用三：工厂函数

```python
from typing import Callable

def create_power_function(exponent: int) -> Callable[[int], int]:
    """创建幂函数工厂"""
    def power(base: int) -> int:
        return base ** exponent
    return power

# 创建不同的幂函数
square = create_power_function(2)
cube = create_power_function(3)

print(square(5))    # 25
print(cube(5))      # 125

# 更复杂的工厂：创建验证器
def create_validator(min_val: int, max_val: int) -> Callable[[int], bool]:
    """创建范围验证器"""
    def validate(value: int) -> bool:
        return min_val <= value <= max_val
    return validate

is_valid_age = create_validator(0, 120)
is_valid_percent = create_validator(0, 100)

print(is_valid_age(25))      # True
print(is_valid_age(150))      # False
print(is_valid_percent(85))   # True
```

### 应用四：延迟计算

```python
from typing import Callable, TypeVar

T = TypeVar('T')

def lazy_evaluation(func: Callable[[], T]) -> Callable[[], T]:
    """延迟计算：只在第一次调用时执行"""
    result: T | None = None
    computed = False
    
    def get_result() -> T:
        nonlocal result, computed
        if not computed:
            result = func()
            computed = True
        return result
    
    return get_result

# 使用
@lazy_evaluation
def expensive_computation() -> int:
    print("执行耗时计算...")
    return 42

print("准备获取结果")
print(expensive_computation())  # 执行耗时计算... 42
print(expensive_computation())  # 42（不再执行）
```

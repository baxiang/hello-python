## 为什么需要闭包？

### 实际场景引入

想象你正在开发一个计数器系统：

```python
# 问题场景：如何保持计数器的状态？
count = 0  # 全局变量，容易被修改

def increment() -> int:
    global count
    count += 1
    return count

print(increment())  # 1
print(increment())  # 2
count = 100  # 外部意外修改！
print(increment())  # 101（错误！）
```

**痛点**：
- 使用全局变量容易被意外修改
- 无法创建多个独立的计数器
- 代码耦合度高，难以维护

**解决思路**：如果能让函数"记住"自己的状态，问题就解决了！

## 第一部分：什么是闭包

### 概念动机

闭包（Closure）是一种特殊的嵌套函数，它能"记住"外层函数的变量，即使外层函数已经执行完毕。

```
┌─────────────────────────────────────────┐
│              闭包的概念                 │
├─────────────────────────────────────────┤
│                                         │
│  闭包（Closure）                        │
│  ─────────────────                      │
│  内层函数引用了外层函数的变量，         │
│  即使外层函数已执行完毕，               │
│  内层函数仍能访问这些变量。             │
│                                         │
│  ─────────────────────────────────      │
│                                         │
│  构成闭包的条件：                       │
│  1. 必须有嵌套函数                      │
│  2. 内层函数引用外层变量                │
│  3. 外层函数返回内层函数                │
│                                         │
└─────────────────────────────────────────┘
```

### 最简示例

```python
def make_multiplier(factor: int) -> callable:
    """创建乘法器工厂函数"""
    def multiply(number: int) -> int:
        # 内层函数引用了外层变量 factor
        return number * factor
    return multiply

# 使用闭包
double = make_multiplier(2)
triple = make_multiplier(3)

# factor 被"记住"了
print(double(5))   # 10（5 * 2）
print(triple(5))   # 15（5 * 3）
```

### 详细讲解：闭包工作原理

```
调用 make_multiplier(2)：

┌─────────────────────────────────┐
│  make_multiplier(factor=2)      │
│  ─────────────────────────      │
│  factor = 2                     │
│                                 │
│  def multiply(number):          │
│      return number * factor  ◄──┼── 引用 factor
│                                 │
│  return multiply ───────────────┼── 返回函数
└─────────────────────────────────│
                                  │
        返回的函数：               │
        double = multiply          │
        factor=2 被"封闭"在其中   │
                                  │
        double(5) → 5 * 2 = 10   │
```

```python
# 验证闭包确实记住了变量
def make_multiplier(factor: int) -> callable:
    def multiply(number: int) -> int:
        return number * factor
    return multiply

double = make_multiplier(2)
print(double.__code__.co_freevars)  # ('factor',) - factor 被保存在闭包中
```

### 渐进复杂化

回到计数器问题，使用闭包解决：

```python
from typing import Callable

def create_counter(start: int = 0) -> Callable[[], int]:
    """创建独立的计数器"""
    count = start  # 私有变量
    
    def increment() -> int:
        nonlocal count
        count += 1
        return count
    
    return increment

# 创建多个独立的计数器
counter_a = create_counter()
counter_b = create_counter(100)

print(counter_a())  # 1
print(counter_a())  # 2
print(counter_b())  # 101（互不影响）
print(counter_a())  # 3
```

### 实际应用：配置管理

```python
from typing import Callable, Any

def create_config_manager(defaults: dict[str, Any]) -> Callable[..., Any]:
    """创建配置管理器"""
    config = defaults.copy()  # 私有配置
    
    def get(key: str, default: Any = None) -> Any:
        """获取配置"""
        return config.get(key, default)
    
    def set_value(key: str, value: Any) -> None:
        """设置配置"""
        config[key] = value
    
    def get_all() -> dict[str, Any]:
        """获取所有配置"""
        return config.copy()
    
    # 返回一个包含多个方法的对象
    class ConfigManager:
        get = staticmethod(get)
        set = staticmethod(set_value)
        all = staticmethod(get_all)
    
    return ConfigManager

# 使用
manager = create_config_manager({"debug": False, "host": "localhost"})
print(manager.get("debug"))  # False
manager.set("debug", True)
print(manager.get("debug"))  # True
```

## 第二部分：nonlocal 关键字

### 概念动机

在闭包中，如果内层函数需要修改外层变量，必须使用 `nonlocal` 关键字声明。

### 最简示例

```python
# ❌ 不使用 nonlocal（会出错）
def counter_wrong() -> Callable[[], int]:
    count = 0
    
    def increment() -> int:
        count += 1  # UnboundLocalError!
        return count
    
    return increment

# ✅ 使用 nonlocal（正确）
def counter() -> Callable[[], int]:
    count = 0
    
    def increment() -> int:
        nonlocal count  # 声明使用外层变量
        count += 1
        return count
    
    return increment

my_counter = counter()
print(my_counter())  # 1
print(my_counter())  # 2
print(my_counter())  # 3
```

### 详细讲解

```python
def demonstrate_nonlocal() -> None:
    """演示 nonlocal 的作用"""
    x = 10
    
    def read_only() -> int:
        # 不修改，直接读取是可以的
        return x * 2
    
    def modify_with_nonlocal() -> int:
        nonlocal x
        x = 20
        return x
    
    def try_modify_without_nonlocal() -> int:
        # 这会创建一个新的局部变量 x
        x = 30  # 这是局部变量，不影响外层 x
        return x
    
    print(f"原始 x: {x}")  # 10
    print(f"read_only: {read_only()}")  # 20
    print(f"修改后 x: {x}")  # 10（未改变）
    
    print(f"modify_with_nonlocal: {modify_with_nonlocal()}")  # 20
    print(f"修改后 x: {x}")  # 20（已改变）
```

### 渐进复杂化：多层嵌套

```python
def outer() -> Callable[[], None]:
    x = "outer"
    
    def middle() -> Callable[[], None]:
        y = "middle"
        
        def inner() -> None:
            nonlocal x, y  # 可以声明多个外层变量
            x = "modified outer"
            y = "modified middle"
            print(f"inner: x={x}, y={y}")
        
        inner()
        print(f"middle: x={x}, y={y}")
    
    middle()
    print(f"outer: x={x}")

# outer() 输出:
# inner: x=modified outer, y=modified middle
# middle: x=modified outer, y=modified middle
# outer: x=modified outer
```

## 第三部分：闭包的应用

### 应用一：数据隐藏/封装

```python
from typing import Callable

def create_account(initial_balance: float) -> dict[str, Callable]:
    """创建银行账户（使用闭包隐藏余额）"""
    balance = initial_balance  # 私有变量，外部无法直接访问
    
    def get_balance() -> float:
        return balance
    
    def deposit(amount: float) -> bool:
        nonlocal balance
        if amount > 0:
            balance += amount
            return True
        return False
    
    def withdraw(amount: float) -> bool:
        nonlocal balance
        if 0 < amount <= balance:
            balance -= amount
            return True
        return False
    
    return {
        "balance": get_balance,
        "deposit": deposit,
        "withdraw": withdraw
    }

# 使用
account = create_account(1000)
print(account["balance"]())  # 1000
account["deposit"](500)
print(account["balance"]())  # 1500
account["withdraw"](200)
print(account["balance"]())  # 1300
# balance 变量无法从外部直接访问！
```

**关键代码说明：**

| 代码 | 含义 | 为什么这样写 |
|------|------|-------------|
| `balance = initial_balance` | 外层函数的私有变量 | `balance` 只存在于闭包内，外部无法通过 `account.balance` 直接读写 |
| `nonlocal balance` | 声明修改外层 balance | `deposit`/`withdraw` 必须用 `nonlocal` 才能修改外层的 `balance` 而非创建局部变量 |
| `return {"balance": get_balance, "deposit": deposit, "withdraw": withdraw}` | 以字典返回多个闭包函数 | 提供类似"方法集合"的接口，无需定义类即可封装状态和行为 |
| `if 0 < amount <= balance` | 取款前验证金额合法性 | 同时检查金额大于 0 和不超过余额，防止负取款和透支 |

### 应用二：缓存（Memoization）

```python
from typing import Callable, TypeVar

T = TypeVar('T')

def memoize(func: Callable[..., T]) -> Callable[..., T]:
    """缓存装饰器：避免重复计算"""
    cache: dict[tuple, T] = {}
    
    def wrapper(*args: tuple) -> T:
        if args in cache:
            print(f"缓存命中: {args}")
            return cache[args]
        result = func(*args)
        cache[args] = result
        return result
    
    return wrapper

@memoize
def fibonacci(n: int) -> int:
    """计算斐波那契数"""
    if n < 2:
        return n
    return fibonacci(n - 1) + fibonacci(n - 2)

print(fibonacci(10))  # 55
print(fibonacci(10))  # 缓存命中: (10,) → 55
```

### 应用三：工厂函数

```python
from typing import Callable

def create_power_function(exponent: int) -> Callable[[int], int]:
    """创建幂函数工厂"""
    def power(base: int) -> int:
        return base ** exponent
    return power

# 创建不同的幂函数
square = create_power_function(2)
cube = create_power_function(3)

print(square(5))    # 25
print(cube(5))      # 125

# 更复杂的工厂：创建验证器
def create_validator(min_val: int, max_val: int) -> Callable[[int], bool]:
    """创建范围验证器"""
    def validate(value: int) -> bool:
        return min_val <= value <= max_val
    return validate

is_valid_age = create_validator(0, 120)
is_valid_percent = create_validator(0, 100)

print(is_valid_age(25))      # True
print(is_valid_age(150))      # False
print(is_valid_percent(85))   # True
```

### 应用四：延迟计算

```python
from typing import Callable, TypeVar

T = TypeVar('T')

def lazy_evaluation(func: Callable[[], T]) -> Callable[[], T]:
    """延迟计算：只在第一次调用时执行"""
    result: T | None = None
    computed = False
    
    def get_result() -> T:
        nonlocal result, computed
        if not computed:
            result = func()
            computed = True
        return result
    
    return get_result

# 使用
@lazy_evaluation
def expensive_computation() -> int:
    print("执行耗时计算...")
    return 42

print("准备获取结果")
print(expensive_computation())  # 执行耗时计算... 42
print(expensive_computation())  # 42（不再执行）
```

## L2 实践层：最佳实践

### 闭包最佳实践

#### 推荐做法

| 做法 | 原因 | 示例 |
|------|------|------|
| 用闭包封装私有状态 | 避免全局变量，数据隔离安全 | `create_counter()` 返回独立计数器 |
| 工厂函数返回定制函数 | 配置一次，多次使用相同设置 | `create_validator(min=0, max=100)` |
| 闭包替代简单类 | 代码更简洁，无类定义开销 | 只有一个方法的"类"用闭包更好 |
| 添加 `nonlocal` 前再修改外层变量 | 否则会创建同名局部变量，外层变量不变 | `nonlocal count; count += 1` |

#### 反模式：闭包陷阱

```python
# ❌ 常见陷阱：忘记 nonlocal（会创建局部变量）
def counter_wrong():
    count = 0

    def increment():
        count += 1  # UnboundLocalError: 局部变量未定义就使用
        return count

    return increment

# 问题：
# Python 认为任何赋值的变量都是局部变量
# count += 1 被解析为 count = count + 1
# 但 count 作为局部变量还未赋值，所以报错
```

```python
# ✅ 正确做法：使用 nonlocal 声明
def counter_correct():
    count = 0

    def increment():
        nonlocal count  # 明确声明使用外层变量
        count += 1
        return count

    return increment

my_counter = counter_correct()
print(my_counter())  # 1
print(my_counter())  # 2
```

#### 反模式：循环闭包陷阱

```python
# ❌ 循环闭包陷阱：所有函数共享同一个变量
def create_multipliers_wrong():
    multipliers = []
    for i in range(5):
        def multiplier(x):
            return x * i  # i 是循环变量，循环结束后 i = 4
        multipliers.append(multiplier)
    return multipliers

funcs = create_multipliers_wrong()
print(funcs[0](10))  # 40（期望 0，实际 4 * 10）
print(funcs[1](10))  # 40（期望 10，实际 4 * 10）
print(funcs[2](10))  # 40（期望 20，实际 4 * 10）

# 问题：
# 闭包捕获的是变量 i，而不是 i 的值
# 循环结束后 i = 4，所有闭包都使用这个值
```

```python
# ✅ 正确做法1：使用默认参数捕获值
def create_multipliers_correct1():
    multipliers = []
    for i in range(5):
        def multiplier(x, n=i):  # 默认参数在定义时求值
            return x * n
        multipliers.append(multiplier)
    return multipliers

funcs = create_multipliers_correct1()
print(funcs[0](10))  # 0
print(funcs[1](10))  # 10
print(funcs[2](10))  # 20

# ✅ 正确做法2：使用工厂函数创建独立闭包
def create_multipliers_correct2():
    def make_multiplier(n):
        def multiplier(x):
            return x * n  # 每次调用 make_multiplier 创建新闭包
        return multiplier

    return [make_multiplier(i) for i in range(5)]

funcs = create_multipliers_correct2()
print(funcs[0](10))  # 0
print(funcs[1](10))  # 10
```

#### 反模式：闭包持有大对象

```python
# ❌ 闭包意外持有大对象（内存泄漏风险）
def process_large_data(data: list):
    # large_data 被闭包捕获，即使只使用其中一小部分

    def get_item(index: int):
        return data[index]  # 整个 data 被闭包持有

    return get_item

# 问题：
# data 可能是巨大的列表，闭包持有它的引用
# 即使外部不再需要 data，它也无法被垃圾回收
# 因为闭包仍然持有它的引用
```

```python
# ✅ 正确做法：只捕获必要的数据
def process_large_data(data: list):
    # 只保存需要的数据，不保存整个列表
    needed_data = data.copy()[:100]  # 只取前 100 个

    def get_item(index: int):
        return needed_data[index] if index < 100 else None

    return get_item

# 或者：只捕获单个值
def create_value_holder(value: int):
    # 闭包只持有单个值，内存占用小
    def get_value():
        return value
    return get_value
```

#### 适用场景

| 场景 | 是否推荐 | 原因 |
|------|---------|------|
| 数据封装/私有变量 | ✅ 推荐 | 实现类似私有属性的效果 |
| 工厂函数 | ✅ 推荐 | 根据参数生成定制函数 |
| 缓存/Memoization | ✅ 推荐 | 缓存字典保存在闭包中 |
| 回调函数带状态 | ✅ 推荐 | GUI 事件处理需要记住状态 |
| 替代简单类 | ✅ 推荐 | 只有一个"方法"时更简洁 |
| 复杂状态管理 | ❌ 不推荐 | 多个状态用类更清晰 |
| 需要继承的场景 | ❌ 不推荐 | 闭包无法继承 |

### 闭包 vs 类的选择

```python
# 场景：需要一个带状态的"加法器"

# 方式1：用闭包
def create_adder(initial: int = 0):
    total = initial

    def add(value: int) -> int:
        nonlocal total
        total += value
        return total

    return add

adder = create_adder(10)
print(adder(5))   # 15
print(adder(3))   # 18

# 方式2：用类
class Adder:
    def __init__(self, initial: int = 0):
        self.total = initial

    def add(self, value: int) -> int:
        self.total += value
        return self.total

adder = Adder(10)
print(adder.add(5))   # 15
print(adder.add(3))   # 18

# 何时选择闭包：
# - 只有一个方法
# - 不需要继承
# - 状态简单（少数变量）

# 何时选择类：
# - 多个方法
# - 需要继承或多态
# - 状态复杂
# - 需要更正式的 API
```

| 对比项 | 闭包 | 类 |
|--------|------|-----|
| 代码量 | 较少 | 较多 |
| 可读性 | 单方法时清晰 | 多方法时更清晰 |
| 继承支持 | ❌ 不支持 | ✅ 支持 |
| 多方法支持 | ❌ 需返回字典 | ✅ 自然支持 |
| 状态访问 | ❌ 外部无法直接访问 | ✅ 可用属性访问 |
| 内存占用 | 略少（无类开销） | 略多（有类对象） |

## L3 专家层：底层原理

### __closure__ 属性详解

#### Python 如何实现

闭包通过 `__closure__` 属性存储捕获的变量：

```
闭包内部结构：
┌───────────────────────────────────────────────────────────────┐
│  闭包函数的 __closure__ 属性                                    │
├───────────────────────────────────────────────────────────────┤
│                                                                │
│  __closure__ 是一个元组，包含 cell 对象                        │
│                                                                │
│  每个 cell 对象结构：                                           │
│  ┌─────────────────────────────────────┐                      │
│  │  cell.cell_contents                 │                      │
│  │  → 存储被捕获变量的值               │                      │
│  └─────────────────────────────────────┘                      │
│                                                                │
│  示例：                                                         │
│  def outer():                                                  │
│      x = 10                                                    │
│      y = 20                                                    │
│      def inner():                                              │
│          return x + y                                          │
│      return inner                                              │
│                                                                │
│  inner.__closure__[0].cell_contents → 10 (x)                  │
│  inner.__closure__[1].cell_contents → 20 (y)                  │
│                                                                │
└───────────────────────────────────────────────────────────────┘
```

```python
# 演示 __closure__ 属性
def outer():
    x = 10
    y = 20

    def inner():
        return x + y  # 引用外层变量

    return inner

func = outer()

# 查看闭包属性
print(func.__closure__)  # (<cell at ...>, <cell at ...>)
print(len(func.__closure__))  # 2（捕获了两个变量）

# 访问捕获的值
print(func.__closure__[0].cell_contents)  # 10 (x)
print(func.__closure__[1].cell_contents)  # 20 (y)

# 对应的自由变量名
print(func.__code__.co_freevars)  # ('x', 'y')
```

### 自由变量捕获机制

#### Python 如何实现

自由变量是函数引用但不属于其局部作用域的变量：

```
自由变量捕获机制：
┌───────────────────────────────────────────────────────────────┐
│  代码对象中的相关属性                                           │
├───────────────────────────────────────────────────────────────┤
│                                                                │
│  co_freevars   → 内层函数的自由变量名元组                      │
│                  （内层引用外层的变量）                         │
│                                                                │
│  co_cellvars   → 外层函数的 cell 变量名元组                    │
│                  （外层被内层引用的变量）                       │
│                                                                │
│  捕获过程：                                                     │
│  ─────────────────────────────────────────                    │
│  1. 编译时检测哪些变量被嵌套函数引用                           │
│  2. 将这些变量标记为"cell变量"（存入 co_cellvars）             │
│  3. 运行时创建 cell 对象包装变量值                             │
│  4. 内层函数的 __closure__ 持有这些 cell 对象                  │
│  5. 内层通过 LOAD_DEREF 指令访问 cell 内容                     │
│                                                                │
└───────────────────────────────────────────────────────────────┘
```

```python
# 演示 co_freevars 和 co_cellvars
def outer():
    x = 10  # 被内层引用，是 cell 变量
    y = 20  # 被内层引用，是 cell 变量
    z = 30  # 不被内层引用，是普通局部变量

    def inner():
        a = 1  # inner 的局部变量
        return x + y  # x, y 是自由变量

    return inner

# 外层函数的 cell 变量
print(outer.__code__.co_cellvars)   # ('x', 'y')，不包括 z
print(outer.__code__.co_varnames)   # ('x', 'y', 'z', 'inner')

# 内层函数的自由变量
inner = outer()
print(inner.__code__.co_freevars)   # ('x', 'y')
print(inner.__code__.co_varnames)   # ('a')

# 使用 dis 查看字节码
import dis
print("=== outer 的字节码 ===")
dis.dis(outer)
# 包含 MAKE_CELL 指令（将 x, y 转为 cell）

print("=== inner 的字节码 ===")
dis.dis(inner)
# 包含 LOAD_DEREF 指令（从 cell 加载 x, y）
```

#### 性能考量

| 操作 | 时间复杂度 | 说明 |
|------|-----------|------|
| 创建闭包 | O(n) | n = 捕获的变量数量 |
| 闭包变量访问 | O(1) | LOAD_DEREF 直接索引 cell |
| 普通局部变量访问 | O(1) | LOAD_FAST 更快 |
| 闭包变量修改 | O(1) | STORE_DEREF 写入 cell |

```python
import timeit

# 测试闭包变量 vs 局部变量访问速度
def test_closure():
    x = 100

    def inner():
        return x + 1  # 访问闭包变量

    return inner

def test_local():
    x = 100
    return x + 1  # 访问局部变量

inner = test_closure()

print("闭包变量访问:", timeit.timeit(inner, number=1000000))
print("局部变量访问:", timeit.timeit(test_local, number=1000000))
# 局部变量访问更快（约快 5-10%）
# 因为 LOAD_FAST 比 LOAD_DEREF 简单
```

```python
# 内存占用对比
import sys

def create_closure():
    data = [1, 2, 3, 4, 5]  # 捕获列表

    def get_data():
        return data

    return get_data

def create_function():
    # 不捕获外层变量

    def get_data():
        return [1, 2, 3, 4, 5]  # 每次创建新列表

    return get_data

closure_func = create_closure()
normal_func = create_function()

# 闭包持有对 data 的引用，额外内存开销
print("闭包函数大小:", sys.getsizeof(closure_func))
print("普通函数大小:", sys.getsizeof(normal_func))
print("闭包 __closure__ 大小:", sys.getsizeof(closure_func.__closure__))

# 注意：闭包会持续持有被捕获对象的引用
# 可能导致对象无法被垃圾回收
```

#### 设计动机

Python 为什么用 cell 对象实现闭包：

| 设计选择 | 原因 | 影响 |
|----------|------|------|
| cell 对象包装变量 | 允许内层和外层共享变量 | `nonlocal` 可以修改外层变量 |
| `__closure__` 元组存储 | 固定大小，快速访问 | O(1) 时间复杂度访问闭包变量 |
| 编译时确定自由变量 | 运行时无需查找 | 性能优于动态查找 |
| cell 而非直接引用 | 支持变量值变化 | 外层修改后内层能看到新值 |

```python
# cell 机制允许内外层共享变量
def outer():
    x = 10

    def inner():
        print(f"inner 看到 x = {x}")

    def modify():
        nonlocal x
        x = 20

    return inner, modify

inner, modify = outer()
inner()  # inner 看到 x = 10
modify()  # 外层修改 x
inner()  # inner 看到 x = 20（cell 共享）

# 验证：inner 和 modify 共享同一个 cell
print(inner.__closure__[0] is modify.__closure__[0])  # True
```

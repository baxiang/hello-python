## 为什么需要函数进阶？

### 实际场景引入

想象你正在开发一个数据处理系统：

```python
# 问题场景：如何灵活地处理不同的计算？
def process_data(data: list[int], operation: str) -> list[int]:
    """根据操作类型处理数据"""
    if operation == "double":
        return [x * 2 for x in data]
    elif operation == "square":
        return [x ** 2 for x in data]
    elif operation == "negate":
        return [-x for x in data]
    # 每次新增操作都要修改这个函数...
```

**痛点**：
- 每增加一个操作就要修改 `process_data` 函数
- 代码变得越来越长，难以维护
- 无法动态添加新操作

**解决思路**：如果能把函数当作参数传递，问题就迎刃而解了！

## 第一部分：函数是对象

### 概念动机

在 Python 中，函数不仅是代码块，更是**一等公民**（First-class Citizen）。这意味着函数可以像普通数据一样被操作。

```
┌─────────────────────────────────────────┐
│       Python 函数是"一等公民"           │
├─────────────────────────────────────────┤
│                                         │
│  一等公民（First-class Citizen）        │
│  ─────────────────────────────          │
│  可以在程序中任意位置使用的对象         │
│                                         │
│  函数可以：                             │
│  ────────────                           │
│  ✅ 赋值给变量                          │
│  ✅ 作为参数传递给其他函数              │
│  ✅ 作为函数的返回值                    │
│  ✅ 存储在数据结构中（列表、字典）      │
│                                         │
└─────────────────────────────────────────┘
```

### 最简示例

```python
# 函数可以赋值给变量
def greet(name: str) -> str:
    return f"Hello, {name}!"

say_hello = greet  # 函数赋值
print(say_hello("Alice"))  # Hello, Alice!
```

### 详细讲解

```python
# ─────────────────────────────────────
# 1. 赋值给变量
# ─────────────────────────────────────
def greet(name: str) -> str:
    return f"Hello, {name}!"

say_hello = greet
print(say_hello("Alice"))  # Hello, Alice!

# ─────────────────────────────────────
# 2. 作为参数传递
# ─────────────────────────────────────
def execute(func: callable, value: str) -> str:
    """执行传入的函数"""
    return func(value)

result = execute(greet, "David")
print(result)  # Hello, David!

# ─────────────────────────────────────
# 3. 作为返回值
# ─────────────────────────────────────
def get_operation(op: str) -> callable:
    """返回不同的操作函数"""
    if op == "add":
        return lambda a, b: a + b
    elif op == "mul":
        return lambda a, b: a * b
    return lambda a, b: None

add_func = get_operation("add")
print(add_func(3, 5))  # 8

# ─────────────────────────────────────
# 4. 存储在数据结构中
# ─────────────────────────────────────
def add(a: int, b: int) -> int: 
    return a + b

def sub(a: int, b: int) -> int: 
    return a - b

def mul(a: int, b: int) -> int: 
    return a * b

op_dict: dict[str, callable] = {"+": add, "-": sub, "*": mul}
print(op_dict["+"](10, 5))  # 15
```

### 渐进复杂化

回到最初的场景，现在可以用函数作为参数来解决：

```python
from typing import Callable

def process_data(data: list[int], operation: Callable[[int], int]) -> list[int]:
    """使用传入的函数处理数据"""
    return [operation(x) for x in data]

# 定义不同的操作
def double(x: int) -> int:
    return x * 2

def square(x: int) -> int:
    return x ** 2

def negate(x: int) -> int:
    return -x

# 使用
data = [1, 2, 3, 4, 5]
print(process_data(data, double))   # [2, 4, 6, 8, 10]
print(process_data(data, square))   # [1, 4, 9, 16, 25]
print(process_data(data, negate))   # [-1, -2, -3, -4, -5]

# 甚至可以用 lambda
print(process_data(data, lambda x: x + 100))  # [101, 102, 103, 104, 105]
```

### 实际应用

```python
from typing import Callable

# 策略模式：灵活切换排序算法
def sort_users(users: list[dict], key_func: Callable[[dict], str | int]) -> list[dict]:
    """按指定键排序用户"""
    return sorted(users, key=key_func)

users = [
    {"name": "Alice", "age": 30},
    {"name": "Bob", "age": 25},
    {"name": "Charlie", "age": 35},
]

# 按年龄排序
by_age = sort_users(users, lambda u: u["age"])
print(by_age)  # [{'name': 'Bob', 'age': 25}, ...]

# 按名字排序
by_name = sort_users(users, lambda u: u["name"])
print(by_name)  # [{'name': 'Alice', 'age': 30}, ...]
```

## 第二部分：嵌套函数

### 概念动机

有时我们需要在函数内部定义辅助函数，这些辅助函数只在外层函数内部使用，不需要暴露给外部。

### 最简示例

```python
def outer() -> None:
    """外层函数"""
    print("进入外层函数")

    def inner() -> None:
        """内层函数"""
        print("  进入内层函数")
        print("  离开内层函数")

    inner()  # 调用内层函数
    print("离开外层函数")

outer()
# 进入外层函数
#   进入内层函数
#   离开内层函数
# 离开外层函数

# ❌ 内层函数在外部不可见
# inner()  # NameError!
```

### 详细讲解

嵌套函数的一个常见用途是**代码组织和封装**：

```python
def process_file(filename: str) -> dict[str, int]:
    """处理文件并返回统计信息"""
    
    def count_lines(content: str) -> int:
        """统计行数"""
        return len(content.splitlines())
    
    def count_words(content: str) -> int:
        """统计单词数"""
        return len(content.split())
    
    def count_chars(content: str) -> int:
        """统计字符数"""
        return len(content)
    
    with open(filename, 'r') as f:
        content = f.read()
    
    return {
        "lines": count_lines(content),
        "words": count_words(content),
        "chars": count_chars(content),
    }
```

### 渐进复杂化

嵌套函数还可以用于**延迟执行**：

```python
from typing import Callable

def create_timer() -> Callable[[], float]:
    """创建一个计时器"""
    import time
    start_time = time.time()
    
    def get_elapsed() -> float:
        """获取已经过的时间"""
        return time.time() - start_time
    
    return get_elapsed

timer = create_timer()
# ... 做一些操作 ...
print(f"已过时间: {timer():.2f} 秒")
```

### 实际应用

```python
from typing import Callable

def validate_input(rules: dict[str, Callable[[str], bool]]) -> Callable[[dict], dict]:
    """创建输入验证器"""
    
    def validate(data: dict) -> dict:
        """验证数据"""
        errors = []
        for field, validator in rules.items():
            if field in data:
                if not validator(data[field]):
                    errors.append(f"字段 '{field}' 验证失败")
            else:
                errors.append(f"缺少字段 '{field}'")
        return {"valid": len(errors) == 0, "errors": errors}
    
    return validate

# 创建验证器
user_validator = validate_input({
    "name": lambda x: len(x) >= 2,
    "email": lambda x: "@" in x,
    "age": lambda x: x.isdigit() and int(x) >= 0,
})

# 使用验证器
result = user_validator({"name": "Alice", "email": "alice@example.com", "age": "25"})
print(result)  # {'valid': True, 'errors': []}
```

## 第三部分：LEGB 作用域规则

### 概念动机

理解变量查找顺序对于编写正确的嵌套函数至关重要。

```
┌─────────────────────────────────────────┐
│         LEGB 作用域查找规则             │
├─────────────────────────────────────────┤
│                                         │
│  L - Local（局部作用域）                │
│      当前函数内部的变量                 │
│                                         │
│  E - Enclosing（封闭作用域）            │
│      外层嵌套函数的变量                 │
│                                         │
│  G - Global（全局作用域）               │
│      模块级别的变量                     │
│                                         │
│  B - Built-in（内置作用域）             │
│      Python 内置的变量和函数            │
│                                         │
│  查找过程：                             │
│  Local → Enclosing → Global → Built-in │
│                                         │
└─────────────────────────────────────────┘
```

### 最简示例

```python
x = "全局变量"  # Global

def outer() -> None:
    y = "外层变量"  # Enclosing

    def inner() -> None:
        z = "内层变量"  # Local
        print(f"inner: {z}")    # Local
        print(f"inner: {y}")    # Enclosing
        print(f"inner: {x}")    # Global

    inner()

outer()
```

### 详细讲解

```python
# 完整的 LEGB 演示
x = "global"

def outer() -> None:
    y = "enclosing"
    
    def inner() -> None:
        z = "local"
        
        # 查找顺序：Local → Enclosing → Global → Built-in
        print(z)  # local (Local)
        print(y)  # enclosing (Enclosing)
        print(x)  # global (Global)
        print(len)  # <built-in function len> (Built-in)
    
    inner()
```

### 渐进复杂化：变量遮蔽

```python
x = "global"

def outer() -> None:
    x = "enclosing"  # 遮蔽全局 x
    
    def inner() -> None:
        x = "local"  # 遮蔽外层 x
        print(x)  # local
    
    inner()
    print(x)  # enclosing

outer()
print(x)  # global
```

### 实际应用

```python
# 配置管理：使用作用域隔离配置
DEFAULT_CONFIG = {"debug": False, "log_level": "INFO"}

def create_service(config: dict) -> Callable[[], dict]:
    """创建服务，配置被封闭在闭包中"""
    # 合并默认配置和用户配置
    final_config = {**DEFAULT_CONFIG, **config}
    
    def get_config() -> dict:
        """获取当前配置"""
        return final_config.copy()
    
    return get_config

service_a = create_service({"debug": True})
service_b = create_service({"log_level": "DEBUG"})

print(service_a())  # {'debug': True, 'log_level': 'INFO'}
print(service_b())  # {'debug': False, 'log_level': 'DEBUG'}
```

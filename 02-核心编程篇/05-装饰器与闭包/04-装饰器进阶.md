# 装饰器进阶

> **Python 版本要求**：本章内容需要 Python 3.11 或更高版本
本章讲解 Python 装饰器的高级用法，包括带参数的装饰器、多层装饰器和类装饰器。

---

## 为什么需要高级装饰器？

### 实际场景引入

基础装饰器虽然强大，但在实际开发中还需要更多灵活性：

```python
# 问题场景：需要让装饰器支持配置
@timer  # 只能固定输出格式
def process_data():
    pass

# 如何实现这些需求？
@retry(times=5, delay=2.0)  # 带参数的装饰器

    pass

@auth_required
@log_execution
@cache_result  # 多个装饰器叠加
def get_user():
    pass

@singleton  # 用类实现的装饰器
class Database:
    pass
```

**痛点**：
- 基础装饰器无法接收配置参数
- 多个装饰器如何组合使用？
- 有些场景需要用类来实现装饰器

## 第一部分：带参数的装饰器

### 概念动机

带参数的装饰器需要三层嵌套：最外层接收装饰器参数，中间层接收函数，最内层执行包装逻辑。

### 最简示例

```python
from typing import Callable, TypeVar
from functools import wraps

T = TypeVar('T')

def repeat(times: int) -> Callable[[Callable[..., T]], Callable[..., T]]:
    """重复执行装饰器"""
    def decorator(func: Callable[..., T]) -> Callable[..., T]:
        @wraps(func)
        def wrapper(*args: tuple, **kwargs: dict) -> T:
            result: T | None = None
            for _ in range(times):
                result = func(*args, **kwargs)
            return result  # type: ignore
        return wrapper
    return decorator

# 使用
@repeat(times=3)
def greet(name: str) -> None:
    print(f"Hello, {name}!")

greet("Alice")
# Hello, Alice!
# Hello, Alice!
# Hello, Alice!
```

### 详细讲解：执行顺序

```
@repeat(times=3)
def greet(name):
    ...

执行过程：

1. 调用 repeat(times=3)
   返回 decorator 函数
          │
2. 用 decorator 包装 greet
   greet = decorator(greet)
   返回 wrapper 函数
          │
3. 调用 greet("Alice")
   实际执行 wrapper("Alice")
```

```python
# 等价写法
def greet(name: str) -> None:
    print(f"Hello, {name}!")

greet = repeat(times=3)(greet)
```

### 渐进复杂化：可配置的日志装饰器

```python
from typing import Callable, TypeVar
from functools import wraps
import logging

T = TypeVar('T')

def log(
    level: int = logging.INFO,
    prefix: str = "[LOG]",
    show_args: bool = True
) -> Callable[[Callable[..., T]], Callable[..., T]]:
    """可配置的日志装饰器"""
    def decorator(func: Callable[..., T]) -> Callable[..., T]:
        @wraps(func)
        def wrapper(*args: tuple, **kwargs: dict) -> T:
            # 前置日志
            if show_args:
                logging.log(level, f"{prefix} 调用 {func.__name__}({args}, {kwargs})")
            else:
                logging.log(level, f"{prefix} 调用 {func.__name__}")
            
            # 执行函数
            result = func(*args, **kwargs)
            
            # 后置日志
            logging.log(level, f"{prefix} {func.__name__} 返回 {result}")
            return result
        return wrapper
    return decorator

# 使用不同配置
@log(level=logging.DEBUG, prefix="[DEBUG]")
def debug_func(x: int) -> int:
    return x * 2

@log(level=logging.WARNING, show_args=False)
def warn_func(x: int) -> int:
    return x * 3
```

### 实际应用：重试装饰器增强版

```python
from typing import Callable, TypeVar
from functools import wraps
import time

T = TypeVar('T')

def retry(
    times: int = 3,
    delay: float = 1.0,
    exceptions: tuple[type[Exception], ...] = (Exception,)
) -> Callable[[Callable[..., T]], Callable[..., T]]:
    """可配置的重试装饰器"""
    def decorator(func: Callable[..., T]) -> Callable[..., T]:
        @wraps(func)
        def wrapper(*args: tuple, **kwargs: dict) -> T:
            last_exception: Exception | None = None
            for attempt in range(times):
                try:
                    return func(*args, **kwargs)
                except exceptions as e:
                    last_exception = e
                    print(f"尝试 {attempt + 1}/{times} 失败：{e}")
                    if attempt < times - 1:
                        time.sleep(delay)
            raise last_exception  # type: ignore
        return wrapper
    return decorator

# 使用
@retry(times=5, delay=0.5, exceptions=(ConnectionError, TimeoutError))
def fetch_api(url: str) -> dict:
    import random
    if random.random() < 0.5:
        raise ConnectionError("网络错误")
    return {"status": "ok"}

# fetch_api("https://api.example.com")
```

## 第二部分：多层装饰器

### 概念动机

多个装饰器可以叠加使用，形成"洋葱模型"的执行顺序。

### 最简示例

```python
from typing import Callable, TypeVar
from functools import wraps

T = TypeVar('T')

def decorator1(func: Callable[..., T]) -> Callable[..., T]:
    @wraps(func)
    def wrapper(*args: tuple, **kwargs: dict) -> T:
        print("[装饰器 1] 前处理")
        result = func(*args, **kwargs)
        print("[装饰器 1] 后处理")
        return result
    return wrapper

def decorator2(func: Callable[..., T]) -> Callable[..., T]:
    @wraps(func)
    def wrapper(*args: tuple, **kwargs: dict) -> T:
        print("[装饰器 2] 前处理")
        result = func(*args, **kwargs)
        print("[装饰器 2] 后处理")
        return result
    return wrapper

@decorator1
@decorator2
def greet(name: str) -> None:
    print(f"  Hello, {name}!")

greet("Alice")
# [装饰器 1] 前处理
# [装饰器 2] 前处理
#   Hello, Alice!
# [装饰器 2] 后处理
# [装饰器 1] 后处理
```

### 详细讲解：装饰顺序与执行顺序

```
@decorator1
@decorator2
def greet(name):
    ...

装饰顺序（从下到上）：
1. greet 先被 decorator2 装饰
2. 结果再被 decorator1 装饰

等价于：
greet = decorator1(decorator2(greet))

调用时的执行顺序（洋葱模型）：
decorator1 前 → decorator2 前 → greet() → decorator2 后 → decorator1 后

┌─────────────────────────────┐
│   decorator1 前处理          │
│   ┌─────────────────────┐   │
│   │   decorator2 前处理  │   │
│   │   ┌─────────────┐   │   │
│   │   │   greet()   │   │   │
│   │   └─────────────┘   │   │
│   │   decorator2 后处理  │   │
│   └─────────────────────┘   │
│   decorator1 后处理          │
└─────────────────────────────┘
```

### 渐进复杂化：实际应用

```python
from typing import Callable, TypeVar
from functools import wraps
import time
import logging

T = TypeVar('T')

def timer(func: Callable[..., T]) -> Callable[..., T]:
    """计时装饰器"""
    @wraps(func)
    def wrapper(*args: tuple, **kwargs: dict) -> T:
        start = time.time()
        result = func(*args, **kwargs)
        print(f"⏱ {func.__name__} 耗时: {time.time() - start:.2f}s")
        return result
    return wrapper

def logger(func: Callable[..., T]) -> Callable[..., T]:
    """日志装饰器"""
    @wraps(func)
    def wrapper(*args: tuple, **kwargs: dict) -> T:
        print(f"📝 调用 {func.__name__}")
        result = func(*args, **kwargs)
        print(f"✅ {func.__name__} 完成")
        return result
    return wrapper

def cache(func: Callable[..., T]) -> Callable[..., T]:
    """缓存装饰器"""
    cached: dict[tuple, T] = {}
    
    @wraps(func)
    def wrapper(*args: tuple, **kwargs: dict) -> T:
        if args in cached:
            print(f"🎯 缓存命中: {args}")
            return cached[args]
        result = func(*args, **kwargs)
        cached[args] = result
        return result
    return wrapper

# 组合使用
@timer
@logger
@cache
def fibonacci(n: int) -> int:
    if n < 2:
        return n
    return fibonacci(n - 1) + fibonacci(n - 2)

print(fibonacci(10))
# 执行顺序：timer → logger → cache → fibonacci
```

### 实际应用：Web 框架中的装饰器组合

```python
from typing import Callable, TypeVar
from functools import wraps

T = TypeVar('T')

class User:
    def __init__(self, name: str, is_authenticated: bool, is_admin: bool):
        self.name = name
        self.is_authenticated = is_authenticated
        self.is_admin = is_admin

def require_auth(func: Callable[..., T]) -> Callable[..., T]:
    """需要登录"""
    @wraps(func)
    def wrapper(user: User, *args: tuple, **kwargs: dict) -> T:
        if not user.is_authenticated:
            raise PermissionError("请先登录")
        return func(user, *args, **kwargs)
    return wrapper

def require_admin(func: Callable[..., T]) -> Callable[..., T]:
    """需要管理员权限"""
    @wraps(func)
    def wrapper(user: User, *args: tuple, **kwargs: dict) -> T:
        if not user.is_admin:
            raise PermissionError("需要管理员权限")
        return func(user, *args, **kwargs)
    return wrapper

def log_action(func: Callable[..., T]) -> Callable[..., T]:
    """记录操作"""
    @wraps(func)
    def wrapper(user: User, *args: tuple, **kwargs: dict) -> T:
        print(f"[审计] {user.name} 执行 {func.__name__}")
        return func(user, *args, **kwargs)
    return wrapper

# 组合使用：权限检查在前，日志在后
@log_action
@require_admin
@require_auth
def delete_user(user: User, target_id: int) -> bool:
    print(f"删除用户 {target_id}")
    return True

# 使用
admin = User("Admin", is_authenticated=True, is_admin=True)
delete_user(admin, 123)  # [审计] Admin 执行 delete_user
```

## 第三部分：类装饰器

### 概念动机

类装饰器使用类的 `__call__` 方法实现，适合需要维护状态的场景。

### 最简示例

```python
from functools import wraps
from typing import Any

class CountCalls:
    """统计函数调用次数的装饰器"""

    def __init__(self, func: Callable[..., Any]) -> None:
        wraps(func)(self)
        self.func = func
        self.count = 0

    def __call__(self, *args: tuple, **kwargs: dict) -> Any:
        self.count += 1
        print(f"第 {self.count} 次调用 {self.__name__}")
        return self.func(*args, **kwargs)

@CountCalls
def say_hello() -> None:
    print("Hello!")

say_hello()  # 第 1 次调用 say_hello
say_hello()  # 第 2 次调用 say_hello
say_hello()  # 第 3 次调用 say_hello
print(f"总调用次数: {say_hello.count}")  # 总调用次数: 3
```

### 详细讲解：带参数的类装饰器

```python
from functools import wraps
from typing import Any, Callable

class Repeat:
    """重复执行指定次数的类装饰器"""

    def __init__(self, times: int) -> None:
        self.times = times

    def __call__(self, func: Callable[..., Any]) -> Callable[..., Any]:
        @wraps(func)
        def wrapper(*args: tuple, **kwargs: dict) -> Any:
            result: Any = None
            for _ in range(self.times):
                result = func(*args, **kwargs)
            return result
        return wrapper

@Repeat(times=3)
def greet(name: str) -> None:
    print(f"Hello, {name}!")

greet("Alice")
# Hello, Alice!
# Hello, Alice!
# Hello, Alice!
```

### 渐进复杂化：状态管理

```python
from typing import Any, Callable

class RateLimiter:
    """限流装饰器：限制函数调用频率"""

    def __init__(self, calls_per_second: float) -> None:
        self.min_interval = 1.0 / calls_per_second
        self.last_call_time = 0.0

    def __call__(self, func: Callable[..., Any]) -> Callable[..., Any]:
        import time
        
        def wrapper(*args: tuple, **kwargs: dict) -> Any:
            current_time = time.time()
            elapsed = current_time - self.last_call_time
            if elapsed < self.min_interval:
                wait_time = self.min_interval - elapsed
                print(f"限流中，等待 {wait_time:.2f}s")
                time.sleep(wait_time)
            
            self.last_call_time = time.time()
            return func(*args, **kwargs)
        
        return wrapper

@RateLimiter(calls_per_second=2)  # 每秒最多2次
def api_call(endpoint: str) -> dict:
    return {"endpoint": endpoint, "status": "ok"}

# 快速调用会被限流
api_call("/users")
api_call("/users")  # 可能需要等待
```

### 实际应用：单例模式

```python
from typing import Any, Callable, TypeVar

T = TypeVar('T')

def singleton(cls: type[T]) -> Callable[..., T]:
    """单例装饰器"""
    instances: dict[type, Any] = {}

    def get_instance(*args: tuple, **kwargs: dict) -> T:
        if cls not in instances:
            instances[cls] = cls(*args, **kwargs)
        return instances[cls]

    return get_instance

@singleton
class Database:
    def __init__(self) -> None:
        print("创建数据库连接")
        self.connection = "db_connection"

# 测试
db1 = Database()  # 创建数据库连接
db2 = Database()  # 不会再次创建
print(db1 is db2)  # True
```

**关键代码说明：**

| 代码 | 含义 | 为什么这样写 |
|------|------|-------------|
| `instances: dict[type, Any] = {}` | 在装饰器函数内保存实例的字典 | 闭包捕获 `instances`，跨调用持久保存每个类的单例，不污染全局命名空间 |
| `if cls not in instances: instances[cls] = cls(...)` | 第一次时才创建实例 | 已有实例则直接返回，保证同一类只有一个实例 |
| `return get_instance` | 返回替换类的调用函数 | `@singleton` 后 `Database` 变量指向 `get_instance`，`Database()` 实际调用 `get_instance()` |
| `@singleton` 装饰类而非函数 | 类装饰器 | 装饰器可以接受类作为参数，对类的行为（实例化）进行包装 |

### 模式二：属性验证装饰器

```python
from typing import Callable, TypeVar

T = TypeVar('T')

def validate(*types: type) -> Callable[[Callable[..., T]], Callable[..., T]]:
    """参数类型验证装饰器"""
    def decorator(func: Callable[..., T]) -> Callable[..., T]:
        def wrapper(*args: tuple, **kwargs: dict) -> T:
            for i, (arg, expected_type) in enumerate(zip(args, types)):
                if not isinstance(arg, expected_type):
                    raise TypeError(
                        f"参数 {i + 1} 应该是 {expected_type.__name__}，"
                        f"但得到的是 {type(arg).__name__}"
                    )
            return func(*args, **kwargs)
        return wrapper
    return decorator

@validate(int, str)
def create_user(user_id: int, name: str) -> dict:
    return {"id": user_id, "name": name}

print(create_user(1, "Alice"))  # {'id': 1, 'name': 'Alice'}
# create_user("1", "Alice")  # TypeError: 参数 1 应该是 int，但得到的是 str
```

**关键代码说明：**

| 代码 | 含义 | 为什么这样写 |
|------|------|-------------|
| `def validate(*types: type)` | 带参数的装饰器最外层接收类型列表 | 三层嵌套：`validate(int, str)` 返回装饰器，装饰器返回包装函数 |
| `for i, (arg, expected_type) in enumerate(zip(args, types))` | 同时遍历实参和期望类型 | `zip` 将两者配对，`enumerate` 提供索引用于错误提示 |
| `if not isinstance(arg, expected_type): raise TypeError(...)` | 类型不匹配时抛出类型错误 | 在函数执行前验证，比运行中途出错提供更清晰的错误信息 |

## L2 实践层：最佳实践

### 带参数装饰器设计

#### 推荐做法

| 做法 | 原因 | 示例 |
|------|------|------|
| 参数有合理默认值 | 使用更灵活，无需每次指定所有参数 | `@retry(times=3, delay=1.0)` |
| 保持三层结构清晰 | 每层职责分明，易于理解 | 参数层 → 装饰器层 → 包装层 |
| 使用 functools.partial 简化 | 减少嵌套层级，代码更简洁 | 复杂场景可用 partial |
| 参数类型明确标注 | IDE 提供更好的类型提示 | `def retry(times: int = 3) -> ...` |

#### 反模式：参数装饰器过于复杂

```python

# ✅ 正确做法：拆分为多个单一职责装饰器
from functools import wraps
import time
import logging

def log(level: int = logging.INFO, prefix: str = "[LOG]"):
    """日志装饰器（单一职责）"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            logging.log(level, f"{prefix} 调用 {func.__name__}")
            result = func(*args, **kwargs)
            logging.log(level, f"{prefix} {func.__name__} 完成")
            return result
        return wrapper
    return decorator

def retry(times: int = 3, delay: float = 1.0):
    """重试装饰器（单一职责）"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(times):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if attempt < times - 1:
                        time.sleep(delay)
                    else:
                        raise
        return wrapper
    return decorator

# 组合使用，按需选择
@log(level=logging.DEBUG)
@retry(times=5, delay=0.5)
def fetch_api(url: str):

# ✅ 正确做法：参数可配置
def retry(times: int = 3):
    """可配置重试次数"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(times):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if attempt == times - 1:
                        raise
                    time.sleep(0.5)
        return wrapper
    return decorator

# 根据场景灵活配置
@retry(times=3)  # 普通接口
def fetch_user_data():
    pass

@retry(times=10)  # 关键接口，更多重试
def fetch_payment_data():
    pass
```

#### 适用场景

| 场景 | 是否推荐 | 原因 |
|------|---------|------|
| 重试机制 | ✅ 推荐 | 不同接口需要不同重试策略 |
| 日志级别配置 | ✅ 推荐 | 开发/生产环境日志级别不同 |
| 缓存大小配置 | ✅ 推荐 | 不同数据量需要不同缓存 |
| 权限级别配置 | ✅ 推荐 | 不同接口权限要求不同 |
| 简单固定功能 | ❌ 不推荐 | 不需要参数时用普通装饰器 |

### 多层装饰器堆叠顺序

#### 推荐做法

| 做法 | 原因 | 示例 |
|------|------|------|
| 外层放通用功能 | 外层先执行，适合通用处理 | `@log` 放在最外层 |
| 内层放核心功能 | 内层离原函数最近 | `@cache` 放在最内层 |
| 权限检查靠外 | 先验证权限，避免无效执行 | `@require_auth` 放在靠外位置 |
| 缓存靠内 | 缓存结果，避免重复执行装饰器 | `@cache` 放在靠内位置 |

#### 装饰器堆叠顺序图解

```
装饰器堆叠的最佳顺序：
┌───────────────────────────────────────────────────────────────┐
│  推荐顺序（从外到内）                                           │
├───────────────────────────────────────────────────────────────┤
│                                                                │
│  @log              ← 最外层：日志（通用监控）                   │
│  @timer            ← 中外层：计时（性能监控）                   │
│  @require_auth     ← 中层：权限（前置检查，失败则终止）         │
│  @retry            ← 中内层：重试（容错处理）                   │
│  @cache            ← 最内层：缓存（结果优化）                   │
│  def my_func():    ← 原函数                                    │
│                                                                │
│  执行顺序（洋葱模型）：                                         │
│  ─────────────────────────────────────────                    │
│  log 前 → timer 前 → auth 前 → retry 前 → cache 前            │
│      → func() → cache 后 → retry 后 → auth 后                 │
│      → timer 后 → log 后                                       │
│                                                                │
│  关键原则：                                                     │
│  ─────────────────────────────────────────                    │
│  • 权限检查失败 → 不执行后续装饰器和原函数                      │
│  • 缓存命中 → 直接返回，不执行原函数                            │
│  • 日志在外 → 记录完整执行过程                                 │
│                                                                │
└───────────────────────────────────────────────────────────────┘
```

```python
# 正确示例：装饰器堆叠顺序
from functools import wraps
import time
import logging

def log_execution(func):
    """日志装饰器"""
    @wraps(func)
    def wrapper(*args, **kwargs):
        logging.info(f"开始执行 {func.__name__}")
        result = func(*args, **kwargs)
        logging.info(f"执行完成 {func.__name__}")
        return result
    return wrapper

def timer(func):
    """计时装饰器"""
    @wraps(func)
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        print(f"耗时: {time.time() - start:.2f}s")
        return result
    return wrapper

def require_auth(func):
    """权限装饰器"""
    @wraps(func)
    def wrapper(user, *args, **kwargs):
        if not user.is_authenticated:
            raise PermissionError("未授权")
        return func(user, *args, **kwargs)
    return wrapper

def cache_result(func):
    """缓存装饰器"""
    cache = {}
    @wraps(func)
    def wrapper(*args, **kwargs):
        key = str(args)
        if key in cache:
            print("缓存命中")
            return cache[key]
        result = func(*args, **kwargs)
        cache[key] = result
        return result
    return wrapper

# 推荐顺序：日志 → 计时 → 权限 → 缓存 → 原函数
@log_execution
@timer
@require_auth
@cache_result
def get_user_data(user, user_id):
    """获取用户数据"""
    return {"id": user_id, "name": "Alice"}
```

#### 反模式：装饰器顺序错误

```python
# ❌ 缓存放在权限外面（安全风险）
@cache_result  # 缓存可能保存未授权访问的结果！
@require_auth
def get_sensitive_data(user, data_id):
    return {"secret": "data"}

# 问题：
# 未授权用户的请求可能被缓存
# 其他用户可能获取到未授权用户缓存的数据
# 安全漏洞！

# ❌ 日志放在最内层（无法记录完整流程）
@require_auth
@timer
@log_execution  # 无法记录权限检查失败的情况
def process_data(user, data):
    return data

# 问题：
# 权限检查失败时，日志不会记录
# 无法追踪完整的请求处理流程
```

```python
# ✅ 正确顺序：安全优先
@log_execution   # 最外层：记录所有请求
@timer           # 计时
@require_auth    # 权限检查（失败则终止）
@cache_result    # 只有授权请求才缓存
def get_sensitive_data(user, data_id):
    return {"secret": "data"}

# 这样：
# 1. 所有请求都被日志记录（包括未授权的）
# 2. 权限检查失败时，后续装饰器不执行
# 3. 只有授权请求的结果被缓存
```

## L3 专家层：底层原理

### 类装饰器的内部实现

#### Python 如何实现

类装饰器利用 `__call__` 方法使类实例可调用：

```
类装饰器内部结构：
┌───────────────────────────────────────────────────────────────┐
│  类装饰器的执行过程                                            │
├───────────────────────────────────────────────────────────────┤
│                                                                │
│  @ClassDecorator                                               │
│  def my_func():                                                │
│      pass                                                      │
│                                                                │
│  执行过程：                                                     │
│  ─────────────────────────────────────────                    │
│  1. Python 定义 my_func                                        │
│  2. 调用 ClassDecorator(my_func)                               │
│     → 创建类实例：instance = ClassDecorator(my_func)           │
│  3. my_func = instance（函数名绑定到实例）                     │
│                                                                │
│  调用过程：                                                     │
│  ─────────────────────────────────────────                    │
│  my_func(args)                                                 │
│     ↓                                                          │
│  调用 instance.__call__(args)                                  │
│     ↓                                                          │
│  __call__ 内部调用 self.func(args)                            │
│                                                                │
│  关键点：                                                       │
│  ─────────────────────────────────────────                    │
│  • __init__ 接收并保存原函数                                   │
│  • __call__ 实现包装逻辑                                       │
│  • 实例属性可用于保存状态                                      │
│                                                                │
└───────────────────────────────────────────────────────────────┘
```

```python
# 演示类装饰器的内部结构
from functools import wraps

class CountCalls:
    """统计调用次数的类装饰器"""

    def __init__(self, func):
        # 使用 wraps 保留元信息
        wraps(func)(self)
        self.func = func
        self.count = 0  # 实例属性保存状态
        print(f"CountCalls.__init__ 被调用，保存 {func.__name__}")

    def __call__(self, *args, **kwargs):
        print(f"CountCalls.__call__ 被调用")
        self.count += 1
        print(f"第 {self.count} 次调用 {self.__name__}")
        return self.func(*args, **kwargs)

@CountCalls
def greet(name):
    """打招呼"""
    return f"Hello, {name}!"

# 模块加载时输出：
# CountCalls.__init__ 被调用，保存 greet

# greet 现在是 CountCalls 实例
print(f"greet 的类型: {type(greet)}")  # <class 'CountCalls'>
print(f"greet.__name__: {greet.__name__}")  # 'greet'（wraps 保留）

# 调用时执行 __call__
result = greet("Alice")
# CountCalls.__call__ 被调用
# 第 1 次调用 greet

# 访问实例属性
print(f"调用次数: {greet.count}")  # 1
```

#### 带参数的类装饰器

```python
# 带参数的类装饰器：__init__ 接收参数，__call__ 接收函数
from functools import wraps

class Repeat:
    """重复执行指定次数"""

    def __init__(self, times: int):
        # __init__ 接收装饰器参数
        self.times = times
        print(f"Repeat.__init__ 参数: times={times}")

    def __call__(self, func):
        # __call__ 接收被装饰的函数
        print(f"Repeat.__call__ 被调用，装饰 {func.__name__}")
        @wraps(func)
        def wrapper(*args, **kwargs):
            result = None
            for _ in range(self.times):
                result = func(*args, **kwargs)
            return result
        return wrapper

@Repeat(times=3)
def greet(name):
    print(f"Hello, {name}!")

# 执行过程：
# 1. Repeat.__init__ 参数: times=3（创建实例）
# 2. Repeat.__call__ 被调用，装饰 greet（实例调用 __call__）
# 3. greet = wrapper（函数被替换）

greet("Alice")
# Hello, Alice!
# Hello, Alice!
# Hello, Alice!
```

#### 性能考量

| 操作 | 时间复杂度 | 说明 |
|------|-----------|------|
| 类装饰器创建 | O(1) | 创建类实例，保存函数 |
| 类装饰器调用 | O(1) + 原函数 | 调用 __call__，再调用原函数 |
| 实例属性访问 | O(1) | 直接访问 self.attr |
| 状态管理 | O(1) | 类天然支持状态管理 |

```python
import timeit

# 性能对比：函数装饰器 vs 类装饰器
def func_decorator(func):
    count = 0
    def wrapper(*args, **kwargs):
        nonlocal count
        count += 1
        return func(*args, **kwargs)
    wrapper.count = lambda: count
    return wrapper

class ClassDecorator:
    def __init__(self, func):
        self.func = func
        self.count = 0

    def __call__(self, *args, **kwargs):
        self.count += 1
        return self.func(*args, **kwargs)

@func_decorator
def func_decorated(n):
    return sum(range(n))

@ClassDecorator
def class_decorated(n):
    return sum(range(n))

# 性能测试
n = 1000
iterations = 100000

print("函数装饰器:", timeit.timeit(lambda: func_decorated(n), number=iterations))
print("类装饰器:", timeit.timeit(lambda: class_decorated(n), number=iterations))

# 结论：
# 1. 类装饰器调用性能与函数装饰器相当
# 2. 类装饰器状态管理更直观（self.count）
# 3. 复杂场景推荐用类装饰器
```

#### 设计动机

为什么需要类装饰器：

| 设计选择 | 原因 | 影响 |
|----------|------|------|
| 类可以有状态 | 实例属性天然支持状态管理 | 调用计数、缓存更直观 |
| __call__ 使实例可调用 | 统一函数和类的调用语法 | `instance()` 等同于 `function()` |
| 可继承和扩展 | 类支持继承和多态 | 装饰器可复用和扩展 |
| 可定义多个方法 | 类可以有辅助方法 | 调用 `decorator.reset()` 等 |

```python
# 类装饰器的优势：可扩展和可继承
from functools import wraps

class BaseDecorator:
    """装饰器基类"""
    def __init__(self, func):
        wraps(func)(self)
        self.func = func

    def __call__(self, *args, **kwargs):
        return self.func(*args, **kwargs)

class CountingDecorator(BaseDecorator):
    """带计数的装饰器（继承基类）"""
    def __init__(self, func):
        super().__init__(func)
        self.count = 0

    def __call__(self, *args, **kwargs):
        self.count += 1
        return super().__call__(*args, **kwargs)

    def reset(self):
        """重置计数（类方法）"""
        self.count = 0

    def get_stats(self):
        """获取统计信息（类方法）"""
        return {"name": self.__name__, "count": self.count}

@CountingDecorator
def process(data):
    return data.upper()

# 调用
process("hello")
process("world")

# 使用类方法
print(process.get_stats())  # {'name': 'process', 'count': 2}
process.reset()
print(process.get_stats())  # {'name': 'process', 'count': 0}
```

### 装饰器堆叠的执行原理

#### Python 如何实现

多个装饰器堆叠时，按从下到上的顺序依次应用：

```
装饰器堆叠的内部执行：
┌───────────────────────────────────────────────────────────────┐
│  @decorator1                                                  │
│  @decorator2                                                  │
│  @decorator3                                                  │
│  def my_func():                                               │
│      pass                                                     │
│                                                                │
│  应用顺序（从下到上）：                                         │
│  ─────────────────────────────────────────                    │
│  1. my_func 被定义                                            │
│  2. my_func = decorator3(my_func)     → wrapper3              │
│  3. my_func = decorator2(wrapper3)    → wrapper2              │
│  4. my_func = decorator1(wrapper2)    → wrapper1              │
│                                                                │
│  最终：my_func = wrapper1                                     │
│                                                                │
│  调用顺序（洋葱模型）：                                         │
│  ─────────────────────────────────────────                    │
│  my_func(args)                                                │
│     ↓                                                          │
│  wrapper1(args)                                               │
│     ↓                                                          │
│  wrapper2(args)                                               │
│     ↓                                                          │
│  wrapper3(args)                                               │
│     ↓                                                          │
│  原函数(args)                                                 │
│     ↓                                                          │
│  返回值层层向上传递                                            │
│                                                                │
└───────────────────────────────────────────────────────────────┘
```

```python
# 演示装饰器堆叠的执行过程
from functools import wraps

def outer(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        print("outer: 前")
        result = func(*args, **kwargs)
        print("outer: 后")
        return result
    return wrapper

def middle(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        print("middle: 前")
        result = func(*args, **kwargs)
        print("middle: 后")
        return result
    return wrapper

def inner(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        print("inner: 前")
        result = func(*args, **kwargs)
        print("inner: 后")
        return result
    return wrapper

@outer  # 最后应用
@middle  # 中间应用
@inner  # 最先应用
def my_func():
    print("my_func: 执行")
    return "结果"

# 等价于：
# my_func = inner(my_func)      → inner_wrapper
# my_func = middle(inner_wrapper) → middle_wrapper
# my_func = outer(middle_wrapper) → outer_wrapper

my_func()
# 输出：
# outer: 前
# middle: 前
# inner: 前
# my_func: 执行
# inner: 后
# middle: 后
# outer: 后
```

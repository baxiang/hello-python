# 多态

> **Python 版本要求：** 3.11+

本章讲解 Python 面向对象编程中的多态机制，包括多态基础、鸭子类型和抽象基类。

---

## 为什么需要多态？

### 问题场景

假设你要开发一个游戏，有不同类型的角色，每个角色都能攻击：

**不使用多态：**

```python
class Warrior:
    def attack(self) -> str:
        return "战士挥剑"

class Mage:
    def attack(self) -> str:
        return "法师施法"

class Archer:
    def attack(self) -> str:
        return "弓箭手射箭"

def perform_attack(character_type: str, name: str) -> None:
    """需要判断类型，很繁琐"""
    if character_type == "warrior":
        print(f"{name} 挥剑")
    elif character_type == "mage":
        print(f"{name} 施法")
    elif character_type == "archer":
        print(f"{name} 射箭")
    else:
        print("未知角色")

# 问题：
# 1. 每次添加新角色都要修改 perform_attack
# 2. 代码重复，难以维护
# 3. 角色类和函数耦合
characters = [
    ("warrior", "张战士"),
    ("mage", "李法师"),
    ("archer", "王弓箭"),
]

for char_type, name in characters:
    perform_attack(char_type, name)
```

**问题：**
- 需要判断类型，代码繁琐
- 添加新角色需要修改函数
- 代码耦合度高
- 难以扩展

**使用多态：**

```python
class Character:
    """角色基类"""
    def __init__(self, name: str) -> None:
        self.name: str = name
    
    def attack(self) -> str:
        raise NotImplementedError

class Warrior(Character):
    def attack(self) -> str:
        return f"{self.name} 挥剑"

class Mage(Character):
    def attack(self) -> str:
        return f"{self.name} 施法"

class Archer(Character):
    def attack(self) -> str:
        return f"{self.name} 射箭"

def perform_attack(character: Character) -> None:
    """统一接口，自动调用正确的方法"""
    print(character.attack())

# 添加新角色无需修改 perform_attack
characters = [
    Warrior("张战士"),
    Mage("李法师"),
    Archer("王弓箭"),
]

for c in characters:
    perform_attack(c)

# 输出：
# 张战士 挥剑
# 李法师 施法
# 王弓箭 射箭
```

**优势：**
- 统一接口，简化代码
- 易于扩展新类型
- 降低耦合度
- 符合开放封闭原则

## 第一部分：什么是多态

### 1.1 概念动机

**多态：同一个接口，不同的实现**

多态允许不同类型的对象对同一消息作出不同的响应。

```
┌─────────────────────────────────────────────────────────────┐
│                      多态的作用                              │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│   1. 统一接口                                               │
│      不同对象可以用相同方式调用                             │
│                                                             │
│   2. 灵活扩展                                               │
│      添加新类型无需修改现有代码                             │
│                                                             │
│   3. 降低耦合                                               │
│      调用者不需要知道具体类型                               │
│                                                             │
│   4. 代码复用                                               │
│      统一处理不同类型的对象                                 │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### 1.2 最简示例

```python
class Animal:
    """动物基类"""
    def speak(self) -> str:
        raise NotImplementedError("子类必须实现此方法")

class Dog(Animal):
    def speak(self) -> str:
        return "汪汪汪"

class Cat(Animal):
    def speak(self) -> str:
        return "喵喵喵"

class Duck(Animal):
    def speak(self) -> str:
        return "嘎嘎嘎"


# ─────────────────────────────────────
# 多态应用：统一的接口处理不同类型的对象
# ─────────────────────────────────────

def make_animal_speak(animal: Animal) -> None:
    """让动物叫（不关心具体类型）"""
    print(f"{type(animal).__name__}: {animal.speak()}")

# 创建不同的动物
animals: list[Animal] = [Dog(), Cat(), Duck()]

# 统一处理
for animal in animals:
    make_animal_speak(animal)

# 输出：
# Dog: 汪汪汪
# Cat: 喵喵喵
# Duck: 嘎嘎嘎
```

### 1.3 详细讲解

**多态的关键要素：**

1. **基类定义接口**
2. **子类实现不同版本**
3. **调用者使用统一接口**

```python

from abc import ABC, abstractmethod
from typing import Optional

class PaymentMethod(ABC):
    """支付方式抽象基类"""
    
    @abstractmethod
    def pay(self, amount: float) -> bool:
        """支付"""
        pass

    def get_name(self) -> str:

        pass

class CreditCard(PaymentMethod):
    """信用卡支付"""

        self.card_number: str = card_number
        self.holder: str = holder
    
    def pay(self, amount: float) -> bool:
        print(f"信用卡支付: {amount}元")
        return True
    
    def get_name(self) -> str:
        return "信用卡"

class PayPal(PaymentMethod):
    """PayPal 支付"""
    
    def __init__(self, email: str) -> None:
        self.email: str = email
    
    def pay(self, amount: float) -> bool:
        print(f"PayPal 支付: {amount}元")
        return True
    
    def get_name(self) -> str:
        return "PayPal"

class Alipay(PaymentMethod):
    """支付宝支付"""
    
    def __init__(self, account: str) -> None:
        self.account: str = account
    
    def pay(self, amount: float) -> bool:
        print(f"支付宝支付: {amount}元")
        return True
    
    def get_name(self) -> str:
        return "支付宝"

class WeChatPay(PaymentMethod):
    """微信支付"""
    
    def __init__(self, openid: str) -> None:
        self.openid: str = openid
    
    def pay(self, amount: float) -> bool:
        print(f"微信支付: {amount}元")
        return True
    
    def get_name(self) -> str:
        return "微信支付"

# 统一的支付处理
def process_payment(payment: PaymentMethod, amount: float) -> None:
    """处理支付（多态）"""
    print(f"使用 {payment.get_name()} 进行支付...")
    if payment.pay(amount):
        print("支付成功")
    else:
        print("支付失败")

# 使用不同的支付方式

    PayPal("zhang@example.com"),
    Alipay("13800138000"),
    WeChatPay("wx123456"),
]

amounts = [100, 200, 150, 80]
```

### 1.5 实际应用

文件处理器示例：

```python
from abc import ABC, abstractmethod
from pathlib import Path
from typing import Any

class FileHandler(ABC):
    """文件处理器抽象基类"""
    
    @abstractmethod
    def read(self, path: Path) -> Any:
        """读取文件"""
        pass
    
    @abstractmethod
    def write(self, path: Path, data: Any) -> bool:
        """写入文件"""
        pass
    
    @abstractmethod
    def get_extension(self) -> str:
        """获取支持的扩展名"""
        pass

class TextHandler(FileHandler):
    """文本文件处理器"""
    
    def read(self, path: Path) -> str:
        return path.read_text()
    
    def write(self, path: Path, data: str) -> bool:
        path.write_text(data)
        return True
    
    def get_extension(self) -> str:
        return ".txt"

class JSONHandler(FileHandler):
    """JSON 文件处理器"""
    
    import json
    
    def read(self, path: Path) -> dict:
        with open(path) as f:
            return json.load(f)
    
    def write(self, path: Path, data: dict) -> bool:
        with open(path, 'w') as f:
            json.dump(data, f)
        return True
    
    def get_extension(self) -> str:
        return ".json"

class CSVHandler(FileHandler):
    """CSV 文件处理器"""
    
    import csv
    
    def read(self, path: Path) -> list[list[str]]:
        with open(path) as f:
            reader = csv.reader(f)
            return list(reader)
    
    def write(self, path: Path, data: list[list[str]]) -> bool:
        with open(path, 'w', newline='') as f:
            writer = csv.writer(f)
            writer.writerows(data)
        return True
    
    def get_extension(self) -> str:
        return ".csv"

# 文件处理器管理器
class FileManager:
    """文件管理器"""
    
    def __init__(self) -> None:
        self.handlers: dict[str, FileHandler] = {}
    
    def register_handler(self, handler: FileHandler) -> None:
        """注册处理器"""
        ext = handler.get_extension()
        self.handlers[ext] = handler
    
    def get_handler(self, path: Path) -> FileHandler:
        """获取合适的处理器"""
        ext = path.suffix
        if ext in self.handlers:
            return self.handlers[ext]
        raise ValueError(f"不支持的文件类型: {ext}")
    
    def read_file(self, path: Path) -> Any:
        """读取文件"""
        handler = self.get_handler(path)
        return handler.read(path)
    
    def write_file(self, path: Path, data: Any) -> bool:
        """写入文件"""
        handler = self.get_handler(path)
        return handler.write(path, data)

# 使用
manager = FileManager()
manager.register_handler(TextHandler())
manager.register_handler(JSONHandler())
manager.register_handler(CSVHandler())

# 自动选择正确的处理器
# data = manager.read_file(Path("data.json"))
# manager.write_file(Path("output.csv"), [["name", "age"], ["张三", 25]])
```

## 第二部分：鸭子类型

### 2.1 概念动机

Python 的多态不依赖于继承，而是依赖于"鸭子类型"。

> "如果它走起来像鸭子，叫起来像鸭子，那它就是鸭子"

```
┌─────────────────────────────────────────────────────────────┐
│              鸭子类型的优势                                  │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│   1. 灵活性                                                 │
│      不需要继承关系，只要有相同的方法即可                   │
│                                                             │
│   2. 可扩展性                                               │
│      新类型只需要实现相同的方法                             │
│                                                             │
│   3. 简洁性                                                 │
│      不需要复杂的继承层次                                   │
│                                                             │
│   4. Pythonic                                               │
│      符合 Python 的设计理念                                 │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### 2.2 最简示例

```python
class Dog:
    def speak(self) -> str:
        return "汪汪"

class Cat:
    def speak(self) -> str:
        return "喵喵"

class Robot:
    def speak(self) -> str:
        return "滴滴"


# 不需要继承同一个父类
def make_speak(thing: object) -> None:
    """只要对象有 speak() 方法就能用"""
    if hasattr(thing, 'speak') and callable(thing.speak):
        print(thing.speak())

# 都能正常工作
make_speak(Dog())    # 汪汪
make_speak(Cat())    # 喵喵
make_speak(Robot())  # 滴滴
```

### 2.3 详细讲解

**鸭子类型 vs 继承多态：**

```python
# 继承多态：需要继承同一个父类
class Animal:
    def speak(self) -> str:
        raise NotImplementedError

class Dog(Animal):
    def speak(self) -> str:
        return "汪汪"

# 鸭子类型：只需要有相同的方法
class Cat:  # 不继承 Animal
    def speak(self) -> str:
        return "喵喵"

class Duck:
    def speak(self) -> str:
        return "嘎嘎"

# 两者都能使用
def make_speak(obj: object) -> None:
    print(obj.speak())

make_speak(Dog())   # 汪汪
make_speak(Cat())   # 喵喵
make_speak(Duck())  # 嘎嘎
```

**鸭子类型的实际应用：**

```python
class FileLike:
    """文件类对象"""
    
    def __init__(self, content: str) -> None:
        self.content: str = content
        self.position: int = 0
    
    def read(self, size: int = -1) -> str:
        """读取"""
        if size == -1:
            result = self.content[self.position:]
            self.position = len(self.content)
        else:
            result = self.content[self.position:self.position + size]
            self.position += size
        return result
    
    def seek(self, position: int) -> None:
        """定位"""
        self.position = position

# 标准文件对象
import io

file_obj = io.StringIO("文件内容")

# 自定义文件类对象
file_like_obj = FileLike("文件内容")

# 鸭子类型：两者都可以用于需要文件对象的场景
def process_file(file: object) -> str:
    """处理文件（接受任何有 read() 方法的对象）"""
    return file.read()

print(process_file(file_obj))       # 文件内容
print(process_file(file_like_obj))  # 文件内容
```

### 2.4 渐进复杂化

序列类型示例：

```python
from typing import Iterator

class MyList:
    """自定义列表类"""
    
    def __init__(self, items: list[int]) -> None:
        self.items: list[int] = items
    
    def __len__(self) -> int:
        """支持 len()"""
        return len(self.items)
    
    def __getitem__(self, index: int) -> int:
        """支持索引访问"""
        return self.items[index]
    
    def __iter__(self) -> Iterator[int]:
        """支持迭代"""
        return iter(self.items)
    
    def __contains__(self, item: int) -> bool:
        """支持 in 操作"""
        return item in self.items

# 使用
my_list = MyList([1, 2, 3, 4, 5])

# 所有列表操作都能用
print(len(my_list))       # 5
print(my_list[0])         # 1
print(3 in my_list)       # True
for item in my_list:
    print(item)           # 1 2 3 4 5

# 鸭子类型：可以用于任何接受序列的函数
def sum_sequence(seq: object) -> int:
    """求和（接受任何可迭代对象）"""
    total = 0
    for item in seq:
        total += item
    return total

print(sum_sequence([1, 2, 3]))      # 6
print(sum_sequence(MyList([1, 2, 3])))  # 6
```

## 第三部分：抽象基类（ABC）

### 3.1 概念动机

抽象基类定义接口规范，强制子类实现特定方法。

```
┌─────────────────────────────────────────────────────────────┐
│              抽象基类的作用                                  │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│   1. 定义接口规范                                           │
│      强制子类实现特定方法                                   │
│                                                             │
│   2. 防止实例化基类                                         │
│      只能通过具体子类创建对象                               │
│                                                             │
│   3. 类型检查                                               │
│      isinstance(obj, Shape)                                 │
│                                                             │
│   4. 文档化接口                                             │
│      明确子类需要实现哪些方法                               │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### 3.2 最简示例

```python
from abc import ABC, abstractmethod

class Shape(ABC):
    """抽象基类：定义所有形状必须实现的方法"""

    @abstractmethod
    def get_area(self) -> float:
        """计算面积 - 子类必须实现"""
        pass

    @abstractmethod
    def get_perimeter(self) -> float:
        """计算周长 - 子类必须实现"""
        pass


# ❌ 不能实例化抽象基类
# shape = Shape()  # TypeError!


# ✅ 必须实现所有抽象方法
class Rectangle(Shape):
    def __init__(self, width: float, height: float) -> None:
        self.width: float = width
        self.height: float = height

    def get_area(self) -> float:
        return self.width * self.height

    def get_perimeter(self) -> float:
        return 2 * (self.width + self.height)


class Circle(Shape):
    def __init__(self, radius: float) -> None:
        self.radius: float = radius

    def get_area(self) -> float:
        return 3.14 * self.radius ** 2

    def get_perimeter(self) -> float:
        return 2 * 3.14 * self.radius


# 使用
shapes: list[Shape] = [Rectangle(10, 5), Circle(7)]
for shape in shapes:
    area = shape.get_area()
    print(f"{type(shape).__name__}: 面积={area:.2f}")

# 输出：
# Rectangle: 面积=50.00
# Circle: 面积=153.86
```

**抽象属性：**

```python
from abc import ABC, abstractmethod

class Database(ABC):
    """数据库抽象基类"""
    
    @property
    @abstractmethod
    def connection_string(self) -> str:
        """连接字符串（抽象属性）"""
        pass
    
    @abstractmethod
    def connect(self) -> bool:
        """连接"""
        pass

class MySQLDatabase(Database):
    """MySQL 数据库"""
```

插件系统：

```python
from abc import ABC, abstractmethod
from typing import Any, Optional
import json

class Plugin(ABC):
    """插件抽象基类"""
    
    @property
    @abstractmethod
    def name(self) -> str:
        """插件名称"""
        pass
    
    @property
    @abstractmethod
    def version(self) -> str:
        """插件版本"""
        pass
    
    @abstractmethod
    def initialize(self, config: dict[str, Any]) -> bool:
        """初始化"""
        pass
    
    @abstractmethod
    def execute(self, data: Any) -> Any:
        """执行"""
        pass
    
    @abstractmethod
    def cleanup(self) -> None:
        """清理"""
        pass
    
    # 可选方法
    def get_info(self) -> dict[str, str]:
        """获取插件信息"""
        return {
            "name": self.name,
            "version": self.version,
        }

class DataProcessorPlugin(Plugin):
    """数据处理插件"""
    
    @property
    def name(self) -> str:
        return "data_processor"
    
    @property
    def version(self) -> str:
        return "1.0.0"
    
    def initialize(self, config: dict[str, Any]) -> bool:
        self.format: str = config.get("format", "json")
        print(f"初始化数据处理插件，格式: {self.format}")
        return True
    
    def execute(self, data: Any) -> Any:
        if self.format == "json":
            return json.dumps(data)
        elif self.format == "text":
            return str(data)
        return data
    
    def cleanup(self) -> None:
        print("清理数据处理插件")

class ValidatorPlugin(Plugin):
    """验证插件"""
    
    @property
    def name(self) -> str:
        return "validator"
    
    @property
    def version(self) -> str:
        return "1.1.0"
    
    def initialize(self, config: dict[str, Any]) -> bool:
        self.strict: bool = config.get("strict", False)
        print(f"初始化验证插件，严格模式: {self.strict}")
        return True
    
    def execute(self, data: Any) -> bool:
        if isinstance(data, dict):
            return len(data) > 0
        return False
    
    def cleanup(self) -> None:
        print("清理验证插件")

class PluginManager:
    """插件管理器"""
    
    def __init__(self) -> None:
        self.plugins: dict[str, Plugin] = {}
    
    def register(self, plugin: Plugin) -> None:
        """注册插件"""
        self.plugins[plugin.name] = plugin
    
    def initialize_all(self, configs: dict[str, dict]) -> None:
        """初始化所有插件"""
        for name, plugin in self.plugins.items():
            config = configs.get(name, {})
            plugin.initialize(config)
    
    def execute(self, plugin_name: str, data: Any) -> Optional[Any]:
        """执行特定插件"""
        if plugin_name in self.plugins:
            return self.plugins[plugin_name].execute(data)
        return None
    
    def cleanup_all(self) -> None:
        """清理所有插件"""
        for plugin in self.plugins.values():
            plugin.cleanup()

# 使用
manager = PluginManager()
manager.register(DataProcessorPlugin())
manager.register(ValidatorPlugin())

manager.initialize_all({
    "data_processor": {"format": "json"},
    "validator": {"strict": True},
})

result = manager.execute("data_processor", {"name": "张三"})
print(result)  # {"name": "张三"}

valid = manager.execute("validator", {"name": "张三"})
print(valid)  # True

manager.cleanup_all()
```

### 3.5 实际应用

消息队列系统：

```python
from abc import ABC, abstractmethod
from typing import Any, Optional
from dataclasses import dataclass
from datetime import datetime

@dataclass
class Message:
    """消息"""
    id: str
    content: Any
    timestamp: datetime
    priority: int = 0

class MessageQueue(ABC):
    """消息队列抽象基类"""
    
    @abstractmethod
    def enqueue(self, message: Message) -> bool:
        """入队"""
        pass
    
    @abstractmethod
    def dequeue(self) -> Optional[Message]:
        """出队"""
        pass
    
    @abstractmethod
    def peek(self) -> Optional[Message]:
        """查看队首"""
        pass
    
    @abstractmethod
    def size(self) -> int:
        """队列大小"""
        pass
    
    @abstractmethod
    def is_empty(self) -> bool:
        """是否为空"""
        pass

class SimpleQueue(MessageQueue):
    """简单队列"""
    
    def __init__(self) -> None:
        self.queue: list[Message] = []
    
    def enqueue(self, message: Message) -> bool:
        self.queue.append(message)
        return True
    
    def dequeue(self) -> Optional[Message]:
        if self.queue:
            return self.queue.pop(0)
        return None
    
    def peek(self) -> Optional[Message]:
        if self.queue:
            return self.queue[0]
        return None
    
    def size(self) -> int:
        return len(self.queue)
    
    def is_empty(self) -> bool:
        return len(self.queue) == 0

class PriorityQueue(MessageQueue):
    """优先队列"""
    
    def __init__(self) -> None:
        self.queue: list[Message] = []
    
    def enqueue(self, message: Message) -> bool:
        self.queue.append(message)
        self.queue.sort(key=lambda m: m.priority, reverse=True)
        return True
    
    def dequeue(self) -> Optional[Message]:
        if self.queue:
            return self.queue.pop(0)
        return None
    
    def peek(self) -> Optional[Message]:
        if self.queue:
            return self.queue[0]
        return None
    
    def size(self) -> int:
        return len(self.queue)
    
    def is_empty(self) -> bool:
        return len(self.queue) == 0

class DelayedQueue(MessageQueue):
    """延迟队列"""
    
    def __init__(self, delay_seconds: int = 5) -> None:
        self.queue: list[Message] = []
        self.delay_seconds: int = delay_seconds
    
    def enqueue(self, message: Message) -> bool:
        self.queue.append(message)
        return True
    
    def dequeue(self) -> Optional[Message]:
        now = datetime.now()
        for i, msg in enumerate(self.queue):
            elapsed = (now - msg.timestamp).total_seconds()
            if elapsed >= self.delay_seconds:
                return self.queue.pop(i)
        return None
    
    def peek(self) -> Optional[Message]:
        now = datetime.now()
        for msg in self.queue:
            elapsed = (now - msg.timestamp).total_seconds()
            if elapsed >= self.delay_seconds:
                return msg
        return None
    
    def size(self) -> int:
        return len(self.queue)
    
    def is_empty(self) -> bool:
        return len(self.queue) == 0

# 统一处理不同队列
def process_queue(queue: MessageQueue) -> None:
    """处理队列（多态）"""
    print(f"队列大小: {queue.size()}")
    
    while not queue.is_empty():
        msg = queue.dequeue()
        if msg:
            print(f"处理消息: {msg.id}")

# 使用不同的队列
queues: list[MessageQueue] = [
    SimpleQueue(),
    PriorityQueue(),
    DelayedQueue(delay_seconds=1),
]

for queue in queues:
    print(f"\n{queue.__class__.__name__}:")
    
    # 入队
    queue.enqueue(Message("1", "消息1", datetime.now(), priority=1))
    queue.enqueue(Message("2", "消息2", datetime.now(), priority=3))
    queue.enqueue(Message("3", "消息3", datetime.now(), priority=2))
    
    # 处理
    process_queue(queue)
```

**关键代码说明：**

| 代码 | 含义 | 为什么这样写 |
|------|------|-------------|
| `class MessageQueue(ABC)` | 抽象基类定义接口规范 | 强制所有子类实现 `enqueue`/`dequeue` 等方法，防止遗漏 |
| `@abstractmethod` | 声明抽象方法 | 子类如果没有实现该方法，实例化时会立即报错，而非运行时才发现 |
| `def process_queue(queue: MessageQueue)` | 参数类型为抽象基类 | 只依赖接口不依赖具体实现，三种队列类型都可传入，体现多态 |
| `queues: list[MessageQueue] = [SimpleQueue(), PriorityQueue(), DelayedQueue()]` | 统一列表存放不同子类 | 多态的核心：用基类类型引用子类对象，统一处理逻辑不变 |
| `PriorityQueue.dequeue` 按优先级弹出 | 不同子类有不同出队行为 | 同一接口 `dequeue()`，不同实现（先进先出 vs 优先级），多态的本质 |

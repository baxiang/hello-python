#### 语法错误（SyntaxError）

```python

# 示例 1：缺少冒号
# ─────────────────────────────────────

#     print("Hello")
# ❌ SyntaxError: invalid syntax

if True:
    print("Hello")

# ─────────────────────────────────────
# 示例 2：缺少括号
# ─────────────────────────────────────
# print("Hello"

print("Hello")

# ─────────────────────────────────────
# 示例 3：缩进错误
# ─────────────────────────────────────

# print("Hello")

def greet():
    print("Hello")

# ─────────────────────────────────────
# 示例 4：无效语法
# ─────────────────────────────────────
# 123abc = 10
# ❌ SyntaxError: invalid decimal literal
# 正确写法：
abc_123 = 10
```

### 10.3 try-except 语句

#### 基本语法

```python
# 基本结构
try:
    # 可能出错的代码
    # 如果这里发生异常，会跳转到 except
    num = int(input("请输入一个数字："))
    result = 10 / num
    print(f"结果：{result}")

except ValueError:
    # 处理 ValueError
    print("输入无效，请输入数字！")

except ZeroDivisionError:
    # 处理 ZeroDivisionError
    print("不能除以零！")
```

#### 捕获多个异常

```python
# ─────────────────────────────────────
# 方式 1：分别处理不同的异常
# ─────────────────────────────────────
try:
    num = int(input("请输入数字："))
    result = 10 / num
    print(f"结果：{result}")

except ValueError:
    print("❌ 输入的不是有效数字！")

except ZeroDivisionError:
    print("❌ 不能除以零！")

# ─────────────────────────────────────
# 方式 2：统一处理多种异常
# ─────────────────────────────────────
try:
    num = int(input("请输入数字："))
    result = 10 / num
    print(f"结果：{result}")

except (ValueError, ZeroDivisionError) as e:
    print(f"❌ 发生错误：{e}")

# ─────────────────────────────────────
# 方式 3：捕获所有异常（不推荐滥用）
# ─────────────────────────────────────
try:
    num = int(input("请输入数字："))
    result = 10 / num
    print(f"结果：{result}")

except Exception as e:
    # Exception 是所有常规异常的基类
    print(f"❌ 发生未知错误：{type(e).__name__}: {e}")
```

### 10.4 else 和 finally 子句

#### else 子句

```python
# else 在没有异常时执行
try:
    num = int(input("请输入数字："))
    result = 10 / num

except ValueError:
    print("请输入有效的数字！")

except ZeroDivisionError:
    print("不能除以零！")

else:
    # 只有 try 块没有抛出异常时才执行
    print(f"✅ 计算成功！结果：{result}")

finally:
    # 无论是否有异常都执行
    print("程序执行完毕")
```

**else 的使用场景：**
```
┌─────────────────────────────────────────┐
│          else 子句的使用场景            │
├─────────────────────────────────────────┤
│                                         │
│  场景：你想执行一段代码，但只有当       │
│       try 块成功时才执行                │
│                                         │
│  例如：读取文件后处理内容               │
│  try:                                   │
│      with open("file.txt") as f:        │
│          content = f.read()             │
│  except FileNotFoundError:              │

│      # 只有读取成功才处理               │
│      process(content)                   │

│  因为 else 里的异常不会被当前 except    │

│                                         │
└─────────────────────────────────────────┘
```

#### finally 子句

```python
# finally 总是执行（无论是否有异常）
try:
    print("尝试打开文件...")
    file = open("data.txt", "r")

    print("尝试读取内容...")
    content = file.read()

    print("尝试转换数字...")
    num = int(content)

except FileNotFoundError:
    print("❌ 文件不存在")

except ValueError:
    print("❌ 内容不是有效数字")

finally:
    # 无论是否发生异常，都会执行
    print("清理资源...")
    if 'file' in locals():
        file.close()
        print("文件已关闭")

print("程序继续执行...")
```

### 10.5 完整的异常处理结构

```python
def divide_numbers():
    """完整的异常处理示例"""

    try:
        # ──────────────────────────────
        # try: 尝试执行的代码
        # ──────────────────────────────
        print("=" * 40)
        print("除法计算器")
```

## 第三部分：抛出异常

### 10.6 raise 语句

#### 抛出异常

```python
# ─────────────────────────────────────
# 基本用法：抛出异常
# ─────────────────────────────────────
def set_age(age):
    if age < 0 or age > 150:
        raise ValueError("年龄必须在 0-150 之间")
    print(f"年龄设置为：{age}")

# 使用
set_age(25)   # ✅ 正常
# set_age(-5)   # ❌ 抛出 ValueError
# set_age(200)  # ❌ 抛出 ValueError

# ─────────────────────────────────────
# 抛出带详细信息的异常
# ─────────────────────────────────────
def withdraw(balance, amount):
    if amount <= 0:
        raise ValueError(f"取款金额必须大于 0，当前输入：{amount}")
    if amount > balance:
        raise ValueError(f"余额不足！余额：{balance}，尝试取款：{amount}")
    return balance - amount

# 使用
# withdraw(100, 200)  # ❌ 抛出详细错误信息
```

#### 重新抛出异常

```python
# ─────────────────────────────────────
# 重新抛出异常
# ─────────────────────────────────────
def process_data(data):
    try:
        result = int(data)
        return result
    except ValueError as e:
        print(f"捕获到错误：{e}")
        print("尝试记录日志...")
        # 记录日志后重新抛出
        raise  # 重新抛出当前异常

# 使用
try:
    process_data("abc")
except ValueError:
    print("最终捕获到异常")

# 输出：
# 捕获到错误：invalid literal for int() with base 10: 'abc'
# 尝试记录日志...
# 最终捕获到异常

# ─────────────────────────────────────
# 捕获一个异常，抛出另一个异常
# ─────────────────────────────────────
def divide(a, b):
    try:
        result = a / b
        return result
    except ZeroDivisionError as e:
        # 包装成更有意义的异常
        raise ValueError(f"除数不能为零：a={a}, b={b}") from e

# 使用
try:
    divide(10, 0)
except ValueError as e:
    print(f"捕获到：{e}")
```

### 10.7 自定义异常类

#### 基础自定义异常

```python
# ─────────────────────────────────────
# 最简单的自定义异常
# ─────────────────────────────────────
class MyError(Exception):
    """自定义异常基类"""
    pass

# 使用
try:
    raise MyError("这是一个自定义错误")
except MyError as e:
    print(f"捕获到自定义错误：{e}")

# ─────────────────────────────────────
# 带属性的自定义异常
# ─────────────────────────────────────
class ValidationError(Exception):
    """验证异常"""

    def __init__(self, field, message):
        self.field = field
        self.message = message
        # 调用父类构造函数
        super().__init__(f"{field}: {message}")

    def __str__(self):
        return f"字段 '{self.field}' 验证失败：{self.message}"

# 使用
try:
    raise ValidationError("email", "邮箱格式不正确")
except ValidationError as e:
    print(e)  # 字段 'email' 验证失败：邮箱格式不正确
    print(f"字段：{e.field}")      # email
    print(f"消息：{e.message}")    # 邮箱格式不正确
```

#### 异常层次结构

```python
# ─────────────────────────────────────
# 创建异常层次结构
# ─────────────────────────────────────

# 应用错误基类
class AppError(Exception):
    """应用错误基类"""
    def __init__(self, message="应用程序错误"):
        super().__init__(message)
        self.message = message

# 数据库相关错误
class DatabaseError(AppError):
    """数据库错误"""
    pass

class ConnectionError(DatabaseError):
    """数据库连接错误"""
    pass

class QueryError(DatabaseError):
    """查询错误"""
    pass

# 用户相关错误
class UserError(AppError):
    """用户相关错误"""
    pass

class AuthenticationError(UserError):
    """认证错误"""
    pass

class AuthorizationError(UserError):
    """授权错误"""
    pass

# 验证错误
class ValidationError(AppError):
    """验证错误"""
    def __init__(self, field, message):
        super().__init__(f"{field}: {message}")
        self.field = field

# ─────────────────────────────────────
# 使用异常层次结构
# ─────────────────────────────────────
def connect_to_database():
    raise ConnectionError("无法连接到数据库")

def execute_query(sql):
    raise QueryError(f"SQL 执行失败：{sql}")

def login(username, password):
    if not username:
        raise ValidationError("username", "用户名不能为空")
    if password != "correct_password":
        raise AuthenticationError("用户名或密码错误")

# 捕获时可以捕获基类或具体异常
try:
    login("", "123456")
except AppError as e:
    print(f"应用错误：{e}")
except Exception as e:
    print(f"其他错误：{e}")
```

### 10.8 assert 断言

#### 断言基础

```python
# ─────────────────────────────────────
# 基本语法
# ─────────────────────────────────────
# assert condition, error_message

def divide(a, b):
    assert b != 0, "除数不能为零"
    return a / b

# 使用
divide(10, 2)  # ✅ 5
# divide(10, 0)  # ❌ AssertionError: 除数不能为零

# ─────────────────────────────────────
# 断言的用途
# ─────────────────────────────────────

# 1. 参数验证（开发阶段）
def calculate_discount(price, discount):
    assert price >= 0, "价格不能为负"
    assert 0 <= discount <= 1, "折扣必须在 0-1 之间"
    return price * (1 - discount)

# 2. 程序不变量
def sort_and_verify(numbers):
    sorted_nums = sorted(numbers)
    # 验证排序结果
    assert len(sorted_nums) == len(numbers), "长度不匹配"
    assert all(sorted_nums[i] <= sorted_nums[i+1]
               for i in range(len(sorted_nums)-1)), "排序失败"
    return sorted_nums

# 3. 类型检查（运行时）
def process_user(user):
    assert isinstance(user, dict), "user 必须是字典"
    assert "name" in user, "user 必须包含 name 字段"
    return f"Hello, {user['name']}"
```

#### 创建上下文管理器

```python
# ─────────────────────────────────────
# 方式 1：使用类实现
# ─────────────────────────────────────
class ManagedFile:
    """文件上下文管理器"""

    def __init__(self, filename, mode):
        self.filename = filename
        self.mode = mode
        self.file = None

    def __enter__(self):
        """进入上下文时调用"""
        print(f"打开文件：{self.filename}")
        self.file = open(self.filename, self.mode, encoding="utf-8")
        return self.file

    def __exit__(self, exc_type, exc_val, exc_tb):
        """离开上下文时调用"""
        print("关闭文件...")
        if self.file:
            self.file.close()

        # 返回 True 表示吞掉异常，False 或 None 表示传播异常
        return False

# 使用
with ManagedFile("data.txt", "r") as f:
    content = f.read()
    print(content)

# ─────────────────────────────────────
# 方式 2：使用装饰器实现
# ─────────────────────────────────────
from contextlib import contextmanager

@contextmanager
def managed_file(filename, mode):
    """文件上下文管理器（生成器版本）"""
    f = None
    try:
        print(f"打开文件：{filename}")
        f = open(filename, mode, encoding="utf-8")
        yield f  # 将文件对象传递给 with 块
    finally:
        if f:
            print("关闭文件...")
            f.close()

# 使用
with managed_file("data.txt", "r") as f:
    content = f.read()
```

```python
# 应用 1：计时器

import time
from contextlib import contextmanager

@contextmanager
def timer(name="操作"):
    """计时代码执行时间"""
    start = time.time()
    try:
        yield
    finally:
        end = time.time()
        print(f"{name}耗时：{end - start:.4f} 秒")

# 使用
with timer("读取文件"):
    with open("data.txt", "r") as f:
        content = f.read()
    time.sleep(0.1)  # 模拟处理

with timer("处理数据"):
    result = [x * 2 for x in range(1000000)]

# ─────────────────────────────────────
# 应用 2：临时切换目录
# ─────────────────────────────────────
import os
from contextlib import contextmanager

@contextmanager
def working_directory(path):
    """临时切换工作目录"""
    original_dir = os.getcwd()
    try:
        os.chdir(path)
        yield
    finally:
        os.chdir(original_dir)

# 使用
print(f"当前目录：{os.getcwd()}")

with working_directory("/tmp"):
    print(f"临时目录：{os.getcwd()}")
    # 在这里操作/tmp 目录

print(f"恢复目录：{os.getcwd()}")

# ─────────────────────────────────────
# 应用 3：抑制异常
# ─────────────────────────────────────
from contextlib import contextmanager

@contextmanager
def suppress_exception(*exceptions):
    """抑制指定的异常"""
    try:
        yield
    except exceptions:
        pass  # 吞掉异常

# 使用
with suppress_exception(FileNotFoundError):
    os.remove("not_exist.txt")  # 不会报错

with suppress_exception(ValueError, TypeError):
    int("abc")  # 不会报错
```

```python
    """
    安全地将字符串转换为整数

    参数:

        default: 转换失败时的默认值
```

### 练习 2：密码验证器

```python
class PasswordError(Exception):
    """密码验证异常"""
    pass

def validate_password(password):
    """
    验证密码强度

    要求：
    - 至少 8 个字符
    - 包含字母
    - 包含数字

    参数:
        password: 要验证的密码

    异常:
        PasswordError: 密码不符合要求
    """
    if len(password) < 8:
        raise PasswordError("密码长度不能少于 8 个字符")

    has_letter = any(c.isalpha() for c in password)
    has_digit = any(c.isdigit() for c in password)

    if not has_letter:
        raise PasswordError("密码必须包含字母")

    if not has_digit:
        raise PasswordError("密码必须包含数字")

    return True

# 测试
try:
    validate_password("abc123")  # 太短
except PasswordError as e:
    print(f"❌ {e}")

try:
    validate_password("abcdefgh")  # 没有数字
except PasswordError as e:
    print(f"❌ {e}")

try:
    validate_password("abc12345")  # ✅
    print("✅ 密码验证通过")
except PasswordError as e:
    print(f"❌ {e}")
```

```python
def safe_read_file(filename, encoding="utf-8"):
    """
    安全地读取文件内容

    参数:
        filename: 文件路径
        encoding: 文件编码

    返回:
        文件内容，或 None（如果失败）
    """
    try:
        with open(filename, "r", encoding=encoding) as f:
            return f.read()
    except FileNotFoundError:
        print(f"错误：文件 '{filename}' 不存在")
        return None
    except PermissionError:
        print(f"错误：没有权限读取 '{filename}'")
        return None
    except UnicodeDecodeError:
        print(f"错误：无法用 {encoding} 解码 '{filename}'")
        return None
    except Exception as e:
        print(f"错误：{type(e).__name__}: {e}")
        return None

# 测试
content = safe_read_file("not_exist.txt")
print(content)  # None
```

```python
    level=logging.INFO,
    format='%(asctime)s - %(levelname)s - %(message)s'

def process_data(data):

    try:
        # 验证输入
        if not isinstance(data, (int, float)):
            raise TypeError(f"期望数字类型，得到 {type(data).__name__}")

        if data < 0:

        result = data * 2
        logging.info(f"处理成功：{data} → {result}")

        logging.error(f"类型错误：{e}")
        raise

    except ValueError as e:
        logging.error(f"值错误：{e}")
        raise

    except Exception as e:
        logging.error(f"未知错误：{type(e).__name__}: {e}")
        raise

    finally:
        logging.info("处理完成")
```

### 异常类型速查

| 异常类型 | 触发条件 | 示例 |
|----------|----------|------|
| `SyntaxError` | 语法错误 | 缺少冒号 |

| `FileNotFoundError` | 文件不存在 | `open("x.txt")` |
| `ZeroDivisionError` | 除以零 | `1/0` |

### 异常处理结构

```
try:
    # 尝试执行的代码
    pass
except SpecificError:
    # 处理特定异常
    pass
except Exception as e:
    # 处理其他异常
    pass
else:
    # 没有异常时执行
    pass
finally:
    # 总是执行
    pass
```

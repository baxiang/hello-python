## 1. 运行时类型检查

### 1.1 get_type_hints

```python
from typing import get_type_hints, get_origin, get_args

def greet(name: str, age: int) -> str:
    return f"Hello {name}, you are {age}"

# 获取类型提示
hints = get_type_hints(greet)
print(hints)
# {'name': <class 'str'>, 'age': <class 'int'>, 'return': <class 'str'>}

# 获取函数注解（不解析前向引用）
print(greet.__annotations__)
# {'name': <class 'str'>, 'age': <class 'int'>, 'return': <class 'str'>}
```

### 1.2 类型检查装饰器

```python
from typing import get_type_hints, Any
import functools

def runtime_check(func):
    """运行时类型检查装饰器"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        hints = get_type_hints(func)
        # 检查位置参数
        for i, (arg, hint) in enumerate(zip(args, hints.items())):
            param_name, expected_type = hint
            if not isinstance(arg, expected_type):
                raise TypeError(
                    f"{param_name} 期望 {expected_type}, 得到 {type(arg)}"
                )
        return func(*args, **kwargs)
    return wrapper

@runtime_check
def add(a: int, b: int) -> int:
    return a + b

add(1, 2)    # ✓
add("1", 2)  # ✗ TypeError
```

### 1.3 自定义运行时检查

```python
from typing import get_type_hints, Any, get_origin, get_args

def validate_args(func):
    """验证函数参数类型"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        hints = get_type_hints(func)
        
        # 检查位置参数
        for i, arg in enumerate(args):
            param_name = list(hints.keys())[i]
            expected = hints[param_name]
            if not _check_type(arg, expected):
                raise TypeError(
                    f"参数 {param_name} 类型错误: "
                    f"期望 {expected}, 得到 {type(arg).__name__}"
                )
        
        return func(*args, **kwargs)
    return wrapper

def _check_type(value: Any, expected: Any) -> bool:
    """检查值是否符合类型"""
    # 处理 Optional
    if get_origin(expected) is Union:
        args = get_args(expected)
        return isinstance(value, args) or value is None
    
    # 处理泛型容器
    if get_origin(expected):
        return isinstance(value, expected)
    
    # 基本类型
    return isinstance(value, expected)

@validate_args
def process(name: str, age: int) -> str:
    return f"{name} is {age}"

process("John", 25)  # ✓
process(123, 25)     # ✗
```

## 2. 泛型高级用法

### 2.1 泛型约束

```python
from typing import TypeVar, Generic

# 约束为数字类型
Number = TypeVar('Number', int, float, bound='numbers.Number')

class Math:
    @staticmethod
    def double(value: Number) -> Number:
        return value * 2

# 使用
Math.double(5)     # int
Math.double(5.0)   # float
# Math.double("5")  # ✗ 错误
```

### 2.2 泛型别名

```python
from typing import TypeVar, List, Dict, Tuple

# 复杂类型别名
Matrix = List[List[float]]
Coordinate = Tuple[float, float]
UserDict = Dict[str, "User"]

# 递归类型别名
Tree = List[dict[str, "Tree"] | None]

# 条件类型别名
from typing import TypeAlias

# Python 3.10+
IntOrStr: TypeAlias = int | str
NumberList: TypeAlias = list[int | float]
```

### 2.3 泛型函数设计

```python
from typing import TypeVar, Generic, Callable, List

T = TypeVar('T')
U = TypeVar('U')

def map_list(func: Callable[[T], U], items: List[T]) -> List[U]:
    """泛型映射函数"""
    return [func(item) for item in items]

def filter_list(predicate: Callable[[T], bool], items: List[T]) -> List[T]:
    """泛型过滤函数"""
    return [item for item in items if predicate(item)]

# 使用
result = map_list(lambda x: x * 2, [1, 2, 3])  # [2, 4, 6]
result = filter_list(lambda x: x > 2, [1, 2, 3])  # [3]
```

## 3. 自定义泛型类

### 3.1 泛型容器

```python
from typing import TypeVar, Generic, Iterator

T = TypeVar('T')

class Box(Generic[T]):
    """泛型盒子"""
    def __init__(self, item: T) -> None:
        self._item = item
    
    def get(self) -> T:
        return self._item
    
    def set(self, item: T) -> None:
        self._item = item

class LinkedList(Generic[T]):
    """泛型链表"""
    def __init__(self) -> None:
        self._items: List[T] = []
    
    def append(self, item: T) -> None:
        self._items.append(item)
    
    def __iter__(self) -> Iterator[T]:
        return iter(self._items)
    
    def __getitem__(self, index: int) -> T:
        return self._items[index]

# 使用
int_box: Box[int] = Box(42)
str_box: Box[str] = Box("hello")

numbers = LinkedList[int]()
numbers.append(1)
numbers.append(2)
```

### 3.2 泛型装饰器

```python
from typing import TypeVar, Generic, Callable, Any
import functools

F = TypeVar('F', bound=Callable[..., Any])

def generic_decorator(func: F) -> F:
    """泛型装饰器"""
    @functools.wraps(func)
    def wrapper(*args: Any, **kwargs: Any) -> Any:
        print(f"Calling {func.__name__}")
        return func(*args, **kwargs)
    return wrapper  # type: ignore
```

### 3.3 泛型上下文管理器

```python
from typing import TypeVar, Generic, ContextManager
from typing_extensions import Self

T = TypeVar('T')

class Transaction(Generic[T]):
    """泛型事务管理器"""
    def __init__(self, resource: T) -> None:
        self._resource = resource
        self._committed = False
    
    def __enter__(self) -> T:
        print("Starting transaction")
        return self._resource
    
    def __exit__(self, exc_type, exc_val, exc_tb) -> bool:
        if exc_type is None and not self._committed:
            self._resource.commit()
            print("Transaction committed")
        else:
            self._resource.rollback()
            print("Transaction rolled back")
        return False
    
    def commit(self) -> None:
        self._committed = True

# 使用
class MockResource:
    def commit(self) -> None: pass
    def rollback(self) -> None: pass

with Transaction(MockResource()) as resource:
    print("Doing work...")
```

## 4. 协议进阶

### 4.1 协议继承

```python
from typing import Protocol, runtime_checkable

class Drawable(Protocol):
    def draw(self) -> None: ...

class Fillable(Protocol):
    def fill(self) -> None: ...

# 继承协议
class Shape(Drawable, Fillable, Protocol):
    def move(self, x: int, y: int) -> None: ...

class Circle:
    def draw(self) -> None: print("Draw circle")
    def fill(self) -> None: print("Fill circle")
    def move(self, x: int, y: int) -> None: print(f"Move to {x}, {y}")

def render(shape: Shape) -> None:
    shape.draw()
    shape.fill()
    shape.move(0, 0)

render(Circle())  # ✓
```

### 4.2 运行时检查

```python
from typing import Protocol, runtime_checkable

@runtime_checkable
class Serializable(Protocol):
    def to_json(self) -> str: ...

class User:
    def to_json(self) -> str:
        return '{"name": "John"}'

# 运行时检查
is_serializable = isinstance(User(), Serializable)  # True
```

### 4.3 复杂协议示例

```python
from typing import Protocol, TypeVar, Generic, List

T = TypeVar('T')

class Repository(Protocol[T]):
    """仓储协议"""
    def get(self, id: int) -> T | None: ...
    def get_all(self) -> List[T]: ...
    def save(self, item: T) -> None: ...
    def delete(self, id: int) -> None: ...

class InMemoryRepository(Generic[T]):
    def __init__(self) -> None:
        self._data: List[T] = []
    
    def get(self, id: int) -> T | None:
        return self._data[id] if id < len(self._data) else None
    
    def get_all(self) -> List[T]:
        return self._data.copy()
    
    def save(self, item: T) -> None:
        self._data.append(item)
    
    def delete(self, id: int) -> None:
        if id < len(self._data):
            self._data.pop(id)

# 使用协议
def get_all_items(repo: Repository[dict]) -> List[dict]:
    return repo.get_all()

repo = InMemoryRepository[dict]()
repo.save({"name": "John"})
get_all_items(repo)  # ✓
```

## 5. 高级类型模式

### 5.1 映射类型

```python
from typing import TypeVar, Generic, Mapping as TypingMapping

K = TypeVar('K')
V = TypeVar('V')

class OrderedDict(Generic[K, V]):
    """有序字典"""
    def __init__(self) -> None:
        self._keys: List[K] = []
        self._values: List[V] = []
    
    def __setitem__(self, key: K, value: V) -> None:
        if key in self._keys:
            idx = self._keys.index(key)
            self._values[idx] = value
        else:
            self._keys.append(key)
            self._values.append(value)
    
    def __getitem__(self, key: K) -> V:
        idx = self._keys.index(key)
        return self._values[idx]
    
    def items(self) -> List[tuple[K, V]]:
        return list(zip(self._keys, self._values))

# 使用
d: OrderedDict[str, int] = OrderedDict()
d["a"] = 1
d["b"] = 2
```

### 5.2 递归类型

```python
from typing import TypeAlias

    "root": {

from typing import Optional

class ListNode:
    def __init__(
        self, 
        value: int, 
        next: Optional["ListNode"] = None
    ) -> None:
        self.value = value
        self.next = next

def traverse(node: Optional[ListNode]) -> None:
    while node:
        print(node.value)
        node = node.next
```

### 5.3 条件类型

```python
from typing import TypeVar, Generic, overload

T = TypeVar('T')

class Container(Generic[T]):
    def __init__(self, value: T) -> None:
        self._value = value
    
    @overload
    def get(self) -> T: ...
    @overload
    def get(self, default: T) -> T: ...
    
    def get(self, default: T | None = None) -> T | None:
        return self._value if self._value is not None else default
    
    def is_set(self) -> bool:
        return self._value is not None

# 使用
c: Container[int] = Container(42)
value: int = c.get()  # 42

c2: Container[int] = Container(None)
value2: int = c2.get(0)  # 0
```

### 6.3 类型标注策略

```python
# ✓ 推荐：显式类型
def process(data: list[int]) -> dict[str, int]:
    return {str(x): x for x in data}

# ✓ 推荐：使用泛型避免 Any
def first(items: list[T]) -> T:  # 泛型
    return items[0]

# ✗ 避免：过度类型化
def simple_add(a: int, b: int) -> int:  # 过于简单不需要
    return a + b

# ✓ 合理：复杂逻辑需要类型
def complex_operation(
    data: list[dict[str, Any]],
    filter_func: Callable[[dict], bool],
    transform: Callable[[dict], dict]
) -> list[dict]:
    return [transform(item) for item in data if filter_func(item)]
```

## 7. 完整示例

### 7.1 类型安全的仓储模式

```python
from typing import TypeVar, Generic, Protocol, Optional, List
from dataclasses import dataclass

T = TypeVar('T')

class Identifiable(Protocol):
    id: int

@dataclass
class User:
    id: int
    name: str
    email: str

class Repository(Generic[T]):
    """泛型仓储基类"""
    def __init__(self) -> None:
        self._storage: List[T] = []
    
    def add(self, item: T) -> int:
        self._storage.append(item)
        return len(self._storage) - 1
    
    def get(self, id: int) -> Optional[T]:
        for item in self._storage:
            if hasattr(item, 'id') and item.id == id:
                return item
        return None
    
    def get_all(self) -> List[T]:
        return self._storage.copy()
    
    def update(self, item: T) -> bool:
        for i, stored in enumerate(self._storage):
            if hasattr(stored, 'id') and stored.id == item.id:
                self._storage[i] = item
                return True
        return False
    
    def delete(self, id: int) -> bool:
        for i, item in enumerate(self._storage):
            if hasattr(item, 'id') and item.id == id:
                self._storage.pop(i)
                return True
        return False

# 使用
user_repo: Repository[User] = Repository()

# 添加
user_id = user_repo.add(User(1, "John", "john@example.com"))
user_repo.add(User(2, "Jane", "jane@example.com"))

# 查询
user: Optional[User] = user_repo.get(1)
if user:
    print(user.name)  # John

# 更新
user_repo.update(User(1, "John Updated", "john@example.com"))

# 删除
user_repo.delete(2)

# 获取全部
all_users: List[User] = user_repo.get_all()
```

# 装饰器基础

本章讲解 Python 装饰器的基本概念和用法。

## 什么是装饰器

### 概念说明

```
┌─────────────────────────────────────────┐
│             装饰器的概念                │
├─────────────────────────────────────────┤
│                                         │
│  装饰器（Decorator）                    │
│  ─────────────────                      │
│  是一个函数，用于修改其他函数的行为    │
│                                         │
│  特点：                                 │
│  • 不修改原函数的代码                  │
│  • 不改变原函数的调用方式              │
│  • 在运行时动态添加功能                │
│                                         │
│  类比：                                 │
│  给房子装修                             │
│  • 房子 = 原函数                        │
│  • 装修 = 装饰器                        │
│  • 装修后房子功能增强，但结构不变       │
│                                         │
└─────────────────────────────────────────┘
```

## 装饰器基本结构

### 基本语法

```python
def my_decorator(func):
    """简单的装饰器"""

    def wrapper(*args, **kwargs):
        # 1. 函数执行前的逻辑
        print("函数执行前")

        # 2. 调用原函数
        result = func(*args, **kwargs)

        # 3. 函数执行后的逻辑
        print("函数执行后")

        return result

    return wrapper

# 使用装饰器
@my_decorator
def say_hello():
    print("Hello!")

say_hello()
# 函数执行前
# Hello!
# 函数执行后
```

### 等价写法

```python
# 方式 1：使用 @ 语法（推荐）
@my_decorator
def say_hello():
    print("Hello!")

# 方式 2：手动包装（等价）
def say_hello():
    print("Hello!")

say_hello = my_decorator(say_hello)
```

## 实用装饰器示例

### 计时装饰器

```python
import time
from functools import wraps

def timer(func):
    """计时装饰器：测量函数执行时间"""
    @wraps(func)
    def wrapper(*args, **kwargs):
        start_time = time.time()
        result = func(*args, **kwargs)
        end_time = time.time()
        print(f"{func.__name__} 执行时间：{end_time - start_time:.4f}秒")
        return result
    return wrapper

@timer
def slow_function():
    time.sleep(1)
    print("任务完成")

slow_function()
# 任务完成
# slow_function 执行时间：1.0012 秒
```

### 日志装饰器

```python
from functools import wraps

def logger(func):
    """日志装饰器：记录函数调用信息"""
    @wraps(func)
    def wrapper(*args, **kwargs):
        print(f"[调用] {func.__name__}({args}, {kwargs})")
        result = func(*args, **kwargs)
        print(f"[返回] {func.__name__} → {result}")
        return result
    return wrapper

@logger
def add(a, b):
    return a + b

add(3, 5)
# [调用] add((3, 5), {})
# [返回] add → 8
```

### 权限验证装饰器

```python
from functools import wraps

def require_auth(func):
    """需要登录验证"""
    @wraps(func)
    def wrapper(user, *args, **kwargs):
        if not user.is_authenticated:
            raise PermissionError("用户未登录")
        return func(user, *args, **kwargs)
    return wrapper

@require_auth
def view_profile(user):
    return f"查看 {user.name} 的个人资料"
```

### 重试装饰器

```python
from functools import wraps
import time

def retry(times=3, delay=1):
    """重试装饰器"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            last_exception = None
            for attempt in range(times):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    last_exception = e
                    print(f"尝试 {attempt + 1}/{times} 失败：{e}")
                    if attempt < times - 1:
                        time.sleep(delay)
            raise last_exception
        return wrapper
    return decorator
```

## @wraps 的作用

```python
from functools import wraps

# ❌ 不使用 @wraps
def decorator_without_wraps(func):
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper

@decorator_without_wraps
def my_func():
    """这是 my_func 的文档"""
    pass

print(my_func.__name__)  # wrapper（不是 my_func！）
print(my_func.__doc__)   # None（文档丢失！）

# ✅ 使用 @wraps
def decorator_with_wraps(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper

@decorator_with_wraps
def my_func():
    """这是 my_func 的文档"""
    pass

print(my_func.__name__)  # my_func
print(my_func.__doc__)   # 这是 my_func 的文档
```

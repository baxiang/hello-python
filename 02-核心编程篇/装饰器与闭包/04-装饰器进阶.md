# 装饰器进阶

本章讲解 Python 装饰器的高级用法，包括带参数的装饰器、多层装饰器和类装饰器。

---

## 带参数的装饰器

### 三层嵌套结构

```python
from functools import wraps

def repeat(times):
    """重复执行装饰器"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            result = None
            for _ in range(times):
                result = func(*args, **kwargs)
            return result
        return wrapper
    return decorator

# 使用
@repeat(times=3)
def greet(name):
    print(f"Hello, {name}!")

greet("Alice")
# Hello, Alice!
# Hello, Alice!
# Hello, Alice!
```

### 执行顺序

```
@repeat(times=3)
def greet(name):
    ...

执行过程：

1. 调用 repeat(times=3)
   返回 decorator 函数
          │
2. 用 decorator 包装 greet
   greet = decorator(greet)
   返回 wrapper 函数
          │
3. 调用 greet("Alice")
   实际执行 wrapper("Alice")
```

## 多层装饰器

### 装饰器堆叠

```python
from functools import wraps

def decorator1(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        print("[装饰器 1] 前处理")
        result = func(*args, **kwargs)
        print("[装饰器 1] 后处理")
        return result
    return wrapper

def decorator2(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        print("[装饰器 2] 前处理")
        result = func(*args, **kwargs)
        print("[装饰器 2] 后处理")
        return result
    return wrapper

@decorator1
@decorator2
def greet(name):
    print(f"  Hello, {name}!")

greet("Alice")
# [装饰器 1] 前处理
# [装饰器 2] 前处理
#   Hello, Alice!
# [装饰器 2] 后处理
# [装饰器 1] 后处理
```

### 执行顺序

```
@decorator1
@decorator2
def greet(name):
    ...

装饰顺序（从下到上）：
1. greet 先被 decorator2 装饰
2. 结果再被 decorator1 装饰

等价于：
greet = decorator1(decorator2(greet))

调用时的执行顺序（洋葱模型）：
decorator1 前 → decorator2 前 → greet() → decorator2 后 → decorator1 后
```

## 类装饰器

### 用类实现装饰器

```python
from functools import wraps

class CountCalls:
    """统计函数调用次数的装饰器"""

    def __init__(self, func):
        wraps(func)(self)
        self.func = func
        self.count = 0

    def __call__(self, *args, **kwargs):
        self.count += 1
        print(f"第 {self.count} 次调用 {self.__name__}")
        return self.func(*args, **kwargs)

@CountCalls
def say_hello():
    print("Hello!")

say_hello()  # 第 1 次调用 say_hello
say_hello()  # 第 2 次调用 say_hello
```

### 带参数的类装饰器

```python
class Repeat:
    """重复执行指定次数的类装饰器"""

    def __init__(self, times):
        self.times = times

    def __call__(self, func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            for _ in range(self.times):
                result = func(*args, **kwargs)
            return result
        return wrapper

@Repeat(times=3)
def greet(name):
    print(f"Hello, {name}!")

greet("Alice")
# Hello, Alice!
# Hello, Alice!
# Hello, Alice!
```

## 常用装饰器模式

### 单例模式

```python
def singleton(cls):
    """单例装饰器"""
    instances = {}

    def get_instance(*args, **kwargs):
        if cls not in instances:
            instances[cls] = cls(*args, **kwargs)
        return instances[cls]

    return get_instance

@singleton
class Database:
    def __init__(self):
        print("创建数据库连接")

db1 = Database()  # 创建数据库连接
db2 = Database()  # 不会再次创建
print(db1 is db2)  # True
```

### 缓存装饰器

```python
from functools import lru_cache

@lru_cache(maxsize=128)
def fibonacci(n):
    if n < 2:
        return n
    return fibonacci(n - 1) + fibonacci(n - 2)

print(fibonacci(100))  # 瞬间完成
```

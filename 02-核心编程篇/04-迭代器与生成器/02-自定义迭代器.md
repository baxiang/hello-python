## 第一部分：为什么需要自定义迭代器？

### 1.1 实际场景引入

假设你需要实现一个倒计时功能：

```python
# 尝试一：使用列表（内存占用大）
def countdown_list(n: int) -> list[int]:
    return list(range(n, 0, -1))

for num in countdown_list(1000000):  # ❌ 生成百万个元素的列表！
    if num == 999999:
        break

# 尝试二：使用自定义迭代器（内存友好）
class Countdown:
    def __init__(self, n: int) -> None:
        self.n = n
    
    def __iter__(self) -> "Countdown":
        return self
    
    def __next__(self) -> int:
        if self.n <= 0:
            raise StopIteration
        self.n -= 1
        return self.n + 1

for num in Countdown(1000000):  # ✅ 每次只计算一个值
    if num == 999999:
        break
```

**问题**：如何创建自定义的迭代器？需要实现哪些方法？

### 1.2 概念动机

Python 的迭代器协议定义了两个必需方法：

```
┌─────────────────────────────────────────┐
│         迭代器协议 (Iterator Protocol)   │
├─────────────────────────────────────────┤
│                                         │
│  __iter__(self)                         │
│  ────────────────                       │
│  • 返回迭代器本身                       │
│  • 允许在 for 循环中使用                │
│                                         │
│  __next__(self)                         │
│  ────────────────                       │
│  • 返回下一个值                         │
│  • 没有更多元素时抛出 StopIteration     │
│                                         │
└─────────────────────────────────────────┘
```

## 第二部分：迭代器协议

### 2.1 最简示例：空迭代器

```python
from typing import Iterator

class EmptyIterator:
    """最简单的迭代器：立即停止"""
    
    def __iter__(self) -> "EmptyIterator":
        """返回迭代器本身（必须）"""
        return self
    
    def __next__(self) -> None:
        """返回下一个值（必须）"""
        # 如果没有更多元素，抛出 StopIteration
        raise StopIteration

# 使用
empty: EmptyIterator = EmptyIterator()
print(list(empty))  # []
```

### 2.2 详细讲解：迭代器协议模板

```python
from typing import Iterator, Generic, TypeVar

T = TypeVar("T")

class MyIterator(Generic[T]):
    """自定义迭代器模板"""
    
    def __init__(self, data: list[T]) -> None:
        self._data: list[T] = data
        self._index: int = 0
    
    def __iter__(self) -> "MyIterator[T]":
        """返回迭代器本身"""
        return self
    
    def __next__(self) -> T:
        """返回下一个值"""
        if self._index >= len(self._data):
            raise StopIteration
        
        value: T = self._data[self._index]
        self._index += 1
        return value

# 使用
my_iter: MyIterator[int] = MyIterator([1, 2, 3])
for item in my_iter:
    print(item, end=" ")
# 1 2 3
```

## 第三部分：实际示例

### 3.1 倒数计时器

```python
from typing import Iterator

class CountDown:
    """倒数计时器迭代器"""
    
    def __init__(self, start: int) -> None:
        self.current: int = start
    
    def __iter__(self) -> "CountDown":
        """返回迭代器本身"""
        return self
    
    def __next__(self) -> int:
        """返回下一个值"""
        if self.current <= 0:
            raise StopIteration
        
        result: int = self.current
        self.current -= 1
        return result

# 使用 for 循环
for num in CountDown(5):
    print(num, end=" ")
# 5 4 3 2 1

# 手动使用
counter: CountDown = CountDown(3)
print(next(counter))  # 3
print(next(counter))  # 2
print(next(counter))  # 1
# print(next(counter))  # ❌ StopIteration
```

### 3.2 渐进复杂化：斐波那契数列

```python
from typing import Iterator, Optional

class Fibonacci:
    """斐波那契数列迭代器"""
    
    def __init__(self, max_count: Optional[int] = None) -> None:
        self.max_count: Optional[int] = max_count
        self.count: int = 0
        self.a: int = 0
        self.b: int = 1
    
    def __iter__(self) -> "Fibonacci":
        return self
    
    def __next__(self) -> int:
        if self.max_count is not None and self.count >= self.max_count:
            raise StopIteration
        
        result: int = self.a
        self.a, self.b = self.b, self.a + self.b
        self.count += 1
        return result

# 使用：前 10 个斐波那契数
fib: Fibonacci = Fibonacci(max_count=10)
for i, num in enumerate(fib, 1):
    print(f"{i}: {num}")
# 1: 0
# 2: 1
# 3: 1
# 4: 2
# 5: 3
# 6: 5
# 7: 8
# 8: 13
# 9: 21
# 10: 34
```

### 3.3 更复杂：范围迭代器

```python
from typing import Iterator, Optional, Union

class MyRange:
    """自定义 range 迭代器"""
    
    def __init__(
        self, 
        start: int, 
        stop: Optional[int] = None, 
        step: int = 1
    ) -> None:
        if stop is None:
            stop = start
            start = 0
        
        self.current: int = start
        self.stop: int = stop
        self.step: int = step
    
    def __iter__(self) -> "MyRange":
        return self
    
    def __next__(self) -> int:
        if (self.step > 0 and self.current >= self.stop) or \
           (self.step < 0 and self.current <= self.stop):
            raise StopIteration
        
        result: int = self.current
        self.current += self.step
        return result

# 使用
print(list(MyRange(5)))           # [0, 1, 2, 3, 4]
print(list(MyRange(1, 5)))        # [1, 2, 3, 4]
print(list(MyRange(1, 10, 2)))    # [1, 3, 5, 7, 9]
print(list(MyRange(10, 0, -2)))   # [10, 8, 6, 4, 2]
```

## 第四部分：实际应用场景

### 4.1 自定义数据结构遍历

```python
from typing import Iterator, Any

class TreeNode:
    """二叉树节点"""
    def __init__(self, value: Any, left: "TreeNode" = None, right: "TreeNode" = None) -> None:
        self.value: Any = value
        self.left: TreeNode | None = left
        self.right: TreeNode | None = right


class TreeIterator:
    """二叉树中序遍历迭代器"""
    
    def __init__(self, root: TreeNode | None) -> None:
        self._stack: list[TreeNode] = []
        self._current: TreeNode | None = root
    
    def __iter__(self) -> "TreeIterator":
        return self
    
    def __next__(self) -> Any:
        # 左子树入栈
        while self._current is not None:
            self._stack.append(self._current)
            self._current = self._current.left
        
        # 栈空则结束
        if not self._stack:
            raise StopIteration
        
        # 访问节点
        node: TreeNode = self._stack.pop()
        self._current = node.right
        return node.value

# 使用
#       4
#      / \
#     2   6
#    / \ / \
#   1  3 5  7
root = TreeNode(4,
    TreeNode(2, TreeNode(1), TreeNode(3)),
    TreeNode(6, TreeNode(5), TreeNode(7))
)

for value in TreeIterator(root):
    print(value, end=" ")
# 1 2 3 4 5 6 7
```

### 4.2 分页数据遍历

```python
from typing import Iterator, Any
import time

class PageIterator:
    """分页数据迭代器"""
    
    def __init__(self, fetch_func: Any, page_size: int = 10) -> None:
        self.fetch_func = fetch_func
        self.page_size: int = page_size
        self.current_page: int = 0
        self.current_items: list[Any] = []
        self.current_index: int = 0
    
    def __iter__(self) -> "PageIterator":
        return self
    
    def __next__(self) -> Any:
        # 当前页数据用完，获取下一页
        if self.current_index >= len(self.current_items):
            self.current_page += 1
            self.current_items = self.fetch_func(self.current_page, self.page_size)
            self.current_index = 0
            
            # 没有更多数据
            if not self.current_items:
                raise StopIteration
        
        item: Any = self.current_items[self.current_index]
        self.current_index += 1
        return item

# 模拟 API 获取数据
def fetch_page(page: int, size: int) -> list[dict[str, Any]]:
    """模拟 API 分页获取"""
    if page > 3:  # 只有 3 页数据
        return []
    return [{"id": (page - 1) * size + i, "page": page} for i in range(size)]

# 使用
for item in PageIterator(fetch_page, page_size=5):
    print(item, end=" ")
# {'id': 0, 'page': 1} {'id': 1, 'page': 1} ... {'id': 14, 'page': 3}
```

### 4.3 文件逐行处理

```python
from typing import Iterator

class FileReader:
    """文件逐行读取迭代器"""
    
    def __init__(self, filename: str, encoding: str = "utf-8") -> None:
        self.filename: str = filename
        self.encoding: str = encoding
        self._file = None
        self._closed: bool = False
    
    def __iter__(self) -> "FileReader":
        self._file = open(self.filename, "r", encoding=self.encoding)
        return self
    
    def __next__(self) -> str:
        if self._file is None:
            raise StopIteration
        
        line: str = self._file.readline()
        if not line:
            self._file.close()
            self._closed = True
            raise StopIteration
        
        return line.strip()
    
    def __del__(self) -> None:
        if hasattr(self, "_file") and self._file and not self._closed:
            self._file.close()

# 使用
for line in FileReader("data.txt"):
    print(line)
```

**关键代码说明：**

| 代码 | 含义 | 为什么这样写 |
|------|------|-------------|
| `def __iter__(self) -> "FileReader"` | `__iter__` 返回 self | 迭代器协议要求 `__iter__` 返回迭代器本身，让对象既可迭代又是迭代器 |
| `self._file = open(...)` 在 `__iter__` 中打开 | 延迟打开文件 | 直到真正开始迭代才打开文件，避免创建对象时就占用资源 |
| `line = self._file.readline(); if not line: raise StopIteration` | 读一行，空行时停止 | `readline()` 在文件末尾返回空字符串，以此触发 `StopIteration` 结束迭代 |
| `def __del__(self)` | 对象销毁时自动关闭文件 | 防止迭代未完成就销毁对象时文件句柄泄漏 |

## L2 实践层：迭代器设计最佳实践

### 推荐做法

| 做法 | 原因 | 示例 |
|------|------|------|
| **优先使用生成器而非迭代器类** | 代码更简洁，Python 自动处理协议 | `def gen(): yield x` vs `class Iter: __iter__, __next__` |
| **__iter__ 返回 self** | 让迭代器本身也是可迭代对象 | `def __iter__(self): return self` |
| **__next__ 抛出 StopIteration** | 标准协议，for 循环自动处理 | `raise StopIteration` |
| **添加类型注解** | 提高代码可读性和 IDE 支持 | `def __next__(self) -> int:` |
| **复杂场景用 yield from** | 简化嵌套迭代器的委托 | `yield from sub_iterator()` |
| **提供 reset 或 rewind 方法** | 允许重新遍历（非标准但有用的扩展） | `def reset(self): self._index = 0` |

### 反模式：不要这样做

```python
# ❌ __iter__ 返回新的迭代器对象（常见错误）
class WrongIterator:
    def __init__(self, data):
        self.data = data
        self.index = 0

    def __iter__(self):
        # 错误：每次调用都创建新迭代器，状态丢失
        return MyIterator(self.data)  # 返回新对象

    def __next__(self):
        if self.index >= len(self.data):
            raise StopIteration
        return self.data[self.index]

# ✅ 正确做法：__iter__ 返回 self
class CorrectIterator:
    def __init__(self, data):
        self.data = data
        self.index = 0

    def __iter__(self):
        return self  # 返回自身

    def __next__(self):
        if self.index >= len(self.data):
            raise StopIteration
        result = self.data[self.index]
        self.index += 1
        return result
```

```python
# ❌ 迭代器类没有实现 __next__
class NotAnIterator:
    def __init__(self, data):
        self.data = data

    def __iter__(self):
        return self  # 只有 __iter__，没有 __next__

# 尝试使用
it = NotAnIterator([1, 2, 3])
for x in it:  # TypeError: 'NotAnIterator' object is not an iterator

# ✅ 正确做法：同时实现 __iter__ 和 __next__
class ProperIterator:
    def __init__(self, data):
        self.data = data
        self.index = 0

    def __iter__(self):
        return self

    def __next__(self):
        if self.index >= len(self.data):
            raise StopIteration
        result = self.data[self.index]
        self.index += 1
        return result
```

```python
# ❌ 使用 return 代替 raise StopIteration
def wrong_generator():
    yield 1
    yield 2
    return "done"  # return 值被忽略（非生成器表达式）

class WrongIterator:
    def __next__(self):
        if self.done:
            return None  # 错误：不抛出 StopIteration
        return self.value

# ✅ 正确做法：使用 raise StopIteration
class CorrectIterator:
    def __next__(self):
        if self.done:
            raise StopIteration  # 正确：抛出异常
        return self.value
```

```python
# ❌ 在 __iter__ 中执行耗时操作
class SlowIterator:
    def __iter__(self):
        # 错误：每次 for 循环都执行耗时操作
        data = fetch_data_from_api()  # 耗时操作
        for item in data:
            yield item

# ✅ 正确做法：耗时操作放在 __init__ 或首次 __next__
class EfficientIterator:
    def __init__(self):
        self._data = None
        self._fetched = False

    def _ensure_data(self):
        if not self._fetched:
            self._data = fetch_data_from_api()
            self._fetched = True

    def __iter__(self):
        return self

    def __next__(self):
        self._ensure_data()  # 首次调用时获取数据
        # ...
```

### 适用场景

| 场景 | 是否推荐 | 原因 |
|------|---------|------|
| 简单序列遍历 | ❌ 不推荐 | 用生成器更简洁 |
| 自定义数据结构遍历（树、图） | ✅ 推荐 | 需要复杂状态管理 |
| 分页数据加载 | ✅ 推荐 | 需要维护页码和索引状态 |
| 文件流处理 | ✅ 推荐 | 需要控制文件句柄生命周期 |
| 无限序列 | ✅ 推荐 | 需要维护计算状态 |
| 需要多次遍历 | ❌ 不推荐迭代器 | 用可迭代对象，每次返回新迭代器 |

### 迭代器 vs 可迭代对象设计模式

```
设计模式选择：
┌─────────────────────────────────────────────────────────────┐
│                                                             │
│  模式一：迭代器（Iterator）                                  │
│  ───────────────────────────                               │
│  • __iter__ 返回 self                                      │
│  • 一次性使用，耗尽后需重新创建                              │
│  • 适合：数据流、文件、无限序列                              │
│                                                             │
│  模式二：可迭代对象（Iterable）                              │
│  ─────────────────────────────                             │
│  • __iter__ 返回新的迭代器对象                              │
│  • 可多次遍历，每次调用 __iter__ 创建新迭代器               │
│  • 适合：容器类（list、dict、自定义容器）                    │
│                                                             │
│  选择依据：                                                  │
│  ─────────────────                                          │
│  • 需要多次遍历 → 使用可迭代对象模式                         │
│  • 只需一次遍历或数据流 → 使用迭代器模式                     │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

```python
# 可迭代对象模式：每次 for 循环都创建新迭代器
class MyContainer:
    """可迭代对象：可以多次遍历"""
    def __init__(self, data):
        self._data = data

    def __iter__(self):
        # 每次调用返回新的迭代器
        return MyContainerIterator(self._data)

class MyContainerIterator:
    """独立的迭代器类"""
    def __init__(self, data):
        self._data = data
        self._index = 0

    def __iter__(self):
        return self

    def __next__(self):
        if self._index >= len(self._data):
            raise StopIteration
        result = self._data[self._index]
        self._index += 1
        return result

# 使用：可以多次遍历
container = MyContainer([1, 2, 3])
for x in container:
    print(x)  # 1 2 3
for x in container:  # 新迭代器
    print(x)  # 1 2 3（再次遍历成功）
```

## L3 专家层：__iter__/__next__ 协议底层原理

### Python 如何实现迭代器协议

```
迭代器协议底层实现：
┌─────────────────────────────────────────────────────────────┐
│  迭代器协议的双方法设计                                      │
│                                                             │
│  __iter__(self)                                             │
│  ─────────────────                                          │
│  • 返回迭代器对象（通常是 self）                            │
│  • 让对象可以在 for 循环中使用                              │
│  • 被 iter() 函数调用                                       │
│  • 对于迭代器：返回 self                                    │
│  • 对于可迭代对象：返回新迭代器                             │
│                                                             │
│  __next__(self)                                             │
│  ─────────────────                                          │
│  • 返回下一个元素                                           │
│  • 被 next() 函数调用                                       │
│  • 没有更多元素时抛出 StopIteration                         │
│  • 必须维护内部状态（位置指针）                             │
│                                                             │
│  协议检查机制：                                              │
│  ─────────────────                                          │
│  isinstance(obj, Iterable) → 检查 __iter__                 │
│  isinstance(obj, Iterator) → 检查 __iter__ + __next__     │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

```python
# 演示迭代器协议的底层检查
from collections.abc import Iterable, Iterator
import inspect

# 查看抽象基类的抽象方法
print(f"Iterable 抽象方法：{Iterable.__abstractmethods__}")
# frozenset({'__iter__'})

print(f"Iterator 抽象方法：{Iterator.__abstractmethods__}")
# frozenset({'__next__'})

# 注意：Iterator 继承 Iterable，所以需要 __iter__ 和 __next__
print(f"Iterator 继承自：{Iterator.__bases__}")
# (Iterable,)

# 手动实现检查
def is_iterator(obj):
    return hasattr(obj, '__iter__') and hasattr(obj, '__next__')

def is_iterable(obj):
    return hasattr(obj, '__iter__')
```

### StopIteration 的工作原理

```
StopIteration 异常机制：
┌─────────────────────────────────────────────────────────────┐
│                                                             │
│  异常作为控制流：                                            │
│  ─────────────────                                          │
│  • StopIteration 是信号，不是错误                           │
│  • for 循环自动捕获并结束循环                                │
│  • 手动调用 next() 时需要处理                                │
│                                                             │
│  异常属性：                                                  │
│  ─────────────────                                          │
│  • value 属性：可携带返回值（生成器 return 值）              │
│  • Python 3.7+：生成器 return 值存于 StopIteration.value   │
│                                                             │
│  捕获位置：                                                  │
│  ─────────────────                                          │
│  • for 循环内部自动捕获                                      │
│  • next() 无默认值时向上传播                                 │
│  • next(gen, default) 时忽略异常返回 default               │
│                                                             │
│  性能考量：                                                  │
│  ─────────────────                                          │
│  • 异常抛出有开销，但只在结束时一次                          │
│  • 比检查返回值更清晰                                        │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

```python
# StopIteration 的 value 属性
def generator_with_return():
    yield 1
    yield 2
    return "完成"  # return 值

gen = generator_with_return()
try:
    while True:
        print(next(gen))
except StopIteration as e:
    print(f"返回值：{e.value}")  # "完成"

# 对比：迭代器的 StopIteration 通常没有 value
class SimpleIterator:
    def __init__(self, data):
        self.data = data
        self.index = 0

    def __iter__(self):
        return self

    def __next__(self):
        if self.index >= len(self.data):
            raise StopIteration  # 没有携带值
        result = self.data[self.index]
        self.index += 1
        return result

it = SimpleIterator([1, 2])
try:
    while True:
        print(next(it))
except StopIteration as e:
    print(f"value: {e.value}")  # None
```

### 性能考量

| 操作 | 时间复杂度 | 空间复杂度 | 说明 |
|------|-----------|-----------|------|
| 创建迭代器对象 | O(1) | O(1) | 只分配对象空间 |
| __next__ 方法调用 | O(1) | O(1) | 单次元素访问 |
| StopIteration 抛出 | O(1) | O(1) | 只在结束时一次 |
| 迭代器类 vs 生成器 | 相似 | 生成器略小 | 生成器帧对象更轻量 |

```python
# 性能对比：迭代器类 vs 生成器函数
import timeit
import sys

# 迭代器类
class CounterIterator:
    def __init__(self, n):
        self.n = n
        self.current = 0

    def __iter__(self):
        return self

    def __next__(self):
        if self.current >= self.n:
            raise StopIteration
        result = self.current
        self.current += 1
        return result

# 生成器函数
def counter_generator(n):
    for i in range(n):
        yield i

# 内存对比
iter_obj = CounterIterator(1000000)
gen_obj = counter_generator(1000000)
print(f"迭代器类内存：{sys.getsizeof(iter_obj)} 字节")  # 约 56 字节
print(f"生成器内存：{sys.getsizeof(gen_obj)} 字节")     # 约 200 字节

# 时间对比（迭代器类通常更快）
def test_iterator():
    return sum(CounterIterator(10000))

def test_generator():
    return sum(counter_generator(10000))

print(f"迭代器类时间：{timeit.timeit(test_iterator, number=1000):.3f}s")
print(f"生成器时间：{timeit.timeit(test_generator, number=1000):.3f}s")
```

### 设计动机

| 设计选择 | 原因 | 替代方案对比 |
|----------|------|-------------|
| 双方法协议（__iter__ + __next__） | 分离可迭代和迭代器概念 | C++ 用单独的 iterator 类型 |
| __iter__ 返回 self | 迭代器本身可迭代，简化 for 循环 | Java Iterator 不支持直接 for |
| StopIteration 作为结束信号 | 异常机制清晰，避免哨兵值 | 返回 None 需额外判断 |
| 迭代器一次性 | 内存效率，支持无限序列 | 可重置迭代器需要额外状态 |

**为什么需要两个方法？**

```
双方法设计原因：
┌─────────────────────────────────────────────────────────────┐
│                                                             │
│  1. 分离概念                                                │
│  ─────────────────                                          │
│  • Iterable：拥有数据，可创建迭代器                         │
│  • Iterator：遍历数据，维护状态                             │
│  • 一个 Iterable 可创建多个 Iterator                        │
│                                                             │
│  2. 支持多次遍历                                            │
│  ─────────────────                                          │
│  • list.__iter__() 每次返回新迭代器                         │
│  • 多个 for 循环可以独立遍历同一列表                        │
│                                                             │
│  3. 状态隔离                                                │
│  ─────────────────                                          │
│  • 不同迭代器有独立的遍历位置                               │
│  • 遍历一个不影响另一个                                     │
│                                                             │
│  示例：                                                     │
│  ─────────────────                                          │
│  lst = [1, 2, 3]                                            │
│  it1 = iter(lst)  # 位置：0                                 │
│  it2 = iter(lst)  # 位置：0（独立）                         │
│  next(it1)        # it1 位置：1                             │
│  next(it2)        # it2 位置：1（不受影响）                 │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

# 自定义迭代器

本章讲解如何创建自定义迭代器类，实现迭代器协议。

---

## 迭代器协议

### 两个必需方法

```python
class MyIterator:
    def __iter__(self):
        """返回迭代器本身（必须）"""
        return self

    def __next__(self):
        """返回下一个值（必须）"""
        # 如果没有更多元素，抛出 StopIteration
        raise StopIteration
```

---

## 示例：倒数计时器

```python
class CountDown:
    """倒数计时器迭代器"""

    def __init__(self, start):
        self.start = start

    def __iter__(self):
        """返回迭代器本身"""
        return self

    def __next__(self):
        """返回下一个值"""
        if self.start <= 0:
            raise StopIteration

        self.start -= 1
        return self.start + 1

# 使用
for num in CountDown(5):
    print(num)
# 输出：5, 4, 3, 2, 1

# 手动使用
counter = CountDown(3)
print(next(counter))  # 3
print(next(counter))  # 2
print(next(counter))  # 1
# print(next(counter))  # ❌ StopIteration
```

## 示例：斐波那契数列

```python
class Fibonacci:
    """斐波那契数列迭代器"""

    def __init__(self, max_count=None):
        self.max_count = max_count
        self.count = 0
        self.a, self.b = 0, 1

    def __iter__(self):
        return self

    def __next__(self):
        if self.max_count and self.count >= self.max_count:
            raise StopIteration

        result = self.a
        self.a, self.b = self.b, self.a + self.b
        self.count += 1
        return result

# 使用：前 10 个斐波那契数
for i, num in enumerate(Fibonacci(max_count=10), 1):
    print(f"{i}: {num}")
```

## 示例：范围迭代器

```python
class MyRange:
    """自定义 range 迭代器"""

    def __init__(self, start, stop=None, step=1):
        if stop is None:
            stop = start
            start = 0
        self.current = start
        self.stop = stop
        self.step = step

    def __iter__(self):
        return self

    def __next__(self):
        if (self.step > 0 and self.current >= self.stop) or \
           (self.step < 0 and self.current <= self.stop):
            raise StopIteration

        result = self.current
        self.current += self.step
        return result

# 使用
print(list(MyRange(5)))           # [0, 1, 2, 3, 4]
print(list(MyRange(1, 5)))        # [1, 2, 3, 4]
print(list(MyRange(1, 10, 2)))    # [1, 3, 5, 7, 9]
```

## 第一部分：为什么需要自定义迭代器？

### 1.1 实际场景引入

假设你需要实现一个倒计时功能：

```python
# 尝试一：使用列表（内存占用大）
def countdown_list(n: int) -> list[int]:
    return list(range(n, 0, -1))

for num in countdown_list(1000000):  # ❌ 生成百万个元素的列表！
    if num == 999999:
        break

# 尝试二：使用自定义迭代器（内存友好）
class Countdown:
    def __init__(self, n: int) -> None:
        self.n = n
    
    def __iter__(self) -> "Countdown":
        return self
    
    def __next__(self) -> int:
        if self.n <= 0:
            raise StopIteration
        self.n -= 1
        return self.n + 1

for num in Countdown(1000000):  # ✅ 每次只计算一个值
    if num == 999999:
        break
```

**问题**：如何创建自定义的迭代器？需要实现哪些方法？

### 1.2 概念动机

Python 的迭代器协议定义了两个必需方法：

```
┌─────────────────────────────────────────┐
│         迭代器协议 (Iterator Protocol)   │
├─────────────────────────────────────────┤
│                                         │
│  __iter__(self)                         │
│  ────────────────                       │
│  • 返回迭代器本身                       │
│  • 允许在 for 循环中使用                │
│                                         │
│  __next__(self)                         │
│  ────────────────                       │
│  • 返回下一个值                         │
│  • 没有更多元素时抛出 StopIteration     │
│                                         │
└─────────────────────────────────────────┘
```

## 第二部分：迭代器协议

### 2.1 最简示例：空迭代器

```python
from typing import Iterator

class EmptyIterator:
    """最简单的迭代器：立即停止"""
    
    def __iter__(self) -> "EmptyIterator":
        """返回迭代器本身（必须）"""
        return self
    
    def __next__(self) -> None:
        """返回下一个值（必须）"""
        # 如果没有更多元素，抛出 StopIteration
        raise StopIteration

# 使用
empty: EmptyIterator = EmptyIterator()
print(list(empty))  # []
```

### 2.2 详细讲解：迭代器协议模板

```python
from typing import Iterator, Generic, TypeVar

T = TypeVar("T")

class MyIterator(Generic[T]):
    """自定义迭代器模板"""
    
    def __init__(self, data: list[T]) -> None:
        self._data: list[T] = data
        self._index: int = 0
    
    def __iter__(self) -> "MyIterator[T]":
        """返回迭代器本身"""
        return self
    
    def __next__(self) -> T:
        """返回下一个值"""
        if self._index >= len(self._data):
            raise StopIteration
        
        value: T = self._data[self._index]
        self._index += 1
        return value

# 使用
my_iter: MyIterator[int] = MyIterator([1, 2, 3])
for item in my_iter:
    print(item, end=" ")
# 1 2 3
```

## 第三部分：实际示例

### 3.1 倒数计时器

```python
from typing import Iterator

class CountDown:
    """倒数计时器迭代器"""
    
    def __init__(self, start: int) -> None:
        self.current: int = start
    
    def __iter__(self) -> "CountDown":
        """返回迭代器本身"""
        return self
    
    def __next__(self) -> int:
        """返回下一个值"""
        if self.current <= 0:
            raise StopIteration
        
        result: int = self.current
        self.current -= 1
        return result

# 使用 for 循环
for num in CountDown(5):
    print(num, end=" ")
# 5 4 3 2 1

# 手动使用
counter: CountDown = CountDown(3)
print(next(counter))  # 3
print(next(counter))  # 2
print(next(counter))  # 1
# print(next(counter))  # ❌ StopIteration
```

### 3.2 渐进复杂化：斐波那契数列

```python
from typing import Iterator, Optional

class Fibonacci:
    """斐波那契数列迭代器"""
    
    def __init__(self, max_count: Optional[int] = None) -> None:
        self.max_count: Optional[int] = max_count
        self.count: int = 0
        self.a: int = 0
        self.b: int = 1
    
    def __iter__(self) -> "Fibonacci":
        return self
    
    def __next__(self) -> int:
        if self.max_count is not None and self.count >= self.max_count:
            raise StopIteration
        
        result: int = self.a
        self.a, self.b = self.b, self.a + self.b
        self.count += 1
        return result

# 使用：前 10 个斐波那契数
fib: Fibonacci = Fibonacci(max_count=10)
for i, num in enumerate(fib, 1):
    print(f"{i}: {num}")
# 1: 0
# 2: 1
# 3: 1
# 4: 2
# 5: 3
# 6: 5
# 7: 8
# 8: 13
# 9: 21
# 10: 34
```

### 3.3 更复杂：范围迭代器

```python
from typing import Iterator, Optional, Union

class MyRange:
    """自定义 range 迭代器"""
    
    def __init__(
        self, 
        start: int, 
        stop: Optional[int] = None, 
        step: int = 1
    ) -> None:
        if stop is None:
            stop = start
            start = 0
        
        self.current: int = start
        self.stop: int = stop
        self.step: int = step
    
    def __iter__(self) -> "MyRange":
        return self
    
    def __next__(self) -> int:
        if (self.step > 0 and self.current >= self.stop) or \
           (self.step < 0 and self.current <= self.stop):
            raise StopIteration
        
        result: int = self.current
        self.current += self.step
        return result

# 使用
print(list(MyRange(5)))           # [0, 1, 2, 3, 4]
print(list(MyRange(1, 5)))        # [1, 2, 3, 4]
print(list(MyRange(1, 10, 2)))    # [1, 3, 5, 7, 9]
print(list(MyRange(10, 0, -2)))   # [10, 8, 6, 4, 2]
```

## 第四部分：实际应用场景

### 4.1 自定义数据结构遍历

```python
from typing import Iterator, Any

class TreeNode:
    """二叉树节点"""
    def __init__(self, value: Any, left: "TreeNode" = None, right: "TreeNode" = None) -> None:
        self.value: Any = value
        self.left: TreeNode | None = left
        self.right: TreeNode | None = right


class TreeIterator:
    """二叉树中序遍历迭代器"""
    
    def __init__(self, root: TreeNode | None) -> None:
        self._stack: list[TreeNode] = []
        self._current: TreeNode | None = root
    
    def __iter__(self) -> "TreeIterator":
        return self
    
    def __next__(self) -> Any:
        # 左子树入栈
        while self._current is not None:
            self._stack.append(self._current)
            self._current = self._current.left
        
        # 栈空则结束
        if not self._stack:
            raise StopIteration
        
        # 访问节点
        node: TreeNode = self._stack.pop()
        self._current = node.right
        return node.value

# 使用
#       4
#      / \
#     2   6
#    / \ / \
#   1  3 5  7
root = TreeNode(4,
    TreeNode(2, TreeNode(1), TreeNode(3)),
    TreeNode(6, TreeNode(5), TreeNode(7))
)

for value in TreeIterator(root):
    print(value, end=" ")
# 1 2 3 4 5 6 7
```

### 4.2 分页数据遍历

```python
from typing import Iterator, Any
import time

class PageIterator:
    """分页数据迭代器"""
    
    def __init__(self, fetch_func: Any, page_size: int = 10) -> None:
        self.fetch_func = fetch_func
        self.page_size: int = page_size
        self.current_page: int = 0
        self.current_items: list[Any] = []
        self.current_index: int = 0
    
    def __iter__(self) -> "PageIterator":
        return self
    
    def __next__(self) -> Any:
        # 当前页数据用完，获取下一页
        if self.current_index >= len(self.current_items):
            self.current_page += 1
            self.current_items = self.fetch_func(self.current_page, self.page_size)
            self.current_index = 0
            
            # 没有更多数据
            if not self.current_items:
                raise StopIteration
        
        item: Any = self.current_items[self.current_index]
        self.current_index += 1
        return item

# 模拟 API 获取数据
def fetch_page(page: int, size: int) -> list[dict[str, Any]]:
    """模拟 API 分页获取"""
    if page > 3:  # 只有 3 页数据
        return []
    return [{"id": (page - 1) * size + i, "page": page} for i in range(size)]

# 使用
for item in PageIterator(fetch_page, page_size=5):
    print(item, end=" ")
# {'id': 0, 'page': 1} {'id': 1, 'page': 1} ... {'id': 14, 'page': 3}
```

### 4.3 文件逐行处理

```python
from typing import Iterator

class FileReader:
    """文件逐行读取迭代器"""
    
    def __init__(self, filename: str, encoding: str = "utf-8") -> None:
        self.filename: str = filename
        self.encoding: str = encoding
        self._file = None
        self._closed: bool = False
    
    def __iter__(self) -> "FileReader":
        self._file = open(self.filename, "r", encoding=self.encoding)
        return self
    
    def __next__(self) -> str:
        if self._file is None:
            raise StopIteration
        
        line: str = self._file.readline()
        if not line:
            self._file.close()
            self._closed = True
            raise StopIteration
        
        return line.strip()
    
    def __del__(self) -> None:
        if hasattr(self, "_file") and self._file and not self._closed:
            self._file.close()

# 使用
for line in FileReader("data.txt"):
    print(line)
```

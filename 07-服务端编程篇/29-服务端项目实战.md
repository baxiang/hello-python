```python
from flask_restful import Resource, request
from flask_jwt_extended import create_access_token, create_refresh_token
from app.models import User
from app.schemas import UserRegisterSchema, UserLoginSchema


class RegisterResource(Resource):
    """用户注册"""

    def post(self):
        data = UserRegisterSchema().load(request.get_json())

        # 检查用户名是否已存在
        if User.query.filter_by(username=data['username']).first():
            return {'error': '用户名已存在'}, 400

        # 检查邮箱是否已存在
        if User.query.filter_by(email=data['email']).first():
            return {'error': '邮箱已被注册'}, 400

        # 创建用户
        user = User(username=data['username'], email=data['email'])
        user.set_password(data['password'])
        db.session.add(user)
        db.session.commit()

        return {'message': '注册成功', 'user_id': user.id}, 201


class LoginResource(Resource):
    """用户登录"""

    def post(self):
        data = UserLoginSchema().load(request.get_json())

        user = User.query.filter_by(username=data['username']).first()

        if not user or not user.check_password(data['password']):
            return {'error': '用户名或密码错误'}, 401

        if not user.is_active:
            return {'error': '账号已被禁用'}, 403

        # 生成 Token
        access_token = create_access_token(identity=user.id)
        refresh_token = create_refresh_token(identity=user.id)

        return {
            'access_token': access_token,
            'refresh_token': refresh_token,
            'user': {
                'id': user.id,
                'username': user.username,
                'email': user.email
            }
        }, 200
```

```python
# app/routes/article.py
from flask_restful import Resource, request, abort
from flask_jwt_extended import jwt_required, get_jwt_identity
from app.extensions import db
from app.models import Article, Tag
from app.schemas import ArticleSchema, ArticleCreateSchema


class ArticleListResource(Resource):
    """文章列表"""

    def get(self):
        """获取文章列表"""
        page = request.args.get('page', 1, type=int)
        per_page = request.args.get('per_page', 10, type=int)
        status = request.args.get('status', 'published')
        tag = request.args.get('tag')

        query = Article.query

        # 过滤已发布
        if status:
            query = query.filter_by(status=status)

        # 按标签过滤
        if tag:
            tag_obj = Tag.query.filter_by(name=tag).first()
            if tag_obj:
                query = query.filter(Article.tags.contains(tag_obj))

        # 分页
        pagination = query.order_by(Article.created_at.desc())\
            .paginate(page=page, per_page=per_page, error_out=False)

        return {
            'articles': ArticleSchema(many=True).dump(pagination.items),
            'pagination': {
                'page': page,
                'per_page': per_page,
                'total': pagination.total,
                'pages': pagination.pages
            }
        }

    @jwt_required()
    def post(self):
        """创建文章"""
        current_user_id = get_jwt_identity()
        data = ArticleCreateSchema().load(request.get_json())

        article = Article(
            title=data['title'],
            content=data['content'],
            summary=data.get('summary'),
            author_id=current_user_id
        )

        # 处理标签
        if 'tags' in data:
            for tag_name in data['tags']:
                tag = Tag.query.filter_by(name=tag_name).first()
                if not tag:
                    tag = Tag(name=tag_name)
                article.tags.append(tag)

        db.session.add(article)
        db.session.commit()

        return ArticleSchema().dump(article), 201


class ArticleResource(Resource):
    """单篇文章"""

    def get(self, article_id):
        """获取文章详情"""
        article = Article.query.get_or_404(article_id)

        # 增加阅读量
        article.view_count += 1
        db.session.commit()

        return ArticleSchema().dump(article)

    @jwt_required()
    def put(self, article_id):
        """更新文章"""
        current_user_id = get_jwt_identity()
        article = Article.query.get_or_404(article_id)

        # 权限检查
        if article.author_id != current_user_id:
            return {'error': '无权限修改此文章'}, 403

        data = request.get_json()
        article.title = data.get('title', article.title)
        article.content = data.get('content', article.content)
        article.summary = data.get('summary', article.summary)
        article.status = data.get('status', article.status)

        db.session.commit()
        return ArticleSchema().dump(article)

    @jwt_required()
    def delete(self, article_id):
        """删除文章"""
        current_user_id = get_jwt_identity()
        article = Article.query.get_or_404(article_id)

        # 权限检查
        if article.author_id != current_user_id:
            return {'error': '无权限删除此文章'}, 403

        db.session.delete(article)
        db.session.commit()
        return {'message': '删除成功'}, 204
```

```python
# app/database.py
from sqlalchemy.ext.asyncio import create_async_engine, AsyncSession
from sqlalchemy.orm import sessionmaker, declarative_base
from sqlalchemy import create_engine
import os

DATABASE_URL = "sqlite+aiosqlite:///./todo.db"

engine = create_async_engine(DATABASE_URL, echo=True)
sync_engine = create_engine(DATABASE_URL.replace("+aiosqlite", ""))

AsyncSessionLocal = sessionmaker(
    engine,
    class_=AsyncSession,
    expire_on_commit=False,
    autocommit=False,
    autoflush=False
)

Base = declarative_base()


async def get_db() -> AsyncSession:
    """数据库会话依赖"""
    async with AsyncSessionLocal() as session:
        try:
            yield session
            await session.commit()
        except Exception:
            await session.rollback()
            raise
        finally:
            await session.close()


async def init_db():
    """初始化数据库"""
    async with engine.begin() as conn:
        await conn.run_sync(Base.metadata.create_all)
```

### 29.9 待办事项路由

```python
# app/routes/todos.py
from fastapi import APIRouter, Depends, HTTPException, Query
from sqlalchemy import select, func
from sqlalchemy.ext.asyncio import AsyncSession
from app.database import get_db
from app.models.todo import Todo
from app.schemas.todo import TodoCreate, TodoUpdate, TodoResponse
from app.auth import get_current_user
from app.models.user import User

router = APIRouter(prefix="/todos", tags=["待办事项"])


@router.get("/", response_model=list[TodoResponse])
async def list_todos(
    skip: int = Query(0, ge=0),
    limit: int = Query(20, ge=1, le=100),
    completed: bool = None,
    db: AsyncSession = Depends(get_db),
    current_user: User = Depends(get_current_user)
):
    """获取待办列表"""
    query = select(Todo).where(Todo.owner_id == current_user.id)

    if completed is not None:
        query = query.where(Todo.completed == completed)

    query = query.offset(skip).limit(limit).order_by(Todo.priority.desc())

    result = await db.execute(query)
    todos = result.scalars().all()
    return todos


@router.post("/", response_model=TodoResponse)
async def create_todo(
    todo: TodoCreate,
    db: AsyncSession = Depends(get_db),
    current_user: User = Depends(get_current_user)
):
    """创建待办"""
    db_todo = Todo(**todo.model_dump(), owner_id=current_user.id)
    db.add(db_todo)
    await db.commit()
    await db.refresh(db_todo)
    return db_todo


@router.get("/{todo_id}", response_model=TodoResponse)
async def get_todo(
    todo_id: int,
    db: AsyncSession = Depends(get_db),
    current_user: User = Depends(get_current_user)
):
    """获取待办详情"""
    result = await db.execute(
        select(Todo).where(Todo.id == todo_id, Todo.owner_id == current_user.id)
    )
    todo = result.scalar_one_or_none()

    if not todo:
        raise HTTPException(status_code=404, detail="待办不存在")

    return todo


@router.patch("/{todo_id}", response_model=TodoResponse)
async def update_todo(
    todo_id: int,
    todo_update: TodoUpdate,
    db: AsyncSession = Depends(get_db),
    current_user: User = Depends(get_current_user)
):
    """更新待办"""
    result = await db.execute(
        select(Todo).where(Todo.id == todo_id, Todo.owner_id == current_user.id)
    )
    todo = result.scalar_one_or_none()

    if not todo:
        raise HTTPException(status_code=404, detail="待办不存在")

    update_data = todo_update.model_dump(exclude_unset=True)
    for field, value in update_data.items():
        setattr(todo, field, value)

    await db.commit()
    await db.refresh(todo)
    return todo


@router.delete("/{todo_id}")
async def delete_todo(
    todo_id: int,
    db: AsyncSession = Depends(get_db),
    current_user: User = Depends(get_current_user)
):
    """删除待办"""
    result = await db.execute(
        select(Todo).where(Todo.id == todo_id, Todo.owner_id == current_user.id)
    )
    todo = result.scalar_one_or_none()

    if not todo:
        raise HTTPException(status_code=404, detail="待办不存在")

    await db.delete(todo)
    await db.commit()
    return {"message": "删除成功"}
```

### 29.11 连接管理器

```python
# app/websocket_manager.py
from fastapi import WebSocket
from typing import Dict, List, Set


class ConnectionManager:
    def __init__(self):
        # 房间 -> 连接集合
        self.rooms: Dict[str, Set[WebSocket]] = {}
        # 连接 -> 用户信息
        self.connections: Dict[WebSocket, dict] = {}

    async def connect(self, websocket: WebSocket, room: str, username: str):
        """接受连接并加入房间"""
        await websocket.accept()

        if room not in self.rooms:
            self.rooms[room] = set()

        self.rooms[room].add(websocket)
        self.connections[websocket] = {
            "room": room,
            "username": username
        }

        # 通知所有人有新用户加入
        await self.broadcast(
            room,
            {"type": "system", "message": f"{username} 加入了聊天室"}
        )

        # 发送欢迎消息
        await self.send_personal(
            websocket,
            {
                "type": "welcome",
                "message": f"欢迎加入 {room} 房间",
                "users": self.get_room_users(room)
            }
        )

    def disconnect(self, websocket: WebSocket):
        """断开连接"""
        if websocket not in self.connections:
            return

        info = self.connections[websocket]
        room = info["room"]
        username = info["username"]

        if room in self.rooms:
            self.rooms[room].discard(websocket)
            if not self.rooms[room]:
                del self.rooms[room]

        del self.connections[websocket]

        # 通知所有人用户离开
        if room in self.rooms:
            import asyncio
            asyncio.create_task(
                self.broadcast(
                    room,
                    {"type": "system", "message": f"{username} 离开了聊天室"}
                )
            )

    async def send_personal(self, websocket: WebSocket, message: dict):
        """发送个人消息"""
        await websocket.send_json(message)

    async def broadcast(self, room: str, message: dict):
        """广播消息给房间内所有人"""
        if room not in self.rooms:
            return

        disconnected = []
        for connection in self.rooms[room]:
            try:
                await connection.send_json(message)
            except:
                disconnected.append(connection)

        # 清理断开的连接
        for conn in disconnected:
            self.disconnect(conn)

    def get_room_users(self, room: str) -> List[str]:
        """获取房间内所有用户"""
        if room not in self.rooms:
            return []
        return [
            self.connections[conn]["username"]
            for conn in self.rooms[room]
            if conn in self.connections
        ]


manager = ConnectionManager()
```

### 29.12 WebSocket 端点

```python
# app/routes/chat.py
from fastapi import APIRouter, WebSocket, WebSocketDisconnect, Query
from app.websocket_manager import manager
from datetime import datetime

router = APIRouter(tags=["聊天室"])


@router.websocket("/ws/{room}")
async def chat_websocket(
    websocket: WebSocket,
    room: str,
    username: str = Query(..., min_length=1, max_length=20)
):
    """WebSocket 聊天端点"""
    await manager.connect(websocket, room, username)

    try:
        while True:
            # 接收消息
            data = await websocket.receive_json()

            # 构建消息
            message = {
                "type": "message",
                "username": username,
                "content": data.get("content", ""),
                "timestamp": datetime.utcnow().isoformat()
            }

            # 广播给房间内所有人
            await manager.broadcast(room, message)

    except WebSocketDisconnect:
        manager.disconnect(websocket)
    except Exception as e:
        manager.disconnect(websocket)
        print(f"WebSocket 错误：{e}")


@router.get("/rooms/{room}/users")
async def get_room_users(room: str):
    """获取房间在线用户"""
    users = manager.get_room_users(room)
    return {"room": room, "users": users, "count": len(users)}
```

```html
        let ws;
        let myUsername;

        function connect() {
            const username = document.getElementById('username').value;
            const room = document.getElementById('room').value;
            myUsername = username;

            ws = new WebSocket(`ws://localhost:8000/ws/${room}?username=${username}`);

            ws.onopen = () => {
                document.getElementById('login').style.display = 'none';
                document.getElementById('chat').style.display = 'block';

                const messagesDiv = document.getElementById('messages');

                    updateUsers(data.users);
                } else if (data.type === 'system') {
                    messagesDiv.innerHTML += `<div class="system">${data.message}</div>`;
                } else if (data.type === 'message') {
                    const isMe = data.username === myUsername;
                    messagesDiv.innerHTML += `
                        <div class="message ${isMe ? 'my-message' : ''}">
                            <strong>${data.username}</strong>: ${data.content}
                        </div>
                    `;
                    messagesDiv.scrollTop = messagesDiv.scrollHeight;
                }

            ws.onclose = () => {
                alert('连接已断开');
                location.reload();
            };
        }

        function send() {
            const input = document.getElementById('messageInput');
            const content = input.value.trim();
            if (content && ws.readyState === WebSocket.OPEN) {
                ws.send(JSON.stringify({content}));
                input.value = '';
            }
        }

        function updateUsers(users) {
            document.getElementById('users').textContent = `在线：${users.join(', ')}`;
        }

        document.getElementById('messageInput').addEventListener('keypress', (e) => {
            if (e.key === 'Enter') send();
        });
```

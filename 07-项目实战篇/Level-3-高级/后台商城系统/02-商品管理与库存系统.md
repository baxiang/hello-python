```python
# app/services/category.py
from sqlalchemy.orm import Session
from sqlalchemy import and_, or_
from typing import List, Optional
from app.models.product import Category
from app.schemas.product import CategoryCreate, CategoryUpdate


class CategoryService:
    """分类服务"""

    def __init__(self, db: Session):
        self.db = db

    def get_category(self, category_id: int) -> Optional[Category]:
        """获取分类"""
        return self.db.query(Category).filter(Category.id == category_id).first()

    def get_category_tree(self, parent_id: Optional[int] = None) -> List[dict]:
        """获取分类树"""
        query = self.db.query(Category).filter(Category.is_enabled == True)

        if parent_id is None:
            query = query.filter(Category.parent_id == None)

        categories = query.order_by(Category.sort, Category.id).all()

        result = []
        for category in categories:
            item = {
                "id": category.id,
                "name": category.name,
                "parent_id": category.parent_id,
                "level": category.level,
                "icon": category.icon,
                "sort": category.sort,
                "children": self.get_category_tree(category.id) if category.level < 3 else []
            }
            result.append(item)

        return result

    def get_all_categories(self, skip: int = 0, limit: int = 100) -> List[Category]:
        """获取所有分类（分页）"""
        return self.db.query(Category).offset(skip).limit(limit).all()

    def create_category(self, data: CategoryCreate) -> Category:
        """创建分类"""
        # 检查层级
        if data.parent_id:
            parent = self.get_category(data.parent_id)
            if not parent:
                raise ValueError("父分类不存在")
            if parent.level >= 3:
                raise ValueError("分类最多支持3级")

        category = Category(
            name=data.name,
            parent_id=data.parent_id,
            level=1 if not data.parent_id else (self.get_category(data.parent_id).level + 1),
            icon=data.icon,
            sort=data.sort or 0,
            is_enabled=data.is_enabled if data.is_enabled is not None else True
        )

        self.db.add(category)
        self.db.commit()
        self.db.refresh(category)
        return category

    def update_category(self, category_id: int, data: CategoryUpdate) -> Category:
        """更新分类"""
        category = self.get_category(category_id)
        if not category:
            raise ValueError("分类不存在")

        update_data = data.model_dump(exclude_unset=True)
        for key, value in update_data.items():
            setattr(category, key, value)

        # 如果修改了父分类，需要更新层级
        if data.parent_id is not None:
            if data.parent_id:
                parent = self.get_category(data.parent_id)
                category.level = parent.level + 1
            else:
                category.level = 1

        self.db.commit()
        self.db.refresh(category)
        return category

    def delete_category(self, category_id: int) -> bool:
        """删除分类"""
        category = self.get_category(category_id)
        if not category:
            raise ValueError("分类不存在")

        # 检查是否有子分类
        if category.children:
            raise ValueError("请先删除子分类")

        # 检查是否有商品
        if category.products:
            raise ValueError("该分类下存在商品，无法删除")

        self.db.delete(category)
        self.db.commit()
        return True
```

```python
from typing import List, Optional
from app.models.product import Brand
from app.schemas.product import BrandCreate, BrandUpdate


class BrandService:
    """品牌服务"""

    def __init__(self, db: Session):
        self.db = db

    def get_brand(self, brand_id: int) -> Optional[Brand]:
        return self.db.query(Brand).filter(Brand.id == brand_id).first()

    def get_brand_by_name(self, name: str) -> Optional[Brand]:
        return self.db.query(Brand).filter(Brand.name == name).first()

    def get_brands(
        self,
        skip: int = 0,
        limit: int = 20,
        is_enabled: Optional[bool] = None
    ) -> tuple[List[Brand], int]:
        """获取品牌列表"""
        query = self.db.query(Brand)

        if is_enabled is not None:
            query = query.filter(Brand.is_enabled == is_enabled)

        total = query.count()
        items = query.order_by(Brand.sort, Brand.id).offset(skip).limit(limit).all()

        return items, total

    def create_brand(self, data: BrandCreate) -> Brand:
        """创建品牌"""
        # 检查名称唯一性
        if self.get_brand_by_name(data.name):
            raise ValueError("品牌名称已存在")

        brand = Brand(
            name=data.name,
            logo=data.logo,
            description=data.description,
            first_letter=data.first_letter,
            sort=data.sort or 0,
            is_enabled=data.is_enabled if data.is_enabled is not None else True
        )

        self.db.add(brand)
        self.db.commit()
        self.db.refresh(brand)
        return brand

    def update_brand(self, brand_id: int, data: BrandUpdate) -> Brand:
        """更新品牌"""
        brand = self.get_brand(brand_id)
        if not brand:
            raise ValueError("品牌不存在")

        # 检查名称唯一性
        if data.name and data.name != brand.name:
            if self.get_brand_by_name(data.name):
                raise ValueError("品牌名称已存在")

        update_data = data.model_dump(exclude_unset=True)
        for key, value in update_data.items():
            setattr(brand, key, value)

        self.db.commit()
        self.db.refresh(brand)
        return brand

    def delete_brand(self, brand_id: int) -> bool:
        """删除品牌"""
        brand = self.get_brand(brand_id)
        if not brand:
            raise ValueError("品牌不存在")

        if brand.products:
            raise ValueError("该品牌下存在商品，无法删除")

        self.db.delete(brand)
        self.db.commit()
        return True
```

```python
# app/services/product.py
from sqlalchemy.orm import Session, joinedload
from sqlalchemy import and_, or_, func
from typing import List, Optional, Tuple
from decimal import Decimal
import json

from app.models.product import Product, ProductSku, ProductStatus
from app.schemas.product import ProductCreate, ProductUpdate, ProductQuery


class ProductService:
    """商品服务"""

    def __init__(self, db: Session):
        self.db = db

    def get_product(self, product_id: int) -> Optional[Product]:
        """获取商品详情"""
        return (
            self.db.query(Product)
            .options(joinedload(Product.category), joinedload(Product.brand))
            .filter(Product.id == product_id)
            .first()
        )

    def get_product_by_code(self, product_code: str) -> Optional[Product]:
        """根据编码获取商品"""
        return self.db.query(Product).filter(Product.product_code == product_code).first()

    def get_products(
        self,
        query: ProductQuery,
        skip: int = 0,
        limit: int = 20
    ) -> Tuple[List[Product], int]:
        """获取商品列表"""
        db_query = self.db.query(Product).options(
            joinedload(Product.category),
            joinedload(Product.brand)
        )

        # 关键词搜索
        if query.keyword:
            db_query = db_query.filter(
                or_(
                    Product.name.ilike(f"%{query.keyword}%"),
                    Product.product_code.ilike(f"%{query.keyword}%")
                )
            )

        # 分类筛选
        if query.category_id:
            db_query = db_query.filter(Product.category_id == query.category_id)

        # 品牌筛选
        if query.brand_id:
            db_query = db_query.filter(Product.brand_id == query.brand_id)

        # 状态筛选
        if query.status:
            db_query = db_query.filter(Product.status == query.status)

        # 价格区间
        if query.min_price:
            db_query = db_query.filter(Product.price >= query.min_price)
        if query.max_price:
            db_query = db_query.filter(Product.price <= query.max_price)

        # 在售状态
        if query.on_sale:
            db_query = db_query.filter(Product.status == ProductStatus.ON_SALE)

        # 推荐/新品
        if query.is_featured is not None:
            db_query = db_query.filter(Product.is_featured == query.is_featured)
        if query.is_new is not None:
            db_query = db_query.filter(Product.is_new == query.is_new)

        # 排序
        if query.sort_by:
            sort_column = getattr(Product, query.sort_by, Product.id)
            if query.sort_order == "desc":
                db_query = db_query.order_by(sort_column.desc())
            else:
                db_query = db_query.order_by(sort_column.asc())
        else:
            db_query = db_query.order_by(Product.id.desc())

        # 统计总数
        total = db_query.count()

        # 分页
        items = db_query.offset(skip).limit(limit).all()

        return items, total

    def create_product(self, data: ProductCreate) -> Product:
        """创建商品"""
        # 检查商品编码唯一性
        if self.get_product_by_code(data.product_code):
            raise ValueError("商品编码已存在")

        # 检查分类存在
        from app.services.category import CategoryService
        category_service = CategoryService(self.db)
        category = category_service.get_category(data.category_id)
        if not category:
            raise ValueError("分类不存在")

        # 解析图片列表
        images_json = json.dumps(data.images) if data.images else None

        product = Product(
            name=data.name,
            product_code=data.product_code,
            category_id=data.category_id,
            brand_id=data.brand_id,
            price=data.price,
            original_price=data.original_price,
            cost_price=data.cost_price,
            stock=data.stock,
            low_stock_threshold=data.low_stock_threshold,
            unit=data.unit,
            main_image=data.main_image,
            images=images_json,
            description=data.description,
            spec_template=data.spec_template,
            status=data.status or ProductStatus.DRAFT
        )

        self.db.add(product)
        self.db.commit()
        self.db.refresh(product)
        return product

    def update_product(self, product_id: int, data: ProductUpdate) -> Product:
        """更新商品"""
        product = self.get_product(product_id)
        if not product:
            raise ValueError("商品不存在")

        update_data = data.model_dump(exclude_unset=True)

        # 处理图片列表
        if 'images' in update_data:
            update_data['images'] = json.dumps(update_data['images'])

        for key, value in update_data.items():
            setattr(product, key, value)

        self.db.commit()
        self.db.refresh(product)
        return product

    def delete_product(self, product_id: int) -> bool:
        """删除商品（软删除）"""
        product = self.get_product(product_id)
        if not product:
            raise ValueError("商品不存在")

        product.status = ProductStatus.DELETED
        self.db.commit()
        return True

    def update_stock(self, product_id: int, quantity: int, operation: str = "subtract") -> bool:
        """更新库存"""
        product = self.get_product(product_id)
        if not product:
            raise ValueError("商品不存在")

        if operation == "subtract":
            if product.stock < quantity:
                raise ValueError("库存不足")
            product.stock -= quantity
        else:
            product.stock += quantity

        self.db.commit()
        return True

    def batch_update_stock(self, stock_data: List[dict]) -> dict:
        """批量更新库存"""
        results = {"success": 0, "failed": 0, "errors": []}

        for item in stock_data:
            try:
                self.update_stock(
                    item["product_id"],
                    item["quantity"],
                    item.get("operation", "subtract")
                )
                results["success"] += 1
            except Exception as e:
                results["failed"] += 1
                results["errors"].append({
                    "product_id": item["product_id"],
                    "error": str(e)
                })

        return results
```

```python
import os
import uuid
import aiofiles
from datetime import datetime
from typing import Optional
from pathlib import Path


class StorageService:
    """文件存储服务"""

    def __init__(self, base_dir: str = "./uploads"):
        self.base_dir = Path(base_dir)
        self.base_dir.mkdir(parents=True, exist_ok=True)

    async def save_file(self, file_content: bytes, filename: str, folder: str = "images") -> str:
        """保存文件

        Args:
            file_content: 文件内容
            filename: 原始文件名
            folder: 存储目录

        Returns:
            文件访问 URL
        """
        # 生成唯一文件名
        ext = Path(filename).suffix.lower()
        unique_name = f"{uuid.uuid4().hex}{ext}"

        # 按日期组织目录
        date_path = datetime.now().strftime("%Y%m%d")
        save_dir = self.base_dir / folder / date_path
        save_dir.mkdir(parents=True, exist_ok=True)

        # 保存文件
        file_path = save_dir / unique_name
        async with aiofiles.open(file_path, 'wb') as f:
            await f.write(file_content)

        # 返回相对路径
        return f"/uploads/{folder}/{date_path}/{unique_name}"

    def delete_file(self, file_path: str) -> bool:
        """删除文件"""
        full_path = self.base_dir / file_path.lstrip("/uploads/")
        if full_path.exists():
            full_path.unlink()
            return True
        return False

    def get_file_url(self, relative_path: str) -> str:
        """获取文件完整 URL"""
        return f"/uploads/{relative_path}"
```

```python
from PIL import Image
import io
from typing import Tuple, Optional
import uuid


class ImageProcessor:
    """图片处理工具"""

    # 缩略图尺寸
    THUMBNAIL_SIZE = (200, 200)
    MEDIUM_SIZE = (400, 400)
    LARGE_SIZE = (800, 800)

    @staticmethod
    def create_thumbnail(image_data: bytes, size: Tuple[int, int] = None) -> bytes:
        """创建缩略图"""
        img = Image.open(io.BytesIO(image_data))

        # 转换为 RGB 模式
        if img.mode in ('RGBA', 'P'):
            img = img.convert('RGB')

        # 保持比例缩放
        img.thumbnail(size or ImageProcessor.THUMBNAIL_SIZE, Image.Resampling.LANCZOS)

        # 输出为 JPEG
        output = io.BytesIO()
        img.save(output, format='JPEG', quality=85, optimize=True)
        return output.getvalue()

    @staticmethod
    def create_watermark(image_data: bytes, text: str = "E-Commerce") -> bytes:
        """添加水印"""
        img = Image.open(io.BytesIO(image_data))

        # 创建水印层
        watermark = Image.new('RGBA', img.size, (255, 255, 255, 0))
        from PIL import ImageDraw, ImageFont

        draw = ImageDraw.Draw(watermark)

        # 尝试加载字体，失败则使用默认
        try:
            font = ImageFont.truetype("/usr/share/fonts/truetype/dejavu/DejaVuSans.ttf", 36)
        except:
            font = ImageFont.load_default()

        # 绘制水印
        text_bbox = draw.textbbox((0, 0), text, font=font)
        text_width = text_bbox[2] - text_bbox[0]
        text_height = text_bbox[3] - text_bbox[1]

        x = img.width - text_width - 10
        y = img.height - text_height - 10

        draw.text((x, y), text, fill=(255, 255, 255, 128), font=font)

        # 合并
        watermarked = Image.alpha_composite(img.convert('RGBA'), watermark)

        output = io.BytesIO()
        watermarked.convert('RGB').save(output, format='JPEG', quality=90)
        return output.getvalue()

    @staticmethod
    def validate_image(image_data: bytes, max_size: int = 10 * 1024 * 1024) -> bool:
        """验证图片"""
        if len(image_data) > max_size:
            return False

        try:
            img = Image.open(io.BytesIO(image_data))
            img.verify()
            return True
        except:
            return False
```

```python
from app.services.storage import storage_service

router = APIRouter(prefix="/products", tags=["商品管理"])


@router.get("", response_model=ProductListResponse)
def get_products(
    keyword: Optional[str] = Query(None, description="关键词搜索"),
    category_id: Optional[int] = Query(None),
    brand_id: Optional[int] = Query(None),
    status: Optional[str] = Query(None),
    min_price: Optional[float] = Query(None),
    max_price: Optional[float] = Query(None),
    on_sale: bool = Query(False),
    is_featured: Optional[bool] = Query(None),
    is_new: Optional[bool] = Query(None),
    sort_by: Optional[str] = Query(None),
    sort_order: str = Query("desc"),
    page: int = Query(1, ge=1),
    page_size: int = Query(20, ge=1, le=100),
    db: Session = Depends(get_db)
):
    """获取商品列表"""
    query = ProductQuery(
        keyword=keyword,
        category_id=category_id,
        brand_id=brand_id,
        status=status,
        min_price=min_price,
        max_price=max_price,
        on_sale=on_sale,
        is_featured=is_featured,
        is_new=is_new,
        sort_by=sort_by,
        sort_order=sort_order
    )

    service = ProductService(db)
    skip = (page - 1) * page_size
    items, total = service.get_products(query, skip, page_size)

    return ProductListResponse(
        items=items,
        total=total,
        page=page,
        page_size=page_size
    )


@router.get("/{product_id}", response_model=ProductResponse)
def get_product(product_id: int, db: Session = Depends(get_db)):
    """获取商品详情"""
    service = ProductService(db)
    product = service.get_product(product_id)
    if not product:
        raise HTTPException(status_code=404, detail="商品不存在")
    return product

def create_product(data: ProductCreate, db: Session = Depends(get_db)):

@router.post("/upload")
async def upload_image(file: UploadFile = File(...)):
    """上传商品图片"""
    # 验证文件类型
    if file.content_type not in ["image/jpeg", "image/png", "image/gif", "image/webp"]:
        raise HTTPException(status_code=400, detail="不支持的图片格式")

    # 读取文件内容
    content = await file.read()

    # 验证图片
    if not ImageProcessor.validate_image(content):
        raise HTTPException(status_code=400, detail="无效的图片文件")

    # 保存文件
    url = await storage_service.save_file(content, file.filename, "products")

    return {"url": url, "filename": file.filename}
```

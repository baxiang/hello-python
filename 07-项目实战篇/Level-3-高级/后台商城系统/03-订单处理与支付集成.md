```python
# app/services/cart.py
from sqlalchemy.orm import Session
from typing import List, Optional, Tuple
from decimal import Decimal

from app.models.product import Product, ProductSku
from app.models.order import Cart


class CartService:
    """购物车服务"""

    def __init__(self, db: Session):
        self.db = db

    def add_item(
        self,
        user_id: int,
        product_id: int,
        quantity: int = 1,
        product_sku_id: Optional[int] = None
    ) -> Cart:
        """添加商品到购物车"""
        # 检查商品是否存在
        product = self.db.query(Product).filter(
            Product.id == product_id,
            Product.status == "on_sale"
        ).first()

        if not product:
            raise ValueError("商品不存在或已下架")

        # 检查库存
        if product.stock < quantity:
            raise ValueError("库存不足")

        # 检查 SKU
        if product_sku_id:
            sku = self.db.query(ProductSku).filter(
                ProductSku.id == product_sku_id,
                ProductSku.product_id == product_id
            ).first()
            if not sku or not sku.is_enabled:
                raise ValueError("SKU 不存在或已禁用")
            if sku.stock < quantity:
                raise ValueError("SKU 库存不足")

        # 检查购物车是否已存在
        cart_item = self.db.query(Cart).filter(
            Cart.user_id == user_id,
            Cart.product_id == product_id,
            Cart.product_sku_id == product_sku_id
        ).first()

        if cart_item:
            # 更新数量
            new_quantity = cart_item.quantity + quantity
            if product.stock < new_quantity:
                raise ValueError("库存不足")
            cart_item.quantity = new_quantity
            self.db.commit()
            self.db.refresh(cart_item)
            return cart_item
        else:
            # 创建新购物车项
            cart_item = Cart(
                user_id=user_id,
                product_id=product_id,
                product_sku_id=product_sku_id,
                quantity=quantity
            )
            self.db.add(cart_item)
            self.db.commit()
            self.db.refresh(cart_item)
            return cart_item

    def get_cart_items(self, user_id: int) -> List[Cart]:
        """获取购物车列表"""
        return (
            self.db.query(Cart)
            .filter(Cart.user_id == user_id)
            .all()
        )

    def get_cart_summary(self, user_id: int) -> dict:
        """获取购物车摘要"""
        items = self.get_cart_items(user_id)

        total_amount = Decimal("0")
        total_count = 0
        product_list = []

        for item in items:
            if not item.checked:
                continue

            product = item.product
            price = product.price

            # 如果有 SKU，使用 SKU 价格
            if item.product_sku_id:
                sku = self.db.query(ProductSku).filter(
                    ProductSku.id == item.product_sku_id
                ).first()
                if sku:
                    price = sku.price

            item_total = price * item.quantity
            total_amount += item_total
            total_count += item.quantity

            product_list.append({
                "id": item.id,
                "product_id": product.id,
                "product_name": product.name,
                "product_image": product.main_image,
                "price": float(price),
                "quantity": item.quantity,
                "checked": item.checked,
                "stock": product.stock,
                "item_total": float(item_total)
            })

        return {
            "items": product_list,
            "total_amount": float(total_amount),
            "total_count": total_count
        }

    def update_quantity(
        self,
        user_id: int,
        cart_id: int,
        quantity: int
    ) -> Cart:
        """更新购物车数量"""
        cart_item = self.db.query(Cart).filter(
            Cart.id == cart_id,
            Cart.user_id == user_id
        ).first()

        if not cart_item:
            raise ValueError("购物车项不存在")

        if quantity <= 0:
            self.db.delete(cart_item)
            self.db.commit()
            return None

        # 检查库存
        product = cart_item.product
        if product.stock < quantity:
            raise ValueError("库存不足")

        cart_item.quantity = quantity
        self.db.commit()
        self.db.refresh(cart_item)
        return cart_item

    def check_items(self, user_id: int, cart_ids: List[int], checked: bool):
        """勾选/取消购物车项"""
        self.db.query(Cart).filter(
            Cart.user_id == user_id,
            Cart.id.in_(cart_ids)
        ).update({"checked": checked}, synchronize_session=False)
        self.db.commit()

    def clear_cart(self, user_id: int):
        """清空购物车"""
        self.db.query(Cart).filter(Cart.user_id == user_id).delete()
        self.db.commit()

    def delete_items(self, user_id: int, cart_ids: List[int]):
        """删除购物车项"""
        self.db.query(Cart).filter(
            Cart.user_id == user_id,
            Cart.id.in_(cart_ids)
        ).delete(synchronize_session=False)
        self.db.commit()
```

### 3.1 分布式锁

```python
# app/utils/lock.py
import redis
import uuid
from contextlib import contextmanager
from typing import Optional
import time

from app.config import settings


class DistributedLock:
    """分布式锁"""

    def __init__(self):
        self.redis_client = redis.Redis.from_url(
            settings.REDIS_URL,
            decode_responses=True
        )

    @contextmanager
    def lock(
        self,
        key: str,
        timeout: int = 10,
        retry_times: int = 3,
        retry_delay: float = 0.2
    ):
        """获取分布式锁

        Args:
            key: 锁的 key
            timeout: 锁超时时间（秒）
            retry_times: 重试次数
            retry_delay: 重试延迟（秒）
        """
        lock_value = str(uuid.uuid4())
        lock_key = f"lock:{key}"

        for _ in range(retry_times):
            # 尝试获取锁
            if self.redis_client.set(lock_key, lock_value, nx=True, ex=timeout):
                try:
                    yield True
                finally:
                    # 释放锁（使用 Lua 脚本确保原子性）
                    lua_script = """
                    if redis.call("get", KEYS[1]) == ARGV[1] then
                        return redis.call("del", KEYS[1])
                    else
                        return 0
                    end
                    """
                    self.redis_client.eval(lua_script, 1, lock_key, lock_value)
                return

            time.sleep(retry_delay)

        raise RuntimeError(f"获取锁失败: {key}")

    def try_lock(self, key: str, timeout: int = 10) -> Optional[str]:
        """尝试获取锁"""
        lock_value = str(uuid.uuid4())
        lock_key = f"lock:{key}"

        if self.redis_client.set(lock_key, lock_value, nx=True, ex=timeout):
            return lock_value
        return None

    def release(self, key: str, lock_value: str):
        """释放锁"""
        lock_key = f"lock:{key}"
        lua_script = """
        if redis.call("get", KEYS[1]) == ARGV[1] then
            return redis.call("del", KEYS[1])
        else
            return 0
        end
        """
        self.redis_client.eval(lua_script, 1, lock_key, lock_value)


# 全局实例
distributed_lock = DistributedLock()
```

### 3.2 订单服务

```python
# app/services/order.py
from sqlalchemy.orm import Session
from sqlalchemy import and_
from typing import List, Optional, Tuple
from decimal import Decimal
from datetime import datetime, timedelta
import json

from app.models.product import Product, ProductSku
from app.models.order import Order, OrderItem, OrderStatus, Cart
from app.models.payment import Payment, PaymentStatus, PaymentMethod
from app.utils.lock import distributed_lock
from app.utils.snowflake import generate_snowflake_id


class OrderService:
    """订单服务"""

    def __init__(self, db: Session):
        self.db = db

    def create_order(
        self,
        user_id: int,
        address_id: int,
        items: List[dict],
        note: str = None,
        buyer_note: str = None,
        coupon_id: Optional[int] = None,
        point_deduction: int = 0
    ) -> Order:
        """创建订单

        Args:
            user_id: 用户ID
            address_id: 地址ID
            items: 商品列表 [{"product_id": 1, "quantity": 2, "sku_id": 1}]
            note: 订单备注
            buyer_note: 买家备注
            coupon_id: 优惠券ID
            point_deduction: 积分抵扣
        """
        # 获取用户地址
        from app.models.user import UserAddress
        address = self.db.query(UserAddress).filter(
            UserAddress.id == address_id,
            UserAddress.user_id == user_id
        ).first()

        if not address:
            raise ValueError("收货地址不存在")

        # 计算订单金额
        total_amount = Decimal("0")
        order_items_data = []

        # 使用分布式锁锁定库存
        lock_keys = [f"stock:{item['product_id']}" for item in items]

        with distributed_lock.lock("create_order", timeout=30):
            for item in items:
                product = self.db.query(Product).filter(
                    Product.id == item["product_id"],
                    Product.status == "on_sale"
                ).first()

                if not product:
                    raise ValueError(f"商品 {item['product_id']} 不存在或已下架")

                if product.stock < item["quantity"]:
                    raise ValueError(f"商品 {product.name} 库存不足")

                # 获取 SKU 价格
                price = product.price
                sku = None
                if item.get("sku_id"):
                    sku = self.db.query(ProductSku).filter(
                        ProductSku.id == item["sku_id"],
                        ProductSku.product_id == item["product_id"]
                    ).first()
                    if sku:
                        price = sku.price

                # 扣减库存
                product.stock -= item["quantity"]

                # 记录订单项数据
                order_items_data.append({
                    "product_id": product.id,
                    "product_sku_id": item.get("sku_id"),
                    "product_name": product.name,
                    "product_image": product.main_image,
                    "product_price": price,
                    "product_code": product.product_code,
                    "sku_spec": sku.spec_data if sku else None,
                    "quantity": item["quantity"],
                    "item_total": price * item["quantity"]
                })

                total_amount += price * item["quantity"]

        # 计算运费（满99免运费）
        freight_amount = Decimal("0") if total_amount >= 99 else Decimal("10")

        # 积分抵扣（100积分=1元）
        point_money = Decimal(point_deduction) / Decimal("100")

        # 优惠金额
        discount_amount = point_money

        # 实付金额
        pay_amount = total_amount + freight_amount - discount_amount

        # 生成订单号
        order_no = self._generate_order_no()

        # 创建订单
        order = Order(
            order_no=order_no,
            user_id=user_id,
            total_amount=total_amount,
            freight_amount=freight_amount,
            discount_amount=discount_amount,
            pay_amount=pay_amount,
            point_deduction=point_deduction,
            receiver_name=address.receiver_name,
            receiver_phone=address.receiver_phone,
            receiver_province=address.province,
            receiver_city=address.city,
            receiver_district=address.district,
            receiver_detail=address.detail_address,
            receiver_postal_code=address.postal_code,
            note=note,
            buyer_note=buyer_note,
            status=OrderStatus.PENDING
        )

        self.db.add(order)
        self.db.flush()  # 获取订单 ID

        # 创建订单项
        for item_data in order_items_data:
            order_item = OrderItem(
                order_id=order.id,
                **item_data
            )
            self.db.add(order_item)

        # 清空购物车
        self.db.query(Cart).filter(Cart.user_id == user_id).delete()

        self.db.commit()
        self.db.refresh(order)
        return order

    def _generate_order_no(self) -> str:
        """生成订单号"""
        # 格式：年月日时分秒 + 6位随机数
        now = datetime.now()
        return now.strftime("%Y%m%d%H%M%S") + str(generate_snowflake_id())[-6:]

    def get_order(self, order_id: int, user_id: int = None) -> Optional[Order]:
        """获取订单详情"""
        query = self.db.query(Order).filter(Order.id == order_id)
        if user_id:
            query = query.filter(Order.user_id == user_id)
        return query.first()

    def get_order_by_no(self, order_no: str, user_id: int = None) -> Optional[Order]:
        """根据订单号获取订单"""
        query = self.db.query(Order).filter(Order.order_no == order_no)
        if user_id:
            query = query.filter(Order.user_id == user_id)
        return query.first()

    def get_orders(
        self,
        user_id: int,
        status: Optional[OrderStatus] = None,
        page: int = 1,
        page_size: int = 20
    ) -> Tuple[List[Order], int]:
        """获取用户订单列表"""
        query = self.db.query(Order).filter(Order.user_id == user_id)

        if status:
            query = query.filter(Order.status == status)

        total = query.count()
        orders = (
            query
            .order_by(Order.created_at.desc())
            .offset((page - 1) * page_size)
            .limit(page_size)
            .all()
        )

        return orders, total

    def cancel_order(self, order_id: int, user_id: int, reason: str = None) -> bool:
        """取消订单"""
        order = self.get_order(order_id, user_id)
        if not order:
            raise ValueError("订单不存在")

        if order.status != OrderStatus.PENDING:
            raise ValueError("订单状态不允许取消")

        # 恢复库存
        with distributed_lock.lock(f"cancel_order:{order_id}", timeout=10):
            for item in order.items:
                product = item.product
                product.stock += item.quantity

            order.status = OrderStatus.CANCELLED
            order.cancel_reason = reason
            order.cancel_time = datetime.now()

            self.db.commit()

        return True

    def confirm_receive(self, order_id: int, user_id: int) -> bool:
        """确认收货"""
        order = self.get_order(order_id, user_id)
        if not order:
            raise ValueError("订单不存在")

        if order.status != OrderStatus.SHIPPED:
            raise ValueError("订单状态不允许确认收货")

        order.status = OrderStatus.COMPLETED
        order.complete_time = datetime.now()

        # 更新商品销量
        for item in order.items:
            product = item.product
            product.sales_count += item.quantity

        self.db.commit()
        return True
```

```python
# app/services/payment.py
from sqlalchemy.orm import Session
from sqlalchemy import and_
from typing import Optional
from decimal import Decimal
from datetime import datetime
import json
import hashlib
import time

from app.models.order import Order, OrderStatus
from app.models.payment import Payment, PaymentStatus, PaymentMethod
from app.utils.snowflake import generate_snowflake_id
from app.config import settings


class PaymentService:
    """支付服务"""

    def __init__(self, db: Session):
        self.db = db

    def create_payment(
        self,
        order_id: int,
        method: PaymentMethod,
        idempotency_key: str = None
    ) -> Payment:
        """创建支付记录"""
        # 检查订单
        order = self.db.query(Order).filter(Order.id == order_id).first()
        if not order:
            raise ValueError("订单不存在")

        if order.status != OrderStatus.PENDING:
            raise ValueError("订单状态不允许支付")

        # 幂等性检查
        if idempotency_key:
            existing = self.db.query(Payment).filter(
                Payment.idempotency_key == idempotency_key,
                Payment.status.in_([PaymentStatus.SUCCESS, PaymentStatus.PROCESSING])
            ).first()
            if existing:
                raise ValueError("该请求已处理，请勿重复提交")

        # 生成支付流水号
        payment_no = self._generate_payment_no()

        payment = Payment(
            payment_no=payment_no,
            order_id=order_id,
            order_no=order.order_no,
            amount=order.pay_amount,
            method=method,
            status=PaymentStatus.PENDING,
            idempotency_key=idempotency_key
        )

        self.db.add(payment)
        self.db.commit()
        self.db.refresh(payment)
        return payment

    def _generate_payment_no(self) -> str:
        """生成支付流水号"""
        return f"PAY{datetime.now().strftime('%Y%m%d%H%M%S')}{generate_snowflake_id()[-8:]}"

    def get_payment(self, payment_id: int) -> Optional[Payment]:
        return self.db.query(Payment).filter(Payment.id == payment_id).first()

    def get_payment_by_no(self, payment_no: str) -> Optional[Payment]:
        return self.db.query(Payment).filter(Payment.payment_no == payment_no).first()

    def process_alipay_callback(self, callback_data: dict) -> bool:
        """处理支付宝回调"""
        # 验签（实际需要使用支付宝 SDK 验签）
        # ...

        payment_no = callback_data.get("out_trade_no")
        trade_no = callback_data.get("trade_no")
        trade_status = callback_data.get("trade_status")

        payment = self.get_payment_by_no(payment_no)
        if not payment:
            return False

        # 检查是否已处理
        if payment.status == PaymentStatus.SUCCESS:
            return True

        # 更新支付状态
        if trade_status in ["TRADE_SUCCESS", "TRADE_FINISHED"]:
            payment.status = PaymentStatus.SUCCESS
            payment.trade_no = trade_no
            payment.transaction_id = callback_data.get("buyer_logon_id")
            payment.raw_response = json.dumps(callback_data)
            payment.callback_time = datetime.now()

            # 更新订单状态
            order = payment.order
            order.status = OrderStatus.PAID
            order.pay_time = datetime.now()

            self.db.commit()

        return True

    def process_wechat_callback(self, callback_data: dict) -> bool:
        """处理微信支付回调"""
        payment_no = callback_data.get("out_trade_no")
        transaction_id = callback_data.get("transaction_id")

        payment = self.get_payment_by_no(payment_no)
        if not payment:
            return False

        if payment.status == PaymentStatus.SUCCESS:
            return True

        # 更新支付状态
        payment.status = PaymentStatus.SUCCESS
        payment.transaction_id = transaction_id
        payment.raw_response = json.dumps(callback_data)
        payment.callback_time = datetime.now()

        # 更新订单状态
        order = payment.order
        order.status = OrderStatus.PAID
        order.pay_time = datetime.now()

        self.db.commit()
        return True

    def create_refund(self, payment_id: int, refund_amount: Decimal, reason: str = None) -> bool:
        """创建退款"""
        payment = self.get_payment(payment_id)
        if not payment:
            raise ValueError("支付记录不存在")

        if payment.status != PaymentStatus.SUCCESS:
            raise ValueError("支付状态不允许退款")

        if refund_amount > payment.amount - payment.refund_amount:
            raise ValueError("退款金额超过可退金额")

        # 实际退款需要调用支付平台 API
        # ...

        payment.refund_amount += refund_amount
        payment.refund_time = datetime.now()
        payment.refund_reason = reason

        if payment.refund_amount >= payment.amount:
            payment.status = PaymentStatus.REFUNDED
            payment.order.status = OrderStatus.REFUNDED

        self.db.commit()
        return True
```

```python
# app/utils/alipay.py
from alipay import AlipayClient, AlipayModel
from alipay.exception import AlipayException
from typing import Optional
from decimal import Decimal

from app.config import settings


class AlipayService:
    """支付宝服务"""

    def __init__(self):
        self.alipay = AlipayClient(
            app_id=settings.ALIPAY_APP_ID,
            private_key=settings.ALIPAY_PRIVATE_KEY,
            alipay_public_key=settings.ALIPAY_PUBLIC_KEY,
            sign_type="RSA2"
        )

    def create_payment_url(
        self,
        out_trade_no: str,
        total_amount: Decimal,
        subject: str,
        notify_url: str = None
    ) -> str:
        """创建支付链接"""
        model = AlipayModel(
            out_trade_no=out_trade_no,
            total_amount=str(total_amount),
            subject=subject,
            timeout_express="30m"
        )

        # 实际返回的是支付页面 URL
        # 这里简化处理，返回 SDK 调用的结果
        try:
            response = self.alipay.trade_page_pay(
                model=model,
                return_url=f"{settings.ALIPAY_NOTIFY_URL}/return",
                notify_url=notify_url or settings.ALIPAY_NOTIFY_URL
            )
            return response
        except AlipayException as e:
            raise RuntimeError(f"支付宝支付创建失败: {e}")

    def verify_callback(self, params: dict) -> bool:
        """验证回调"""
        try:
            return self.alipay.verify_callback(params)
        except:
            return False

    def query_payment(self, out_trade_no: str) -> dict:
        """查询支付状态"""
        try:
            response = self.alipay.trade_query(out_trade_no=out_trade_no)
            return response
```

## 第六部分 - 订单超时处理

### 6.1 订单超时取消任务

```python
# app/tasks/order.py
from celery import Celery
from datetime import datetime, timedelta

from app.database import get_db_context
from app.models.order import Order, OrderStatus
from app.services.order import OrderService

celery_app = Celery("tasks", broker="redis://localhost:6379/0")


@celery_app.task
def cancel_expired_orders():
    """取消超时未支付订单"""
    with get_db_context() as db:
        # 查找超时订单（30分钟未支付）
        timeout = datetime.now() - timedelta(minutes=30)

        orders = db.query(Order).filter(
            Order.status == OrderStatus.PENDING,
            Order.created_at < timeout
        ).all()

        for order in orders:
            try:
                # 恢复库存
                for item in order.items:
                    product = item.product
                    product.stock += item.quantity

                order.status = OrderStatus.CANCELLED
                order.cancel_reason = "订单超时未支付"
                order.cancel_time = datetime.now()

                db.commit()
                print(f"订单 {order.order_no} 已自动取消")

            except Exception as e:
                db.rollback()
                print(f"取消订单 {order.order_no} 失败: {e}")
```

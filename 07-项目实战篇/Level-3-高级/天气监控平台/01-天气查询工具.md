### 2.2 基础查询功能

```python
# weather_query.py
import requests

API_KEY = "your_api_key"
URL = "https://devapi.qweather.com/v7/weather/now"

def get_weather(city: str) -> dict:
    """获取城市天气"""
    # 1. 先获取城市 ID
    city_url = "https://geoapi.qweather.com/v2/city/lookup"
    params = {
        "location": city,
        "key": API_KEY
    }
    response = requests.get(city_url, params=params)
    city_data = response.json()
    
    if city_data["code"] != "200":
        raise Exception(f"城市查找失败：{city_data['code']}")
    
    city_id = city_data["location"][0]["id"]
    
    # 2. 获取天气数据
    params = {
        "location": city_id,
        "key": API_KEY
    }
    response = requests.get(URL, params=params)
    weather_data = response.json()
    
    if weather_data["code"] != "200":
        raise Exception(f"天气查询失败：{weather_data['code']}")
    
    return {
        "city": city,
        "temp": weather_data["now"]["temp"],
        "feels_like": weather_data["now"]["feelsLike"],
        "condition": weather_data["now"]["text"],
        "humidity": weather_data["now"]["humidity"],
        "wind_dir": weather_data["now"]["windDir"],
        "wind_scale": weather_data["now"]["windScale"],
    }

if __name__ == "__main__":
    weather = get_weather("北京")
    print(f"城市：{weather['city']}")
    print(f"温度：{weather['temp']}°C")
    print(f"体感：{weather['feels_like']}°C")
    print(f"天气：{weather['condition']}")
    print(f"湿度：{weather['humidity']}%")
    print(f"风向：{weather['wind_dir']}")
    print(f"风力：{weather['wind_scale']}级")
```

### 2.3 批量查询

```python
def query_multiple_cities(cities: list) -> list:
    """批量查询多个城市天气"""
    results = []
    for city in cities:
        try:
            weather = get_weather(city)
            results.append(weather)
        except Exception as e:
            print(f"查询 {city} 失败：{e}")
            results.append({"city": city, "error": str(e)})
    return results

if __name__ == "__main__":
    cities = ["北京", "上海", "广州", "深圳", "杭州"]
    results = query_multiple_cities(cities)
    
    # 格式化输出
    print(f"{'城市':<8} {'温度':<6} {'天气':<8} {'湿度':<6}")
    print("-" * 35)
    for r in results:
        if "error" in r:
            print(f"{r['city']:<8} {'--':<6} {'错误':<8} {'--':<6}")
        else:
            print(f"{r['city']:<8} {r['temp']:<6} {r['condition']:<8} {r['humidity']:<6}")
```

### 3.1 添加缓存

```python
import time
from functools import wraps

def cache_result(ttl: int = 300):
    """结果缓存装饰器，默认 5 分钟过期"""
    cache = {}
    
    def decorator(func):
        @wraps(func)
        def wrapper(*args):
            key = args[0]  # 城市名作为缓存 key
            if key in cache:
                timestamp, result = cache[key]
                if time.time() - timestamp < ttl:
                    return result
            result = func(*args)
            cache[key] = (time.time(), result)
            return result
        return wrapper
    return decorator

@cache_result(ttl=600)  # 10 分钟缓存
```

### 3.2 导出功能

```python
import csv
from datetime import datetime

def export_to_csv(results: list, filename: str = None):
    """导出查询结果到 CSV"""
    if filename is None:
        filename = f"weather_{datetime.now().strftime('%Y%m%d_%H%M%S')}.csv"
    
    with open(filename, 'w', newline='', encoding='utf-8') as f:
        writer = csv.DictWriter(f, fieldnames=[
            'city', 'temp', 'feels_like', 'condition', 
            'humidity', 'wind_dir', 'wind_scale'
        ])
        writer.writeheader()
        for r in results:
            if "error" not in r:
                writer.writerow(r)
    
    print(f"数据已导出到：{filename}")
    return filename
```

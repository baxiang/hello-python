### 2.2 Docker 部署架构

```yaml
# docker-compose.yml
version: '3.8'

services:
  db:
    image: mysql:8.0
    environment:
      MYSQL_ROOT_PASSWORD: rootpass
      MYSQL_DATABASE: weather_db
      MYSQL_USER: weather
      MYSQL_PASSWORD: weatherpass
    volumes:
      - mysql_data:/var/lib/mysql
    ports:
      - "3306:3306"

  api:
    build: ./weather_monitor
    environment:
      DATABASE_URL: mysql+pymysql://weather:weatherpass@db:3306/weather_db
      WEATHER_API_KEY: ${WEATHER_API_KEY}
    ports:
      - "8000:8000"
    depends_on:
      - db

  frontend:
    build: ./frontend
    ports:
      - "80:80"
    depends_on:
      - api

volumes:
  mysql_data:
```

## 第三部分 - Dockerfile

### 3.1 API 服务 Dockerfile

```dockerfile
# weather_monitor/Dockerfile
FROM python:3.11-slim

WORKDIR /app

# 安装依赖
COPY pyproject.toml .
RUN pip install uv && uv sync --no-dev

# 复制代码
COPY app ./app

# 启动服务
CMD ["uv", "run", "uvicorn", "app.main:app", "--host", "0.0.0.0", "--port", "8000"]
```

### 3.2 前端 Dockerfile

```dockerfile
# frontend/Dockerfile
FROM node:18-alpine AS builder

WORKDIR /app
COPY package*.json ./
RUN npm install
COPY . .
RUN npm run build

FROM nginx:alpine
COPY --from=builder /app/dist /usr/share/nginx/html
COPY nginx.conf /etc/nginx/conf.d/default.conf
EXPOSE 80
CMD ["nginx", "-g", "daemon off;"]
```

## 第四部分 - 部署运行

### 4.1 本地开发

```bash
# 启动数据库
docker-compose up db -d

# 启动 API
cd weather_monitor
uv run uvicorn app.main:app --reload

# 启动前端
cd frontend
npm run dev
```

### 4.2 生产部署

```bash
# 构建并启动所有服务
docker-compose up -d --build

# 查看服务状态
docker-compose ps

# 查看日志
docker-compose logs -f api

# 停止服务
docker-compose down
```

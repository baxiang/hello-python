### 1.2 用户名验证

```python
# manager.py
import re

def validate_username(username: str) -> tuple[bool, str]:
    """验证用户名"""
    # 长度检查
    if len(username) < 2 or len(username) > 20:
        return False, "用户名长度应为 2-20 个字符"
    
    # 格式检查（只允许字母、数字、下划线、中文）
    if not re.match(r'^[\w\u4e00-\u9fa5]+$', username):
        return False, "用户名只能包含字母、数字、下划线和中文"
    
    return True, ""

class ConnectionManager:
    def __init__(self):
        self.active_connections: Dict[str, WebSocket] = {}
    
    async def connect(self, websocket: WebSocket, username: str) -> tuple[bool, str]:
        """接受新连接"""
        # 验证用户名
        valid, msg = validate_username(username)
        if not valid:
            return False, msg
        
        # 检查是否已存在
        if username in self.active_connections:
            return False, "用户名已被使用"
        
        await websocket.accept()
        self.active_connections[username] = websocket
        return True, ""
```

## 2. 在线状态管理

### 2.1 用户状态模型

```python
from enum import Enum
from datetime import datetime

class UserStatus(str, Enum):
    ONLINE = "online"      # 在线
    AWAY = "away"          # 离开
    BUSY = "busy"          # 忙碌
    OFFLINE = "offline"    # 离线

class OnlineUser:
    def __init__(self, username: str, websocket: WebSocket):
        self.username = username
        self.websocket = websocket
        self.status = UserStatus.ONLINE
        self.connected_at = datetime.now()
        self.last_activity = datetime.now()
    
    def update_activity(self):
        """更新活动时间"""
        self.last_activity = datetime.now()
    
    def to_dict(self) -> dict:
        return {
            "username": self.username,
            "status": self.status.value,
            "connected_at": self.connected_at.isoformat()
        }
```

### 2.2 自动离开检测

```python
import asyncio

class ConnectionManager:
    def __init__(self):
        self.active_connections: Dict[str, OnlineUser] = {}
    
    async def check_idle_users(self):
        """定期检查空闲用户"""
        while True:
            await asyncio.sleep(60)  # 每分钟检查一次
            
            now = datetime.now()
            for username, user in self.active_connections.items():
                idle_time = (now - user.last_activity).total_seconds()
                
                # 超过 5 分钟无活动，标记为离开
                if idle_time > 300 and user.status == UserStatus.ONLINE:
                    user.status = UserStatus.AWAY
                    await self.broadcast_status(username, UserStatus.AWAY)
    
    async def broadcast_status(self, username: str, status: UserStatus):
        """广播用户状态变化"""
        await self.broadcast({
            "type": "status",
            "username": username,
            "status": status.value
        })
```

## 3. 用户列表 API

### 3.1 REST API

```python
from fastapi import APIRouter

router = APIRouter(prefix="/api/users", tags=["users"])

@router.get("")
async def get_users():
    """获取在线用户列表"""
    return {
        "users": [user.to_dict() for user in manager.active_connections.values()],
        "count": len(manager.active_connections)
    }

@router.get("/{username}/status")
async def get_user_status(username: str):
    """获取用户状态"""
    user = manager.active_connections.get(username)
    if user:
        return user.to_dict()
    return {"error": "用户不在线"}

@router.post("/{username}/status")
async def set_user_status(username: str, status: UserStatus):
    """设置用户状态"""
    user = manager.active_connections.get(username)
    if user:
        user.status = status
        await manager.broadcast_status(username, status)
        return {"success": True}
    return {"error": "用户不在线"}
```

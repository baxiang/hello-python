### 2.2 基础查询功能

```python
# stock_query.py
import requests
import json

def get_stock_price(stock_code: str) -> dict:
    """获取股票实时行情"""
    # 腾讯财经 API
    url = f"https://qt.gtimg.cn/q={stock_code}"

    response = requests.get(url)
    response.encoding = 'gbk'

    if response.status_code != 200:
        raise Exception(f"请求失败：{response.status_code}")

    # 解析返回数据
    data = response.text
    if not data or data == "null":
        raise Exception(f"股票代码 {stock_code} 不存在")

    # 解析股票数据
    stock_data = data.split("~")
    return {
        "code": stock_code,
        "name": stock_data[1],           # 股票名称
        "price": stock_data[3],          # 当前价格
        "change": stock_data[4],         # 涨跌额
        "percent": stock_data[5],        # 涨跌幅
        "volume": stock_data[6],         # 成交量
        "amount": stock_data[7],         # 成交额
        "open": stock_data[8],           # 开盘价
        "high": stock_data[9],           # 最高价
        "low": stock_data[10],           # 最低价
        "close": stock_data[30],         # 昨收价
    }

if __name__ == "__main__":
    # 测试查询
    stock = get_stock_price("sh600519")  # 贵州茅台
    print(f"股票：{stock['name']}")
    print(f"代码：{stock['code']}")
    print(f"价格：{stock['price']} 元")
    print(f"涨跌：{stock['change']} 元 ({stock['percent']}%)")
    print(f"开盘：{stock['open']} 最高：{stock['high']} 最低：{stock['low']}")
```

### 2.3 批量查询

```python
def query_multiple_stocks(codes: list) -> list:
    """批量查询多只股票"""
    results = []
    for code in codes:
        try:
            stock = get_stock_price(code)
            results.append(stock)
        except Exception as e:
            print(f"查询 {code} 失败：{e}")
            results.append({"code": code, "error": str(e)})
    return results

if __name__ == "__main__":
    stocks = ["sh600519", "sh000001", "sz000001"]  # 茅台、上证、深证
    results = query_multiple_stocks(stocks)

    # 格式化输出
    print(f"{'代码':<12} {'名称':<10} {'价格':<10} {'涨跌':<12} {'涨跌幅':<10}")
    print("-" * 60)
    for r in results:
        if "error" in r:
            print(f"{r['code']:<12} {'--':<10} {'--':<10} {'错误':<12} {'--':<10}")
        else:
            print(f"{r['code']:<12} {r['name']:<10} {r['price']:<10} "
                  f"{r['change']:<12} {r['percent']:<10}")
```

### 3.1 添加缓存

```python
import time
from functools import wraps

def cache_result(ttl: int = 60):
    """结果缓存装饰器，默认 60 秒过期"""
    cache = {}

    def decorator(func):
        @wraps(func)
        def wrapper(*args):
            key = args[0]  # 股票代码作为缓存 key
            if key in cache:
                timestamp, result = cache[key]
                if time.time() - timestamp < ttl:
                    return result
            result = func(*args)
            cache[key] = (time.time(), result)
            return result
        return wrapper
    return decorator

@cache_result(ttl=30)  # 30 秒缓存
def get_stock_price(stock_code: str) -> dict:
    # 原有代码不变

```

### 3.2 导出功能

```python
import csv
from datetime import datetime

def export_to_csv(results: list, filename: str = None):
    """导出查询结果到 CSV"""
    if filename is None:
        filename = f"stock_{datetime.now().strftime('%Y%m%d_%H%M%S')}.csv"

    with open(filename, 'w', newline='', encoding='utf-8-sig') as f:
        writer = csv.DictWriter(f, fieldnames=[
            'code', 'name', 'price', 'change', 'percent',
            'volume', 'open', 'high', 'low', 'close'
        ])
        writer.writeheader()
        for r in results:
            if "error" not in r:
                writer.writerow(r)

    print(f"数据已导出到：{filename}")
    return filename
```

### 3.3 价格提醒

```python
def check_price_alert(stock: dict, target_price: float, condition: str = "above"):
    """检查是否触发价格提醒"""
    price = float(stock['price'])

    if condition == "above" and price > target_price:
        return True, f"{stock['name']} 突破 {target_price} 元，当前 {price} 元"
    elif condition == "below" and price < target_price:
        return True, f"{stock['name']} 跌破 {target_price} 元，当前 {price} 元"

    return False, None
```

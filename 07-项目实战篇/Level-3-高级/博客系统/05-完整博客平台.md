### 1.2 点赞服务

```python
# like_services.py
from sqlalchemy.orm import Session
from models import Like, Collection, Post


class LikeService:
    """点赞服务"""

    def __init__(self, db: Session):
        self.db = db

    def like_post(self, user_id: int, post_id: int) -> bool:
        """点赞文章"""
        # 检查是否已点赞
        existing = self.db.query(Like).filter(
            Like.user_id == user_id,
            Like.post_id == post_id
        ).first()

        if existing:
            return False  # 已点赞

        like = Like(user_id=user_id, post_id=post_id)
        self.db.add(like)
        self.db.commit()
        return True

    def unlike_post(self, user_id: int, post_id: int) -> bool:
        """取消点赞"""
        like = self.db.query(Like).filter(
            Like.user_id == user_id,
            Like.post_id == post_id
        ).first()

        if not like:
            return False

        self.db.delete(like)
        self.db.commit()
        return True

    def get_post_likes(self, post_id: int) -> int:
        """获取文章点赞数"""
        return self.db.query(Like).filter(Like.post_id == post_id).count()

    def has_liked(self, user_id: int, post_id: int) -> bool:
        """检查是否已点赞"""
        return self.db.query(Like).filter(
            Like.user_id == user_id,
            Like.post_id == post_id
        ).first() is not None


class CollectionService:
    """收藏服务"""

    def __init__(self, db: Session):
        self.db = db

    def collect_post(self, user_id: int, post_id: int) -> bool:
        """收藏文章"""
        existing = self.db.query(Collection).filter(
            Collection.user_id == user_id,
            Collection.post_id == post_id
        ).first()

        if existing:
            return False

        collection = Collection(user_id=user_id, post_id=post_id)
        self.db.add(collection)
        self.db.commit()
        return True

    def uncollect_post(self, user_id: int, post_id: int) -> bool:
        """取消收藏"""
        collection = self.db.query(Collection).filter(
            Collection.user_id == user_id,
            Collection.post_id == post_id
        ).first()

        if not collection:
            return False

        self.db.delete(collection)
        self.db.commit()
        return True

    def get_user_collections(self, user_id: int) -> list:
        """获取用户收藏"""
        return self.db.query(Collection).filter(
            Collection.user_id == user_id
        ).order_by(Collection.created_at.desc()).all()
```

### 2.2 关注服务

```python
# follow_services.py
from sqlalchemy.orm import Session
from models import Follow, User


class FollowService:
    """关注服务"""

    def __init__(self, db: Session):
        self.db = db

    def follow_user(self, follower_id: int, following_id: int) -> bool:
        """关注用户"""
        if follower_id == following_id:
            return False  # 不能关注自己

        existing = self.db.query(Follow).filter(
            Follow.follower_id == follower_id,
            Follow.following_id == following_id
        ).first()

        if existing:
            return False

        follow = Follow(follower_id=follower_id, following_id=following_id)
        self.db.add(follow)
        self.db.commit()
        return True

    def unfollow_user(self, follower_id: int, following_id: int) -> bool:
        """取消关注"""
        follow = self.db.query(Follow).filter(
            Follow.follower_id == follower_id,
            Follow.following_id == following_id
        ).first()

        if not follow:
            return False

        self.db.delete(follow)
        self.db.commit()
        return True

    def get_followers(self, user_id: int) -> list:
        """获取粉丝列表"""
        follows = self.db.query(Follow).filter(
            Follow.following_id == user_id
        ).all()
        return [f.follower for f in follows]

    def get_following(self, user_id: int) -> list:
        """获取关注列表"""
        follows = self.db.query(Follow).filter(
            Follow.follower_id == user_id
        ).all()
        return [f.following for f in follows]

    def get_follower_count(self, user_id: int) -> int:
        """获取粉丝数"""
        return self.db.query(Follow).filter(
            Follow.following_id == user_id
        ).count()

    def get_following_count(self, user_id: int) -> int:
        """获取关注数"""
        return self.db.query(Follow).filter(
            Follow.follower_id == user_id
        ).count()
```

```python
        "code_theme": "solarizedlight"
    }
}


class ThemeService:
    """主题服务"""

    def __init__(self):
        self.themes = THEMES

    def get_theme(self, theme_name: str) -> dict:
        """获取主题配置"""
        return self.themes.get(theme_name, self.themes["default"])

    def get_all_themes(self) -> list:
        """获取所有主题"""
        return [
            {"name": name, "display_name": config["name"]}
            for name, config in self.themes.items()
        ]
```

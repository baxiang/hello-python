## 第一部分 - 数据模型

### MySQL 表结构

```sql
CREATE TABLE access_logs (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    timestamp DATETIME NOT NULL,
    level VARCHAR(10) DEFAULT 'INFO',
    ip VARCHAR(45) NOT NULL,
    method VARCHAR(10) NOT NULL,
    path VARCHAR(500) NOT NULL,
    status INT NOT NULL,
    response_time INT DEFAULT 0,
    user_agent VARCHAR(500),
    size BIGINT DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_timestamp (timestamp),
    INDEX idx_ip (ip),
    INDEX idx_status (status),
    INDEX idx_path (path)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

```python
# models.py
from sqlalchemy import Column, Integer, String, BigInteger, DateTime, Index, Text
from sqlalchemy.sql import func
from database import Base


class AccessLog(Base):
    """访问日志表"""
    __tablename__ = "access_logs"

    id = Column(BigInteger, primary_key=True, autoincrement=True)
    timestamp = Column(DateTime, nullable=False, index=True)
    level = Column(String(10), default='INFO')
    ip = Column(String(45), nullable=False, index=True)
    method = Column(String(10), nullable=False)
    path = Column(String(500), nullable=False, index=True)
    status = Column(Integer, nullable=False, index=True)
    response_time = Column(Integer, default=0)
    user_agent = Column(String(500))
    size = Column(BigInteger, default=0)
    error_message = Column(Text, nullable=True)

    # 复合索引
    __table_args__ = (
        Index('idx_timestamp_status', 'timestamp', 'status'),
        Index('idx_ip_timestamp', 'ip', 'timestamp'),
    )

    def __repr__(self):
        return f"<AccessLog {self.id}: {self.ip} {self.method} {self.path}>"


class ErrorLog(Base):
    """错误日志表"""
    __tablename__ = "error_logs"

    id = Column(BigInteger, primary_key=True, autoincrement=True)
    timestamp = Column(DateTime, nullable=False, index=True)
    level = Column(String(10), nullable=False)
    ip = Column(String(45))
    method = Column(String(10))
    path = Column(String(500))
    status = Column(Integer)
    error = Column(Text, nullable=False)
    traceback = Column(Text)
    created_at = Column(DateTime, default=func.now())

    def __repr__(self):
        return f"<ErrorLog {self.id}: {self.error}>"
```

### 2.4 日志存储服务

```python
# services.py
from sqlalchemy.orm import Session
from sqlalchemy import and_, func, desc
from typing import List, Optional
from datetime import datetime, timedelta

from models import AccessLog, ErrorLog


class LogStorageService:
    """日志存储服务"""

    def __init__(self, db: Session):
        self.db = db

    def save_log(self, log_data: dict) -> AccessLog:
        """保存单条日志"""
        log = AccessLog(
            timestamp=datetime.fromisoformat(log_data['timestamp']),
            level=log_data.get('level', 'INFO'),
            ip=log_data.get('ip', ''),
            method=log_data.get('method', 'GET'),
            path=log_data.get('path', '/'),
            status=log_data.get('status', 200),
            response_time=log_data.get('response_time', 0),
            user_agent=log_data.get('user_agent', ''),
            size=log_data.get('size', 0)
        )

        self.db.add(log)
        self.db.commit()
        self.db.refresh(log)
        return log

    def batch_save_logs(self, logs: List[dict]) -> int:
        """批量保存日志"""
        count = 0

        for log_data in logs:
            try:
                log = AccessLog(
                    timestamp=datetime.fromisoformat(log_data['timestamp']),
                    level=log_data.get('level', 'INFO'),
                    ip=log_data.get('ip', ''),
                    method=log_data.get('method', 'GET'),
                    path=log_data.get('path', '/'),
                    status=log_data.get('status', 200),
                    response_time=log_data.get('response_time', 0),
                    user_agent=log_data.get('user_agent', ''),
                    size=log_data.get('size', 0)
                )
                self.db.add(log)
                count += 1

                # 每1000条提交一次
                if count % 1000 == 0:
                    self.db.commit()
            except Exception as e:
                print(f"保存日志失败: {e}")

        self.db.commit()
        return count

    def query_logs(
        self,
        start_time: datetime = None,
        end_time: datetime = None,
        ip: str = None,
        status: int = None,
        path: str = None,
        keyword: str = None,
        limit: int = 100,
        offset: int = 0
    ) -> tuple[List[AccessLog], int]:
        """查询日志"""
        query = self.db.query(AccessLog)

        if start_time:
            query = query.filter(AccessLog.timestamp >= start_time)
        if end_time:
            query = query.filter(AccessLog.timestamp <= end_time)
        if ip:
            query = query.filter(AccessLog.ip == ip)
        if status:
            query = query.filter(AccessLog.status == status)
        if path:
            query = query.filter(AccessLog.path.like(f"%{path}%"))
        if keyword:
            query = query.filter(
                AccessLog.path.like(f"%{keyword}%") |
                AccessLog.user_agent.like(f"%{keyword}%")
            )

        total = query.count()

        logs = query.order_by(desc(AccessLog.timestamp)).offset(offset).limit(limit).all()

        return logs, total

    def get_logs_by_status(self, status: int, limit: int = 100) -> List[AccessLog]:
        """按状态码查询"""
        return self.db.query(AccessLog).filter(
            AccessLog.status == status
        ).order_by(desc(AccessLog.timestamp)).limit(limit).all()

    def get_error_logs(self, limit: int = 100) -> List[AccessLog]:
        """获取错误日志（4xx/5xx）"""
        return self.db.query(AccessLog).filter(
            AccessLog.status >= 400
        ).order_by(desc(AccessLog.timestamp)).limit(limit).all()

    def get_slow_requests(self, threshold: int = 1000, limit: int = 100) -> List[AccessLog]:
        """获取慢请求"""
        return self.db.query(AccessLog).filter(
            AccessLog.response_time >= threshold
        ).order_by(desc(AccessLog.response_time)).limit(limit).all()

    def delete_old_logs(self, days: int = 30) -> int:
        """删除旧日志"""
        cutoff_date = datetime.now() - timedelta(days=days)

        count = self.db.query(AccessLog).filter(
            AccessLog.timestamp < cutoff_date
        ).delete(synchronize_session=False)

        self.db.commit()
        return count


class StatisticsService:
    """统计服务"""

    def __init__(self, db: Session):
        self.db = db

    def get_total_count(self, start_time: datetime = None, end_time: datetime = None) -> int:
        """获取日志总数"""
        query = self.db.query(func.count(AccessLog.id))

        if start_time:
            query = query.filter(AccessLog.timestamp >= start_time)
        if end_time:
            query = query.filter(AccessLog.timestamp <= end_time)

        return query.scalar()

    def get_status_distribution(self, start_time: datetime = None, end_time: datetime = None) -> dict:
        """状态码分布"""
        query = self.db.query(
            AccessLog.status,
            func.count(AccessLog.id)
        )

        if start_time:
            query = query.filter(AccessLog.timestamp >= start_time)
        if end_time:
            query = query.filter(AccessLog.timestamp <= end_time)

        results = query.group_by(AccessLog.status).all()
        return {status: count for status, count in results}

    def get_top_ips(self, limit: int = 10, start_time: datetime = None) -> list:
        """Top IP"""
        query = self.db.query(
            AccessLog.ip,
            func.count(AccessLog.id).label('count')
        )

        if start_time:
            query = query.filter(AccessLog.timestamp >= start_time)

        results = query.group_by(AccessLog.ip).order_by(
            func.count(AccessLog.id).desc()
        ).limit(limit).all()

        return [{"ip": ip, "count": count} for ip, count in results]

    def get_top_paths(self, limit: int = 10, start_time: datetime = None) -> list:
        """Top 路径"""
        query = self.db.query(
            AccessLog.path,
            func.count(AccessLog.id).label('count')
        )

        if start_time:
            query = query.filter(AccessLog.timestamp >= start_time)

        results = query.group_by(AccessLog.path).order_by(
            func.count(AccessLog.id).desc()
        ).limit(limit).all()

        return [{"path": path, "count": count} for path, count in results]

    def get_response_time_stats(self, start_time: datetime = None) -> dict:
        """响应时间统计"""
        query = self.db.query(
            func.min(AccessLog.response_time),
            func.max(AccessLog.response_time),
            func.avg(AccessLog.response_time),
            func.percentile_cont(0.5).within_group(AccessLog.response_time),
            func.percentile_cont(0.9).within_group(AccessLog.response_time),
            func.percentile_cont(0.99).within_group(AccessLog.response_time)
        )

        if start_time:
            query = query.filter(AccessLog.timestamp >= start_time)

        result = query.first()
        return {
            "min": result[0] or 0,
            "max": result[1] or 0,
            "avg": float(result[2] or 0),
            "p50": result[3] or 0,
            "p90": result[4] or 0,
            "p99": result[5] or 0
        }
```

## 第三部分 - Elasticsearch 集成

### 3.1 Elasticsearch 服务

```python
# elasticsearch_service.py
from elasticsearch import Elasticsearch
from typing import List, Dict, Optional
from datetime import datetime


class ElasticsearchService:
    """Elasticsearch 服务"""

    INDEX_NAME = "access-logs"

    def __init__(self, hosts: List[str] = None):
        self.es = Elasticsearch(hosts or ["http://localhost:9200"])

    def create_index(self):
        """创建索引"""
        if self.es.indices.exists(index=self.INDEX_NAME):
            return

        mapping = {
            "mappings": {
                "properties": {
                    "timestamp": {"type": "date"},
                    "level": {"type": "keyword"},
                    "ip": {"type": "ip"},
                    "method": {"type": "keyword"},
                    "path": {"type": "text", "fields": {"keyword": {"type": "keyword"}}},
                    "status": {"type": "integer"},
                    "response_time": {"type": "integer"},
                    "user_agent": {"type": "text"},
                    "size": {"type": "long"}
                }
            },
            "settings": {
                "number_of_shards": 1,
                "number_of_replicas": 0
            }
        }

        self.es.indices.create(index=self.INDEX_NAME, body=mapping)
        print(f"索引 {self.INDEX_NAME} 创建成功")

    def bulk_index_logs(self, logs: List[Dict]) -> int:
        """批量索引日志"""
        if not logs:
            return 0

        actions = []
        for log in logs:
            action = {"index": {"_index": self.INDEX_NAME}}
            actions.append(action)
            actions.append(log)

        result = self.es.bulk(operations=actions, refresh=True)
        return len(logs) - len(result.get('errors', []))

    def search(
        self,
        keyword: str = None,
        ip: str = None,
        status: int = None,
        start_time: datetime = None,
        end_time: datetime = None,
        from_: int = 0,
        size: int = 100
    ) -> Dict:
        """搜索日志"""
        must = []

        if keyword:
            must.append({"match": {"path": keyword}})

        if ip:
            must.append({"term": {"ip": ip}})

        if status:
            must.append({"term": {"status": status}})

        if start_time or end_time:
            range_filter = {"timestamp": {}}
            if start_time:
                range_filter["timestamp"]["gte"] = start_time.isoformat()
            if end_time:
                range_filter["timestamp"]["lte"] = end_time.isoformat()
            must.append({"range": range_filter})

        query = {"bool": {"must": must}} if must else {"match_all": {}}

        result = self.es.search(
            index=self.INDEX_NAME,
            body={
                "query": query,
                "from": from_,
                "size": size,
                "sort": [{"timestamp": "desc"}]
            }
        )

        return {
            "total": result["hits"]["total"]["value"],
            "hits": [hit["_source"] for hit in result["hits"]["hits"]]
        }

    def delete_index(self):
        """删除索引"""
        if self.es.indices.exists(index=self.INDEX_NAME):
            self.es.indices.delete(index=self.INDEX_NAME)
            print(f"索引 {self.INDEX_NAME} 已删除")
```

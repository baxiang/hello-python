### 1.1 统计服务

```python
# analytics_services.py
from sqlalchemy.orm import Session
from sqlalchemy import func, and_
from typing import Dict, List
from datetime import datetime, timedelta

from models import AccessLog


class AnalyticsService:
    """分析服务"""

    def __init__(self, db: Session):
        self.db = db

    def get_realtime_stats(self, minutes: int = 5) -> Dict:
        """实时统计（最近N分钟）"""
        start_time = datetime.now() - timedelta(minutes=minutes)

        # 总请求数
        total = self.db.query(func.count(AccessLog.id)).filter(
            AccessLog.timestamp >= start_time
        ).scalar()

        # 错误数
        errors = self.db.query(func.count(AccessLog.id)).filter(
            AccessLog.timestamp >= start_time,
            AccessLog.status >= 400
        ).scalar()

        # 慢请求数
        slow = self.db.query(func.count(AccessLog.id)).filter(
            AccessLog.timestamp >= start_time,
            AccessLog.response_time >= 1000
        ).scalar()

        # 平均响应时间
        avg_time = self.db.query(func.avg(AccessLog.response_time)).filter(
            AccessLog.timestamp >= start_time
        ).scalar()

        # QPS
        qps = total / (minutes * 60) if minutes > 0 else 0

        return {
            "time_range": f"{minutes} 分钟",
            "total_requests": total,
            "error_count": errors,
            "error_rate": round(errors / total * 100, 2) if total > 0 else 0,
            "slow_request_count": slow,
            "avg_response_time": round(avg_time or 0, 2),
            "qps": round(qps, 2)
        }

    def get_hourly_stats(self, hours: int = 24) -> List[Dict]:
        """每小时统计"""
        start_time = datetime.now() - timedelta(hours=hours)

        results = self.db.query(
            func.date_format(AccessLog.timestamp, '%Y-%m-%d %H:00:00').label('hour'),
            func.count(AccessLog.id).label('count'),
            func.avg(AccessLog.response_time).label('avg_time')
        ).filter(
            AccessLog.timestamp >= start_time
        ).group_by(
            func.date_format(AccessLog.timestamp, '%Y-%m-%d %H')
        ).all()

        return [
            {
                "hour": r.hour,
                "requests": r.count,
                "avg_response_time": round(r.avg_time or 0, 2)
            }
            for r in results
        ]

    def get_status_trend(self, hours: int = 24) -> Dict:
        """状态码趋势"""
        start_time = datetime.now() - timedelta(hours=hours)

        results = self.db.query(
            AccessLog.status,
            func.count(AccessLog.id).label('count')
        ).filter(
            AccessLog.timestamp >= start_time
        ).group_by(AccessLog.status).all()

        return {status: count for status, count in results}

    def get_top_slow_ips(self, limit: int = 10) -> List[Dict]:
        """最慢 IP"""
        results = self.db.query(
            AccessLog.ip,
            func.avg(AccessLog.response_time).label('avg_time'),
            func.count(AccessLog.id).label('count')
        ).group_by(AccessLog.ip).order_by(
            func.avg(AccessLog.response_time).desc()
        ).limit(limit).all()

        return [
            {
                "ip": r.ip,
                "avg_response_time": round(r.avg_time, 2),
                "request_count": r.count
            }
            for r in results
        ]

    def get_error_analysis(self) -> List[Dict]:
        """错误分析"""
        results = self.db.query(
            AccessLog.path,
            AccessLog.status,
            func.count(AccessLog.id).label('count')
        ).filter(
            AccessLog.status >= 400
        ).group_by(
            AccessLog.path, AccessLog.status
        ).order_by(
            func.count(AccessLog.id).desc()
        ).limit(20).all()

        return [
            {
                "path": r.path,
                "status": r.status,
                "count": r.count
            }
            for r in results
        ]
```

# alert_config.py
from pydantic import BaseModel
from typing import Optional
from datetime import datetime


class AlertRule(BaseModel):
    """告警规则"""
    name: str
    metric: str  # error_rate, slow_rate, qps
    threshold: float
    operator: str  # gt, lt, gte, lte
    duration: int = 5  # 持续时间（分钟）
    enabled: bool = True


# 默认告警规则
DEFAULT_ALERT_RULES = [
    AlertRule(
        name="错误率过高",
        metric="error_rate",
        threshold=5.0,
        operator="gte",
        duration=5
    ),
    AlertRule(
        name="响应时间过长",
        metric="avg_response_time",
        threshold=3000,
        operator="gte",
        duration=5
    ),
    AlertRule(
        name="QPS 异常",
        metric="qps",
        threshold=1000,
        operator="gt",
        duration=5
    ),
]

### 2.2 告警服务

```python
# alert_service.py
from typing import List, Dict, Optional
from datetime import datetime
import time

from alert_config import AlertRule


class AlertService:
    """告警服务"""

    def __init__(self):
        self.rules: List[AlertRule] = []
        self.alert_history: List[Dict] = []
        self.last_alert_time: Dict[str, datetime] = {}

    def add_rule(self, rule: AlertRule):
        """添加告警规则"""
        self.rules.append(rule)

    def remove_rule(self, name: str):
        """移除告警规则"""
        self.rules = [r for r in self.rules if r.name != name]

    def check_condition(self, value: float, rule: AlertRule) -> bool:
        """检查条件"""
        if rule.operator == "gt":
            return value > rule.threshold
        elif rule.operator == "lt":
            return value < rule.threshold
        elif rule.operator == "gte":
            return value >= rule.threshold
        elif rule.operator == "lte":
            return value <= rule.threshold
        return False

    def should_alert(self, rule: AlertRule) -> bool:
        """检查是否应该告警（避免重复告警）"""
        now = datetime.now()
        last_time = self.last_alert_time.get(rule.name)

        # 5分钟内不重复告警
        if last_time and (now - last_time).seconds < 300:
            return False

        return True

    def trigger_alert(self, rule: AlertRule, value: float, message: str):
        """触发告警"""
        if not self.should_alert(rule):
            return

        alert = {
            "rule": rule.name,
            "metric": rule.metric,
            "value": value,
            "threshold": rule.threshold,
            "message": message,
            "timestamp": datetime.now().isoformat()
        }

        self.alert_history.append(alert)
        self.last_alert_time[rule.name] = datetime.now()

        # 发送告警通知
        self.send_notification(alert)

    def send_notification(self, alert: Dict):
        """发送告警通知"""
        print(f"\n{'='*50}")
        print(f"🚨 告警通知")
        print(f"{'='*50}")
        print(f"规则: {alert['rule']}")
        print(f"指标: {alert['metric']}")
        print(f"当前值: {alert['value']}")
        print(f"阈值: {alert['threshold']}")
        print(f"消息: {alert['message']}")
        print(f"时间: {alert['timestamp']}")
        print(f"{'='*50}\n")

    def check_alerts(self, stats: Dict):
        """检查所有告警规则"""
        metrics = {
            "error_rate": stats.get("error_rate", 0),
            "avg_response_time": stats.get("avg_response_time", 0),
            "qps": stats.get("qps", 0)
        }

        for rule in self.rules:
            if not rule.enabled:
                continue

            value = metrics.get(rule.metric, 0)
            if self.check_condition(value, rule):
                message = f"{rule.name}: 当前值 {value}, 阈值 {rule.threshold}"
                self.trigger_alert(rule, value, message)

    def get_alert_history(self, limit: int = 50) -> List[Dict]:
        """获取告警历史"""
        return self.alert_history[-limit:]

    def clear_alert_history(self):
        """清空告警历史"""
        self.alert_history = []
```

## 第三部分 - 定时任务

### 3.1 Celery 任务

```python
# tasks.py
from celery import Celery
from datetime import datetime, timedelta
from sqlalchemy import func

from database import SessionLocal
from models import AccessLog

celery_app = Celery("log_analytics", broker="redis://localhost:6379/0")


@celery_app.task
def cleanup_old_logs(days: int = 30):
    """清理旧日志"""
    db = SessionLocal()
    try:
        cutoff = datetime.now() - timedelta(days=days)
        count = db.query(AccessLog).filter(
            AccessLog.timestamp < cutoff
        ).delete(synchronize_session=False)
        db.commit()
        print(f"已清理 {count} 条旧日志")
        return count
    finally:
        db.close()


@celery_app.task
def generate_hourly_report():
    """生成小时报告"""
    db = SessionLocal()
    try:
        now = datetime.now()
        start_time = now - timedelta(hours=1)

        # 统计
        total = db.query(func.count(AccessLog.id)).filter(
            AccessLog.timestamp >= start_time
        ).scalar()

        errors = db.query(func.count(AccessLog.id)).filter(
            AccessLog.timestamp >= start_time,
            AccessLog.status >= 400
        ).scalar()

        avg_time = db.query(func.avg(AccessLog.response_time)).filter(
            AccessLog.timestamp >= start_time
        ).scalar()

        report = {
            "time": now.isoformat(),
            "total_requests": total,
            "error_count": errors,
            "error_rate": round(errors / total * 100, 2) if total > 0 else 0,
            "avg_response_time": round(avg_time or 0, 2)
        }

        print(f"小时报告: {report}")
        return report
    finally:
        db.close()


@celery_app.task
def check_alerts():
    """检查告警"""
    from analytics_services import AnalyticsService

    db = SessionLocal()
    try:
        service = AnalyticsService(db)
        stats = service.get_realtime_stats(minutes=5)

        # 添加错误率和 QPS
        stats["error_rate"] = stats.get("error_rate", 0)
        stats["qps"] = stats.get("qps", 0)

        from alert_service import AlertService
        alert_service = AlertService()
        alert_service.check_alerts(stats)

        return alert_service.get_alert_history()
    finally:
        db.close()
```

from celery.schedules import crontab

celery_app.conf.beat_schedule = {
    # 每5分钟检查告警
    "check-alerts": {
        "task": "tasks.check_alerts",
        "schedule": 300,  # 5分钟
    },

    # 每小时生成报告
    "hourly-report": {
        "task": "tasks.generate_hourly_report",
        "schedule": crontab(minute=0),  # 每小时整点

    # 每天凌晨3点清理日志
    "cleanup-logs": {
        "task": "tasks.cleanup_old_logs",

### 4.1 主程序

```python
# main.py
from fastapi import FastAPI, Depends
from sqlalchemy.orm import Session
from datetime import datetime, timedelta

from database import get_db, init_db
from analytics_services import AnalyticsService
from alert_service import AlertService
from alert_config import DEFAULT_ALERT_RULES

app = FastAPI(title="日志分析系统", version="1.0.0")

# 初始化告警服务
alert_service = AlertService()
for rule in DEFAULT_ALERT_RULES:
    alert_service.add_rule(rule)


@app.on_event("startup")
def startup():
    init_db()


@app.get("/analytics/realtime")
def get_realtime_stats(
    minutes: int = 5,
    db: Session = Depends(get_db)
):
    """实时统计"""
    service = AnalyticsService(db)
    stats = service.get_realtime_stats(minutes)

    # 检查告警
    alert_service.check_alerts(stats)

    return stats


@app.get("/analytics/hourly")
def get_hourly_stats(
    hours: int = 24,
    db: Session = Depends(get_db)
):
    """每小时统计"""
    service = AnalyticsService(db)
    return service.get_hourly_stats(hours)


@app.get("/analytics/trend")
def get_status_trend(
    hours: int = 24,
    db: Session = Depends(get_db)
):
    """状态码趋势"""
    service = AnalyticsService(db)
    return service.get_status_trend(hours)


@app.get("/analytics/slow-ips")
def get_slow_ips(
    limit: int = 10,
    db: Session = Depends(get_db)
):
    """最慢 IP"""
    service = AnalyticsService(db)
    return service.get_top_slow_ips(limit)


@app.get("/analytics/errors")
def get_error_analysis(db: Session = Depends(get_db)):
    """错误分析"""
    service = AnalyticsService(db)
    return service.get_error_analysis()


@app.get("/alerts/history")
def get_alert_history(limit: int = 50):
    """告警历史"""
    return alert_service.get_alert_history(limit)


@app.post("/alerts/rules")
def add_alert_rule(rule: AlertRule):
    """添加告警规则"""
    alert_service.add_rule(rule)
    return {"message": "规则添加成功"}


@app.get("/health")
def health_check():
    return {"status": "ok"}


if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)
```

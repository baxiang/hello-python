## 2. 数据模型 (models.py)

```python
from pydantic import BaseModel
from enum import Enum
from datetime import datetime
from typing import Optional

class MessageType(str, Enum):
    JOIN = "join"
    LEAVE = "leave"
    MESSAGE = "message"
    USERS = "users"
    HISTORY = "history"

class ChatMessage(BaseModel):
    """聊天消息"""
    type: MessageType
    username: str
    content: str
    timestamp: datetime = None
    
    def __init__(self, **data):
        super().__init__(**data)
        if self.timestamp is None:
            self.timestamp = datetime.now()

class User(BaseModel):
    """用户信息"""
    username: str
    connected_at: datetime = None
```

## 3. 连接管理 (manager.py)

```python
from fastapi import WebSocket
from typing import Dict, List
from datetime import datetime
import json

class ConnectionManager:
    """WebSocket 连接管理器"""
    
    def __init__(self):
        # username -> WebSocket
        self.active_connections: Dict[str, WebSocket] = {}
    
    async def connect(self, websocket: WebSocket, username: str):
        """接受新连接"""
        await websocket.accept()
        self.active_connections[username] = websocket
    
    def disconnect(self, username: str):
        """断开连接"""
        if username in self.active_connections:
            del self.active_connections[username]
    
    def get_users(self) -> List[str]:
        """获取在线用户列表"""
        return list(self.active_connections.keys())
    
    async def send_to(self, username: str, message: dict):
        """发送消息给指定用户"""
        if username in self.active_connections:
            await self.active_connections[username].send_json(message)
    
    async def broadcast(self, message: dict, exclude: str = None):
        """广播消息给所有用户"""
        for username, connection in self.active_connections.items():
            if username != exclude:
                await connection.send_json(message)
    
    async def broadcast_users(self):
        """广播用户列表"""
        users = self.get_users()
        message = {
            "type": "users",
            "users": users,
            "count": len(users)
        }
        await self.broadcast(message)
```

## 4. 主程序 (main.py)

```python
from fastapi import FastAPI, WebSocket, WebSocketDisconnect
from fastapi.staticfiles import StaticFiles
from fastapi.responses import HTMLResponse
from datetime import datetime
import json

from manager import ConnectionManager
from models import MessageType

app = FastAPI(title="实时聊天")

# 静态文件
app.mount("/static", StaticFiles(directory="static"), name="static")

# 连接管理器
manager = ConnectionManager()

# 消息历史（内存存储，生产环境应使用数据库）
message_history = []


@app.get("/", response_class=HTMLResponse)
async def get():
    """返回聊天页面"""
    with open("static/index.html", "r", encoding="utf-8") as f:
        return f.read()


@app.websocket("/ws/{username}")
async def websocket_endpoint(websocket: WebSocket, username: str):
    """WebSocket 聊天端点"""
    await manager.connect(websocket, username)
    
    # 发送历史消息
    await manager.send_to(username, {
        "type": "history",
        "messages": message_history[-50:]  # 最近50条
    })
    
    # 广播用户加入
    join_msg = {
        "type": "join",
        "username": username,
        "timestamp": datetime.now().isoformat(),
        "users": manager.get_users()
    }
    await manager.broadcast(join_msg)
    
    try:
        while True:
            # 接收消息
            data = await websocket.receive_text()
            message_data = json.loads(data)
            
            # 构建消息
            message = {
                "type": "message",
                "username": username,
                "content": message_data.get("content", ""),
                "timestamp": datetime.now().isoformat()
            }
            
            # 保存到历史
            message_history.append(message)
            
            # 广播给所有人
            await manager.broadcast(message)
    
    except WebSocketDisconnect:
        # 用户断开连接
        manager.disconnect(username)
        
        # 广播用户离开
        leave_msg = {
            "type": "leave",
            "username": username,
            "timestamp": datetime.now().isoformat(),
            "users": manager.get_users()
        }
        await manager.broadcast(leave_msg)


@app.get("/api/users")
async def get_users():
    """获取在线用户列表"""
    return {
        "users": manager.get_users(),
        "count": len(manager.get_users())
    }


if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)
```

## 5. 运行服务器

```bash
uv init chat-server
cd chat-server

# 安装依赖
uv add fastapi uvicorn

# 创建静态目录
mkdir -p static

# 运行
uv run python main.py
```

### 2.2 定时数据采集

```python
# scheduler.py
from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.triggers.interval import IntervalTrigger
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class StockScheduler:
    """股票数据定时采集器"""

    def __init__(self):
        self.scheduler = BackgroundScheduler()
        self.repo = StockRepository()

    def crawl_stock_data(self, stock_code: str):
        """采集单只股票数据"""
        try:
            logger.info(f"开始采集 {stock_code} 数据...")
            from stock_crawler import get_stock_history, clean_stock_data

            df = get_stock_history(stock_code, period="daily")
            df = clean_stock_data(df)

            # 存储到数据库
            self.repo.batch_add_daily(stock_code, df)
            logger.info(f"{stock_code} 数据采集完成，新增 {len(df)} 条记录")
        except Exception as e:
            logger.error(f"采集 {stock_code} 失败：{e}")

    def start_crawl_job(self, stock_codes: list, interval_minutes: int = 60):
        """启动定时采集任务"""
        self.scheduler.add_job(
            func=self._crawl_all_stocks,
            trigger=IntervalTrigger(minutes=interval_minutes),
            args=[stock_codes],
            id="stock_crawl",
            name="股票数据采集",
            replace_existing=True
        )
        logger.info(f"定时采集任务已启动，每 {interval_minutes} 分钟执行一次")

    def _crawl_all_stocks(self, stock_codes: list):
        """采集所有股票数据"""
        for code in stock_codes:
            self.crawl_stock_data(code)

    def start(self):
        """启动调度器"""
        self.scheduler.start()
        logger.info("调度器已启动")

    def stop(self):
        """停止调度器"""
        self.scheduler.shutdown()
        logger.info("调度器已停止")

if __name__ == "__main__":
    scheduler = StockScheduler()
    stocks = ["sh600519", "sh600036", "sh601318"]
    scheduler.start_crawl_job(stocks, interval_minutes=60)
    scheduler.start()

    try:
        import time
        while True:
            time.sleep(1)
    except KeyboardInterrupt:
        scheduler.stop()
```

### 2.3 价格监控告警

```python
# monitor.py
import requests
import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart
from datetime import datetime
import logging

logger = logging.getLogger(__name__)

class StockMonitor:
    """股票价格监控器"""

    def __init__(self):
        self.alerts = []  # 告警配置列表

    def add_alert(self, stock_code: str, condition: str, threshold: float, notification: dict):
        """添加告警配置

        Args:
            stock_code: 股票代码
            condition: 条件 (above/below/change)
            threshold: 阈值
            notification: 通知配置
        """
        self.alerts.append({
            'stock_code': stock_code,
            'condition': condition,
            'threshold': threshold,
            'notification': notification
        })

    def check_price(self, stock_code: str) -> dict:
        """获取实时价格"""
        url = f"https://qt.gtimg.cn/q={stock_code}"
        response = requests.get(url)
        response.encoding = 'gbk'

        data = response.text.split("~")
        return {
            'code': stock_code,
            'name': data[1],
            'price': float(data[3]),
            'change': float(data[4]),
            'percent': float(data[5]),
            'open': float(data[8]),
            'high': float(data[9]),
            'low': float(data[10]),
            'close': float(data[30])
        }

    def check_alerts(self):
        """检查所有告警"""
        for alert in self.alerts:
            try:
                stock = self.check_price(alert['stock_code'])
                triggered = False
                message = ""

                if alert['condition'] == 'above' and stock['price'] > alert['threshold']:
                    triggered = True
                    message = f"{stock['name']} 突破 {alert['threshold']} 元，当前价格 {stock['price']} 元"
                elif alert['condition'] == 'below' and stock['price'] < alert['threshold']:
                    triggered = True
                    message = f"{stock['name']} 跌破 {alert['threshold']} 元，当前价格 {stock['price']} 元"
                elif alert['condition'] == 'change' and abs(stock['percent']) > alert['threshold']:
                    triggered = True
                    message = f"{stock['name']} 涨跌幅超过 {alert['threshold']}%，当前涨跌幅 {stock['percent']}%"

                if triggered:
                    logger.warning(f"触发告警：{message}")
                    self.send_notification(alert['notification'], message)
            except Exception as e:
                logger.error(f"检查告警失败：{e}")

    def send_notification(self, notification: dict, message: str):
        """发送通知"""
        if notification.get('type') == 'email':
            self.send_email(
                notification.get('smtp_server'),
                notification.get('smtp_port'),
                notification.get('sender'),
                notification.get('password'),
                notification.get('receivers', []),
                message
            )
        elif notification.get('type') == 'console':
            print(f"【告警】{message}")

    def send_email(self, smtp_server, smtp_port, sender, password, receivers, message):
        """发送邮件"""
        msg = MIMEMultipart()
        msg['From'] = sender
        msg['To'] = ','.join(receivers)
        msg['Subject'] = f"股票告警 - {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}"

        body = f"""
        <h2>股票价格告警</h2>
        <p>{message}</p>
        <p>告警时间：{datetime.now().strftime('%Y-%m-%d %H:%M:%S')}</p>
        """
        msg.attach(MIMEText(body, 'html', 'utf-8'))

        try:
            with smtplib.SMTP(smtp_server, smtp_port) as server:
                server.starttls()
                server.login(sender, password)
                server.send_message(msg)
            logger.info("邮件发送成功")
        except Exception as e:
            logger.error(f"邮件发送失败：{e}")

if __name__ == "__main__":
    monitor = StockMonitor()

    # 添加告警配置
    monitor.add_alert(
        stock_code="sh600519",
        condition="above",
        threshold=2000,
        notification={
            "type": "email",
            "smtp_server": "smtp.qq.com",
            "smtp_port": 587,
            "sender": "your_email@qq.com",
            "password": "your_password",
            "receivers": ["alert@example.com"]
        }
    )

    # 启动监控
    import time
    while True:
        monitor.check_alerts()
        time.sleep(60)  # 每分钟检查一次
```

### 2.4 整合服务

```python
# app.py
from scheduler import StockScheduler
from monitor import StockMonitor

def main():
    """主程序入口"""
    # 初始化
    scheduler = StockScheduler()
    monitor = StockMonitor()

    # 配置监控
    monitor.add_alert(
        stock_code="sh600519",
        condition="above",
        threshold=2000,
        notification={"type": "console"}
    )

    # 启动定时采集
    stocks = ["sh600519", "sh600036", "sh601318"]
    scheduler.start_crawl_job(stocks, interval_minutes=60)

    # 启动服务
    scheduler.start()

    try:
        import time
        while True:
            # 每分钟检查告警
            monitor.check_alerts()
            time.sleep(60)
    except KeyboardInterrupt:
        scheduler.stop()
        print("服务已停止")

if __name__ == "__main__":
    main()
```

### 3.2 Systemd 服务

```bash
# /etc/systemd/system/stock-monitor.service
[Unit]
Description=Stock Monitor Service
After=network.target

[Service]
Type=simple
User=ubuntu
WorkingDirectory=/path/to/project
ExecStart=/usr/bin/python3 app.py
Restart=always
RestartSec=10

[Install]
WantedBy=multi-user.target
```

```bash
# 启动服务
sudo systemctl start stock-monitor
sudo systemctl enable stock-monitor  # 开机自启
```

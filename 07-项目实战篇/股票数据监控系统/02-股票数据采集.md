### 2.2 爬取历史数据

```python
# stock_crawler.py
import requests
import pandas as pd
from datetime import datetime
import time

def get_stock_history(stock_code: str, period: str = "daily") -> pd.DataFrame:
    """获取股票历史数据

    Args:
        stock_code: 股票代码，如 sh600519
        period: 日/周/月线 (daily/weekly/monthly)
    """
    # 东方财富 API
    if stock_code.startswith("sh"):
        secid = f"1.{stock_code[2:]}"
    else:
        secid = f"0.{stock_code[2:]}"

    url = "https://push2his.eastmoney.com/api/qt/stock/kline/get"

    # 周期参数
    period_map = {
        "daily": "101",
        "weekly": "102",
        "monthly": "103"
    }

    params = {
        "secid": secid,
        "fields1": "f1,f2,f3,f4,f5,f6",
        "fields2": "f51,f52,f53,f54,f55,f56,f57,f58,f59,f60,f61",
        "klt": period_map.get(period, "101"),
        "fqt": "1",  # 前复权
        "beg": "20200101",  # 开始日期
        "end": "20500101",  # 结束日期
    }

    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36"
    }

    response = requests.get(url, params=params, headers=headers)

    if response.status_code != 200:
        raise Exception(f"请求失败：{response.status_code}")

    data = response.json()

    if data["data"]["klines"] is None:
        raise Exception(f"未获取到数据：{stock_code}")

    # 解析数据
    klines = data["data"]["klines"]
    records = []

    for kline in klines:
        items = kline.split(",")
        records.append({
            "date": items[0],
            "open": float(items[1]),
            "close": float(items[2]),
            "high": float(items[3]),
            "low": float(items[4]),
            "volume": float(items[5]),
            "amount": float(items[6]),
        })

    df = pd.DataFrame(records)
    return df

if __name__ == "__main__":
    # 获取茅台历史数据
    df = get_stock_history("sh600519", period="daily")
    print(f"获取到 {len(df)} 条数据")
    print(df.head())
```

### 2.3 数据清洗

```python
def clean_stock_data(df: pd.DataFrame) -> pd.DataFrame:
    """清洗股票数据"""
    # 转换日期格式
    df['date'] = pd.to_datetime(df['date'])

    # 处理缺失值
    df = df.dropna()

    # 处理异常值
    df = df[df['close'] > 0]
    df = df[df['volume'] >= 0]

    # 添加衍生指标
    df['change'] = df['close'].pct_change()  # 日涨跌幅
    df['ma5'] = df['close'].rolling(window=5).mean()   # 5日均线
    df['ma10'] = df['close'].rolling(window=10).mean() # 10日均线
    df['ma20'] = df['close'].rolling(window=20).mean() # 20日均线

    # 重置索引
    df = df.reset_index(drop=True)

    return df

# 清洗数据
df_clean = clean_stock_data(df)
print(df_clean.tail())
```

### 2.4 批量采集

```python
def batch_crawl_stocks(codes: list, period: str = "daily") -> dict:
    """批量采集多只股票数据"""
    results = {}

    for code in codes:
        try:
            print(f"正在采集 {code}...")
            df = get_stock_history(code, period)
            df = clean_stock_data(df)
            results[code] = df
            time.sleep(1)  # 避免请求过快
        except Exception as e:
            print(f"采集 {code} 失败：{e}")
            results[code] = None

    return results

if __name__ == "__main__":
    stocks = ["sh600519", "sh600036", "sh601318"]  # 茅台、招商银行、中国平安
    results = batch_crawl_stocks(stocks)

    for code, df in results.items():
        if df is not None:
            print(f"{code}: {len(df)} 条记录")
```

```python
def export_to_csv(df: pd.DataFrame, stock_code: str, period: str = "daily"):
    """导出数据到 CSV"""
    filename = f"stock_{stock_code}_{period}_{datetime.now().strftime('%Y%m%d')}.csv"
    df.to_csv(filename, index=False, encoding='utf-8-sig')
    print(f"数据已导出到：{filename}")
    return filename

def export_to_json(df: pd.DataFrame, stock_code: str):
    """导出数据到 JSON"""
    filename = f"stock_{stock_code}_{datetime.now().strftime('%Y%m%d')}.json"
    df.to_json(filename, orient='records', date_format='iso', force_ascii=False, indent=2)
    print(f"数据已导出到：{filename}")
    return filename
```

### 3.2 计算技术指标

```python
def calculate_indicators(df: pd.DataFrame) -> pd.DataFrame:
    """计算技术指标"""
    # RSI 相对强弱指标
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))

    # MACD
    exp12 = df['close'].ewm(span=12, adjust=False).mean()
    exp26 = df['close'].ewm(span=26, adjust=False).mean()
    df['macd'] = exp12 - exp26
    df['signal'] = df['macd'].ewm(span=9, adjust=False).mean()
    df['histogram'] = df['macd'] - df['signal']

    # 布林带
    df['bb_middle'] = df['close'].rolling(window=20).mean()
    df['bb_std'] = df['close'].rolling(window=20).std()
    df['bb_upper'] = df['bb_middle'] + 2 * df['bb_std']
    df['bb_lower'] = df['bb_middle'] - 2 * df['bb_std']

    return df
```

### 3.3 增量更新

```python
def update_stock_data(stock_code: str, existing_df: pd.DataFrame) -> pd.DataFrame:
    """增量更新股票数据"""
    # 获取最新数据
    latest_date = existing_df['date'].max()
    new_df = get_stock_history(stock_code)
    new_df = clean_stock_data(new_df)

    # 过滤出新数据
    new_data = new_df[new_df['date'] > latest_date]

    if len(new_data) > 0:
        # 合并数据
        updated_df = pd.concat([existing_df, new_data], ignore_index=True)
        print(f"新增 {len(new_data)} 条数据")
        return updated_df
    else:
        print("没有新数据")
        return existing_df
```

### 2.2 创建数据库模型

```python
# models.py
from sqlalchemy import create_engine, Column, Integer, String, Float, DateTime, UniqueConstraint
from sqlalchemy.ext.declarative import declarative_base
from sqlalchemy.orm import sessionmaker
from datetime import datetime

Base = declarative_base()

class StockDaily(Base):
    """股票日线数据表"""
    __tablename__ = 'stock_daily'

    id = Column(Integer, primary_key=True, autoincrement=True)
    stock_code = Column(String(10), nullable=False, index=True)
    trade_date = Column(DateTime, nullable=False, index=True)
    open = Column(Float)
    high = Column(Float)
    low = Column(Float)
    close = Column(Float)
    volume = Column(Float)
    amount = Column(Float)
    change = Column(Float)
    created_at = Column(DateTime, default=datetime.now)

    # 联合唯一索引，防止重复数据
    __table_args__ = (
        UniqueConstraint('stock_code', 'trade_date', name='uq_stock_date'),
    )

    def __repr__(self):
        return f"<StockDaily({self.stock_code}, {self.trade_date})>"

class StockInfo(Base):
    """股票基本信息表"""
    __tablename__ = 'stock_info'

    id = Column(Integer, primary_key=True, autoincrement=True)
    stock_code = Column(String(10), unique=True, nullable=False, index=True)
    stock_name = Column(String(50))
    industry = Column(String(50))
    list_date = Column(DateTime)
    created_at = Column(DateTime, default=datetime.now)

    def __repr__(self):
        return f"<StockInfo({self.stock_code}, {self.stock_name})>"
```

### 2.3 数据库连接

```python
# database.py
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker
from dbutils.pooled_db import PooledDB
import pymysql

# 数据库配置
DB_CONFIG = {
    "host": "localhost",
    "port": 3306,
    "user": "root",
    "password": "your_password",
    "database": "stock_db",
    "charset": "utf8mb4"
}

# 创建连接池
engine = create_engine(
    f"mysql+pymysql://{DB_CONFIG['user']}:{DB_CONFIG['password']}"
    f"@{DB_CONFIG['host']}:{DB_CONFIG['port']}/{DB_CONFIG['database']}"
    f"?charset={DB_CONFIG['charset']}",
    poolclass=PooledDB,
    maxconnections=10,
    pool_pre_ping=True
)

# 创建 Session
Session = sessionmaker(bind=engine)

def get_session():
    """获取数据库会话"""
    return Session()

def init_db():
    """初始化数据库"""
    from models import Base
    Base.metadata.create_all(engine)
    print("数据库表创建成功")
```

### 2.4 数据操作

```python
# repository.py
from models import StockDaily, StockInfo
from database import get_session
from datetime import datetime
import pandas as pd

class StockRepository:
    """股票数据仓库"""

    def __init__(self):
        self.session = get_session()

    def add_stock_info(self, stock_code: str, stock_name: str, industry: str = None):
        """添加股票基本信息"""
        info = StockInfo(
            stock_code=stock_code,
            stock_name=stock_name,
            industry=industry
        )
        self.session.add(info)
        self.session.commit()
        return info

    def add_daily_data(self, stock_code: str, trade_date: str, data: dict):
        """添加日线数据"""
        daily = StockDaily(
            stock_code=stock_code,
            trade_date=datetime.strptime(trade_date, "%Y-%m-%d"),
            open=data.get('open'),
            high=data.get('high'),
            low=data.get('low'),
            close=data.get('close'),
            volume=data.get('volume'),
            amount=data.get('amount'),
            change=data.get('change')
        )
        self.session.add(daily)
        try:
            self.session.commit()
            return True
        except Exception as e:
            self.session.rollback()
            if "Duplicate" in str(e):
                return False  # 数据已存在
            raise e

    def batch_add_daily(self, stock_code: str, df: pd.DataFrame):
        """批量添加日线数据"""
        success_count = 0
        for _, row in df.iterrows():
            try:
                result = self.add_daily_data(
                    stock_code=stock_code,
                    trade_date=row['date'].strftime("%Y-%m-%d") if isinstance(row['date'], pd.Timestamp) else str(row['date']),
                    data={
                        'open': row['open'],
                        'high': row['high'],
                        'low': row['low'],
                        'close': row['close'],
                        'volume': row['volume'],
                        'amount': row.get('amount'),
                        'change': row.get('change')
                    }
                )
                if result:
                    success_count += 1
            except Exception as e:
                print(f"添加数据失败：{e}")
        print(f"成功添加 {success_count}/{len(df)} 条数据")
        return success_count

    def get_daily_data(self, stock_code: str, start_date: str = None, end_date: str = None) -> pd.DataFrame:
        """查询日线数据"""
        query = self.session.query(StockDaily).filter(StockDaily.stock_code == stock_code)

        if start_date:
            query = query.filter(StockDaily.trade_date >= datetime.strptime(start_date, "%Y-%m-%d"))
        if end_date:
            query = query.filter(StockDaily.trade_date <= datetime.strptime(end_date, "%Y-%m-%d"))

        results = query.order_by(StockDaily.trade_date).all()

        data = [{
            'date': r.trade_date,
            'open': r.open,
            'high': r.high,
            'low': r.low,
            'close': r.close,
            'volume': r.volume,
            'amount': r.amount,
            'change': r.change
        } for r in results]

        return pd.DataFrame(data)

    def get_all_stocks(self):
        """获取所有股票"""
        return self.session.query(StockInfo).all()

    def close(self):
        """关闭会话"""
        self.session.close()
```

### 3.1 数据更新

```python
def update_daily_data(self, stock_code: str, trade_date: str, data: dict):
    """更新日线数据"""
    query = self.session.query(StockDaily).filter(
        StockDaily.stock_code == stock_code,
        StockDaily.trade_date == datetime.strptime(trade_date, "%Y-%m-%d")
    )
    daily = query.first()

    if daily:
        for key, value in data.items():
            if hasattr(daily, key):
                setattr(daily, key, value)
        self.session.commit()
        return True
    return False
```

### 3.2 数据统计

```python
def get_stock_stats(self, stock_code: str, days: int = 30):
    """获取股票统计信息"""
    from sqlalchemy import func

    query = self.session.query(
        func.avg(StockDaily.close).label('avg_price'),
        func.max(StockDaily.high).label('max_price'),
        func.min(StockDaily.low).label('min_price'),
        func.sum(StockDaily.volume).label('total_volume'),
        func.count(StockDaily.id).label('days')
    ).filter(StockDaily.stock_code == stock_code)

    if days:
        from datetime import timedelta
        start_date = datetime.now() - timedelta(days=days)
        query = query.filter(StockDaily.trade_date >= start_date)

    result = query.first()
    return {
        'avg_price': result.avg_price,
        'max_price': result.max_price,
        'min_price': result.min_price,
        'total_volume': result.total_volume,
        'days': result.days
    }
```

def create_database():
    """创建数据库"""
    import pymysql

    connection = pymysql.connect(
        host=DB_CONFIG['host'],
        port=DB_CONFIG['port'],
        user=DB_CONFIG['user'],
        password=DB_CONFIG['password']
    )

    try:
        with connection.cursor() as cursor:
            cursor.execute(f"CREATE DATABASE IF NOT EXISTS {DB_CONFIG['database']}")
        connection.commit()
        print(f"数据库 {DB_CONFIG['database']} 创建成功")
    finally:
        connection.close()

if __name__ == "__main__":
    create_database()
    init_db()

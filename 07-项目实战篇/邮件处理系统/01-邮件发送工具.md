### 2.2 基础发送功能

```python
# mail_sender.py
import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart

class MailSender:
    """邮件发送器"""

    def __init__(self, smtp_server: str, smtp_port: int, username: str, password: str):
        self.smtp_server = smtp_server
        self.smtp_port = smtp_port
        self.username = username
        self.password = password

    def send_email(self, to_addrs: list, subject: str, body: str,
                   mail_type: str = "plain", cc: list = None, bcc: list = None):
        """发送邮件

        Args:
            to_addrs: 收件人列表
            subject: 邮件主题
            body: 邮件正文
            mail_type: 邮件类型 (plain/html)
            cc: 抄送列表
            bcc: 密送列表
        """
        # 创建邮件对象
        msg = MIMEMultipart()
        msg['From'] = self.username
        msg['To'] = ', '.join(to_addrs)
        msg['Subject'] = subject

        if cc:
            msg['Cc'] = ', '.join(cc)
        if bcc:
            msg['Bcc'] = ', '.join(bcc)

        # 添加正文
        if mail_type == "html":
            msg.attach(MIMEText(body, 'html', 'utf-8'))
        else:
            msg.attach(MIMEText(body, 'plain', 'utf-8'))

        # 合并所有收件人
        all_recipients = to_addrs + (cc or []) + (bcc or [])

        # 发送邮件
        try:
            with smtplib.SMTP(self.smtp_server, self.smtp_port) as server:
                server.starttls()  # 启用 TLS
                server.login(self.username, self.password)
                server.send_message(msg, self.username, all_recipients)
            print(f"邮件发送成功！发送给：{all_recipients}")
            return True
        except Exception as e:
            print(f"邮件发送失败：{e}")
            return False

if __name__ == "__main__":
    # 配置邮件服务器
    sender = MailSender(
        smtp_server="smtp.qq.com",
        smtp_port=587,
        username="your_email@qq.com",
        password="your_auth_code"  # QQ 邮箱需要使用授权码
    )

    # 发送邮件
    sender.send_email(
        to_addrs=["recipient@example.com"],
        subject="测试邮件",
        body="这是一封测试邮件"
    )
```

### 2.3 发送附件

```python
from email.mime.base import MIMEBase
from email import encoders
import os

def send_email_with_attachment(self, to_addrs: list, subject: str, body: str,
                                attachments: list):
    """发送带附件的邮件"""
    msg = MIMEMultipart()
    msg['From'] = self.username
    msg['To'] = ', '.join(to_addrs)
    msg['Subject'] = subject

    # 添加正文
    msg.attach(MIMEText(body, 'plain', 'utf-8'))

    # 添加附件
    for file_path in attachments:
        if not os.path.exists(file_path):
            print(f"文件不存在：{file_path}")
            continue

        # 读取文件
        with open(file_path, 'rb') as f:
            file_data = f.read()

        # 创建附件
        filename = os.path.basename(file_path)
        part = MIMEBase('application', 'octet-stream')
        part.set_payload(file_data)
        encoders.encode_base64(part)

        # 设置附件头
        part.add_header(
            'Content-Disposition',
            f'attachment; filename= {filename}'
        )
        msg.attach(part)

    # 发送邮件
    with smtplib.SMTP(self.smtp_server, self.smtp_port) as server:
        server.starttls()
        server.login(self.username, self.password)
        server.send_message(msg, self.username, to_addrs)

    print(f"邮件发送成功，附件：{attachments}")

# 使用
sender.send_email_with_attachment(
    to_addrs=["recipient@example.com"],
    subject="带附件的邮件",
    body="请查收附件",
    attachments=["/path/to/file.pdf", "/path/to/image.png"]
)
```

## 第三部分 - 扩展功能

### 3.1 批量发送

```python
def batch_send(self, recipients: list, subject: str, body: str):
    """批量发送邮件"""
    success = 0
    failed = 0

    for recipient in recipients:
        try:
            if self.send_email([recipient], subject, body):
                success += 1
            else:
                failed += 1
        except Exception as e:
            print(f"发送失败：{recipient}, 错误：{e}")
            failed += 1

    print(f"批量发送完成：成功 {success}，失败 {failed}")
    return success, failed
```

### 3.2 邮件模板

```python
def send_template_email(self, to_addrs: list, subject: str, template: str, **kwargs):
    """发送模板邮件"""
    # 简单模板替换
    body = template
    for key, value in kwargs.items():
        body = body.replace(f"{{{key}}}", str(value))

    return self.send_email(to_addrs, subject, body)

# 使用
template = """
您好，{name}！

您的订单 {order_id} 已确认，
订单金额：{amount} 元，
预计发货时间：{delivery_date}

感谢您的购买！
"""

sender.send_template_email(
    to_addrs=["customer@example.com"],
    subject="订单确认",
    template=template,
    name="张三",
    order_id="ORD20240101001",
    amount="299.00",
    delivery_date="2024-01-05"
)
```

### 3.3 发送记录

```python
import json
from datetime import datetime

def save_send_log(self, to_addrs: list, subject: str, status: str):
    """保存发送日志"""
    log = {
        "time": datetime.now().isoformat(),
        "recipients": to_addrs,
        "subject": subject,
        "status": status
    }

    with open("send_log.json", "a", encoding="utf-8") as f:
        f.write(json.dumps(log, ensure_ascii=False) + "\n")
```

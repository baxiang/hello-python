### 2.2 附件处理器

```python
# attachment_handler.py
import os
import shutil
import zipfile
import tarfile
from pathlib import Path

class AttachmentHandler:
    """邮件附件处理器"""

    def __init__(self, base_dir: str = "./attachments"):
        self.base_dir = base_dir
        os.makedirs(base_dir, exist_ok=True)

    def process_attachments(self, attachments: list, process_type: str = "organize") -> list:
        """处理附件

        Args:
            attachments: 附件路径列表
            process_type: 处理类型 (organize/compress/convert)
        """
        results = []

        for attachment in attachments:
            if not os.path.exists(attachment):
                continue

            if process_type == "organize":
                result = self.organize_file(attachment)
            elif process_type == "compress":
                result = self.compress_file(attachment)
            elif process_type == "convert":
                result = self.convert_file(attachment)
            else:
                result = self.copy_to_processed(attachment)

            results.append(result)

        return results

    def organize_file(self, file_path: str) -> str:
        """整理文件 - 按类型分类"""
        ext = Path(file_path).suffix.lower()

        # 定义分类目录
        categories = {
            'images': ['.jpg', '.jpeg', '.png', '.gif', '.bmp', '.webp'],
            'documents': ['.pdf', '.doc', '.docx', '.txt', '.md'],
            'spreadsheets': ['.xls', '.xlsx', '.csv'],
            'archives': ['.zip', '.tar', '.gz', '.rar'],
            'others': []
        }

        # 查找分类
        category = 'others'
        for cat, extensions in categories.items():
            if ext in extensions:
                category = cat
                break

        # 移动文件
        target_dir = os.path.join(self.base_dir, category)
        os.makedirs(target_dir, exist_ok=True)

        filename = os.path.basename(file_path)
        target_path = os.path.join(target_dir, filename)

        # 避免覆盖
        if os.path.exists(target_path):
            name, ext = os.path.splitext(filename)
            target_path = os.path.join(target_dir, f"{name}_copy{ext}")

        shutil.move(file_path, target_path)
        print(f"已整理：{filename} -> {category}/")

        return target_path
```

### 2.3 文件压缩

```python
    def compress_file(self, file_path: str, compression: str = "zip") -> str:
        """压缩文件

        Args:
            file_path: 文件路径
            compression: 压缩格式 (zip/tar)
        """
        filename = os.path.basename(file_path)
        name, _ = os.path.splitext(filename)

        if compression == "zip":
            archive_path = os.path.join(self.base_dir, f"{name}.zip")
            with zipfile.ZipFile(archive_path, 'w', zipfile.ZIP_DEFLATED) as zf:
                zf.write(file_path, filename)
        elif compression == "tar":
            archive_path = os.path.join(self.base_dir, f"{name}.tar.gz")
            with tarfile.open(archive_path, 'w:gz') as tf:
                tf.add(file_path, arcname=filename)

        print(f"已压缩：{filename} -> {archive_path}")
        return archive_path

    def extract_archive(self, archive_path: str, extract_dir: str = None) -> list:
        """解压文件"""
        if extract_dir is None:
            extract_dir = os.path.join(self.base_dir, "extracted")

        os.makedirs(extract_dir, exist_ok=True)

        extracted_files = []

        if archive_path.endswith('.zip'):
            with zipfile.ZipFile(archive_path, 'r') as zf:
                zf.extractall(extract_dir)
                extracted_files = [os.path.join(extract_dir, name) for name in zf.namelist()]
        elif archive_path.endswith(('.tar', '.tar.gz', '.tgz')):
            with tarfile.open(archive_path, 'r:*') as tf:
                tf.extractall(extract_dir)
                extracted_files = [os.path.join(extract_dir, member.name) for member in tf.getmembers()]

        print(f"已解压：{os.path.basename(archive_path)} -> {extract_dir}")
        return extracted_files
```

### 2.4 文件转换

```python
    def convert_file(self, file_path: str, target_format: str = None) -> str:
        """转换文件格式"""
        ext = Path(file_path).suffix.lower()

        if ext == '.pdf':
            return self.convert_pdf(file_path, target_format)
        elif ext in ['.jpg', '.jpeg', '.png']:
            return self.convert_image(file_path, target_format)
        else:
            print(f"不支持转换：{ext}")
            return file_path

    def convert_pdf(self, pdf_path: str, target_format: str = "text") -> str:
        """PDF 转换"""
        from PyPDF2 import PdfReader

        output_path = os.path.join(
            self.base_dir,
            f"converted_{Path(pdf_path).stem}.{target_format}"
        )

        if target_format == "text":
            # 提取文本
            reader = PdfReader(pdf_path)
            text = ""
            for page in reader.pages:
                text += page.extract_text() + "\n"

            with open(output_path, 'w', encoding='utf-8') as f:
                f.write(text)

            print(f"PDF 转文本完成：{output_path}")
            return output_path

        return pdf_path

    def convert_image(self, image_path: str, target_format: str = "png") -> str:
        """图片格式转换"""
        from PIL import Image

        output_path = os.path.join(
            self.base_dir,
            f"converted_{Path(image_path).stem}.{target_format}"
        )

        with Image.open(image_path) as img:
            # 转换为 RGB 模式（支持 JPEG）
            if target_format.lower() in ['jpg', 'jpeg'] and img.mode != 'RGB':
                img = img.convert('RGB')
            img.save(output_path)

        print(f"图片转换完成：{output_path}")
        return output_path
```

### 3.1 批量重命名

```python
    def batch_rename(self, directory: str, pattern: str, replacement: str) -> int:
        """批量重命名文件

        Args:
            directory: 目录路径
            pattern: 匹配模式
            replacement: 替换内容
        """
        count = 0

        for filename in os.listdir(directory):
            if pattern in filename:
                old_path = os.path.join(directory, filename)
                new_filename = filename.replace(pattern, replacement)
                new_path = os.path.join(directory, new_filename)

                os.rename(old_path, new_path)
                count += 1
                print(f"重命名：{filename} -> {new_filename}")

        return count
```

### 3.2 文件去重

```python
    def remove_duplicates(self, directory: str) -> int:
        """删除重复文件（基于内容哈希）"""
        import hashlib

        hash_dict = {}
        removed_count = 0

        for filename in os.listdir(directory):
            file_path = os.path.join(directory, filename)

            if not os.path.isfile(file_path):
                continue

            # 计算文件哈希
            with open(file_path, 'rb') as f:
                file_hash = hashlib.md5(f.read()).hexdigest()

            if file_hash in hash_dict:
                # 删除重复文件
                os.remove(file_path)
                removed_count += 1
                print(f"删除重复：{filename} (与 {hash_dict[file_hash]} 相同)")
            else:
                hash_dict[file_hash] = filename

        return removed_count
```

### 3.3 文件统计

```python
    def get_file_stats(self, directory: str) -> dict:
        """获取文件统计信息"""
        stats = {
            'total_files': 0,
            'total_size': 0,
            'by_extension': {}
        }

        for root, dirs, files in os.walk(directory):
            for filename in files:
                file_path = os.path.join(root, filename)
                ext = Path(filename).suffix.lower() or 'no_extension'

                stats['total_files'] += 1
                stats['total_size'] += os.path.getsize(file_path)

                if ext not in stats['by_extension']:
                    stats['by_extension'][ext] = {'count': 0, 'size': 0}

                stats['by_extension'][ext]['count'] += 1
                stats['by_extension'][ext]['size'] += os.path.getsize(file_path)

        return stats
```

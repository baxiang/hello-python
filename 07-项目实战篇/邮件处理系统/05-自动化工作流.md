```python
# workflow.py
from mail_receiver import MailReceiver
from mail_sender import MailSender
from mail_template import MailTemplate
from attachment_handler import AttachmentHandler
import logging
import json
from datetime import datetime
from typing import List, Dict, Any

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class MailRule:
    """邮件处理规则"""

    def __init__(self, name: str, condition: dict, action: dict):
        self.name = name
        self.condition = condition  # 匹配条件
        self.action = action        # 执行动作

    def match(self, email: dict) -> bool:
        """检查邮件是否匹配规则"""
        # 匹配发件人
        if 'from' in self.condition:
            if self.condition['from'] not in email.get('from', ''):
                return False

        # 匹配主题
        if 'subject' in self.condition:
            if self.condition['subject'] not in email.get('subject', ''):
                return False

        # 包含关键词
        if 'contains' in self.condition:
            content = email.get('subject', '') + email.get('body', '')
            if self.condition['contains'] not in content:
                return False

        return True

    def execute(self, email: dict, context: dict):
        """执行规则动作"""
        action_type = self.action.get('type')

        if action_type == 'reply':
            self._send_reply(email, context)
        elif action_type == 'forward':
            self._forward(email, context)
        elif action_type == 'download':
            self._download_attachments(email, context)
        elif action_type == 'move':
            self._move_folder(email, context)
        elif action_type == 'custom':
            self._custom_action(email, context)

    def _send_reply(self, email: dict, context: dict):
        """自动回复"""
        sender = context['sender']
        template = context.get('template')

        if template:
            template_engine = context['template_engine']
            body = template_engine.render_string(template, email=email)
        else:
            body = f"收到您的邮件：{email.get('subject')}"

        sender.send_email(
            to_addrs=[email['from']],
            subject=f"Re: {email.get('subject', '')}",
            body=body
        )
        logger.info(f"自动回复已发送：{email['from']}")

    def _forward(self, email: dict, context: dict):
        """转发邮件"""
        forward_to = self.action.get('forward_to')
        if forward_to:
            sender = context['sender']
            sender.send_email(
                to_addrs=[forward_to],
                subject=f"Fwd: {email.get('subject', '')}",
                body=email.get('body', '')
            )
            logger.info(f"邮件已转发到：{forward_to}")

    def _download_attachments(self, email: dict, context: dict):
        """下载附件"""
        handler = context['attachment_handler']
        # 这里需要从邮件服务器下载
        logger.info(f"准备下载附件：{email.get('subject')}")

    def _move_folder(self, email: dict, context: dict):
        """移动邮件到文件夹"""
        receiver = context['receiver']
        folder = self.action.get('folder')
        if folder and 'id' in email:
            receiver.move_email(email['id'], folder)
            logger.info(f"邮件已移动到：{folder}")

    def _custom_action(self, email: dict, context: dict):
```

### 2.3 工作流引擎

```python
class MailWorkflow:
    """邮件自动化工作流"""

    def __init__(self, config: dict):
        self.config = config
        self.rules: List[MailRule] = []
        self.receiver = None
        self.sender = None
        self.template_engine = None
        self.attachment_handler = None
        self.processed_ids = set()

    def initialize(self):
        """初始化组件"""
        # 初始化接收器
        self.receiver = MailReceiver(
            imap_server=self.config['imap_server'],
            imap_port=self.config['imap_port'],
            username=self.config['username'],
            password=self.config['password']
        )
        self.receiver.connect()

        # 初始化发送器
        self.sender = MailSender(
            smtp_server=self.config['smtp_server'],
            smtp_port=self.config['smtp_port'],
            username=self.config['username'],
            password=self.config['password']
        )

        # 初始化模板引擎
        self.template_engine = MailTemplate(self.config.get('template_dir', './templates'))

        # 初始化附件处理器
        self.attachment_handler = AttachmentHandler(self.config.get('attachment_dir', './attachments'))

        # 加载规则
        self._load_rules()

    def _load_rules(self):
        """加载处理规则"""
        rules_config = self.config.get('rules', [])

        for rule_config in rules_config:
            rule = MailRule(
                name=rule_config['name'],
                condition=rule_config['condition'],
                action=rule_config['action']
            )
            self.rules.append(rule)
            logger.info(f"加载规则：{rule.name}")

    def process_new_emails(self):
        """处理新邮件"""
        logger.info("开始检查新邮件...")

        # 获取未读邮件
        emails = self.receiver.get_unread_emails()
        logger.info(f"发现 {len(emails)} 封未读邮件")

        for email in emails:
            # 跳过已处理的
            if email['id'] in self.processed_ids:
                continue

            # 应用规则
            for rule in self.rules:
                if rule.match(email):
                    logger.info(f"邮件匹配规则：{rule.name}")

                    context = {
                        'sender': self.sender,
                        'receiver': self.receiver,
                        'template_engine': self.template_engine,
                        'attachment_handler': self.attachment_handler,
                        'template': rule.action.get('template')
                    }

                    try:
                        rule.execute(email, context)
                        self.processed_ids.add(email['id'])
                        self.receiver.mark_as_read(email['id'])
                    except Exception as e:
                        logger.error(f"执行规则失败：{e}")

        logger.info("邮件处理完成")

    def run(self):
        """运行工作流"""
        logger.info("邮件自动化工作流启动")

        try:
            while True:
                self.process_new_emails()
                import time
                time.sleep(60)  # 每分钟检查一次
        except KeyboardInterrupt:
            logger.info("工作流停止")
            self.stop()

    def stop(self):
        """停止工作流"""
        if self.receiver:
            self.receiver.disconnect()
```

```python
import json

def main():
```

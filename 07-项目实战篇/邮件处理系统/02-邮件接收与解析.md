### 2.2 连接和登录

```python
# mail_receiver.py
import imaplib
import email
from email import policy
from email.parser import BytesParser
from email.utils import parseaddr, parsedate_to_datetime
import os

class MailReceiver:
    """邮件接收器"""

    def __init__(self, imap_server: str, imap_port: int, username: str, password: str):
        self.imap_server = imap_server
        self.imap_port = imap_port
        self.username = username
        self.password = password
        self.connection = None

    def connect(self):
        """连接邮件服务器"""
        try:
            self.connection = imaplib.IMAP4_SSL(self.imap_server, self.imap_port)
            self.connection.login(self.username, self.password)
            print(f"连接成功：{self.username}")
            return True
        except Exception as e:
            print(f"连接失败：{e}")
            return False

    def disconnect(self):
        """断开连接"""
        if self.connection:
            try:
                self.connection.logout()
                print("已断开连接")
            except:
                pass

    def select_folder(self, folder: str = "INBOX"):
        """选择邮件文件夹"""
        status, _ = self.connection.select(folder)
        return status == 'OK'

    def get_mail_count(self, folder: str = "INBOX") -> int:
        """获取邮件数量"""
        self.select_folder(folder)
        status, data = self.connection.select(folder)
        if status == 'OK':
            return int(data[0])
        return 0
```

### 2.3 接收和解析邮件

```python
    def fetch_email(self, email_id: int) -> dict:
        """获取单封邮件"""
        status, data = self.connection.fetch(str(email_id), '(RFC822)')

        if status != 'OK':
            return None

        # 解析邮件
        msg = email.message_from_bytes(data[0][1], policy=policy.default)

        # 提取邮件信息
        mail_info = {
            'id': email_id,
            'subject': msg['subject'],
            'from': parseaddr(msg['from'])[1],
            'to': parseaddr(msg['to'])[1],
            'date': msg['date'],
            'date_parsed': parsedate_to_datetime(msg['date']) if msg['date'] else None,
        }

        # 提取邮件正文
        body = ""
        html_body = ""

        if msg.is_multipart():
            # 多部分邮件
            for part in msg.walk():
                content_type = part.get_content_type()
                content_disposition = str(part.get('Content-Disposition'))

                # 纯文本正文
                if content_type == 'text/plain' and 'attachment' not in content_disposition:
                    try:
                        body = part.get_content()
                    except:
                        body = part.get_payload(decode=True).decode(part.get_content_charset() or 'utf-8')

                # HTML 正文
                elif content_type == 'text/html' and 'attachment' not in content_disposition:
                    try:
                        html_body = part.get_content()
                    except:
                        html_body = part.get_payload(decode=True).decode(part.get_content_charset() or 'utf-8')
        else:
            # 单部分邮件
            content_type = msg.get_content_type()
            try:
                content = msg.get_content()
            except:
                content = msg.get_payload(decode=True).decode(msg.get_content_charset() or 'utf-8')

            if content_type == 'text/plain':
                body = content
            elif content_type == 'text/html':
                html_body = content

        mail_info['body'] = body
        mail_info['html_body'] = html_body

        return mail_info

    def fetch_recent_emails(self, count: int = 10) -> list:
        """获取最近的邮件"""
        self.select_folder("INBOX")

        # 获取最新邮件 ID
        status, data = self.connection.search(None, "ALL")
        if status != 'OK':
            return []

        email_ids = data[0].split()
        recent_ids = email_ids[-count:] if len(email_ids) > count else email_ids

        emails = []
        for email_id in reversed(recent_ids):
            mail = self.fetch_email(int(email_id))
            if mail:
                emails.append(mail)

        return emails
```

### 2.4 邮件搜索

```python
    def search_emails(self, criteria: str) -> list:
        """搜索邮件

        Args:
            criteria: 搜索条件
                - ALL: 所有邮件
                - UNSEEN: 未读邮件
                - SEEN: 已读邮件
                - FROM sender@example.com: 发件人
                - SUBJECT keyword: 主题
                - SINCE date: 日期之后
                - BEFORE date: 日期之前
        """
        self.select_folder("INBOX")
        status, data = self.connection.search(None, criteria)

        if status != 'OK':
            return []

        email_ids = data[0].split()
        emails = []

        for email_id in email_ids:
            mail = self.fetch_email(int(email_id))
            if mail:
                emails.append(mail)

        return emails

    def get_unread_emails(self) -> list:
        """获取未读邮件"""
        return self.search_emails("UNSEEN")

    def get_emails_from(self, sender: str) -> list:
        """获取指定发件人的邮件"""
        return self.search_emails(f'FROM "{sender}"')

    def get_emails_by_subject(self, keyword: str) -> list:
        """获取指定主题的邮件"""
        return self.search_emails(f'SUBJECT "{keyword}"')
```

## 第三部分 - 扩展功能

### 3.1 邮件操作

```python
    def mark_as_read(self, email_id: int):
        """标记为已读"""
        self.connection.store(str(email_id), '+FLAGS', '\\Seen')

    def mark_as_unread(self, email_id: int):
        """标记为未读"""
        self.connection.store(str(email_id), '-FLAGS', '\\Seen')

    def delete_email(self, email_id: int):
        """删除邮件"""
        self.connection.store(str(email_id), '+FLAGS', '\\Deleted')
        self.connection.expunge()

    def move_email(self, email_id: int, target_folder: str):
        """移动邮件到其他文件夹"""
        self.connection.copy(str(email_id), target_folder)
        self.connection.store(str(email_id), '+FLAGS', '\\Deleted')
        self.connection.expunge()
```

### 3.2 提取邮件头

```python
    def get_email_headers(self, email_id: int) -> dict:
        """获取邮件头部信息"""
        status, data = self.connection.fetch(str(email_id), '(RFC822.HEADER)')
        msg = email.message_from_bytes(data[0][1])

        headers = {}
        for header in ['subject', 'from', 'to', 'cc', 'bcc', 'date',
                       'message-id', 'reply-to', 'return-path']:
            headers[header] = msg.get(header)

        return headers
```

### 3.3 下载附件

```python
    def download_attachments(self, email_id: int, save_dir: str = "./attachments"):
        """下载邮件附件"""
        os.makedirs(save_dir, exist_ok=True)

        status, data = self.connection.fetch(str(email_id), '(RFC822)')
        msg = email.message_from_bytes(data[0][1])

        attachments = []

        if msg.is_multipart():
            for part in msg.walk():
                content_disposition = part.get('Content-Disposition')

                if content_disposition and 'attachment' in content_disposition:
                    filename = part.get_filename()
                    if filename:
                        # 解码文件名
                        filename = email.header.decode_header(filename)[0]
                        if isinstance(filename, bytes):
                            filename = filename.decode('utf-8')

                        # 保存附件
                        file_path = os.path.join(save_dir, filename)
                        with open(file_path, 'wb') as f:
                            f.write(part.get_payload(decode=True))

                        attachments.append(file_path)
                        print(f"已下载：{filename}")

        return attachments
```

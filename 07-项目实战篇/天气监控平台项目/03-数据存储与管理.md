### 1.1 数据表设计

```sql
-- 创建数据库
CREATE DATABASE weather_db CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

USE weather_db;

-- 城市表
CREATE TABLE cities (
    id INT PRIMARY KEY AUTO_INCREMENT,
    city_name VARCHAR(50) NOT NULL UNIQUE,
    city_code VARCHAR(20),
    province VARCHAR(50),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 天气数据表
CREATE TABLE weather_records (
    id INT PRIMARY KEY AUTO_INCREMENT,
    city_id INT NOT NULL,
    record_date DATE NOT NULL,
    high_temp INT,
    low_temp INT,
    weather_condition VARCHAR(50),
    wind_direction VARCHAR(50),
    wind_power VARCHAR(20),
    humidity INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (city_id) REFERENCES cities(id),
    UNIQUE KEY unique_city_date (city_id, record_date)
);

-- 创建索引
CREATE INDEX idx_record_date ON weather_records(record_date);
CREATE INDEX idx_city_id ON weather_records(city_id);
```

## 第二部分 - SQLAlchemy ORM

### 2.1 模型定义

```python
# models.py
from sqlalchemy import Column, Integer, String, Date, DateTime, ForeignKey, create_engine
from sqlalchemy.orm import relationship, declarative_base, sessionmaker
from datetime import datetime

Base = declarative_base()

class City(Base):
    __tablename__ = "cities"
    
    id = Column(Integer, primary_key=True, autoincrement=True)
    city_name = Column(String(50), nullable=False, unique=True)
    city_code = Column(String(20))
    province = Column(String(50))
    created_at = Column(DateTime, default=datetime.now)
    
    # 关联天气记录
    weather_records = relationship("WeatherRecord", back_populates="city")
    
    def __repr__(self):
        return f"<City(id={self.id}, name='{self.city_name}')>"

class WeatherRecord(Base):
    __tablename__ = "weather_records"
    
    id = Column(Integer, primary_key=True, autoincrement=True)
    city_id = Column(Integer, ForeignKey("cities.id"), nullable=False)
    record_date = Column(Date, nullable=False)
    high_temp = Column(Integer)
    low_temp = Column(Integer)
    weather_condition = Column(String(50))
    wind_direction = Column(String(50))
    wind_power = Column(String(20))
    humidity = Column(Integer)
    created_at = Column(DateTime, default=datetime.now)
    updated_at = Column(DateTime, default=datetime.now, onupdate=datetime.now)
    
    # 关联城市
    city = relationship("City", back_populates="weather_records")
    
    def __repr__(self):
        return f"<WeatherRecord(city_id={self.city_id}, date={self.record_date}, temp={self.low_temp}~{self.high_temp})>"
```

### 2.2 数据库连接

```python
# database.py
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker
from models import Base

DATABASE_URL = "mysql+pymysql://user:password@localhost:3306/weather_db"

# 创建引擎
engine = create_engine(
    DATABASE_URL,
    echo=False,  # 打印 SQL 语句
    pool_size=10,
    pool_recycle=3600
)

# 创建会话
SessionLocal = sessionmaker(autocommit=False, autoflush=False, bind=engine)

def init_db():
    """初始化数据库，创建所有表"""
    Base.metadata.create_all(bind=engine)

def get_session():
    """获取数据库会话"""
    session = SessionLocal()
    try:
        return session
    except Exception:
        session.close()
        raise

if __name__ == "__main__":
    init_db()
    print("数据库初始化完成")
```

## 第三部分 - 数据仓库类

### 3.1 CRUD 操作

```python
# weather_repository.py
from sqlalchemy.orm import Session
from models import City, WeatherRecord
from datetime import date
from typing import List, Optional

class WeatherRepository:
    def __init__(self, session: Session):
        self.session = session
    
    # === 城市操作 ===
    
    def add_city(self, city_name: str, city_code: str, province: str = None) -> City:
        """添加城市"""
        city = City(city_name=city_name, city_code=city_code, province=province)
        self.session.add(city)
        self.session.commit()
        self.session.refresh(city)
        return city
    
    def get_city_by_name(self, city_name: str) -> Optional[City]:
        """根据名称获取城市"""
        return self.session.query(City).filter(City.city_name == city_name).first()
    
    def get_or_create_city(self, city_name: str, city_code: str) -> City:
        """获取或创建城市"""
        city = self.get_city_by_name(city_name)
        if not city:
            city = self.add_city(city_name, city_code)
        return city
    
    # === 天气记录操作 ===
    
    def add_weather_record(self, city_id: int, record_date: date, 
                          high_temp: int, low_temp: int,
                          weather_condition: str = None,
                          wind_direction: str = None,
                          wind_power: str = None,
                          humidity: int = None) -> WeatherRecord:
        """添加天气记录"""
        record = WeatherRecord(
            city_id=city_id,
            record_date=record_date,
            high_temp=high_temp,
            low_temp=low_temp,
            weather_condition=weather_condition,
            wind_direction=wind_direction,
            wind_power=wind_power,
            humidity=humidity
        )
        self.session.add(record)
        self.session.commit()
        self.session.refresh(record)
        return record
    
    def upsert_weather_record(self, city_id: int, record_date: date, **kwargs) -> WeatherRecord:
        """更新或插入天气记录（Upsert）"""
        record = self.session.query(WeatherRecord).filter(
            WeatherRecord.city_id == city_id,
            WeatherRecord.record_date == record_date
        ).first()
        
        if record:
            # 更新
            for key, value in kwargs.items():
                if hasattr(record, key) and value is not None:
                    setattr(record, key, value)
            self.session.commit()
            self.session.refresh(record)
        else:
            # 插入
            record = self.add_weather_record(city_id, record_date, **kwargs)
        
        return record
    
    def get_records_by_city(self, city_id: int, start_date: date = None, 
                           end_date: date = None) -> List[WeatherRecord]:
        """获取城市的天气记录"""
        query = self.session.query(WeatherRecord).filter(WeatherRecord.city_id == city_id)
        
        if start_date:
            query = query.filter(WeatherRecord.record_date >= start_date)
        if end_date:
            query = query.filter(WeatherRecord.record_date <= end_date)
        
        return query.order_by(WeatherRecord.record_date.desc()).all()
    
    def get_latest_record(self, city_id: int) -> Optional[WeatherRecord]:
        """获取最新的天气记录"""
        return self.session.query(WeatherRecord).filter(
            WeatherRecord.city_id == city_id
        ).order_by(WeatherRecord.record_date.desc()).first()
    
    def delete_old_records(self, city_id: int, before_date: date) -> int:
        """删除旧记录"""
        count = self.session.query(WeatherRecord).filter(
            WeatherRecord.city_id == city_id,
            WeatherRecord.record_date < before_date
        ).delete()
        self.session.commit()
        return count
```

## 第四部分 - 数据导入

### 4.1 从 CSV 导入

```python
# data_importer.py
import pandas as pd
from datetime import datetime
from database import get_session
from weather_repository import WeatherRepository

def import_from_csv(filename: str):
    """从 CSV 文件导入数据"""
    session = get_session()
    repo = WeatherRepository(session)
    
    df = pd.read_csv(filename)
    
    imported = 0
    skipped = 0
    
    for _, row in df.iterrows():
        try:
            # 获取或创建城市
            city = repo.get_or_create_city(row['city'], city_code="")
            
            # 解析日期
            record_date = datetime.strptime(row['date'], '%Y-%m-%d').date()
            
            #  Upsert 天气记录
            repo.upsert_weather_record(
                city_id=city.id,
                record_date=record_date,
                high_temp=int(row['high_temp']) if pd.notna(row.get('high_temp')) else None,
                low_temp=int(row['low_temp']) if pd.notna(row.get('low_temp')) else None,
                weather_condition=row.get('weather'),
                wind_direction=row.get('wind_dir'),
                wind_power=row.get('wind_power'),
                humidity=int(row['humidity']) if pd.notna(row.get('humidity')) else None
            )
            imported += 1
        except Exception as e:
            print(f"导入失败：{row} - {e}")
            skipped += 1
    
    session.close()
    print(f"导入完成：成功 {imported} 条，跳过 {skipped} 条")
    return imported, skipped

if __name__ == "__main__":
    import_from_csv("weather_北京_20240101.csv")
```

### 4.2 从爬虫导入

```python
def import_from_spider(city: str, year: int = 2024):
    """直接从爬虫导入数据"""
    from weather_spider import crawl_weather_history, clean_weather_data
    
    session = get_session()
    repo = WeatherRepository(session)
    
    # 爬取数据
    raw_data = crawl_weather_history(city, year)
    cleaned_data = clean_weather_data(raw_data)
    
    # 获取城市
    city_obj = repo.get_or_create_city(city, city_code="")
    
    # 导入数据
    for item in cleaned_data:
        try:
            record_date = datetime.strptime(item['date'], '%Y-%m-%d').date()
            repo.upsert_weather_record(
                city_id=city_obj.id,
                record_date=record_date,
                high_temp=item['high_temp'],
                low_temp=item['low_temp'],
                weather_condition=item['weather'],
                wind_direction=item['wind_dir'],
                wind_power=item['wind_power']
            )
        except Exception as e:
            print(f"导入失败：{item} - {e}")
    
    session.close()
    print(f"导入完成：{len(cleaned_data)} 条数据")

if __name__ == "__main__":
    import_from_spider("北京", 2024)
```

## 第五部分 - 数据查询

### 5.1 统计查询

```python
# weather_stats.py
from sqlalchemy import func
from database import get_session
from weather_repository import WeatherRepository
from models import City, WeatherRecord

def get_city_stats(city_name: str):
    """获取城市统计信息"""
    session = get_session()
    repo = WeatherRepository(session)
    
    city = repo.get_city_by_name(city_name)
    if not city:
        print(f"城市不存在：{city_name}")
        return
    
    stats = session.query(
        func.count(WeatherRecord.id).label('total_days'),
        func.max(WeatherRecord.high_temp).label('max_high'),
        func.min(WeatherRecord.low_temp).label('min_low'),
        func.avg(WeatherRecord.high_temp).label('avg_high'),
        func.avg(WeatherRecord.low_temp).label('avg_low'),
    ).filter(WeatherRecord.city_id == city.id).first()
    
    print(f"\n=== {city_name} 天气统计 ===")
    print(f"总记录数：{stats.total_days} 天")
    print(f"最高温度：{stats.max_high}°C")
    print(f"最低温度：{stats.min_low}°C")
    print(f"平均高温：{stats.avg_high:.1f}°C")
    print(f"平均低温：{stats.avg_low:.1f}°C")
    
    session.close()
    return stats

def get_weather_distribution(city_name: str):
    """获取天气状况分布"""
    session = get_session()
    repo = WeatherRepository(session)
    
    city = repo.get_city_by_name(city_name)
    if not city:
        return
    
    distribution = session.query(
        WeatherRecord.weather_condition,
        func.count(WeatherRecord.id).label('count')
    ).filter(
        WeatherRecord.city_id == city.id
    ).group_by(WeatherRecord.weather_condition).order_by(func.count(WeatherRecord.id).desc()).all()
    
    print(f"\n=== {city_name} 天气分布 ===")
    for weather, count in distribution[:10]:
        bar = "█" * (count // 5)
        print(f"{weather:<10} {count:>3}天 {bar}")
    
    session.close()

if __name__ == "__main__":
    get_city_stats("北京")
    get_weather_distribution("北京")
```

### 2.2 基础爬虫

```python
# weather_spider.py
import requests
from bs4 import BeautifulSoup
import pandas as pd

def get_city_code(city: str) -> str:
    """获取城市代码"""
    city_codes = {
        "北京": "54511",
        "上海": "58362",
        "广州": "59287",
        "深圳": "59493",
        "杭州": "58457",
    }
    return city_codes.get(city, "54511")

def crawl_weather_history(city: str, year: int = 2024) -> list:
    """爬取城市历史天气"""
    city_code = get_city_code(city)
    url = f"http://tianqi.2345.com/wea_history/{city_code}.htm"
    
    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36"
    }
    
    response = requests.get(url, headers=headers)
    response.encoding = "utf-8"
    
    soup = BeautifulSoup(response.text, "html.parser")
    
    # 解析数据（根据实际 HTML 结构调整）
    weather_data = []
    table = soup.find("table", {"class": "t4"})
    if table:
        rows = table.find_all("tr")[1:]  # 跳过表头
        for row in rows:
            cols = row.find_all("td")
            if len(cols) >= 7:
                date_str = cols[0].get_text().strip()
                # 检查年份是否匹配
                if str(year) in date_str:
                    weather_data.append({
                        "city": city,
                        "date": date_str,
                        "high_temp": cols[1].get_text().strip().replace("℃", ""),
                        "low_temp": cols[2].get_text().strip().replace("℃", ""),
                        "weather": cols[3].get_text().strip(),
                        "wind_dir": cols[4].get_text().strip(),
                        "wind_power": cols[5].get_text().strip(),
                    })
    
    return weather_data

if __name__ == "__main__":
    data = crawl_weather_history("北京", 2024)
    print(f"采集到 {len(data)} 条数据")
    for item in data[:5]:  # 显示前 5 条
        print(f"{item['date']}: {item['low_temp']}~{item['high_temp']}°C {item['weather']}")
```

### 2.3 数据清洗

```python
def clean_weather_data(data: list) -> list:
    """清洗天气数据"""
    cleaned = []
    for item in data:
        try:
            # 清洗温度数据
            high = item["high_temp"]
            low = item["low_temp"]
            
            # 处理特殊字符
            high = high.replace("℃", "").replace(" ", "")
            low = low.replace("℃", "").replace(" ", "")
            
            # 处理范围值（如 "15~25"）
            if "~" in high:
                high = high.split("~")[1]
            if "~" in low:
                low = low.split("~")[0]
            
            # 转换为数字
            high_temp = int(high) if high else None
            low_temp = int(low) if low else None
            
            cleaned.append({
                "city": item["city"],
                "date": item["date"],
                "high_temp": high_temp,
                "low_temp": low_temp,
                "weather": item["weather"],
                "wind_dir": item["wind_dir"],
                "wind_power": item["wind_power"],
            })
        except (ValueError, IndexError) as e:
            print(f"清洗失败：{item} - {e}")
    
    return cleaned
```

### 2.4 保存数据

```python
def save_to_csv(data: list, city: str):
    """保存数据到 CSV"""
    if not data:
        print("没有数据可保存")
        return
    
    df = pd.DataFrame(data)
    filename = f"weather_{city}_{pd.Timestamp.now().strftime('%Y%m%d')}.csv"
    df.to_csv(filename, index=False, encoding="utf-8-sig")
    print(f"数据已保存到：{filename}")
    return filename

def save_to_excel(data: list, city: str):
    """保存数据到 Excel"""
    if not data:
        return
    
    df = pd.DataFrame(data)
    filename = f"weather_{city}_{pd.Timestamp.now().strftime('%Y%m%d')}.xlsx"
    df.to_excel(filename, index=False)
    print(f"数据已保存到：{filename}")
    return filename

if __name__ == "__main__":
    # 采集并保存
    raw_data = crawl_weather_history("北京", 2024)
    print(f"原始数据：{len(raw_data)} 条")
    
    cleaned_data = clean_weather_data(raw_data)
    print(f"清洗后：{len(cleaned_data)} 条")
    
    save_to_csv(cleaned_data, "北京")
```

```python
def batch_crawl(cities: list, year: int = 2024) -> dict:
    """批量采集多个城市"""
    all_data = {}
    for city in cities:
        print(f"正在采集 {city}...")
        try:
            data = crawl_weather_history(city, year)
            cleaned = clean_weather_data(data)
            all_data[city] = cleaned
            save_to_csv(cleaned, city)
            print(f"  ✓ 采集到 {len(cleaned)} 条数据")
        except Exception as e:
            print(f"  ✗ {city} 采集失败：{e}")
            all_data[city] = []
    return all_data

if __name__ == "__main__":
    cities = ["北京", "上海", "广州", "深圳", "杭州"]
    all_data = batch_crawl(cities, 2024)
    
    # 汇总统计
    total = sum(len(d) for d in all_data.values())
    print(f"\n总计采集：{total} 条数据")
```

### 3.2 添加延时和重试

```python
import time
from functools import wraps

def retry(max_attempts: int = 3, delay: float = 1.0):
    """重试装饰器"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            for i in range(max_attempts):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if i == max_attempts - 1:
                        raise
                    print(f"重试 {i+1}/{max_attempts}: {e}")
                    time.sleep(delay * (i + 1))
        return wrapper
    return decorator

@retry(max_attempts=3, delay=2.0)
```

```python
def analyze_weather(data: list):
    """分析天气数据"""
    df = pd.DataFrame(data)
    
    print(f"\n=== {df['city'].iloc[0]} 天气分析 ===")
    print(f"数据量：{len(df)} 条")
    print(f"最高温度：{df['high_temp'].max()}°C")
    print(f"最低温度：{df['low_temp'].min()}°C")
    print(f"平均温度：{(df['high_temp'].mean() + df['low_temp'].mean()) / 2:.1f}°C")
    
    # 天气状况统计
    print("\n天气状况分布:")
    weather_counts = df['weather'].value_counts()
    for weather, count in weather_counts.head(5).items():
        print(f"  {weather}: {count} 天")
```

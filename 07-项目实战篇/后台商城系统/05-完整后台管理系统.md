### 1.1 Redis 缓存

```python
# app/services/cache.py
import redis
import json
from typing import Optional, Any
from functools import wraps
import hashlib

from app.config import settings


class CacheService:
    """缓存服务"""

    def __init__(self):
        self.client = redis.Redis.from_url(
            settings.REDIS_URL,
            decode_responses=True
        )

    def get(self, key: str) -> Optional[Any]:
        """获取缓存"""
        value = self.client.get(key)
        if value:
            try:
                return json.loads(value)
            except:
                return value
        return None

    def set(self, key: str, value: Any, ttl: int = 3600):
        """设置缓存"""
        if isinstance(value, (dict, list)):
            value = json.dumps(value, ensure_ascii=False)
        self.client.setex(key, ttl, value)

    def delete(self, key: str):
        """删除缓存"""
        self.client.delete(key)

    def exists(self, key: str) -> bool:
        """检查缓存是否存在"""
        return self.client.exists(key) > 0

    def clear_pattern(self, pattern: str):
        """清除匹配的所有缓存"""
        keys = self.client.keys(pattern)
        if keys:
            self.client.delete(*keys)

    def get_hash(self, key: str, field: str) -> Optional[Any]:
        """获取哈希缓存"""
        value = self.client.hget(key, field)
        if value:
            try:
                return json.loads(value)
            except:
                return value
        return None

    def set_hash(self, key: str, field: str, value: Any, ttl: int = 3600):
        """设置哈希缓存"""
        if isinstance(value, (dict, list)):
            value = json.dumps(value, ensure_ascii=False)
        self.client.hset(key, field, value)
        if ttl:
            self.client.expire(key, ttl)


# 全局实例
cache_service = CacheService()


# 缓存装饰器
def cached(key_prefix: str, ttl: int = 3600):
    """缓存装饰器"""

    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            # 生成缓存 key
            key_parts = [key_prefix]
            for arg in args:
                if isinstance(arg, (str, int, float)):
                    key_parts.append(str(arg))
            for k, v in sorted(kwargs.items()):
                if isinstance(v, (str, int, float)):
                    key_parts.append(f"{k}:{v}")

            cache_key = ":".join(key_parts)

            # 尝试从缓存获取
            cached_value = cache_service.get(cache_key)
            if cached_value is not None:
                return cached_value

            # 执行函数
            result = await func(*args, **kwargs)

            # 存入缓存
            cache_service.set(cache_key, result, ttl)

            return result

        return wrapper

    return decorator
```

### 1.2 会话管理

```python
# app/services/session.py
from typing import Optional
import json
from datetime import timedelta

from app.services.cache import cache_service


class SessionService:
    """会话服务"""

    SESSION_PREFIX = "session:"
    SESSION_TTL = 86400  # 24小时

    @classmethod
    def create_session(cls, user_id: int, user_data: dict) -> str:
        """创建会话"""
        import uuid
        session_id = str(uuid.uuid4())
        session_key = f"{cls.SESSION_PREFIX}{session_id}"

        session_data = {
            "user_id": user_id,
            "username": user_data.get("username"),
            "nickname": user_data.get("nickname"),
            "role": user_data.get("role")
        }

        cache_service.set(session_key, session_data, cls.SESSION_TTL)
        return session_id

    @classmethod
    def get_session(cls, session_id: str) -> Optional[dict]:
        """获取会话"""
        session_key = f"{cls.SESSION_PREFIX}{session_id}"
        return cache_service.get(session_key)

    @classmethod
    def refresh_session(cls, session_id: str) -> bool:
        """刷新会话"""
        session_key = f"{cls.SESSION_PREFIX}{session_id}"
        if cache_service.exists(session_key):
            cache_service.client.expire(session_key, cls.SESSION_TTL)
            return True
        return False

    @classmethod
    def destroy_session(cls, session_id: str):
        """销毁会话"""
        session_key = f"{cls.SESSION_PREFIX}{session_id}"
        cache_service.delete(session_key)
```

```python
# app/tasks/order.py
from app.celery_app import celery_app
from app.database import SessionLocal
from datetime import datetime, timedelta


@celery_app.task(bind=True, max_retries=3)
def cancel_expired_orders(self):
    """取消超时未支付订单"""
    db = SessionLocal()
    try:
        from app.models.order import Order, OrderStatus

        # 查找超时订单（30分钟未支付）
        timeout = datetime.now() - timedelta(minutes=30)

        orders = db.query(Order).filter(
            Order.status == OrderStatus.PENDING,
            Order.created_at < timeout
        ).all()

        cancelled_count = 0
        for order in orders:
            try:
                # 恢复库存
                for item in order.items:
                    product = item.product
                    product.stock += item.quantity

                order.status = OrderStatus.CANCELLED
                order.cancel_reason = "订单超时未支付"
                order.cancel_time = datetime.now()

                db.commit()
                cancelled_count += 1

            except Exception as e:
                db.rollback()
                print(f"取消订单 {order.order_no} 失败: {e}")

        return {"cancelled_count": cancelled_count}

    finally:
        db.close()


@celery_app.task
def daily_sales_statistics():
    """每日销售统计"""
    db = SessionLocal()
    try:
        from app.models.order import Order, OrderStatus
        from sqlalchemy import func

        today = datetime.now().date()
        tomorrow = today + timedelta(days=1)

        # 统计今日订单
        stats = db.query(
            func.count(Order.id).label("order_count"),
            func.sum(Order.pay_amount).label("total_amount")
        ).filter(
            Order.status == OrderStatus.COMPLETED,
            Order.complete_time >= today,
            Order.complete_time < tomorrow
        ).first()

        result = {
            "date": str(today),
            "order_count": stats.order_count or 0,
            "total_amount": float(stats.total_amount or 0)
        }

        # 可以发送到消息队列或存储
        print(f"今日销售统计: {result}")
        return result

    finally:
        db.close()
```

```python
# app/main.py
from fastapi import FastAPI, Request, status
from fastapi.responses import JSONResponse
from fastapi.middleware.cors import CORSMiddleware

from fastapi.exceptions import RequestValidationError
from contextlib import asynccontextmanager

from app.config import settings
from app.database import init_db
from app.routers import auth, user, product, category, brand, order, payment


@asynccontextmanager
async def lifespan(app: FastAPI):
    """应用生命周期"""
    # 启动
    print("应用启动中...")
    init_db()
    print("数据库初始化完成")
    yield

    print("应用关闭")

    docs_url="/docs",
    redoc_url="/redoc",
    openapi_url="/openapi.json",
    lifespan=lifespan
)

app.add_middleware(

    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],

# 静态文件

@app.exception_handler(RequestValidationError)
async def validation_exception_handler(request: Request, exc: RequestValidationError):
    return JSONResponse(
        status_code=status.HTTP_422_UNPROCESSABLE_ENTITY,
        content={
            "code": 422,
            "message": "参数验证失败",
            "errors": exc.errors()

@app.exception_handler(Exception)
async def global_exception_handler(request: Request, exc: Exception):

        status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
        content={
            "code": 500,
            "message": str(exc),

app.include_router(auth.router, prefix="/api")
app.include_router(user.router, prefix="/api")
app.include_router(product.router, prefix="/api")
app.include_router(category.router, prefix="/api")
```

### 4.3 Nginx 配置

```nginx
# nginx.conf
events {
    worker_connections 1024;
}

http {
    include /etc/nginx/mime.types;
    default_type application/octet-stream;

    upstream app {
        server app:8000;
    }

    server {
        listen 80;
        server_name localhost;

        client_max_body_size 20M;

        # 静态文件
        location /uploads/ {
            alias /app/uploads/;
            expires 7d;
        }

        # API 代理
        location /api/ {
            proxy_pass http://app/;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }

        # WebSocket 支持
        location /ws/ {
            proxy_pass http://app;
            proxy_http_version 1.1;
            proxy_set_header Upgrade $http_upgrade;
            proxy_set_header Connection "upgrade";
        }

        # 健康检查
        location /health {
            proxy_pass http://app/health;
        }
    }
}
```

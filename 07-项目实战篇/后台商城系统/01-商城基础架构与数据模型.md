### 3.2 数据库连接

```python
# app/database.py
from sqlalchemy import create_engine
from sqlalchemy.ext.declarative import declarative_base
from sqlalchemy.orm import sessionmaker, Session
from contextlib import contextmanager
from typing import Generator

from app.config import settings

# 创建引擎
engine = create_engine(
    settings.DATABASE_URL,
    pool_pre_ping=True,          # 连接前测试
    pool_recycle=3600,           # 连接回收时间
    pool_size=10,                # 连接池大小
    max_overflow=20,             # 最大溢出连接数
    echo=settings.DEBUG,         # SQL 日志
)

# 创建会话工厂
SessionLocal = sessionmaker(
    autocommit=False,
    autoflush=False,
    bind=engine
)

# 声明基类
Base = declarative_base()


def get_db() -> Generator[Session, None, None]:
    """获取数据库会话"""
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()


@contextmanager
def get_db_context() -> Generator[Session, None, None]:
    """上下文管理器方式的数据库会话"""
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()


def init_db():
    """初始化数据库表"""
    from app.models import user, product, order, payment
    Base.metadata.create_all(bind=engine)
```

```python
from sqlalchemy import Column, Integer, String, Boolean, DateTime, Text, Enum as SQLEnum
from sqlalchemy.orm import relationship
from sqlalchemy.sql import func
import enum

from app.database import Base


class UserStatus(str, enum.Enum):
    """用户状态"""
    ACTIVE = "active"
    INACTIVE = "inactive"
    BANNED = "banned"


class User(Base):
    """用户表"""
    __tablename__ = "users"

    id = Column(Integer, primary_key=True, index=True, comment="用户ID")
    username = Column(String(50), unique=True, index=True, nullable=False, comment="用户名")
    email = Column(String(255), unique=True, index=True, nullable=False, comment="邮箱")
    phone = Column(String(20), unique=True, index=True, nullable=True, comment="手机号")
    hashed_password = Column(String(255), nullable=False, comment="加密密码")
    nickname = Column(String(50), nullable=True, comment="昵称")
    avatar = Column(String(500), nullable=True, comment="头像URL")
    status = Column(SQLEnum(UserStatus), default=UserStatus.ACTIVE, comment="用户状态")
    is_superuser = Column(Boolean, default=False, comment="是否超级管理员")
    last_login = Column(DateTime, nullable=True, comment="最后登录时间")
    login_ip = Column(String(50), nullable=True, comment="登录IP")
    created_at = Column(DateTime, server_default=func.now(), comment="创建时间")
    updated_at = Column(DateTime, server_default=func.now(), onupdate=func.now(), comment="更新时间")

    # 关系
    addresses = relationship("UserAddress", back_populates="user", cascade="all, delete-orphan")
    orders = relationship("Order", back_populates="user")
    carts = relationship("Cart", back_populates="user")
    favorites = relationship("Favorite", back_populates="user")

    def __repr__(self):
        return f"<User(id={self.id}, username={self.username})>"


class UserAddress(Base):
    """用户地址表"""
    __tablename__ = "user_addresses"

    id = Column(Integer, primary_key=True, index=True)
    user_id = Column(Integer, ForeignKey("users.id", ondelete="CASCADE"), nullable=False, index=True)
    receiver_name = Column(String(50), nullable=False, comment="收货人姓名")
    receiver_phone = Column(String(20), nullable=False, comment="收货人电话")
    province = Column(String(50), nullable=False, comment="省份")
    city = Column(String(50), nullable=False, comment="城市")
    district = Column(String(50), nullable=False, comment="区县")
    detail_address = Column(String(255), nullable=False, comment="详细地址")
    postal_code = Column(String(10), nullable=True, comment="邮政编码")
    is_default = Column(Boolean, default=False, comment="是否默认地址")
    created_at = Column(DateTime, server_default=func.now())
    updated_at = Column(DateTime, server_default=func.now(), onupdate=func.now())

    # 关系
```

```python
# app/models/product.py
from sqlalchemy import Column, Integer, String, Boolean, DateTime, Text, Numeric, ForeignKey, Enum as SQLEnum
from sqlalchemy.orm import relationship
from sqlalchemy.sql import func
from sqlalchemy.dialects.mysql import LONGTEXT
import enum

from app.database import Base


class ProductStatus(str, enum.Enum):
    """商品状态"""
    DRAFT = "draft"           # 草稿
    ON_SALE = "on_sale"       # 上架
    OFF_SALE = "off_sale"     # 下架
    DELETED = "deleted"       # 已删除


class Category(Base):
    """商品分类表"""
    __tablename__ = "categories"

    id = Column(Integer, primary_key=True, index=True)
    name = Column(String(100), nullable=False, comment="分类名称")
    parent_id = Column(Integer, ForeignKey("categories.id"), nullable=True, comment="父分类ID")
    level = Column(Integer, default=1, comment="层级")
    sort = Column(Integer, default=0, comment="排序")
    icon = Column(String(500), nullable=True, comment="图标")
    is_enabled = Column(Boolean, default=True, comment="是否启用")
    created_at = Column(DateTime, server_default=func.now())
    updated_at = Column(DateTime, server_default=func.now(), onupdate=func.now())

    # 自关联
    parent = relationship("Category", remote_side=[id], backref="children")
    products = relationship("Product", back_populates="category")

    def __repr__(self):
        return f"<Category(id={self.id}, name={self.name})>"


class Brand(Base):
    """品牌表"""
    __tablename__ = "brands"

    id = Column(Integer, primary_key=True, index=True)
    name = Column(String(100), nullable=False, unique=True, comment="品牌名称")
    logo = Column(String(500), nullable=True, comment="品牌Logo")
    description = Column(Text, nullable=True, comment="品牌描述")
    first_letter = Column(String(1), nullable=True, comment="首字母")
    sort = Column(Integer, default=0, comment="排序")
    is_enabled = Column(Boolean, default=True, comment="是否启用")
    created_at = Column(DateTime, server_default=func.now())
    updated_at = Column(DateTime, server_default=func.now(), onupdate=func.now())

    # 关系
    products = relationship("Product", back_populates="brand")

    def __repr__(self):
        return f"<Brand(id={self.id}, name={self.name})>"


class Product(Base):
    """商品表"""
    __tablename__ = "products"

    id = Column(Integer, primary_key=True, index=True)
    name = Column(String(255), nullable=False, comment="商品名称")
    product_code = Column(String(50), unique=True, nullable=False, index=True, comment="商品编码")
    category_id = Column(Integer, ForeignKey("categories.id"), nullable=False, index=True)
    brand_id = Column(Integer, ForeignKey("brands.id"), nullable=True, index=True)

    # 价格相关
    price = Column(Numeric(10, 2), nullable=False, comment="销售价格")
    original_price = Column(Numeric(10, 2), nullable=True, comment="原价")
    cost_price = Column(Numeric(10, 2), nullable=True, comment="成本价")

    # 库存相关
    stock = Column(Integer, default=0, comment="库存数量")
    low_stock_threshold = Column(Integer, default=10, comment="库存预警阈值")
    stock_status = Column(String(20), computed="CASE WHEN stock <= 0 THEN 'out_of_stock' WHEN stock <= low_stock_threshold THEN 'low_stock' ELSE 'in_stock' END", comment="库存状态")

    # 商品属性
    unit = Column(String(20), default="件", comment="单位")
    weight = Column(Numeric(10, 2), nullable=True, comment="重量(kg)")
    volume = Column(Numeric(10, 2), nullable=True, comment="体积(m³)")

    # 图片
    main_image = Column(String(500), nullable=True, comment="主图")
    images = Column(LONGTEXT, nullable=True, comment="商品图片列表(JSON)")

    # 详情
    description = Column(LONGTEXT, nullable=True, comment="商品详情")
    spec_template = Column(Text, nullable=True, comment="规格模板(JSON)")

    # 状态
    status = Column(SQLEnum(ProductStatus), default=ProductStatus.DRAFT, comment="商品状态")
    is_featured = Column(Boolean, default=False, comment="是否推荐")
    is_new = Column(Boolean, default=False, comment="是否新品")
    sort = Column(Integer, default=0, comment="排序")

    # 统计
    sales_count = Column(Integer, default=0, comment="销量")
    view_count = Column(Integer, default=0, comment="浏览量")

    created_at = Column(DateTime, server_default=func.now())
    updated_at = Column(DateTime, server_default=func.now(), onupdate=func.now())

    # 关系
    category = relationship("Category", back_populates="products")
    brand = relationship("Brand", back_populates="products")
    items = relationship("OrderItem", back_populates="product")
    skus = relationship("ProductSku", back_populates="product", cascade="all, delete-orphan")
    cart_items = relationship("Cart", back_populates="product")
    favorites = relationship("Favorite", back_populates="product")

    def __repr__(self):
        return f"<Product(id={self.id}, name={self.name})>"


class ProductSku(Base):
    """商品SKU表"""
    __tablename__ = "product_skus"

    id = Column(Integer, primary_key=True, index=True)
    product_id = Column(Integer, ForeignKey("products.id", ondelete="CASCADE"), nullable=False, index=True)
    sku_code = Column(String(50), unique=True, nullable=False, comment="SKU编码")
    spec_data = Column(String(500), nullable=True, comment="规格数据(JSON)")
    price = Column(Numeric(10, 2), nullable=False, comment="价格")
    stock = Column(Integer, default=0, comment="库存")

    image = Column(String(500), nullable=True, comment="SKU图片")
    is_enabled = Column(Boolean, default=True, comment="是否启用")
    created_at = Column(DateTime, server_default=func.now())
    updated_at = Column(DateTime, server_default=func.now(), onupdate=func.now())
```

```python
    SUCCESS = "success"       # 支付成功
    FAILED = "failed"         # 支付失败
    CANCELLED = "cancelled"   # 已取消
    REFUNDED = "refunded"     # 已退款


class PaymentMethod(str, enum.Enum):
    """支付方式"""
    ALIPAY = "alipay"         # 支付宝
    WECHAT = "wechat"         # 微信支付
    BALANCE = "balance"        # 余额支付
    UNIONPAY = "unionpay"     # 银联


class Payment(Base):
    """支付记录表"""
    __tablename__ = "payments"

    id = Column(Integer, primary_key=True, index=True)
    payment_no = Column(String(64), unique=True, nullable=False, index=True, comment="支付流水号")
    order_id = Column(Integer, ForeignKey("orders.id"), nullable=False, index=True)
    order_no = Column(String(50), nullable=False, comment="订单号")

    # 支付信息
    amount = Column(Numeric(10, 2), nullable=False, comment="支付金额")
    method = Column(SQLEnum(PaymentMethod), nullable=False, comment="支付方式")
    status = Column(SQLEnum(PaymentStatus), default=PaymentStatus.PENDING, comment="支付状态")

    # 第三方支付信息
    trade_no = Column(String(64), nullable=True, comment="第三方交易号")
    transaction_id = Column(String(64), nullable=True, comment="微信/支付宝交易号")
    buyer_logon_id = Column(String(100), nullable=True, comment="买家账号")

    # 回调信息
    raw_response = Column(Text, nullable=True, comment="原始回调数据")
    callback_time = Column(DateTime, nullable=True, comment="回调时间")

    # 退款信息
    refund_amount = Column(Numeric(10, 2), default=0, comment="退款金额")
    refund_time = Column(DateTime, nullable=True, comment="退款时间")
    refund_reason = Column(String(500), nullable=True, comment="退款原因")

    # 幂等性
    idempotency_key = Column(String(64), nullable=True, index=True, comment="幂等Key")

    created_at = Column(DateTime, server_default=func.now())
    updated_at = Column(DateTime, server_default=func.now(), onupdate=func.now())

    # 关系
    order = relationship("Order", back_populates="payments")

    def __repr__(self):
        return f"<Payment(id={self.id}, payment_no={self.payment_no})>"
```

```python
# alembic/env.py
from logging.config import fileConfig
from sqlalchemy import engine_from_config, pool
from alembic import context

# 导入模型
from app.config import settings
from app.database import Base
from app.models import *

config = context.config
config.set_main_option("sqlalchemy.url", settings.DATABASE_URL)

if config.config_file_name is not None:
    fileConfig(config.config_file_name)

target_metadata = Base.metadata


def run_migrations_offline() -> None:
    """离线迁移"""
    url = config.get_main_option("sqlalchemy.url")
    context.configure(
        url=url,
        target_metadata=target_metadata,
        literal_binds=True,
        dialect_opts={"paramstyle": "named"},
    )

    with context.begin_transaction():
        context.run_migrations()


def run_migrations_online() -> None:
    """在线迁移"""
    connectable = engine_from_config(
        config.get_section(config.config_ini_section, {}),
        prefix="sqlalchemy.",
        poolclass=pool.NullPool,
    )

    with connectable.connect() as connection:
        context.configure(
            connection=connection,
            target_metadata=target_metadata
        )

        with context.begin_transaction():
            context.run_migrations()


if context.is_offline_mode():
    run_migrations_offline()
else:
    run_migrations_online()
```

```python
# file_manager.py
import os
import shutil
import hashlib
from pathlib import Path
from typing import List, Dict, Optional
from datetime import datetime


class FileManager:
    """文件管理器"""

    @staticmethod
    def list_directory(path: str = ".") -> List[Dict]:
        """列出目录内容"""
        items = []

        try:
            for item in os.listdir(path):
                item_path = os.path.join(path, item)
                stat = os.stat(item_path)

                items.append({
                    "name": item,
                    "path": item_path,
                    "is_dir": os.path.isdir(item_path),
                    "size": stat.st_size,
                    "modified": datetime.fromtimestamp(stat.st_mtime).isoformat(),
                    "created": datetime.fromtimestamp(stat.st_ctime).isoformat()
                })
        except PermissionError:
            print(f"无权限访问: {path}")

        # 按目录优先排序
        items.sort(key=lambda x: (not x["is_dir"], x["name"].lower()))
        return items

    @staticmethod
    def create_directory(path: str) -> bool:
        """创建目录"""
        try:
            os.makedirs(path, exist_ok=True)
            return True
        except Exception as e:
            print(f"创建目录失败: {e}")
            return False

    @staticmethod
    def delete_file(path: str) -> bool:
        """删除文件或目录"""
        try:
            if os.path.isdir(path):
                shutil.rmtree(path)
            else:
                os.remove(path)
            return True
        except Exception as e:
            print(f"删除失败: {e}")
            return False

    @staticmethod
    def copy_file(source: str, destination: str) -> bool:
        """复制文件或目录"""
        try:
            if os.path.isdir(source):
                shutil.copytree(source, destination)
            else:
                shutil.copy2(source, destination)
            return True
        except Exception as e:
            print(f"复制失败: {e}")
            return False

    @staticmethod
    def move_file(source: str, destination: str) -> bool:
        """移动文件或目录"""
        try:
            shutil.move(source, destination)
            return True
        except Exception as e:
            print(f"移动失败: {e}")
            return False

    @staticmethod
    def get_file_info(path: str) -> Optional[Dict]:
        """获取文件详细信息"""
        if not os.path.exists(path):
            return None

        stat = os.stat(path)
        path_obj = Path(path)

        return {
            "name": path_obj.name,
            "path": os.path.abspath(path),
            "parent": path_obj.parent.name,
            "is_dir": path_obj.is_dir(),
            "is_file": path_obj.is_file(),
            "is_link": path_obj.is_symlink(),
            "size": stat.st_size,
            "size_formatted": FileManager.format_size(stat.st_size),
            "modified": datetime.fromtimestamp(stat.st_mtime).isoformat(),
            "created": datetime.fromtimestamp(stat.st_ctime).isoformat(),
            "accessed": datetime.fromtimestamp(stat.st_atime).isoformat(),
            "extension": path_obj.suffix,
            "permissions": oct(stat.st_mode)[-3:]
        }

    @staticmethod
    def format_size(size: int) -> str:
        """格式化文件大小"""
        for unit in ['B', 'KB', 'MB', 'GB', 'TB']:
            if size < 1024:
                return f"{size:.2f} {unit}"
            size /= 1024
        return f"{size:.2f} PB"

    @staticmethod
    def calculate_hash(path: str, algorithm: str = "md5") -> Optional[str]:
        """计算文件哈希"""
        if not os.path.isfile(path):
            return None

        hash_func = hashlib.new(algorithm)

        try:
            with open(path, 'rb') as f:
                for chunk in iter(lambda: f.read(4096), b''):
                    hash_func.update(chunk)
            return hash_func.hexdigest()
        except Exception as e:
            print(f"计算哈希失败: {e}")
            return None


class FileSearcher:
    """文件搜索器"""

    @staticmethod
    def search_by_name(directory: str, pattern: str, recursive: bool = True) -> List[str]:
        """按名称搜索"""
        results = []
        path = Path(directory)

        if recursive:
            results = [str(p) for p in path.rglob(pattern)]
        else:
            results = [str(p) for p in path.glob(pattern)]

        return results

    @staticmethod
    def search_by_extension(directory: str, extension: str) -> List[str]:
        """按扩展名搜索"""
        if not extension.startswith('.'):
            extension = '.' + extension

        path = Path(directory)
        return [str(p) for p in path.rglob(f'*{extension}')]

    @staticmethod
    def search_by_size(directory: str, min_size: int = 0, max_size: int = None) -> List[str]:
        """按大小搜索"""
        results = []
        path = Path(directory)

        for p in path.rglob('*'):
            if p.is_file():
                size = p.stat().st_size
                if size >= min_size:
                    if max_size is None or size <= max_size:
                        results.append(str(p))

        return results

    @staticmethod
    def search_by_date(directory: str, days: int = 7, modified_after: bool = True) -> List[str]:
        """按日期搜索"""
        import time
        results = []
        path = Path(directory)
        cutoff = time.time() - (days * 86400)

        for p in path.rglob('*'):
            if p.is_file():
                mtime = p.stat().st_mtime
                if modified_after and mtime >= cutoff:
                    results.append(str(p))
                elif not modified_after and mtime <= cutoff:
                    results.append(str(p))

        return results
```

### 2.2 命令行界面

```python
# cli.py
import os
from file_manager import FileManager, FileSearcher


def print_items(items: list):
    """打印文件列表"""
    if not items:
        print("目录为空")
        return

    print(f"{'名称':<30} {'类型':<8} {'大小':<12} {'修改时间':<20}")
    print("-" * 80)

    for item in items:
        item_type = "目录" if item["is_dir"] else "文件"
        size = FileManager.format_size(item["size"]) if not item["is_dir"] else "-"
        print(f"{item['name']:<30} {item_type:<8} {size:<12} {item['modified'][:19]:<20}")


def main():
    current_dir = os.getcwd()

    while True:
        print(f"\n当前目录: {current_dir}")
        print("1. 列出文件")
        print("2. 创建目录")
        print("3. 删除文件/目录")
        print("4. 复制文件/目录")
        print("5. 移动文件/目录")
        print("6. 查看文件信息")
        print("7. 搜索文件")
        print("8. 切换目录")
        print("0. 退出")

        choice = input("\n请选择: ").strip()

        if choice == "1":
            items = FileManager.list_directory(current_dir)
            print_items(items)

        elif choice == "2":
            name = input("请输入目录名: ").strip()
            if name:
                path = os.path.join(current_dir, name)
                if FileManager.create_directory(path):
                    print("创建成功")

        elif choice == "3":
            name = input("请输入要删除的文件/目录名: ").strip()
            if name:
                path = os.path.join(current_dir, name)
                if FileManager.delete_file(path):
                    print("删除成功")

        elif choice == "4":
            source = input("源路径: ").strip()
            dest = input("目标路径: ").strip()
            if source and dest:
                if FileManager.copy_file(source, dest):
                    print("复制成功")

        elif choice == "5":
            source = input("源路径: ").strip()
            dest = input("目标路径: ").strip()
            if source and dest:
                if FileManager.move_file(source, dest):
                    print("移动成功")

        elif choice == "6":
            name = input("请输入文件名: ").strip()
            if name:
                path = os.path.join(current_dir, name)
                info = FileManager.get_file_info(path)
                if info:
                    for k, v in info.items():
                        print(f"{k}: {v}")
                else:
                    print("文件不存在")

        elif choice == "7":
            pattern = input("搜索模式 (如 *.py): ").strip()
            if pattern:
                results = FileSearcher.search_by_name(current_dir, pattern)
                print(f"找到 {len(results)} 个结果:")
                for r in results[:20]:
                    print(r)

        elif choice == "8":
            path = input("请输入目录路径: ").strip()
            if path and os.path.isdir(path):
                current_dir = os.path.abspath(path)
            else:
                print("目录不存在")

        elif choice == "0":
            print("再见!")
            break


if __name__ == "__main__":
    main()
```

### 3.1 批量重命名

```python
@staticmethod
def batch_rename(directory: str, old_pattern: str, new_pattern: str) -> int:
    """批量重命名"""
    count = 0

    for filename in os.listdir(directory):
        if old_pattern in filename:
            old_path = os.path.join(directory, filename)
            new_filename = filename.replace(old_pattern, new_pattern)
            new_path = os.path.join(directory, new_filename)

            os.rename(old_path, new_path)
            count += 1
            print(f"重命名: {filename} -> {new_filename}")

    return count
```

### 3.2 重复文件检测

```python
@staticmethod
def find_duplicates(directory: str) -> Dict[str, List[str]]:
    """查找重复文件"""
    hash_dict = {}
    path = Path(directory)

    for p in path.rglob('*'):
        if p.is_file():
            file_hash = FileManager.calculate_hash(str(p))
            if file_hash:
                if file_hash not in hash_dict:
                    hash_dict[file_hash] = []
                hash_dict[file_hash].append(str(p))

    # 返回重复文件
    return {h: files for h, files in hash_dict.items() if len(files) > 1}
```

```python
class FilePermission(Base):
    """文件权限表"""
    __tablename__ = "file_permissions"

    id = Column(Integer, primary_key=True, autoincrement=True)
    file_id = Column(Integer, ForeignKey("files.id"), nullable=False)
    user_id = Column(Integer, ForeignKey("users.id"), nullable=False)
    permission = Column(String(20), nullable=False)  # read, write, admin
    created_at = Column(DateTime, default=func.now())

    # 关系
    file = relationship("FileRecord", backref="permissions")
    user = relationship("User")


class FileShare(Base):
    """文件分享表"""
    __tablename__ = "file_shares"

    id = Column(Integer, primary_key=True, autoincrement=True)
    file_id = Column(Integer, ForeignKey("files.id"), nullable=False)
    share_code = Column(String(32), unique=True, nullable=False)
    password = Column(String(100), nullable=True)  # 可选密码
    expires_at = Column(DateTime, nullable=True)
    download_limit = Column(Integer, default=0)  # 0 = 无限制
    download_count = Column(Integer, default=0)
    created_at = Column(DateTime, default=func.now())
    created_by = Column(Integer, ForeignKey("users.id"))

    # 关系
    file = relationship("FileRecord", backref="shares")
```

### 1.2 权限服务

```python
# permission_services.py
from sqlalchemy.orm import Session
from typing import List, Optional
import uuid
from datetime import datetime, timedelta

from models import FileRecord, FilePermission, FileShare, User


class PermissionService:
    """权限服务"""

    def __init__(self, db: Session):
        self.db = db

    def grant_permission(
        self,
        file_id: int,
        user_id: int,
        permission: str
    ) -> FilePermission:
        """授予权限"""
        # 检查是否已存在
        existing = self.db.query(FilePermission).filter(
            FilePermission.file_id == file_id,
            FilePermission.user_id == user_id
        ).first()

        if existing:
            existing.permission = permission
            self.db.commit()
            self.db.refresh(existing)
            return existing

        perm = FilePermission(
            file_id=file_id,
            user_id=user_id,
            permission=permission
        )

        self.db.add(perm)
        self.db.commit()
        self.db.refresh(perm)
        return perm

    def revoke_permission(self, file_id: int, user_id: int) -> bool:
        """撤销权限"""
        perm = self.db.query(FilePermission).filter(
            FilePermission.file_id == file_id,
            FilePermission.user_id == user_id
        ).first()

        if perm:
            self.db.delete(perm)
            self.db.commit()
            return True

        return False

    def check_permission(
        self,
        file_id: int,
        user_id: int,
        required_permission: str
    ) -> bool:
        """检查权限"""
        file = self.db.query(FileRecord).filter(FileRecord.id == file_id).first()

        if not file:
            return False

        # 所有者拥有所有权限
        if file.owner_id == user_id:
            return True

        # 检查权限
        perm = self.db.query(FilePermission).filter(
            FilePermission.file_id == file_id,
            FilePermission.user_id == user_id
        ).first()

        if not perm:
            return False

        # 权限等级
        level = {"read": 1, "write": 2, "admin": 3}
        required_level = level.get(required_permission, 0)
        user_level = level.get(perm.permission, 0)

        return user_level >= required_level


class ShareService:
    """分享服务"""

    def __init__(self, db: Session):
        self.db = db

    def create_share(
        self,
        file_id: int,
        user_id: int,
        password: str = None,
        expires_days: int = 7,
        download_limit: int = 0
    ) -> FileShare:
        """创建分享链接"""
        # 生成分享码
        share_code = uuid.uuid4().hex[:16]

        # 计算过期时间
        expires_at = None
        if expires_days > 0:
            expires_at = datetime.now() + timedelta(days=expires_days)

        share = FileShare(
            file_id=file_id,
            share_code=share_code,
            password=password,  # 实际应该加密
            expires_at=expires_at,
            download_limit=download_limit,
            created_by=user_id
        )

        self.db.add(share)
        self.db.commit()
        self.db.refresh(share)
        return share

    def get_share(self, share_code: str) -> Optional[FileShare]:
        """获取分享"""
        return self.db.query(FileShare).filter(
            FileShare.share_code == share_code
        ).first()

    def access_share(
        self,
        share_code: str,
        password: str = None
    ) -> Optional[FileRecord]:
        """访问分享"""
        share = self.get_share(share_code)

        if not share:
            return None

        # 检查过期
        if share.expires_at and share.expires_at < datetime.now():
            return None

        # 检查下载限制
        if share.download_limit > 0 and share.download_count >= share.download_limit:
            return None

        # 检查密码
        if share.password and share.password != password:
            return None

        # 增加下载计数
        share.download_count += 1
        self.db.commit()

        return share.file
```

### 2.1 回收站服务

```python
# trash_service.py
from sqlalchemy.orm import Session
from typing import List
from datetime import datetime

from models import FileRecord


class TrashService:
    """回收站服务"""

    def __init__(self, db: Session):
        self.db = db

    def get_trash_files(self, user_id: int = None) -> List[FileRecord]:
        """获取回收站文件"""
        query = self.db.query(FileRecord).filter(
            FileRecord.deleted_at != None
        )

        if user_id:
            query = query.filter(FileRecord.owner_id == user_id)

        return query.order_by(FileRecord.deleted_at.desc()).all()

    def restore_file(self, file_id: int) -> bool:
        """恢复文件"""
        file = self.db.query(FileRecord).filter(
            FileRecord.id == file_id
        ).first()

        if not file or not file.deleted_at:
            return False

        file.deleted_at = None
        self.db.commit()
        return True

    def empty_trash(self, user_id: int = None) -> int:
        """清空回收站"""
        query = self.db.query(FileRecord).filter(
            FileRecord.deleted_at != None
        )

        if user_id:
            query = query.filter(FileRecord.owner_id == user_id)

        # 获取要删除的文件
        files = query.all()
        count = len(files)

        # 删除实际文件
        for file in files:
            if file.is_directory and file.path:
                import shutil
                if shutil.os.path.exists(file.path):
                    shutil.rmtree(file.path)
            elif file.path and shutil.os.path.exists(file.path):
                shutil.os.remove(file.path)

        # 删除数据库记录
        query.delete(synchronize_session=False)
        self.db.commit()

        return count

    def get_trash_size(self, user_id: int = None) -> int:
        """获取回收站大小"""
        query = self.db.query(
            func.sum(FileRecord.size)
        ).filter(
            FileRecord.deleted_at != None
        )

        if user_id:
            query = query.filter(FileRecord.owner_id == user_id)

        result = query.scalar()
        return result or 0
```

### 3.1 权限和分享 API

```python
# router.py
from fastapi import APIRouter, Depends, HTTPException, Query
from sqlalchemy.orm import Session
from typing import Optional

from database import get_db
from models import FileRecord
from services.permission_services import PermissionService, ShareService
from services.trash_service import TrashService

router = APIRouter(prefix="/files", tags=["文件管理"])


# 权限管理
@router.post("/{file_id}/permissions")
def grant_permission(
    file_id: int,
    user_id: int,
    permission: str = Query(..., regex="^(read|write|admin)$"),
    db: Session = Depends(get_db)
):
    """授予权限"""
    service = PermissionService(db)
    perm = service.grant_permission(file_id, user_id, permission)
    return {"message": "权限授予成功"}


@router.delete("/{file_id}/permissions/{user_id}")
def revoke_permission(
    file_id: int,
    user_id: int,
    db: Session = Depends(get_db)
):
    """撤销权限"""
    service = PermissionService(db)
    if service.revoke_permission(file_id, user_id):
        return {"message": "权限已撤销"}
    raise HTTPException(status_code=404, detail="权限不存在")


# 文件分享
@router.post("/{file_id}/share")
def create_share(
    file_id: int,
    password: Optional[str] = None,
    expires_days: int = Query(7, ge=1, le=30),
    download_limit: int = Query(0),
    user_id: int = Query(1),
    db: Session = Depends(get_db)
):
    """创建分享链接"""
    service = ShareService(db)
    share = service.create_share(
        file_id=file_id,
        user_id=user_id,
        password=password,
        expires_days=expires_days,
        download_limit=download_limit
    )

    return {
        "share_code": share.share_code,
        "expires_at": share.expires_at,
        "download_limit": share.download_limit
    }


@router.get("/share/{share_code}")
def access_share(
    share_code: str,
    password: Optional[str] = None,
    db: Session = Depends(get_db)
):
    """访问分享"""
    service = ShareService(db)
    file = service.access_share(share_code, password)

    if not file:
        raise HTTPException(status_code=404, detail="分享不存在或已过期")

    return {
        "name": file.name,
        "size": file.size,
        "mime_type": file.mime_type
    }


# 回收站
@router.get("/trash")
def get_trash(
    user_id: int = Query(1),
    db: Session = Depends(get_db)
):
    """获取回收站"""
    service = TrashService(db)
    files = service.get_trash_files(user_id)

    return {
        "files": [
            {
                "id": f.id,
                "name": f.name,
                "deleted_at": f.deleted_at,
                "size": f.size
            }
            for f in files
        ],
        "total_size": service.get_trash_size(user_id)
    }


@router.post("/trash/{file_id}/restore")
def restore_from_trash(
    file_id: int,
    db: Session = Depends(get_db)
):
    """恢复文件"""
    service = TrashService(db)
    if service.restore_file(file_id):
        return {"message": "恢复成功"}
    raise HTTPException(status_code=404, detail="文件不存在")


@router.delete("/trash")
def empty_trash(
    user_id: int = Query(1),
    db: Session = Depends(get_db)
):
    """清空回收站"""
    service = TrashService(db)
    count = service.empty_trash(user_id)
    return {"message": f"已清空 {count} 个文件"}
```

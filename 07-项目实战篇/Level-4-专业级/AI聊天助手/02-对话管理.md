## 1. 对话历史管理

### 1.1 基本对话循环

```python
from openai import OpenAI

client = OpenAI()

# 存储对话历史
messages = [
    {"role": "system", "content": "你是一个友好的助手"}
]

def chat(user_input: str) -> str:
    # 添加用户消息
    messages.append({"role": "user", "content": user_input})
    
    # 调用 API
    response = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=messages
    )
    
    # 获取回复
    assistant_message = response.choices[0].message.content
    
    # 添加到历史
    messages.append({"role": "assistant", "content": assistant_message})
    
    return assistant_message

# 多轮对话
print(chat("你好"))                    # 你好！有什么可以帮助你的？
print(chat("我叫 Alice"))              # 很高兴认识你，Alice！
print(chat("你还记得我的名字吗？"))    # 当然，你叫 Alice
```

## 2. 上下文限制

### 2.1 Token 限制问题

**每个模型有最大 Token 限制：**

| 模型 | 最大 Token |
|------|-----------|
| gpt-4o-mini | 128K |
| gpt-4o | 128K |
| qwen-plus | 32K |

**当对话过长时，需要截断历史：**

```python
def trim_history(messages: list, max_messages: int = 20) -> list:
    """保留最近的对话历史"""
    if len(messages) > max_messages:
        # 保留 system 消息和最近的对话
        system = messages[0] if messages[0]["role"] == "system" else None
        recent = messages[-(max_messages-1):]
        
        if system:
            return [system] + recent
        return recent
    return messages

# 使用
messages = trim_history(messages, max_messages=10)
```

### 2.2 Token 计数

```python
# 使用 tiktoken 计算 Token 数
import tiktoken

def count_tokens(messages: list, model: str = "gpt-4o-mini") -> int:
    """计算消息列表的 Token 数"""
    encoding = tiktoken.encoding_for_model(model)
    
    total = 0
    for msg in messages:
        # 每条消息有格式开销
        total += 4  # role + content 格式
        total += len(encoding.encode(msg["content"]))
    
    return total

# 检查是否超限
if count_tokens(messages) > 100000:
    messages = trim_history(messages)
```

## 3. 对话持久化

### 3.1 保存到文件

```python
import json

def save_conversation(messages: list, filepath: str):
    """保存对话到文件"""
    with open(filepath, "w", encoding="utf-8") as f:
        json.dump(messages, f, ensure_ascii=False, indent=2)

def load_conversation(filepath: str) -> list:
    """从文件加载对话"""
    try:
        with open(filepath, "r", encoding="utf-8") as f:
            return json.load(f)
    except FileNotFoundError:
        return [{"role": "system", "content": "你是一个友好的助手"}]

# 使用
messages = load_conversation("chat_history.json")
# ... 对话 ...
save_conversation(messages, "chat_history.json")
```

### 3.2 多会话管理

```python
import os
from datetime import datetime

class ConversationManager:
    def __init__(self, save_dir: str = "conversations"):
        self.save_dir = save_dir
        os.makedirs(save_dir, exist_ok=True)
    
    def new_session(self) -> str:
        """创建新会话"""
        session_id = datetime.now().strftime("%Y%m%d_%H%M%S")
        filepath = os.path.join(self.save_dir, f"{session_id}.json")
        
        messages = [{"role": "system", "content": "你是一个友好的助手"}]
        self.save(filepath, messages)
        
        return session_id
    
    def save(self, session_id: str, messages: list):
        filepath = os.path.join(self.save_dir, f"{session_id}.json")
        with open(filepath, "w", encoding="utf-8") as f:
            json.dump(messages, f, ensure_ascii=False, indent=2)
    
    def load(self, session_id: str) -> list:
        filepath = os.path.join(self.save_dir, f"{session_id}.json")
        with open(filepath, "r", encoding="utf-8") as f:
            return json.load(f)
    
    def list_sessions(self) -> list:
        """列出所有会话"""
        files = os.listdir(self.save_dir)
        return [f.replace(".json", "") for f in files if f.endswith(".json")]
```

## 4. System Prompt 设计

### 4.1 角色设定

```python
# 编程助手
SYSTEM_PROMPT_CODING = """
你是一个专业的 Python 编程助手。
- 回答简洁准确，提供代码示例
- 解释代码时使用中文注释
- 遇到错误时给出具体解决方案
"""

# 翻译助手
SYSTEM_PROMPT_TRANSLATE = """
你是一个翻译助手，精通中文和英文。
- 翻译时保持原文风格
- 专业术语保留英文
- 提供翻译解释
"""

# 学习辅导
SYSTEM_PROMPT_TUTOR = """
你是一个耐心的学习辅导老师。
- 用简单易懂的语言解释概念
- 提供类比和例子帮助理解
- 鼓励学生提问
"""
```

### 4.2 动态 System Prompt

```python
def get_system_prompt(role: str, context: dict = None) -> str:
    """根据角色生成 System Prompt"""
    prompts = {
        "coding": SYSTEM_PROMPT_CODING,
        "translate": SYSTEM_PROMPT_TRANSLATE,
        "tutor": SYSTEM_PROMPT_TUTOR
    }
    
    base = prompts.get(role, "你是一个友好的助手")
    
    # 添加额外上下文
    if context:
        extra = f"\n\n当前上下文：{context}"
        return base + extra
    
    return base
```

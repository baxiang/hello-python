# 流式响应

学习如何实现流式输出，让用户实时看到 AI 的回复。

---

## 1. 为什么需要流式响应

```
┌─────────────────────────────────────────────────────────────┐
│                  普通响应 vs 流式响应                        │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│   普通响应：                                                 │
│   用户发送 → 等待... → 完整回复一次性显示                    │
│   问题：长回复时等待时间长，用户体验差                       │
│                                                             │
│   流式响应：                                                 │
│   用户发送 → 逐字显示... → 完成回复                         │
│   优点：实时反馈，用户感知更快                               │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

## 2. 同步流式响应

### 2.1 基本用法

```python
from openai import OpenAI

client = OpenAI()

def stream_chat(message: str):
    """流式输出聊天回复"""
    stream = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": message}],
        stream=True  # 启用流式响应
    )
    
    # 逐块处理
    for chunk in stream:
        if chunk.choices[0].delta.content:
            print(chunk.choices[0].delta.content, end="", flush=True)
    
    print()  # 换行

# 使用
stream_chat("介绍一下 Python 的特点")
```

### 2.2 收集完整回复

```python
def stream_and_collect(message: str) -> str:
    """流式输出并收集完整回复"""
    stream = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": message}],
        stream=True
    )
    
    full_content = ""
    
    for chunk in stream:
        delta = chunk.choices[0].delta
        if delta.content:
            print(delta.content, end="", flush=True)
            full_content += delta.content
    
    print()
    return full_content
```

## 3. 异步流式响应

### 3.1 使用 AsyncOpenAI

```python
import asyncio
from openai import AsyncOpenAI

async_client = AsyncOpenAI()

async def async_stream_chat(message: str):
    """异步流式聊天"""
    stream = await async_client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": message}],
        stream=True
    )
    
    async for chunk in stream:
        if chunk.choices[0].delta.content:
            print(chunk.choices[0].delta.content, end="", flush=True)
    
    print()

# 运行
asyncio.run(async_stream_chat("你好"))
```

### 3.2 并发处理多个请求

```python
async def process_multiple(prompts: list):
    """并发处理多个请求"""
    tasks = [async_stream_chat(prompt) for prompt in prompts]
    await asyncio.gather(*tasks)

# 使用
prompts = [
    "什么是机器学习？",
    "什么是深度学习？",
    "什么是自然语言处理？"
]
asyncio.run(process_multiple(prompts))
```

## 4. 流式响应进度显示

### 4.1 使用 rich 库

```bash
uv add rich
```

```python
from rich.console import Console
from rich.live import Live
from rich.markdown import Markdown
from openai import OpenAI

console = Console()
client = OpenAI()

def stream_with_markdown(message: str):
    """流式输出并渲染 Markdown"""
    stream = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": message}],
        stream=True
    )
    
    content = ""
    
    with Live(console=console, refresh_per_second=10) as live:
        for chunk in stream:
            if chunk.choices[0].delta.content:
                content += chunk.choices[0].delta.content
                # 实时渲染 Markdown
                live.update(Markdown(content))
    
    return content

# 使用
stream_with_markdown("用 Markdown 格式介绍 Python")
```

### 4.2 打字机效果

```python
import time
import random

def typewriter_effect(text: str, delay: float = 0.02):
    """打字机效果显示文本"""
    for char in text:
        print(char, end="", flush=True)
        # 随机延迟，更自然
        time.sleep(delay * random.uniform(0.5, 1.5))
    print()

def stream_with_typewriter(message: str):
    """流式输出 + 打字机效果"""
    stream = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": message}],
        stream=True
    )
    
    for chunk in stream:
        if chunk.choices[0].delta.content:
            text = chunk.choices[0].delta.content
            typewriter_effect(text, delay=0.01)
```

## 5. 流式响应 + 对话管理

```python
from openai import OpenAI
from typing import Generator

client = OpenAI()

class StreamingChat:
    def __init__(self, system_prompt: str = "你是一个友好的助手"):
        self.messages = [{"role": "system", "content": system_prompt}]
    
    def chat(self, user_input: str) -> Generator[str, None, None]:
        """流式聊天，返回生成器"""
        self.messages.append({"role": "user", "content": user_input})
        
        stream = client.chat.completions.create(
            model="gpt-4o-mini",
            messages=self.messages,
            stream=True
        )
        
        full_response = ""
        
        for chunk in stream:
            delta = chunk.choices[0].delta
            if delta.content:
                full_response += delta.content
                yield delta.content
        
        # 保存完整回复
        self.messages.append({"role": "assistant", "content": full_response})
    
    def chat_and_print(self, user_input: str) -> str:
        """流式聊天并打印"""
        full = ""
        for chunk in self.chat(user_input):
            print(chunk, end="", flush=True)
            full += chunk
        print()
        return full

# 使用
chat = StreamingChat()
chat.chat_and_print("你好")
chat.chat_and_print("介绍一下你自己")
```

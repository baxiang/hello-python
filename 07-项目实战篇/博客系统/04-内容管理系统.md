```python
# dashboard_services.py
from sqlalchemy.orm import Session
from sqlalchemy import func
from datetime import datetime, timedelta

from models import Post, Comment, User


class DashboardService:
    """仪表盘服务"""

    def __init__(self, db: Session):
        self.db = db

    def get_overview(self) -> dict:
        """获取概览数据"""
        # 文章统计
        total_posts = self.db.query(Post).count()
        published_posts = self.db.query(Post).filter(Post.is_published == True).count()
        draft_posts = total_posts - published_posts

        # 评论统计
        total_comments = self.db.query(Comment).count()
        pending_comments = self.db.query(Comment).filter(Comment.is_approved == False).count()

        # 用户统计
        total_users = self.db.query(User).count()

        # 今日数据
        today = datetime.now().date()
        today_posts = self.db.query(Post).filter(
            func.date(Post.created_at) == today
        ).count()

        today_comments = self.db.query(Comment).filter(
            func.date(Comment.created_at) == today
        ).count()

        return {
            "posts": {
                "total": total_posts,
                "published": published_posts,
                "draft": draft_posts
            },
            "comments": {
                "total": total_comments,
                "pending": pending_comments
            },
            "users": {
                "total": total_users
            },
            "today": {
                "posts": today_posts,
                "comments": today_comments
            }
        }

    def get_recent_activities(self, limit: int = 10) -> list:
        """获取最近活动"""
        # 最近文章
        recent_posts = self.db.query(Post).order_by(
            Post.created_at.desc()
        ).limit(5).all()

        # 最近评论
        recent_comments = self.db.query(Comment).order_by(
            Comment.created_at.desc()
        ).limit(5).all()

        activities = []

        for post in recent_posts:
            activities.append({
                "type": "post",
                "action": "created",
                "title": post.title,
                "time": post.created_at
            })

        for comment in recent_comments:
            activities.append({
                "type": "comment",
                "action": "created",
                "content": comment.content[:50],
                "time": comment.created_at
            })

        # 按时间排序
        activities.sort(key=lambda x: x["time"], reverse=True)
        return activities[:limit]

    def get_popular_posts(self, limit: int = 10) -> list:
        """获取热门文章"""
        return self.db.query(Post).filter(
            Post.is_published == True
        ).order_by(Post.view_count.desc()).limit(limit).all()
```

### 1.2 文章管理

```python
# admin_post_router.py
from fastapi import APIRouter, Depends, HTTPException, Query
from sqlalchemy.orm import Session
from typing import Optional

from database import get_db
from models import Post
from services import PostService
from schemas import PostCreate, PostUpdate, PostResponse

router = APIRouter(prefix="/admin/posts", tags=["文章管理"])


@router.get("")
def get_posts(
    page: int = Query(1, ge=1),
    page_size: int = Query(20, ge=1, le=100),
    keyword: Optional[str] = None,
    category_id: Optional[int] = None,
    status: Optional[str] = None,  # published, draft, all
    db: Session = Depends(get_db)
):
    """获取文章列表（管理）"""
    service = PostService(db)

    query = db.query(Post)

    if keyword:
        query = query.filter(Post.title.ilike(f"%{keyword}%"))

    if category_id:
        query = query.filter(Post.category_id == category_id)

    if status == "published":
        query = query.filter(Post.is_published == True)
    elif status == "draft":
        query = query.filter(Post.is_published == False)

    total = query.count()
    posts = query.order_by(Post.created_at.desc()).offset(
        (page - 1) * page_size
    ).limit(page_size).all()

    return {
        "items": posts,
        "total": total,
        "page": page,
        "page_size": page_size
    }


@router.post("")
def create_post(data: PostCreate, db: Session = Depends(get_db)):
    """创建文章"""
    service = PostService(db)

    post = service.create_post(
        title=data.title,
        content=data.content,
        author_id=1,  # 从认证获取
        summary=data.summary,
        cover_image=data.cover_image,
        category_id=data.category_id,
        tag_ids=data.tag_ids,
        is_published=data.is_published
    )

    return post


@router.put("/{post_id}")
def update_post(post_id: int, data: PostUpdate, db: Session = Depends(get_db)):
    """更新文章"""
    service = PostService(db)

    update_data = data.model_dump(exclude_unset=True)
    post = service.update_post(post_id, **update_data)

    if not post:
        raise HTTPException(status_code=404, detail="文章不存在")

    return post


@router.delete("/{post_id}")
def delete_post(post_id: int, db: Session = Depends(get_db)):
    """删除文章"""
    service = PostService(db)

    if not service.delete_post(post_id):
        raise HTTPException(status_code=404, detail="文章不存在")

    return {"message": "删除成功"}


@router.post("/{post_id}/publish")
def publish_post(post_id: int, db: Session = Depends(get_db)):
    """发布文章"""
    service = PostService(db)
    post = service.update_post(post_id, is_published=True)

    if not post:
        raise HTTPException(status_code=404, detail="文章不存在")

    return {"message": "发布成功"}


@router.post("/{post_id}/unpublish")
def unpublish_post(post_id: int, db: Session = Depends(get_db)):
    """取消发布"""
    service = PostService(db)
    post = service.update_post(post_id, is_published=False)

    if not post:
        raise HTTPException(status_code=404, detail="文章不存在")

    return {"message": "取消发布成功"}
```

### 1.3 分类和标签管理

```python
# category_router.py
from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy.orm import Session

from database import get_db
from models import Category, Tag
from schemas import CategoryCreate, CategoryResponse, TagCreate, TagResponse

router = APIRouter(prefix="/admin", tags=["内容管理"])


@router.post("/categories", response_model=CategoryResponse)
def create_category(data: CategoryCreate, db: Session = Depends(get_db)):
    """创建分类"""
    # 检查唯一性
    existing = db.query(Category).filter(Category.slug == data.slug).first()
    if existing:
        raise HTTPException(status_code=400, detail="分类 slug 已存在")

    category = Category(**data.model_dump())
    db.add(category)
    db.commit()
    db.refresh(category)
    return category


@router.put("/categories/{category_id}", response_model=CategoryResponse)
def update_category(category_id: int, data: CategoryCreate, db: Session = Depends(get_db)):
    """更新分类"""
    category = db.query(Category).filter(Category.id == category_id).first()
    if not category:
        raise HTTPException(status_code=404, detail="分类不存在")

    for key, value in data.model_dump().items():
        setattr(category, key, value)

    db.commit()
    db.refresh(category)
    return category


@router.delete("/categories/{category_id}")
def delete_category(category_id: int, db: Session = Depends(get_db)):
    """删除分类"""
    category = db.query(Category).filter(Category.id == category_id).first()
    if not category:
        raise HTTPException(status_code=404, detail="分类不存在")

    # 检查是否有文章
    if category.posts:
        raise HTTPException(status_code=400, detail="分类下有文章，无法删除")

    db.delete(category)
    db.commit()
    return {"message": "删除成功"}


@router.post("/tags", response_model=TagResponse)
def create_tag(data: TagCreate, db: Session = Depends(get_db)):
    """创建标签"""
    existing = db.query(Tag).filter(Tag.slug == data.slug).first()
    if existing:
        raise HTTPException(status_code=400, detail="标签 slug 已存在")

    tag = Tag(**data.model_dump())
    db.add(tag)
    db.commit()
    db.refresh(tag)
    return tag


@router.delete("/tags/{tag_id}")
def delete_tag(tag_id: int, db: Session = Depends(get_db)):
    """删除标签"""
    tag = db.query(Tag).filter(Tag.id == tag_id).first()
    if not tag:
        raise HTTPException(status_code=404, detail="标签不存在")

    db.delete(tag)
    db.commit()
    return {"message": "删除成功"}
```

```python
# comment_router.py
from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy.orm import Session

from database import get_db
from models import Comment

router = APIRouter(prefix="/admin/comments", tags=["评论管理"])


@router.get("")
def get_comments(
    post_id: int = None,
    is_approved: bool = None,
    page: int = 1,
    page_size: int = 20,
    db: Session = Depends(get_db)
):
    """获取评论列表"""
    query = db.query(Comment)

    if post_id:
        query = query.filter(Comment.post_id == post_id)

    if is_approved is not None:
        query = query.filter(Comment.is_approved == is_approved)

    total = query.count()
    comments = query.order_by(Comment.created_at.desc()).offset(
        (page - 1) * page_size
    ).limit(page_size).all()

    return {
        "items": comments,
        "total": total,
        "page": page,
        "page_size": page_size
    }


@router.post("/{comment_id}/approve")
def approve_comment(comment_id: int, db: Session = Depends(get_db)):
    """审核通过评论"""
    comment = db.query(Comment).filter(Comment.id == comment_id).first()
    if not comment:
        raise HTTPException(status_code=404, detail="评论不存在")

    comment.is_approved = True
    db.commit()

    return {"message": "审核通过"}


@router.post("/{comment_id}/reject")
def reject_comment(comment_id: int, db: Session = Depends(get_db)):
    """拒绝评论"""

    """删除评论"""
```

### 3.1 用户管理

```python
# user_router.py
from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy.orm import Session

from database import get_db
from models import User
from utils.security import get_password_hash

router = APIRouter(prefix="/admin/users", tags=["用户管理"])


@router.get("")
def get_users(
    page: int = 1,
    page_size: int = 20,
    keyword: str = None,
    db: Session = Depends(get_db)
):
    """获取用户列表"""
    query = db.query(User)

    if keyword:
        query = query.filter(
            User.username.ilike(f"%{keyword}%") |
            User.email.ilike(f"%{keyword}%")
        )

    total = query.count()
    users = query.order_by(User.created_at.desc()).offset(
        (page - 1) * page_size
    ).limit(page_size).all()

    return {
        "items": users,
        "total": total,
        "page": page,
        "page_size": page_size
    }


@router.post("")
def create_user(
    username: str,
    email: str,
    password: str,
    nickname: str = None,
    db: Session = Depends(get_db)
):
    """创建用户"""
    # 检查唯一性
    if db.query(User).filter(User.username == username).first():
        raise HTTPException(status_code=400, detail="用户名已存在")

    if db.query(User).filter(User.email == email).first():
        raise HTTPException(status_code=400, detail="邮箱已存在")

    user = User(
        username=username,
        email=email,
        password_hash=get_password_hash(password),
        nickname=nickname
    )

    db.add(user)
    db.commit()
    db.refresh(user)

    return user


@router.put("/{user_id}/disable")
def disable_user(user_id: int, db: Session = Depends(get_db)):
    """禁用用户"""
    user = db.query(User).filter(User.id == user_id).first()
    if not user:
        raise HTTPException(status_code=404, detail="用户不存在")

    user.is_active = False
    db.commit()

    return {"message": "用户已禁用"}


@router.put("/{user_id}/enable")
def enable_user(user_id: int, db: Session = Depends(get_db)):
    """启用用户"""
    user = db.query(User).filter(User.id == user_id).first()
    if not user:
        raise HTTPException(status_code=404, detail="用户不存在")

    user.is_active = True
    db.commit()

    return {"message": "用户已启用"}
```

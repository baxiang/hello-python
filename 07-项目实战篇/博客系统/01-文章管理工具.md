### 2.1 文章模型

```python
# post.py
import os
import re
from datetime import datetime
from typing import Optional, List
import markdown


class Post:
    """文章类"""

    def __init__(
        self,
        title: str,
        content: str,
        slug: str = None,
        tags: List[str] = None,
        category: str = None,
        created_at: datetime = None,
        updated_at: datetime = None
    ):
        self.title = title
        self.content = content
        self.slug = slug or self._generate_slug(title)
        self.tags = tags or []
        self.category = category
        self.created_at = created_at or datetime.now()
        self.updated_at = updated_at or datetime.now()

    def _generate_slug(self, title: str) -> str:
        """生成 URL 友好的 slug"""
        # 转换为小写，替换空格为连字符
        slug = title.lower()
        slug = re.sub(r'[^\w\s-]', '', slug)
        slug = re.sub(r'[-\s]+', '-', slug)
        return slug.strip('-')

    def to_dict(self) -> dict:
        return {
            "title": self.title,
            "slug": self.slug,
            "content": self.content,
            "tags": self.tags,
            "category": self.category,
            "created_at": self.created_at.isoformat(),
            "updated_at": self.updated_at.isoformat()
        }

    def to_html(self) -> str:
        """转换为 HTML"""
        md = markdown.Markdown(extensions=['extra', 'codehilite', 'toc'])
        return md.convert(self.content)

    @staticmethod
    def from_file(filepath: str) -> "Post":
        """从文件读取文章"""
        with open(filepath, 'r', encoding='utf-8') as f:
            content = f.read()

        # 解析元数据
        metadata = {}
        if content.startswith('---'):
            end_idx = content.find('---', 3)
            if end_idx != -1:
                meta_section = content[3:end_idx]
                content = content[end_idx + 3:]

                for line in meta_section.strip().split('\n'):
                    if ':' in line:
                        key, value = line.split(':', 1)
                        metadata[key.strip()] = value.strip()

        title = metadata.get('title', os.path.splitext(os.path.basename(filepath))[0])
        tags = metadata.get('tags', '').split(',') if metadata.get('tags') else []
        tags = [t.strip() for t in tags if t.strip()]
        category = metadata.get('category')

        return Post(
            title=title,
            content=content.strip(),
            slug=metadata.get('slug'),
            tags=tags,
            category=category
        )
```

### 2.2 文章管理器

```python
# post_manager.py
import os
import json
from typing import List, Optional
from datetime import datetime
from post import Post


class PostManager:
    """文章管理器"""

    def __init__(self, posts_dir: str = "posts"):
        self.posts_dir = posts_dir
        os.makedirs(posts_dir, exist_ok=True)

    def create_post(self, title: str, content: str, tags: List[str] = None, category: str = None) -> Post:
        """创建文章"""
        post = Post(title, content, tags=tags, category=category)

        # 保存到文件
        filename = f"{post.slug}.md"
        filepath = os.path.join(self.posts_dir, filename)

        # 添加元数据
        metadata = f"""---
title: {post.title}
slug: {post.slug}
tags: {','.join(post.tags) if post.tags else ''}
category: {post.category or ''}
created_at: {post.created_at.isoformat()}
---

{post.content}"""

        with open(filepath, 'w', encoding='utf-8') as f:
            f.write(metadata)

        return post

    def get_post(self, slug: str) -> Optional[Post]:
        """获取文章"""
        filepath = os.path.join(self.posts_dir, f"{slug}.md")

        if not os.path.exists(filepath):
            return None

        return Post.from_file(filepath)

    def get_all_posts(self) -> List[Post]:
        """获取所有文章"""
        posts = []

        for filename in os.listdir(self.posts_dir):
            if filename.endswith('.md'):
                filepath = os.path.join(self.posts_dir, filename)
                post = Post.from_file(filepath)
                posts.append(post)

        # 按日期排序
        posts.sort(key=lambda p: p.created_at, reverse=True)
        return posts

    def search_posts(self, keyword: str) -> List[Post]:
        """搜索文章"""
        posts = self.get_all_posts()
        keyword = keyword.lower()

        return [
            p for p in posts
            if keyword in p.title.lower() or keyword in p.content.lower()
        ]

    def get_posts_by_tag(self, tag: str) -> List[Post]:
        """按标签获取文章"""
        posts = self.get_all_posts()
        return [p for p in posts if tag in p.tags]

    def get_posts_by_category(self, category: str) -> List[Post]:
        """按分类获取文章"""
        posts = self.get_all_posts()
        return [p for p in posts if p.category == category]

    def update_post(self, slug: str, **kwargs) -> Optional[Post]:
        """更新文章"""
        post = self.get_post(slug)
        if not post:
            return None

        # 更新字段
        if 'title' in kwargs:
            post.title = kwargs['title']
        if 'content' in kwargs:
            post.content = kwargs['content']
        if 'tags' in kwargs:
            post.tags = kwargs['tags']
        if 'category' in kwargs:
            post.category = kwargs['category']

        post.updated_at = datetime.now()

        # 重新保存
        self._save_post(post)
        return post

    def delete_post(self, slug: str) -> bool:
        """删除文章"""
        filepath = os.path.join(self.posts_dir, f"{slug}.md")

        if os.path.exists(filepath):
            os.remove(filepath)
            return True

        return False

    def _save_post(self, post: Post):
        """保存文章到文件"""
        filename = f"{post.slug}.md"
        filepath = os.path.join(self.posts_dir, filename)

        metadata = f"""---
title: {post.title}
slug: {post.slug}
tags: {','.join(post.tags) if post.tags else ''}
category: {post.category or ''}
created_at: {post.created_at.isoformat()}
updated_at: {post.updated_at.isoformat()}
---

{post.content}"""

        with open(filepath, 'w', encoding='utf-8') as f:
            f.write(metadata)
```

### 3.1 标签统计

```python
def get_tag_cloud(self) -> List[dict]:
    """获取标签云"""
    posts = self.get_all_posts()

    tag_counts = {}
    for post in posts:
        for tag in post.tags:
            tag_counts[tag] = tag_counts.get(tag, 0) + 1

    return [
        {"name": tag, "count": count}
        for tag, count in sorted(tag_counts.items(), key=lambda x: x[1], reverse=True)
    ]
```

### 3.2 归档功能

```python
def get_archives(self) -> dict:
    """获取文章归档"""
    posts = self.get_all_posts()

    archives = {}
    for post in posts:
        year = post.created_at.year
        month = post.created_at.month

        if year not in archives:
            archives[year] = {}

        if month not in archives[year]:
            archives[year][month] = []

        archives[year][month].append({
            "title": post.title,
            "slug": post.slug,
            "day": post.created_at.day
        })

    return archives
```

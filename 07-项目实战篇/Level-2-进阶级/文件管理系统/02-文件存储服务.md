```sql
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(255) NOT NULL,
    path VARCHAR(1000) NOT NULL,
    parent_id INT,
    size BIGINT,
    mime_type VARCHAR(100),
    extension VARCHAR(20),
    hash VARCHAR(64),
    is_directory BOOLEAN DEFAULT FALSE,
    owner_id INT,
    created_at DATETIME,
    updated_at DATETIME,
    deleted_at DATETIME,
    INDEX idx_parent (parent_id),
    INDEX idx_owner (owner_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

```python
# models.py
from sqlalchemy import Column, Integer, String, BigInteger, Boolean, DateTime, ForeignKey, Text
from sqlalchemy.orm import relationship
from sqlalchemy.sql import func
from database import Base


class FileRecord(Base):
    """文件记录表"""
    __tablename__ = "files"

    id = Column(Integer, primary_key=True, autoincrement=True)
    name = Column(String(255), nullable=False)
    path = Column(String(1000), nullable=False, unique=True)
    parent_id = Column(Integer, ForeignKey("files.id"), nullable=True)
    size = Column(BigInteger, default=0)
    mime_type = Column(String(100))
    extension = Column(String(20))
    hash = Column(String(64))
    description = Column(Text)
    is_directory = Column(Boolean, default=False)
    is_hidden = Column(Boolean, default=False)
    is_starred = Column(Boolean, default=False)

    # 所有者
    owner_id = Column(Integer, ForeignKey("users.id"), nullable=True)

    # 时间
    created_at = Column(DateTime, default=func.now())
    updated_at = Column(DateTime, default=func.now(), onupdate=func.now())
    deleted_at = Column(DateTime, nullable=True)  # 软删除

    # 关系
    parent = relationship("FileRecord", remote_side=[id], backref="children")
    owner = relationship("User", backref="files")

    def __repr__(self):
        return f"<FileRecord {self.name}>"


class FileVersion(Base):
    """文件版本表"""
    __tablename__ = "file_versions"

    id = Column(Integer, primary_key=True, autoincrement=True)
    file_id = Column(Integer, ForeignKey("files.id"), nullable=False)
    version = Column(Integer, nullable=False)
    size = Column(BigInteger)
    hash = Column(String(64))
    created_at = Column(DateTime, default=func.now())
    created_by = Column(Integer, ForeignKey("users.id"))

    # 关系
    file = relationship("FileRecord", backref="versions")

    def __repr__(self):
        return f"<FileVersion {self.file_id} v{self.version}>"


class User(Base):
    """用户表"""
    __tablename__ = "users"

    id = Column(Integer, primary_key=True, autoincrement=True)
    username = Column(String(50), unique=True, nullable=False)
    email = Column(String(255), unique=True, nullable=False)
    password_hash = Column(String(255), nullable=False)
    created_at = Column(DateTime, default=func.now())

    def __repr__(self):
        return f"<User {self.username}>"
```

### 2.2 文件服务

```python
# services.py
from sqlalchemy.orm import Session
from sqlalchemy import and_, or_, func
from typing import List, Optional
from datetime import datetime
import os
import hashlib
from pathlib import Path

from models import FileRecord, FileVersion


class FileStorageService:
    """文件存储服务"""

    def __init__(self, db: Session):
        self.db = db

    def create_file(
        self,
        name: str,
        path: str,
        parent_id: int = None,
        size: int = 0,
        mime_type: str = None,
        owner_id: int = None,
        is_directory: bool = False
    ) -> FileRecord:
        """创建文件记录"""
        # 计算哈希
        file_hash = None
        if not is_directory and os.path.exists(path):
            file_hash = self.calculate_file_hash(path)

        # 获取扩展名
        extension = Path(name).suffix.lstrip('.') if not is_directory else None

        file_record = FileRecord(
            name=name,
            path=path,
            parent_id=parent_id,
            size=size,
            mime_type=mime_type,
            extension=extension,
            hash=file_hash,
            is_directory=is_directory,
            owner_id=owner_id
        )

        self.db.add(file_record)
        self.db.commit()
        self.db.refresh(file_record)
        return file_record

    def calculate_file_hash(self, path: str) -> str:
        """计算文件哈希"""
        hash_md5 = hashlib.md5()

        try:
            with open(path, 'rb') as f:
                for chunk in iter(lambda: f.read(4096), b''):
                    hash_md5.update(chunk)
            return hash_md5.hexdigest()
        except:
            return None

    def get_file(self, file_id: int) -> Optional[FileRecord]:
        """获取文件"""
        return self.db.query(FileRecord).filter(
            FileRecord.id == file_id,
            FileRecord.deleted_at == None
        ).first()

    def get_files_by_parent(self, parent_id: int = None) -> List[FileRecord]:
        """获取目录下的文件"""
        query = self.db.query(FileRecord).filter(
            FileRecord.deleted_at == None
        )

        if parent_id is None:
            query = query.filter(FileRecord.parent_id == None)
        else:
            query = query.filter(FileRecord.parent_id == parent_id)

        return query.order_by(
            FileRecord.is_directory.desc(),
            FileRecord.name.asc()
        ).all()

    def search_files(
        self,
        keyword: str = None,
        owner_id: int = None,
        file_type: str = None,
        limit: int = 100
    ) -> List[FileRecord]:
        """搜索文件"""
        query = self.db.query(FileRecord).filter(
            FileRecord.deleted_at == None
        )

        if keyword:
            query = query.filter(FileRecord.name.ilike(f"%{keyword}%"))

        if owner_id:
            query = query.filter(FileRecord.owner_id == owner_id)

        if file_type == "folder":
            query = query.filter(FileRecord.is_directory == True)
        elif file_type == "file":
            query = query.filter(FileRecord.is_directory == False)

        return query.order_by(FileRecord.updated_at.desc()).limit(limit).all()

    def update_file(self, file_id: int, **kwargs) -> Optional[FileRecord]:
        """更新文件"""
        file_record = self.get_file(file_id)
        if not file_record:
            return None

        for key, value in kwargs.items():
            if hasattr(file_record, key):
                setattr(file_record, key, value)

        self.db.commit()
        self.db.refresh(file_record)
        return file_record

    def delete_file(self, file_id: int, permanent: bool = False) -> bool:
        """删除文件"""
        file_record = self.get_file(file_id)
        if not file_record:
            return False

        if permanent:
            self.db.delete(file_record)
        else:
            file_record.deleted_at = datetime.now()

        self.db.commit()
        return True

    def restore_file(self, file_id: int) -> bool:
        """恢复文件"""
        file_record = self.db.query(FileRecord).filter(
            FileRecord.id == file_id
        ).first()

        if not file_record:
            return False

        file_record.deleted_at = None
        self.db.commit()
        return True

    def get_deleted_files(self, owner_id: int = None) -> List[FileRecord]:
        """获取回收站文件"""
        query = self.db.query(FileRecord).filter(
            FileRecord.deleted_at != None
        )

        if owner_id:
            query = query.filter(FileRecord.owner_id == owner_id)

        return query.order_by(FileRecord.deleted_at.desc()).all()

    def create_version(self, file_id: int, user_id: int) -> FileVersion:
        """创建文件版本"""
        file_record = self.get_file(file_id)
        if not file_record:
            return None

        # 获取最新版本号
        latest = self.db.query(FileVersion).filter(
            FileVersion.file_id == file_id
        ).order_by(FileVersion.version.desc()).first()

        version_num = (latest.version + 1) if latest else 1

        version = FileVersion(
            file_id=file_id,
            version=version_num,
            size=file_record.size,
            hash=file_record.hash,
            created_by=user_id
        )

        self.db.add(version)
        self.db.commit()
        self.db.refresh(version)
        return version


class FileStatisticsService:
    """文件统计服务"""

    def __init__(self, db: Session):
        self.db = db

    def get_storage_usage(self, owner_id: int = None) -> dict:
        """获取存储使用情况"""
        query = self.db.query(
            func.sum(FileRecord.size).label('total_size'),
            func.count(FileRecord.id).label('file_count')
        ).filter(
            FileRecord.deleted_at == None,
            FileRecord.is_directory == False
        )

        if owner_id:
            query = query.filter(FileRecord.owner_id == owner_id)

        result = query.first()

        return {
            "total_size": result.total_size or 0,
            "file_count": result.file_count or 0
        }

    def get_file_type_distribution(self, owner_id: int = None) -> dict:
        """获取文件类型分布"""
        query = self.db.query(
            FileRecord.extension,
            func.count(FileRecord.id).label('count'),
            func.sum(FileRecord.size).label('size')
        ).filter(
            FileRecord.deleted_at == None,
            FileRecord.is_directory == False
        )

        if owner_id:
            query = query.filter(FileRecord.owner_id == owner_id)

        results = query.group_by(FileRecord.extension).all()

        return {
            r.extension or "unknown": {
                "count": r.count,
                "size": r.size or 0
            }
            for r in results
        }
```

## 第三部分 - 扩展功能

### 3.1 收藏和隐藏

```python
def star_file(self, file_id: int) -> bool:
    """收藏文件"""
    return self.update_file(file_id, is_starred=True) is not None

def unstar_file(self, file_id: int) -> bool:
    """取消收藏"""
    return self.update_file(file_id, is_starred=False) is not None

def hide_file(self, file_id: int) -> bool:
    """隐藏文件"""
    return self.update_file(file_id, is_hidden=True) is not None
```

### 3.2 移动和复制

```python
def move_file(self, file_id: int, new_parent_id: int) -> bool:
    """移动文件"""
    file_record = self.get_file(file_id)
    if not file_record:
        return False

    # 检查目标目录
    if new_parent_id:
        target = self.get_file(new_parent_id)
        if not target or not target.is_directory:
            return False

    file_record.parent_id = new_parent_id
    self.db.commit()
    return True
```

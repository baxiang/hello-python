# 动态页面爬取

学习使用 Playwright 处理 JavaScript 渲染的动态页面。

---

## 1. 为什么需要 Playwright

```
┌─────────────────────────────────────────────────────────────┐
│                  静态页面 vs 动态页面                        │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│   静态页面：                                                 │
│   requests → HTML 内容完整 → BeautifulSoup 解析             │
│                                                             │
│   动态页面：                                                 │
│   requests → HTML 框架（无数据）→ 需要执行 JS               │
│                                                             │
│   Playwright：                                               │
│   启动浏览器 → 执行 JS → 获取完整 HTML                      │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

## 2. Playwright 安装

### 2.1 安装依赖

```bash
uv add playwright
playwright install chromium
```

### 2.2 基本使用

```python
from playwright.sync_api import sync_playwright

with sync_playwright() as p:
    # 启动浏览器
    browser = p.chromium.launch(headless=True)
    page = browser.new_page()
    
    # 访问页面
    page.goto("https://example.com")
    
    # 等待内容加载
    page.wait_for_selector(".content")
    
    # 获取内容
    content = page.content()
    print(content)
    
    browser.close()
```

## 3. 页面交互

### 3.1 点击和输入

```python
from playwright.sync_api import sync_playwright

with sync_playwright() as p:
    browser = p.chromium.launch(headless=False)  # 显示浏览器
    page = browser.new_page()
    
    page.goto("https://example.com/login")
    
    # 输入文本
    page.fill("#username", "alice")
    page.fill("#password", "123456")
    
    # 点击按钮
    page.click("button[type=submit]")
    
    # 等待导航完成
    page.wait_for_url("**/dashboard")
    
    browser.close()
```

### 3.2 等待策略

```python
# 等待元素出现
page.wait_for_selector(".item", state="visible")

# 等待元素消失（如加载动画）
page.wait_for_selector(".loading", state="hidden")

# 等待网络请求完成
page.wait_for_load_state("networkidle")

# 等待特定时间
page.wait_for_timeout(1000)  # 1 秒
```

### 3.3 滚动加载

```python
def scrape_infinite_scroll(page, selector: str, max_items: int = 100):
    """爬取无限滚动页面"""
    items = []
    
    while len(items) < max_items:
        # 获取当前项目
        new_items = page.query_selector_all(selector)
        items = new_items
        
        # 滚动到底部
        page.evaluate("window.scrollTo(0, document.body.scrollHeight)")
        
        # 等待新内容加载
        page.wait_for_timeout(1000)
        
        # 检查是否到底
        if len(new_items) == len(items):
            break
    
    return items[:max_items]
```

## 4. 数据提取

### 4.1 提取文本和属性

```python
# 提取单个元素文本
title = page.locator("h1").text_content()

# 提取属性
href = page.locator("a.link").get_attribute("href")

# 提取多个元素
items = page.locator(".item").all()
for item in items:
    title = item.locator(".title").text_content()
    price = item.locator(".price").text_content()
    print(f"{title}: {price}")
```

### 4.2 结合 BeautifulSoup

```python
from playwright.sync_api import sync_playwright
from bs4 import BeautifulSoup

with sync_playwright() as p:
    browser = p.chromium.launch()
    page = browser.new_page()
    
    page.goto("https://example.com")
    page.wait_for_load_state("networkidle")
    
    # 获取完整 HTML
    html = page.content()
    
    # 用 BeautifulSoup 解析
    soup = BeautifulSoup(html, "lxml")
    items = soup.select(".item")
    
    browser.close()
```

## 5. 异步 Playwright

```python
import asyncio
from playwright.async_api import async_playwright

async def scrape_multiple(urls: list):
    """并发爬取多个页面"""
    async with async_playwright() as p:
        browser = await p.chromium.launch()
        
        tasks = [scrape_page(browser, url) for url in urls]
        results = await asyncio.gather(*tasks)
        
        await browser.close()
        return results

async def scrape_page(browser, url: str):
    """爬取单个页面"""
    page = await browser.new_page()
    await page.goto(url)
    await page.wait_for_load_state("networkidle")
    
    title = await page.title()
    content = await page.content()
    
    await page.close()
    return {"url": url, "title": title, "content": content}

# 运行
urls = ["https://example.com/page1", "https://example.com/page2"]
results = asyncio.run(scrape_multiple(urls))
```

## 6. 实战示例

### 6.1 爬取 SPA 应用

```python
from playwright.sync_api import sync_playwright

def scrape_spa(url: str) -> list:
    """爬取单页应用"""
    with sync_playwright() as p:
        browser = p.chromium.launch()
        page = browser.new_page()
        
        # 拦截不必要的请求（加速加载）
        page.route("**/*.{png,jpg,jpeg,gif,svg,css}", lambda route: route.abort())
        
        page.goto(url)
        page.wait_for_selector(".data-list")
        
        # 点击加载更多
        while True:
            load_more = page.query_selector(".load-more")
            if not load_more:
                break
            
            load_more.click()
            page.wait_for_timeout(1000)
        
        # 提取数据
        items = page.query_selector_all(".item")
        data = []
        
        for item in items:
            data.append({
                "title": item.query_selector(".title").text_content(),
                "desc": item.query_selector(".desc").text_content()
            })
        
        browser.close()
        return data
```

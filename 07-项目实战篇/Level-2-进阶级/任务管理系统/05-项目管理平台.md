```python
# models.py 扩展
class Board(Base):
    """看板表"""
    __tablename__ = "boards"

    id = Column(Integer, primary_key=True, autoincrement=True)
    name = Column(String(100), nullable=False)
    project_id = Column(Integer, ForeignKey("projects.id"), nullable=False)
    created_at = Column(DateTime, default=func.now())

    # 关系
    project = relationship("Project", back_populates="boards")
    columns = relationship("BoardColumn", back_populates="board", cascade="all, delete-orphan")


class BoardColumn(Base):
    """看板列"""
    __tablename__ = "board_columns"

    id = Column(Integer, primary_key=True, autoincrement=True)
    board_id = Column(Integer, ForeignKey("boards.id", ondelete="CASCADE"), nullable=False)
    name = Column(String(50), nullable=False)
    status = Column(String(20), nullable=False)  # 对应 TaskStatus
    position = Column(Integer, default=0)  # 排序位置
    color = Column(String(7), default="#3498db")

    # 关系
    board = relationship("Board", back_populates="columns")

    def __repr__(self):
        return f"<BoardColumn {self.name}>"
```

### 1.2 看板服务

```python
# board_services.py
from sqlalchemy.orm import Session
from typing import List, Dict
import json

from models import Board, BoardColumn, Task, TaskStatus


class BoardService:
    """看板服务"""

    def __init__(self, db: Session):
        self.db = db

    def create_board(self, project_id: int, name: str) -> Board:
        """创建看板"""
        board = Board(name=name, project_id=project_id)
        self.db.add(board)
        self.db.flush()

        # 创建默认列
        default_columns = [
            ("待办", "pending", "#95a5a6"),
            ("进行中", "in_progress", "#3498db"),
            ("已完成", "completed", "#2ecc71"),
            ("已取消", "cancelled", "#e74c3c")
        ]

        for i, (col_name, status, color) in enumerate(default_columns):
            column = BoardColumn(
                board_id=board.id,
                name=col_name,
                status=status,
                position=i,
                color=color
            )
            self.db.add(column)

        self.db.commit()
        self.db.refresh(board)
        return board

    def get_board(self, board_id: int) -> Board:
        """获取看板"""
        return self.db.query(Board).filter(Board.id == board_id).first()

    def get_board_data(self, board_id: int) -> Dict:
        """获取看板数据（包含任务）"""
        board = self.get_board(board_id)
        if not board:
            return None

        # 获取列
        columns = self.db.query(BoardColumn).filter(
            BoardColumn.board_id == board_id
        ).order_by(BoardColumn.position).all()

        # 获取每个列的任务
        result = {
            "id": board.id,
            "name": board.name,
            "columns": []
        }

        for column in columns:
            tasks = self.db.query(Task).filter(
                Task.project_id == board.project_id,
                Task.status == TaskStatus[column.status.upper()]
            ).all()

            result["columns"].append({
                "id": column.id,
                "name": column.name,
                "status": column.status,
                "color": column.color,
                "position": column.position,
                "tasks": [
                    {
                        "id": task.id,
                        "title": task.title,
                        "priority": task.priority.value,
                        "assignee": {
                            "id": task.assignee.id,
                            "username": task.assignee.username,
                            "avatar": task.assignee.avatar
                        } if task.assignee else None,
                        "due_date": task.due_date.isoformat() if task.due_date else None
                    }
                    for task in tasks
                ]
            })

        return result

    def move_task(self, task_id: int, new_status: str) -> Task:
        """移动任务到其他列"""
        task = self.db.query(Task).filter(Task.id == task_id).first()
        if not task:
            raise ValueError("任务不存在")

        task.status = TaskStatus[new_status.upper()]
        self.db.commit()
        self.db.refresh(task)
        return task
```

```python
# gantt_services.py
from sqlalchemy.orm import Session
from typing import List, Dict
from datetime import datetime

from models import Task, Project


class GanttService:
    """甘特图服务"""

    def __init__(self, db: Session):
        self.db = db

    def get_project_gantt(self, project_id: int) -> List[Dict]:
        """获取项目甘特图数据"""
        tasks = self.db.query(Task).filter(
            Task.project_id == project_id,
            Task.due_date.isnot(None)
        ).order_by(Task.due_date).all()

        result = []
        for task in tasks:
            result.append({
                "id": task.id,
                "title": task.title,
                "start": task.created_at.isoformat() if task.created_at else task.due_date.isoformat(),
                "end": task.due_date.isoformat(),
                "progress": 100 if task.completed else 0,
                "status": task.status.value,
                "priority": task.priority.value,
                "assignee": task.assignee.username if task.assignee else None,
                "dependencies": []  # 可以扩展依赖关系
            })

        return result

    def get_timeline_view(self, project_id: int, start_date: datetime, end_date: datetime) -> List[Dict]:
        """获取时间线视图"""
        tasks = self.db.query(Task).filter(
            Task.project_id == project_id,
            Task.due_date >= start_date,
            Task.due_date <= end_date
        ).all()

        return [
            {
                "id": task.id,
                "title": task.title,
                "start": task.created_at.isoformat() if task.created_at else task.due_date.isoformat(),
                "end": task.due_date.isoformat(),
                "completed": task.completed
            }
            for task in tasks
        ]
```

## 第三部分 - 统计报表

### 3.1 数据统计

```python
# statistics_services.py
from sqlalchemy.orm import Session
from sqlalchemy import func, and_
from typing import Dict, List
from datetime import datetime, timedelta

from models import Task, Project, User, TaskStatus, Priority


class StatisticsService:
    """统计服务"""

    def __init__(self, db: Session):
        self.db = db

    def get_project_overview(self, project_id: int) -> Dict:
        """项目概览"""
        tasks = self.db.query(Task).filter(Task.project_id == project_id)

        total = tasks.count()
        completed = tasks.filter(Task.completed == True).count()
        in_progress = tasks.filter(Task.status == TaskStatus.IN_PROGRESS).count()
        pending = total - completed - in_progress

        # 逾期
        overdue = tasks.filter(
            and_(
                Task.due_date < datetime.now(),
                Task.completed == False
            )
        ).count()

        # 优先级分布
        priority_dist = {}
        for priority in Priority:
            count = tasks.filter(Task.priority == priority).count()
            priority_dist[priority.value] = count

        return {
            "total": total,
            "completed": completed,
            "in_progress": in_progress,
            "pending": pending,
            "overdue": overdue,
            "completion_rate": round(completed / total * 100, 1) if total > 0 else 0,
            "priority_distribution": priority_dist
        }

    def get_member_statistics(self, project_id: int) -> List[Dict]:
        """成员统计"""
        members = self.db.query(User).join(Task, Task.assignee_id == User.id).filter(
            Task.project_id == project_id
        ).distinct().all()

        result = []
        for member in members:
            tasks = self.db.query(Task).filter(
                Task.project_id == project_id,
                Task.assignee_id == member.id
            )

            total = tasks.count()
            completed = tasks.filter(Task.completed == True).count()

            result.append({
                "id": member.id,
                "username": member.username,
                "avatar": member.avatar,
                "total_tasks": total,
                "completed_tasks": completed,
                "completion_rate": round(completed / total * 100, 1) if total > 0 else 0
            })

        return result

    def get_burndown_data(self, project_id: int, days: int = 30) -> List[Dict]:
        """燃尽图数据"""
        end_date = datetime.now()
        start_date = end_date - timedelta(days=days)

        result = []
        current_date = start_date

        # 获取项目创建时的任务总数
        total_tasks = self.db.query(Task).filter(
            Task.project_id == project_id,
            Task.created_at <= end_date
        ).count()

        remaining = total_tasks

        while current_date <= end_date:
            # 当天完成的任务数
            completed_on_day = self.db.query(Task).filter(
                Task.project_id == project_id,
                Task.completed == True,
                Task.completed_at >= current_date,
                Task.completed_at < current_date + timedelta(days=1)
            ).count()

            remaining -= completed_on_day

            result.append({
                "date": current_date.strftime("%Y-%m-%d"),
                "remaining": remaining,
                "completed": total_tasks - remaining
            })

            current_date += timedelta(days=1)

        return result
```

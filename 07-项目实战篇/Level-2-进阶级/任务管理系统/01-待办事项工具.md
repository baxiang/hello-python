### 2.1 基础结构

```python
# task_manager.py
import json
import os
from datetime import datetime
from typing import List, Optional

DATA_FILE = "tasks.json"


class Task:
    """任务类"""

    def __init__(self, title: str, priority: str = "medium", task_id: int = None):
        self.id = task_id or self._generate_id()
        self.title = title
        self.completed = False
        self.priority = priority  # high, medium, low
        self.created_at = datetime.now().strftime("%Y-%m-%d %H:%M:%S")

    def _generate_id(self) -> int:
        """生成唯一 ID"""
        tasks = self.load_tasks()
        if not tasks:
            return 1
        return max(t["id"] for t in tasks) + 1

    def to_dict(self) -> dict:
        return {
            "id": self.id,
            "title": self.title,
            "completed": self.completed,
            "priority": self.priority,
            "created_at": self.created_at
        }

    @staticmethod
    def from_dict(data: dict) -> "Task":
        task = Task(data["title"], data["priority"], data["id"])
        task.completed = data["completed"]
        task.created_at = data["created_at"]
        return task


class TaskManager:
    """任务管理器"""

    def __init__(self, data_file: str = DATA_FILE):
        self.data_file = data_file

    def load_tasks(self) -> List[dict]:
        """加载任务列表"""
        if not os.path.exists(self.data_file):
            return []

        with open(self.data_file, "r", encoding="utf-8") as f:
            data = json.load(f)
            return data.get("tasks", [])

    def save_tasks(self, tasks: List[dict]):
        """保存任务列表"""
        data = {"tasks": tasks}

        with open(self.data_file, "w", encoding="utf-8") as f:
            json.dump(data, f, ensure_ascii=False, indent=2)

    def add_task(self, title: str, priority: str = "medium") -> Task:
        """添加任务"""
        tasks = self.load_tasks()
        task = Task(title, priority)
        tasks.append(task.to_dict())
        self.save_tasks(tasks)
        return task

    def list_tasks(self, show_completed: bool = True) -> List[dict]:
        """列出任务"""
        tasks = self.load_tasks()

        if not show_completed:
            tasks = [t for t in tasks if not t["completed"]]

        # 按优先级和创建时间排序
        priority_order = {"high": 0, "medium": 1, "low": 2}
        tasks.sort(key=lambda x: (priority_order[x["priority"]], x["id"]))

        return tasks

    def complete_task(self, task_id: int) -> bool:
        """完成任务"""
        tasks = self.load_tasks()

        for task in tasks:
            if task["id"] == task_id:
                task["completed"] = True
                self.save_tasks(tasks)
                return True

        return False

    def uncomplete_task(self, task_id: int) -> bool:
        """取消完成"""
        tasks = self.load_tasks()

        for task in tasks:
            if task["id"] == task_id:
                task["completed"] = False
                self.save_tasks(tasks)
                return True

        return False

    def delete_task(self, task_id: int) -> bool:
        """删除任务"""
        tasks = self.load_tasks()
        original_count = len(tasks)
        tasks = [t for t in tasks if t["id"] != task_id]

        if len(tasks) < original_count:
            self.save_tasks(tasks)
            return True

        return False

    def update_task(self, task_id: int, title: str = None, priority: str = None) -> bool:
        """更新任务"""
        tasks = self.load_tasks()

        for task in tasks:
            if task["id"] == task_id:
                if title:
                    task["title"] = title
                if priority:
                    task["priority"] = priority
                self.save_tasks(tasks)
                return True

        return False
```

### 2.2 命令行界面

```python
# cli.py
from task_manager import TaskManager


def print_tasks(tasks: list):
    """打印任务列表"""
    if not tasks:
        print("暂无任务")
        return

    print(f"{'ID':<5} {'优先级':<8} {'任务':<30} {'状态':<10}")
    print("-" * 60)

    for task in tasks:
        status = "✓ 完成" if task["completed"] else "○ 待办"
        priority_symbol = {"high": "🔴", "medium": "🟡", "low": "🟢"}
        symbol = priority_symbol.get(task["priority"], "")

        print(f"{task['id']:<5} {symbol}{task['priority']:<6} {task['title']:<30} {status:<10}")


def main():
    manager = TaskManager()

    while True:
        print("\n=== 任务管理器 ===")
        print("1. 查看任务列表")
        print("2. 添加任务")
        print("3. 完成任务")
        print("4. 取消完成")
        print("5. 删除任务")
        print("6. 更新任务")
        print("0. 退出")

        choice = input("\n请选择操作: ").strip()

        if choice == "1":
            tasks = manager.list_tasks(show_completed=False)
            print_tasks(tasks)

        elif choice == "2":
            title = input("请输入任务标题: ").strip()
            if title:
                priority = input("优先级 (high/medium/low): ").strip() or "medium"
                task = manager.add_task(title, priority)
                print(f"任务已添加: ID={task.id}")

        elif choice == "3":
            task_id = input("请输入任务 ID: ").strip()
            if task_id.isdigit():
                if manager.complete_task(int(task_id)):
                    print("任务已完成")
                else:
                    print("任务不存在")

        elif choice == "4":
            task_id = input("请输入任务 ID: ").strip()
            if task_id.isdigit():
                if manager.uncomplete_task(int(task_id)):
                    print("已取消完成")
                else:
                    print("任务不存在")

        elif choice == "5":
            task_id = input("请输入任务 ID: ").strip()
            if task_id.isdigit():
                if manager.delete_task(int(task_id)):
                    print("任务已删除")
                else:
                    print("任务不存在")

        elif choice == "6":
            task_id = input("请输入任务 ID: ").strip()
            if task_id.isdigit():
                title = input("新标题 (回车跳过): ").strip()
                priority = input("新优先级 (high/medium/low, 回车跳过): ").strip()
                if manager.update_task(int(task_id), title or None, priority or None):
                    print("任务已更新")
                else:
                    print("任务不存在")

        elif choice == "0":
            print("再见!")
            break

        else:
            print("无效选择")


if __name__ == "__main__":
    main()
```

def add_category(self, name: str) -> dict:
    """添加分类"""
    data = self._load_data()
    categories = data.get("categories", [])

    category_id = max([c["id"] for c in categories], default=0) + 1
    category = {
        "id": category_id,
        "name": name
    }
    categories.append(category)
    data["categories"] = categories
    self._save_data(data)
    return category


def add_task_with_category(self, title: str, category_id: int, priority: str = "medium") -> Task:
    """添加带分类的任务"""
    task = Task(title, priority)
    task.category_id = category_id

    tasks = self.load_tasks()
    tasks.append(task.to_dict())
    self.save_tasks(tasks)
    return task

### 3.2 统计功能

```python
def get_statistics(self) -> dict:
    """获取统计信息"""
    tasks = self.load_tasks()

    total = len(tasks)
    completed = sum(1 for t in tasks if t["completed"])
    pending = total - completed

    by_priority = {}
    for p in ["high", "medium", "low"]:
        by_priority[p] = sum(1 for t in tasks if t["priority"] == p)

    return {
        "total": total,
        "completed": completed,
        "pending": pending,
        "completion_rate": round(completed / total * 100, 1) if total > 0 else 0,
        "by_priority": by_priority
    }
```

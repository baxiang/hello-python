### 2.1 用户和团队模型

```python
# models.py
from sqlalchemy import Column, Integer, String, Boolean, DateTime, ForeignKey, Text, Enum
from sqlalchemy.orm import relationship
from sqlalchemy.sql import func
import enum

from database import Base


class UserRole(str, enum.Enum):
    """用户角色"""
    OWNER = "owner"      # 所有者
    ADMIN = "admin"      # 管理员
    MEMBER = "member"    # 成员


class Team(Base):
    """团队表"""
    __tablename__ = "teams"

    id = Column(Integer, primary_key=True, autoincrement=True)
    name = Column(String(100), nullable=False)
    description = Column(Text, nullable=True)
    owner_id = Column(Integer, ForeignKey("users.id"), nullable=False)
    created_at = Column(DateTime, default=func.now())
    updated_at = Column(DateTime, default=func.now(), onupdate=func.now())

    # 关系
    owner = relationship("User", back_populates="owned_teams")
    members = relationship("User", back_populates="team")
    projects = relationship("Project", back_populates="team")

    def __repr__(self):
        return f"<Team {self.name}>"


class User(Base):
    """用户表"""
    __tablename__ = "users"

    id = Column(Integer, primary_key=True, autoincrement=True)
    username = Column(String(50), unique=True, nullable=False)
    email = Column(String(255), unique=True, nullable=False)
    password_hash = Column(String(255), nullable=False)
    avatar = Column(String(500), nullable=True)
    team_id = Column(Integer, ForeignKey("teams.id"), nullable=True)
    role = Column(Enum(UserRole), default=UserRole.MEMBER)
    is_active = Column(Boolean, default=True)
    created_at = Column(DateTime, default=func.now())

    # 关系
    team = relationship("Team", back_populates="members")
    owned_teams = relationship("Team", back_populates="owner")
    assigned_tasks = relationship("Task", back_populates="assignee", foreign_keys="Task.assignee_id")
    created_tasks = relationship("Task", back_populates="creator", foreign_keys="Task.creator_id")
    comments = relationship("Comment", back_populates="user")
    watched_tasks = relationship("TaskWatcher", back_populates="user")

    def __repr__(self):
        return f"<User {self.username}>"


class Project(Base):
    """项目表"""
    __tablename__ = "projects"

    id = Column(Integer, primary_key=True, autoincrement=True)
    name = Column(String(100), nullable=False)
    description = Column(Text, nullable=True)
    team_id = Column(Integer, ForeignKey("teams.id"), nullable=False)
    status = Column(String(20), default="active")  # active, completed, archived
    start_date = Column(DateTime, nullable=True)
    end_date = Column(DateTime, nullable=True)
    created_at = Column(DateTime, default=func.now())

    # 关系
    team = relationship("Team", back_populates="projects")
    tasks = relationship("Task", back_populates="project")

    def __repr__(self):
        return f"<Project {self.name}>"


# 扩展 Task 模型
class Task(Base):
    """任务表 - 添加团队相关字段"""
    __tablename__ = "tasks"

    id = Column(Integer, primary_key=True, autoincrement=True)
    title = Column(String(200), nullable=False)
    description = Column(Text, nullable=True)
    completed = Column(Boolean, default=False)
    priority = Column(Enum(Priority), default=Priority.MEDIUM)
    status = Column(Enum(TaskStatus), default=TaskStatus.PENDING)
    due_date = Column(DateTime, nullable=True)
    completed_at = Column(DateTime, nullable=True)

    # 团队相关
    team_id = Column(Integer, ForeignKey("teams.id"), nullable=True)
    project_id = Column(Integer, ForeignKey("projects.id"), nullable=True)
    assignee_id = Column(Integer, ForeignKey("users.id"), nullable=True)
    creator_id = Column(Integer, ForeignKey("users.id"), nullable=False)

    created_at = Column(DateTime, default=func.now())
    updated_at = Column(DateTime, default=func.now(), onupdate=func.now())

    # 关系
    team = relationship("Team")
    project = relationship("Project", back_populates="tasks")
    assignee = relationship("User", back_populates="assigned_tasks", foreign_keys=[assignee_id])
    creator = relationship("User", back_populates="created_tasks", foreign_keys=[creator_id])
    comments = relationship("Comment", back_populates="task", cascade="all, delete-orphan")
    watchers = relationship("TaskWatcher", back_populates="task", cascade="all, delete-orphan")


class Comment(Base):
    """评论表"""
    __tablename__ = "comments"

    id = Column(Integer, primary_key=True, autoincrement=True)
    task_id = Column(Integer, ForeignKey("tasks.id", ondelete="CASCADE"), nullable=False)
    user_id = Column(Integer, ForeignKey("users.id"), nullable=False)
    content = Column(Text, nullable=False)
    created_at = Column(DateTime, default=func.now())
    updated_at = Column(DateTime, default=func.now(), onupdate=func.now())

    # 关系
    task = relationship("Task", back_populates="comments")
    user = relationship("User", back_populates="comments")

    def __repr__(self):
        return f"<Comment {self.id}>"


class TaskWatcher(Base):
    """任务关注者表"""
    __tablename__ = "task_watchers"

    id = Column(Integer, primary_key=True, autoincrement=True)
    task_id = Column(Integer, ForeignKey("tasks.id", ondelete="CASCADE"), nullable=False)
    user_id = Column(Integer, ForeignKey("users.id"), nullable=False)
    created_at = Column(DateTime, default=func.now())

    # 关系
    task = relationship("Task", back_populates="watchers")
    user = relationship("User", back_populates="watched_tasks")

    def __repr__(self):
        return f"<TaskWatcher task={self.task_id} user={self.user_id}>"
```

### 2.2 团队服务

```python
# team_services.py
from sqlalchemy.orm import Session
from typing import List, Optional
from datetime import datetime

from models import Team, User, Project, Task, Comment, TaskWatcher, UserRole
from utils.security import get_password_hash, verify_password


class TeamService:
    """团队服务"""

    def __init__(self, db: Session):
        self.db = db

    def create_team(self, name: str, description: str, owner_id: int) -> Team:
        """创建团队"""
        team = Team(
            name=name,
            description=description,
            owner_id=owner_id
        )
        self.db.add(team)
        self.db.flush()

        # 将创建者设为所有者
        user = self.db.query(User).filter(User.id == owner_id).first()
        user.team_id = team.id
        user.role = UserRole.OWNER

        self.db.commit()
        self.db.refresh(team)
        return team

    def add_member(self, team_id: int, user_id: int, role: UserRole = UserRole.MEMBER) -> User:
        """添加成员"""
        user = self.db.query(User).filter(User.id == user_id).first()
        if not user:
            raise ValueError("用户不存在")

        user.team_id = team_id
        user.role = role

        self.db.commit()
        self.db.refresh(user)
        return user

    def remove_member(self, team_id: int, user_id: int) -> bool:
        """移除成员"""
        user = self.db.query(User).filter(
            User.id == user_id,
            User.team_id == team_id
        ).first()

        if not user:
            return False

        user.team_id = None
        user.role = UserRole.MEMBER

        self.db.commit()
        return True

    def get_team_members(self, team_id: int) -> List[User]:
        """获取团队成员"""
        return self.db.query(User).filter(User.team_id == team_id).all()


class TaskCollaborationService:
    """任务协作服务"""

    def __init__(self, db: Session):
        self.db = db

    def assign_task(self, task_id: int, assignee_id: int) -> Task:
        """指派任务"""
        task = self.db.query(Task).filter(Task.id == task_id).first()
        if not task:
            raise ValueError("任务不存在")

        task.assignee_id = assignee_id
        self.db.commit()
        self.db.refresh(task)
        return task

    def add_comment(self, task_id: int, user_id: int, content: str) -> Comment:
        """添加评论"""
        comment = Comment(
            task_id=task_id,
            user_id=user_id,
            content=content
        )
        self.db.add(comment)
        self.db.commit()
        self.db.refresh(comment)
        return comment

    def get_comments(self, task_id: int) -> List[Comment]:
        """获取任务评论"""
        return self.db.query(Comment).filter(
            Comment.task_id == task_id
        ).order_by(Comment.created_at.desc()).all()

    def watch_task(self, task_id: int, user_id: int) -> TaskWatcher:
        """关注任务"""
        # 检查是否已关注
        existing = self.db.query(TaskWatcher).filter(
            TaskWatcher.task_id == task_id,
            TaskWatcher.user_id == user_id
        ).first()

        if existing:
            return existing

        watcher = TaskWatcher(task_id=task_id, user_id=user_id)
        self.db.add(watcher)
        self.db.commit()
        self.db.refresh(watcher)
        return watcher

    def unwatch_task(self, task_id: int, user_id: int) -> bool:
        """取消关注"""
        watcher = self.db.query(TaskWatcher).filter(
            TaskWatcher.task_id == task_id,
            TaskWatcher.user_id == user_id
        ).first()

        if not watcher:
            return False

        self.db.delete(watcher)
        self.db.commit()
        return True

    def get_watchers(self, task_id: int) -> List[User]:
        """获取任务关注者"""
        watchers = self.db.query(TaskWatcher).filter(
            TaskWatcher.task_id == task_id
        ).all()
        return [w.user for w in watchers]
```

### 3.1 权限装饰器

```python
# permissions.py
from functools import wraps
from fastapi import HTTPException, Depends
from sqlalchemy.orm import Session

from database import get_db
from models import User, Team, UserRole


def check_team_permission(required_roles: List[UserRole] = None):
    """检查团队权限"""
    if required_roles is None:
        required_roles = [UserRole.OWNER, UserRole.ADMIN, UserRole.MEMBER]

    def decorator(func):
        @wraps(func)
        async def wrapper(team_id: int, current_user: User = Depends(get_current_user), db: Session = Depends(get_db)):
            # 检查用户是否属于团队
            if current_user.team_id != team_id:
                raise HTTPException(status_code=403, detail="无权限访问")

            # 检查角色
            if current_user.role not in required_roles:
                raise HTTPException(status_code=403, detail="权限不足")

            return await func(team_id, current_user, db)

        return wrapper

    return decorator


def check_task_owner(func):
    """检查任务所有权"""
    @wraps(func)
    async def wrapper(task_id: int, current_user: User = Depends(get_current_user), db: Session = Depends(get_db)):
        from models import Task
        task = db.query(Task).filter(Task.id == task_id).first()

        if not task:
            raise HTTPException(status_code=404, detail="任务不存在")

        # 创建人、负责人或有管理员权限可以操作
        if task.creator_id != current_user.id and \
           task.assignee_id != current_user.id and \
           current_user.role not in [UserRole.OWNER, UserRole.ADMIN]:
            raise HTTPException(status_code=403, detail="无权限操作")

        return await func(task_id, current_user, db)

    return wrapper
```

### 3.2 任务协作 API

```python
# collaboration_router.py
from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy.orm import Session
from typing import List

from database import get_db
from models import User, Task, Comment, TaskWatcher
from schemas import CommentCreate, CommentResponse
from services.team_services import TaskCollaborationService
from dependencies.auth import get_current_user

router = APIRouter(prefix="/tasks/{task_id}", tags=["任务协作"])


@router.post("/assign")
def assign_task(
    task_id: int,
    assignee_id: int,
    db: Session = Depends(get_db),
    current_user: User = Depends(get_current_user)
):
    """指派任务"""
    service = TaskCollaborationService(db)
    task = service.assign_task(task_id, assignee_id)
    return task


@router.post("/comments", response_model=CommentResponse)
def add_comment(
    task_id: int,
    data: CommentCreate,
    db: Session = Depends(get_db),
    current_user: User = Depends(get_current_user)
):
    """添加评论"""
    service = TaskCollaborationService(db)
    comment = service.add_comment(task_id, current_user.id, data.content)
    return comment


@router.get("/comments", response_model=List[CommentResponse])
def get_comments(
    task_id: int,
    db: Session = Depends(get_db)
):
    """获取评论列表"""
    service = TaskCollaborationService(db)
    return service.get_comments(task_id)


@router.post("/watch")
def watch_task(
    task_id: int,
    db: Session = Depends(get_db),
    current_user: User = Depends(get_current_user)
):
    """关注任务"""
    service = TaskCollaborationService(db)
    service.watch_task(task_id, current_user.id)
    return {"message": "关注成功"}


@router.delete("/watch")
def unwatch_task(
    task_id: int,
    db: Session = Depends(get_db),
    current_user: User = Depends(get_current_user)
):
    """取消关注"""
    service = TaskCollaborationService(db)
    service.unwatch_task(task_id, current_user.id)
    return {"message": "取消关注成功"}


@router.get("/watchers")
def get_watchers(
    task_id: int,
    db: Session = Depends(get_db)
):
    """获取关注者列表"""
    service = TaskCollaborationService(db)
    watchers = service.get_watchers(task_id)
    return [{"id": w.id, "username": w.username} for w in watchers]
```

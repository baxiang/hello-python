### 2.2 数据模型

```python
# models.py
from sqlalchemy import create_engine, Column, Integer, String, Boolean, DateTime, ForeignKey, Text, Enum
from sqlalchemy.orm import relationship, sessionmaker, declarative_base
from datetime import datetime
import enum

Base = declarative_base()


class Priority(enum.Enum):
    """优先级枚举"""
    HIGH = "high"
    MEDIUM = "medium"
    LOW = "low"


class TaskStatus(enum.Enum):
    """任务状态"""
    PENDING = "pending"
    IN_PROGRESS = "in_progress"
    COMPLETED = "completed"
    CANCELLED = "cancelled"


class Category(Base):
    """分类表"""
    __tablename__ = "categories"

    id = Column(Integer, primary_key=True, autoincrement=True)
    name = Column(String(50), nullable=False, unique=True)
    color = Column(String(7), default="#3498db")  # 颜色代码
    created_at = Column(DateTime, default=datetime.now)

    # 关系
    tasks = relationship("Task", back_populates="category")

    def __repr__(self):
        return f"<Category {self.name}>"


class Task(Base):
    """任务表"""
    __tablename__ = "tasks"

    id = Column(Integer, primary_key=True, autoincrement=True)
    title = Column(String(200), nullable=False)
    description = Column(Text, nullable=True)
    completed = Column(Boolean, default=False)
    priority = Column(Enum(Priority), default=Priority.MEDIUM)
    status = Column(Enum(TaskStatus), default=TaskStatus.PENDING)
    due_date = Column(DateTime, nullable=True)
    completed_at = Column(DateTime, nullable=True)

    # 外键
    category_id = Column(Integer, ForeignKey("categories.id"), nullable=True)

    # 时间戳
    created_at = Column(DateTime, default=datetime.now)
    updated_at = Column(DateTime, default=datetime.now, onupdate=datetime.now)

    # 关系
    category = relationship("Category", back_populates="tasks")
    tags = relationship("TaskTag", back_populates="task", cascade="all, delete-orphan")

    def __repr__(self):
        return f"<Task {self.id}: {self.title}>"


class Tag(Base):
    """标签表"""
    __tablename__ = "tags"

    id = Column(Integer, primary_key=True, autoincrement=True)
    name = Column(String(50), nullable=False, unique=True)
    color = Column(String(7), default="#95a5a6")
    created_at = Column(DateTime, default=datetime.now)

    def __repr__(self):
        return f"<Tag {self.name}>"


class TaskTag(Base):
    """任务标签关联表"""
    __tablename__ = "task_tags"

    id = Column(Integer, primary_key=True, autoincrement=True)
    task_id = Column(Integer, ForeignKey("tasks.id", ondelete="CASCADE"))
    tag_id = Column(Integer, ForeignKey("tags.id", ondelete="CASCADE"))

    # 关系
    task = relationship("Task", back_populates="tags")
    tag = relationship("Tag")

    def __repr__(self):
        return f"<TaskTag task={self.task_id} tag={self.tag_id}>"
```

### 2.3 数据库服务

```python
# database.py
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker
from models import Base

DATABASE_URL = "sqlite:///tasks.db"

engine = create_engine(
    DATABASE_URL,
    connect_args={"check_same_thread": False}
)

SessionLocal = sessionmaker(autocommit=False, autoflush=False, bind=engine)


def init_db():
    """初始化数据库"""
    Base.metadata.create_all(bind=engine)


def get_db():
    """获取数据库会话"""
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()
```

### 2.4 服务层

```python
# services.py
from sqlalchemy.orm import Session
from sqlalchemy import and_, or_, func
from typing import List, Optional
from datetime import datetime, timedelta

from models import Task, Category, Tag, TaskTag, Priority, TaskStatus


class TaskService:
    """任务服务"""

    def __init__(self, db: Session):
        self.db = db

    def create_task(
        self,
        title: str,
        description: str = None,
        priority: Priority = Priority.MEDIUM,
        category_id: int = None,
        due_date: datetime = None,
        tag_ids: List[int] = None
    ) -> Task:
        """创建任务"""
        task = Task(
            title=title,
            description=description,
            priority=priority,
            category_id=category_id,
            due_date=due_date
        )

        self.db.add(task)
        self.db.flush()  # 获取 ID

        # 添加标签
        if tag_ids:
            for tag_id in tag_ids:
                task_tag = TaskTag(task_id=task.id, tag_id=tag_id)
                self.db.add(task_tag)

        self.db.commit()
        self.db.refresh(task)
        return task

    def get_task(self, task_id: int) -> Optional[Task]:
        """获取任务"""
        return self.db.query(Task).filter(Task.id == task_id).first()

    def get_tasks(
        self,
        category_id: int = None,
        priority: Priority = None,
        status: TaskStatus = None,
        completed: bool = None,
        keyword: str = None,
        tag_id: int = None,
        due_date_before: datetime = None,
        due_date_after: datetime = None,
        skip: int = 0,
        limit: int = 20,
        order_by: str = "created_at",
        desc: bool = True
    ) -> tuple[List[Task], int]:
        """获取任务列表（高级查询）"""
        query = self.db.query(Task)

        # 条件过滤
        if category_id:
            query = query.filter(Task.category_id == category_id)

        if priority:
            query = query.filter(Task.priority == priority)

        if status:
            query = query.filter(Task.status == status)

        if completed is not None:
            query = query.filter(Task.completed == completed)

        if keyword:
            query = query.filter(
                or_(
                    Task.title.ilike(f"%{keyword}%"),
                    Task.description.ilike(f"%{keyword}%")
                )
            )

        if tag_id:
            query = query.join(TaskTag).filter(TaskTag.tag_id == tag_id)

        if due_date_before:
            query = query.filter(Task.due_date <= due_date_before)

        if due_date_after:
            query = query.filter(Task.due_date >= due_date_after)

        # 统计总数
        total = query.count()

        # 排序
        order_column = getattr(Task, order_by, Task.created_at)
        if desc:
            query = query.order_by(order_column.desc())
        else:
            query = query.order_by(order_column.asc())

        # 分页
        tasks = query.offset(skip).limit(limit).all()

        return tasks, total

    def update_task(
        self,
        task_id: int,
        title: str = None,
        description: str = None,
        priority: Priority = None,
        category_id: int = None,
        due_date: datetime = None,
        tag_ids: List[int] = None
    ) -> Optional[Task]:
        """更新任务"""
        task = self.get_task(task_id)
        if not task:
            return None

        # 更新字段
        if title is not None:
            task.title = title
        if description is not None:
            task.description = description
        if priority is not None:
            task.priority = priority
        if category_id is not None:
            task.category_id = category_id
        if due_date is not None:
            task.due_date = due_date

        # 更新标签
        if tag_ids is not None:
            # 删除旧标签
            self.db.query(TaskTag).filter(TaskTag.task_id == task_id).delete()
            # 添加新标签
            for tag_id in tag_ids:
                task_tag = TaskTag(task_id=task_id, tag_id=tag_id)
                self.db.add(task_tag)

        self.db.commit()
        self.db.refresh(task)
        return task

    def complete_task(self, task_id: int) -> Optional[Task]:
        """完成任务"""
        task = self.get_task(task_id)
        if not task:
            return None

        task.completed = True
        task.status = TaskStatus.COMPLETED
        task.completed_at = datetime.now()

        self.db.commit()
        self.db.refresh(task)
        return task

    def delete_task(self, task_id: int) -> bool:
        """删除任务"""
        task = self.get_task(task_id)
        if not task:
            return False

        self.db.delete(task)
        self.db.commit()
        return True

    def get_overdue_tasks(self) -> List[Task]:
        """获取逾期任务"""
        return self.db.query(Task).filter(
            and_(
                Task.due_date < datetime.now(),
                Task.completed == False
            )
        ).all()

    def get_today_tasks(self) -> List[Task]:
        """获取今日任务"""
        today = datetime.now().date()
        tomorrow = today + timedelta(days=1)

        return self.db.query(Task).filter(
            and_(
                Task.due_date >= today,
                Task.due_date < tomorrow,
                Task.completed == False
            )
        ).all()


class CategoryService:
    """分类服务"""

    def __init__(self, db: Session):
        self.db = db

    def create_category(self, name: str, color: str = "#3498db") -> Category:
        """创建分类"""
        category = Category(name=name, color=color)
        self.db.add(category)
        self.db.commit()
        self.db.refresh(category)
        return category

    def get_categories(self) -> List[Category]:
        """获取所有分类"""
        return self.db.query(Category).all()

    def get_category_tasks(self, category_id: int) -> List[Task]:
        """获取分类下的任务"""
        return self.db.query(Task).filter(Task.category_id == category_id).all()


class TagService:
    """标签服务"""

    def __init__(self, db: Session):
        self.db = db

    def create_tag(self, name: str, color: str = "#95a5a6") -> Tag:
        """创建标签"""
        tag = Tag(name=name, color=color)
        self.db.add(tag)
        self.db.commit()
        self.db.refresh(tag)
        return tag

    def get_tags(self) -> List[Tag]:
        """获取所有标签"""
        return self.db.query(Tag).all()
```

### 3.1 统计功能

```python
def get_statistics(self, category_id: int = None) -> dict:
    """获取统计信息"""
    query = self.db.query(Task)

    if category_id:
        query = query.filter(Task.category_id == category_id)

    total = query.count()
    completed = query.filter(Task.completed == True).count()
    pending = total - completed

    # 按优先级统计
    priority_stats = {}
    for priority in Priority:
        count = query.filter(Task.priority == priority).count()
        priority_stats[priority.value] = count

    # 逾期任务
    overdue = query.filter(
        and_(
            Task.due_date < datetime.now(),
            Task.completed == False
        )
    ).count()

    return {
        "total": total,
        "completed": completed,
        "pending": pending,
        "completion_rate": round(completed / total * 100, 1) if total > 0 else 0,
        "priority_stats": priority_stats,
        "overdue": overdue
    }
```

### 3.2 批量操作

```python
def batch_complete(self, task_ids: List[int]) -> int:
    """批量完成任务"""
    count = self.db.query(Task).filter(
        Task.id.in_(task_ids)
    ).update({
        "completed": True,
        "status": TaskStatus.COMPLETED,
        "completed_at": datetime.now()
    }, synchronize_session=False)

    self.db.commit()
    return count

def batch_delete(self, task_ids: List[int]) -> int:
    """批量删除任务"""
    count = self.db.query(Task).filter(
        Task.id.in_(task_ids)
    ).delete(synchronize_session=False)

    self.db.commit()
    return count

def batch_move_category(self, task_ids: List[int], category_id: int) -> int:
    """批量移动分类"""
    count = self.db.query(Task).filter(
        Task.id.in_(task_ids)
    ).update({"category_id": category_id}, synchronize_session=False)

    self.db.commit()
    return count
```

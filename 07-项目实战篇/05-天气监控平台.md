## 第三部分 - 监控告警

### 3.1 告警规则

```python
# alert_rules.py
from dataclasses import dataclass
from typing import Optional, Callable
from datetime import datetime

@dataclass
class AlertRule:
    name: str
    condition: Callable[[dict], bool]
    message: str
    level: str = "warning"  # info, warning, critical

# 预定义规则
ALERT_RULES = [
    AlertRule(
        name="高温告警",
        condition=lambda w: w.get('high_temp', 0) > 35,
        message="高温预警：{city} 今日最高温度 {high_temp}°C",
        level="warning"
    ),
    AlertRule(
        name="低温告警",
        condition=lambda w: w.get('low_temp', 100) < 0,
        message="低温预警：{city} 今日最低温度 {low_temp}°C",
        level="warning"
    ),
    AlertRule(
        name="极端温差",
        condition=lambda w: (w.get('high_temp', 0) - w.get('low_temp', 0)) > 15,
        message="温差过大：{city} 今日温差 {diff}°C",
        level="info"
    ),
]

def check_alerts(weather_data: dict) -> list:
    """检查告警"""
    alerts = []
    for rule in ALERT_RULES:
        if rule.condition(weather_data):
            message = rule.message.format(
                city=weather_data.get('city', '未知'),
                high_temp=weather_data.get('high_temp'),
                low_temp=weather_data.get('low_temp'),
                diff=weather_data.get('high_temp', 0) - weather_data.get('low_temp', 0)
            )
            alerts.append({
                "rule": rule.name,
                "message": message,
                "level": rule.level,
                "time": datetime.now().isoformat()
            })
    return alerts
```

### 3.2 定时任务

```python
# scheduler.py
from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.triggers.cron import CronTrigger
import logging

scheduler = BackgroundScheduler()

def check_weather_alerts():
    """定时检查天气告警"""
    from database import get_session
    from repository import WeatherRepository
    
    session = get_session()
    repo = WeatherRepository(session)
    
    # 获取所有城市
    cities = session.query(City).all()
    
    for city in cities:
        latest = repo.get_latest_record(city.id)
        if latest:
            weather_data = {
                "city": city.city_name,
                "high_temp": latest.high_temp,
                "low_temp": latest.low_temp,
                "weather_condition": latest.weather_condition
            }
            alerts = check_alerts(weather_data)
            
            for alert in alerts:
                logging.warning(f"告警：{alert['message']}")
                # 发送邮件/微信通知
                send_notification(alert)

# 每天早上 8 点检查
scheduler.add_job(
    check_weather_alerts,
    CronTrigger(hour=8, minute=0),
    id='daily_alert_check'
)

scheduler.start()
```

## 第四部分 - Docker 部署

### 4.1 Dockerfile

```dockerfile
# Dockerfile
FROM python:3.11-slim

WORKDIR /app

# 安装 uv
COPY --from=ghcr.io/astral-sh/uv:latest /uv /usr/local/bin/uv

# 复制项目文件
COPY pyproject.toml uv.lock ./
COPY . .

# 安装依赖
RUN uv sync --frozen --no-dev

# 设置环境变量
ENV PATH="/app/.venv/bin:$PATH"

# 暴露端口
EXPOSE 8000

# 启动命令
CMD ["uv", "run", "uvicorn", "main:app", "--host", "0.0.0.0", "--port", "8000"]
```

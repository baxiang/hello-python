## 第一部分 - pytest 基础

### 1.1 安装与配置

```bash
# 安装 pytest
uv add --dev pytest pytest-cov

# 运行测试
pytest

# 运行特定测试
pytest tests/test_module.py
pytest tests/test_module.py::test_function

# 显示覆盖率
pytest --cov=src/my_package --cov-report=html

# 运行匹配模式的测试
pytest -k "test_user"

# 显示详细输出
pytest -v
```

### 1.2 编写测试

```python
# tests/test_calculator.py
import pytest
from src.my_package.calculator import add, divide

def test_add():
    """测试加法函数"""
    assert add(2, 3) == 5
    assert add(-1, 1) == 0
    assert add(0, 0) == 0

def test_add_strings():
    """测试字符串拼接"""
    assert add("hello", "world") == "helloworld"

def test_divide():
    """测试除法函数"""
    assert divide(10, 2) == 5
    assert divide(7, 2) == 3.5

def test_divide_by_zero():
    """测试除以零异常"""
    with pytest.raises(ZeroDivisionError):
        divide(10, 0)
```

### 1.3 参数化测试

```python
import pytest
from src.my_package.calculator import multiply

@pytest.mark.parametrize("a,b,expected", [
    (2, 3, 6),
    (5, 5, 25),
    (0, 10, 0),
    (-2, 3, -6),
    (2.5, 4, 10.0),
])
def test_multiply(a, b, expected):
    """参数化测试乘法"""
    assert multiply(a, b) == expected
```

## 第二部分 - 测试夹具 (Fixtures)

### 2.1 基础 Fixture

```python
# tests/conftest.py
import pytest
from src.my_package.database import Database

@pytest.fixture
def db():
    """创建数据库连接"""
    db = Database(":memory:")
    yield db
    db.close()

@pytest.fixture
def sample_user():
    """示例用户数据"""
    return {
        "id": 1,
        "name": "Alice",
        "email": "alice@example.com"
    }
```

```python
# tests/test_database.py
import pytest
from src.my_package.user_service import UserService

def test_create_user(db, sample_user):
    """测试创建用户"""
    service = UserService(db)
    user = service.create(sample_user)
    assert user["name"] == "Alice"
    assert user["id"] == 1

def test_get_user(db, sample_user):
    """测试获取用户"""
    service = UserService(db)
    service.create(sample_user)
    retrieved = service.get_by_id(1)
    assert retrieved["email"] == "alice@example.com"
```

## 第三部分 - Mock 与补丁

### 3.1 使用 unittest.mock

```python
from unittest.mock import Mock, patch, MagicMock
import pytest
from src.my_package.api_client import APIClient, WeatherService

def test_api_client():
    """测试 API 客户端"""
    mock_response = Mock()
    mock_response.status_code = 200
    mock_response.json.return_value = {"temp": 25}
    
    with patch('requests.get', return_value=mock_response):
        client = APIClient()
        result = client.get_weather("Beijing")
        assert result["temp"] == 25

def test_weather_service():
    """测试天气服务"""
    # 创建 mock 对象
    mock_client = Mock(spec=APIClient)
    mock_client.get_weather.return_value = {"temp": 30, "city": "Shanghai"}
    
    service = WeatherService(mock_client)
    result = service.get_today_weather("Shanghai")
    
    assert result["temp"] == 30
    mock_client.get_weather.assert_called_once_with("Shanghai")
```

### 3.2 补丁装饰器

```python
import pytest
from unittest.mock import patch
from src.my_package.payment import PaymentProcessor

class TestPaymentProcessor:
    
    @patch('src.my_package.payment.stripe.Charge.create')
    def test_process_payment(self, mock_create):
        """测试支付处理"""
        mock_create.return_value = {"id": "ch_123", "status": "succeeded"}
        
        processor = PaymentProcessor()
        result = processor.process(100.00, "user_123")
        
        assert result["status"] == "succeeded"
        mock_create.assert_called_once()
    
    @patch('src.my_package.payment.send_email')
    @patch('src.my_package.payment.log_transaction')
    def test_process_with_notifications(self, mock_log, mock_email):
        """测试带通知的支付"""
        processor = PaymentProcessor()
        processor.process_with_notifications(100.00, "user_123")
        
        mock_log.assert_called_once()
        mock_email.assert_called_once()
```

## 第四部分 - 测试覆盖率

### 4.1 配置覆盖率

```toml
# pyproject.toml
[tool.coverage.run]
source = ["src/my_package"]
omit = [
    "*/tests/*",
    "*/__init__.py",
    "*/__main__.py",
]

[tool.coverage.report]
exclude_lines = [
    "pragma: no cover",
    "def __repr__",
    "raise AssertionError",
    "raise NotImplementedError",
    "if __name__ == .__main__.:",
    "if TYPE_CHECKING:",
    "@abstractmethod",
]

[tool.coverage.html]
directory = "htmlcov"
```

### 4.2 运行覆盖率报告

```bash
# 运行并生成 HTML 报告
pytest --cov=src/my_package --cov-report=html

# 在浏览器中打开报告
open htmlcov/index.html  # macOS
xdg-open htmlcov/index.html  # Linux

# 生成终端报告
pytest --cov=src/my_package --cov-report=term-missing

# 生成 XML 报告（用于 CI）
pytest --cov=src/my_package --cov-report=xml
```

## 第五部分 - 测试最佳实践

### 5.1 测试命名

```python
# ✅ 清晰的测试名
def test_create_user_with_valid_email():
    pass

def test_login_fails_with_invalid_password():
    pass

def test_calculate_total_includes_tax():
    pass

# ❌ 模糊的测试名
def test_user():
    pass

def test_stuff():
    pass
```

### 5.2 AAA 模式

```python
def test_withdraw_money():
    """使用 Arrange-Act-Assert 模式"""
    # Arrange - 准备
    account = BankAccount(balance=100)
    
    # Act - 执行
    account.withdraw(50)
    
    # Assert - 断言
    assert account.balance == 50
```

### 5.3 测试隔离

```python
# ✅ 每个测试独立
def test_create():
    db = create_test_db()
    # 测试创建

def test_update():
    db = create_test_db()
    # 测试更新

# ❌ 测试间有依赖
def test_create():
    global db
    db = create_test_db()
    
def test_update():
    # 依赖 test_create 创建的 db
    update(db)
```

### 5.4 测试数据工厂

```python
# tests/factories.py

from typing import Optional
import faker

fake = faker.Faker()
```

## 第六部分 - CI 集成

### 6.1 GitHub Actions 配置

```yaml
# .github/workflows/test.yml
name: Tests

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  test:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        python-version: ["3.11", "3.12"]
    
    steps:
      - uses: actions/checkout@v4
      
      - name: Install uv
        uses: astral-sh/setup-uv@v3
      
      - name: Install dependencies
        run: uv sync --dev
      
      - name: Run tests
        run: uv run pytest --cov=src/my_package --cov-report=xml
      
      - name: Upload coverage
        uses: codecov/codecov-action@v3
        with:
          files: ./coverage.xml
```

## 最佳实践清单

### ✅ 必须遵守

1. 为所有公共 API 编写测试
2. 保持测试独立，可并行运行
3. 使用有意义的测试名称
4. 测试正常路径和异常路径

### ✅ 推荐做法

1. 使用 pytest 而非 unittest
2. 使用 fixture 管理测试资源
3. 保持测试简洁（< 20 行）
4. 使用参数化减少重复
5. 在 CI 中运行测试并检查覆盖率

### ❌ 避免

1. 测试之间相互依赖
2. 测试真实的外部服务
3. 过于复杂的测试逻辑
4. 只测试 happy path

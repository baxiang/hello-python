```python
        total += i * i
    return total

def main():

        slow_function()

# 命令行分析
# python -m cProfile -o output.prof script.py
```

## 第二部分 - 代码优化技巧

### 2.1 使用内置函数和库

```python
# ❌ 慢：手动循环
result = []
for i in range(1000):
    result.append(i * 2)

# ✅ 快：列表推导式
result = [i * 2 for i in range(1000)]

# ✅ 更快：map
result = list(map(lambda x: x * 2, range(1000)))

# ❌ 慢：手动求和
total = 0
for i in range(1000):
    total += i

# ✅ 快：内置 sum
total = sum(range(1000))

# ❌ 慢：手动查找
min_val = data[0]
for x in data:
    if x < min_val:
        min_val = x

# ✅ 快：内置 min
min_val = min(data)
```

### 2.2 使用生成器

```python
# ❌ 占用大量内存
def get_squares(n):
    result = []
    for i in range(n):
        result.append(i * i)
    return result

# ✅ 使用生成器，节省内存
def get_squares_gen(n):
    for i in range(n):
        yield i * i

# 使用
for square in get_squares_gen(1000000):
    process(square)

# 生成器表达式
squares = (i * i for i in range(1000000))
```

### 2.3 使用局部变量

```python
# ❌ 慢：频繁访问全局变量
count = 0

def increment_slow(n):
    global count
    for i in range(n):
        count += 1

# ✅ 快：使用局部变量
def increment_fast(n):
    local_count = 0
    for i in range(n):
        local_count += 1
    return local_count
```

### 2.4 字符串拼接优化

```python
# ❌ 慢：O(n²) 复杂度
result = ""
for i in range(10000):
    result += str(i)

# ✅ 快：O(n) 复杂度
result = "".join(str(i) for i in range(10000))

# ✅ 或使用列表
parts = []
for i in range(10000):
    parts.append(str(i))
result = "".join(parts)
```

### 2.5 使用集合进行查找

```python
# ❌ 慢：列表查找 O(n)
items = [1, 2, 3, 4, 5]
if 3 in items:  # 需要遍历列表
    pass

# ✅ 快：集合查找 O(1)
items_set = {1, 2, 3, 4, 5}
if 3 in items_set:  # 哈希查找
    pass

# 转换
items_list = [1, 2, 3, 4, 5]
items_set = set(items_list)
```

## 第三部分 - 数据结构优化

### 3.1 选择合适的数据结构

```python
from collections import deque, defaultdict, Counter

# 频繁在两端添加/删除：使用 deque
queue = deque()
queue.append(1)
queue.appendleft(0)
queue.pop()
queue.popleft()

# 默认值：使用 defaultdict
d = defaultdict(list)
d['key'].append(1)  # 无需检查 key 是否存在

# 计数：使用 Counter
counter = Counter(['a', 'b', 'a', 'c', 'b', 'a'])
print(counter['a'])  # 3
print(counter.most_common(1))  # [('a', 3)]
```

### 3.2 使用 __slots__

```python
# ❌ 占用更多内存
class Person:
    def __init__(self, name, age):
        self.name = name
        self.age = age

# ✅ 节省内存
class Person:
    __slots__ = ['name', 'age']
    
    def __init__(self, name, age):
        self.name = name
        self.age = age

# 内存对比
# 普通类：~200 字节/实例
# __slots__: ~50 字节/实例
```

### 3.3 缓存优化

```python
from functools import lru_cache

# ❌ 重复计算
def fibonacci(n):
    if n < 2:
        return n
    return fibonacci(n-1) + fibonacci(n-2)

# ✅ 使用缓存
@lru_cache(maxsize=None)
def fibonacci_cached(n):
    if n < 2:
        return n
    return fibonacci_cached(n-1) + fibonacci_cached(n-2)

# 手动缓存
cache = {}
def fibonacci_manual(n):
    if n in cache:
        return cache[n]
    if n < 2:
        return n
    result = fibonacci_manual(n-1) + fibonacci_manual(n-2)
    cache[n] = result
    return result
```

## 第四部分 - 并发优化

### 4.1 多线程（IO 密集型）

```python
from concurrent.futures import ThreadPoolExecutor
import requests

def fetch_url(url):
    return requests.get(url).text

urls = ['http://example.com'] * 10

# 串行（慢）
# results = [fetch_url(url) for url in urls]

# 并行（快）
with ThreadPoolExecutor(max_workers=5) as executor:
    results = list(executor.map(fetch_url, urls))
```

### 4.2 多进程（CPU 密集型）

```python
from concurrent.futures import ProcessPoolExecutor
import multiprocessing

def cpu_intensive(n):
    total = 0
    for i in range(n):
        total += i * i
    return total

# 使用多进程
if __name__ == '__main__':
    workers = multiprocessing.cpu_count()
    with ProcessPoolExecutor(max_workers=workers) as executor:
        results = list(executor.map(cpu_intensive, [1000000] * 10))
```

### 4.3 异步 IO

```python
import asyncio
import aiohttp

async def fetch(session, url):
    async with session.get(url) as response:
        return await response.text()

async def main():
    urls = ['http://example.com'] * 10
    async with aiohttp.ClientSession() as session:
        tasks = [fetch(session, url) for url in urls]
        results = await asyncio.gather(*tasks)
        return results

# 运行
# asyncio.run(main())
```

### 5.2 批量操作

```python
# ❌ 慢：逐条插入
for user in users:
    cursor.execute(
        "INSERT INTO users (name, email) VALUES (%s, %s)",
        (user.name, user.email)
    )

# ✅ 快：批量插入
data = [(u.name, u.email) for u in users]
cursor.executemany(
    "INSERT INTO users (name, email) VALUES (%s, %s)",

)
```

## 第六部分 - 缓存策略

### 6.1 Redis 缓存

```python
import redis
import json
from functools import wraps

r = redis.Redis(host='localhost', port=6379, db=0)

def cache_result(key_prefix: str, ttl: int = 3600):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            key = f"{key_prefix}:{args}:{kwargs}"
            
            # 尝试从缓存获取
            cached = r.get(key)
            if cached:
                return json.loads(cached)
            
            # 执行函数并缓存结果
            result = func(*args, **kwargs)
            r.setex(key, ttl, json.dumps(result))
            return result
        return wrapper
    return decorator

@cache_result("user_profile", ttl=1800)
def get_user_profile(user_id: int):
    # 从数据库查询
    return db.query("SELECT * FROM users WHERE id = %s", user_id)
```

## 最佳实践清单

### ✅ 优化原则

1. 先分析，后优化（不要猜测瓶颈）
2. 80/20 法则：20% 的代码占用 80% 的时间
3. 优先优化最频繁执行的代码
4. 保持代码可读性，不要过度优化

### ✅ 推荐做法

1. 使用内置函数和库
2. 使用生成器处理大数据
3. 使用合适的数据结构
4. 对 IO 密集型任务使用多线程
5. 对 CPU 密集型任务使用多进程
6. 实现适当的缓存策略

# Python 类型提示（Type Hints）详解

## 第一部分：类型提示基础

### 1.1 什么是类型提示

类型提示（Type Hints）是 Python 3.5+ 引入的特性，用于在代码中声明变量、函数参数和返回值的类型。

#### 类型提示的好处

| 好处 | 说明 |
|------|------|
| 🔍 **早期发现错误** | 静态类型检查工具可以在运行前发现类型错误 |
| 📖 **更好的文档** | 代码即文档，一目了然知道函数接受什么返回什么 |
| 💡 **智能补全** | IDE 可以提供更准确的代码补全和建议 |
| 🛠️ **重构更安全** | 修改代码时类型检查器可以找出潜在问题 |
| 📊 **代码质量** | 强制思考类型，写出更清晰的代码 |

### 2.1 Optional 和 Union

```python
from typing import Optional, Union

# ========== Optional ==========

# Optional[X] 等价于 Union[X, None]
def find_user(user_id: int) -> Optional[str]:
    """查找用户，可能返回 None"""
    users = {1: "Alice", 2: "Bob"}
    return users.get(user_id)

# 以下两种写法等价
def get_name1() -> Optional[str]:
    return None

def get_name2() -> Union[str, None]:
    return None

# ========== Union ==========

# 可以是多种类型之一
def process_value(value: Union[int, str]) -> str:
    """处理整数或字符串"""
    return str(value)

# Python 3.10+ 可以使用 | 语法
def process_value_new(value: int | str) -> str:
    return str(value)

# 多个类型
def parse(data: Union[int, str, float, bool]) -> str:
    return str(data)

# Python 3.10+ 更简洁
def parse_new(data: int | str | float | bool) -> str:
    return str(data)

# ========== 实际应用 ==========

from typing import Optional, List

class UserService:
    def __init__(self):
        self.users: dict[int, str] = {1: "Alice", 2: "Bob"}

    def get_user(self, user_id: int) -> Optional[str]:
        """获取用户，不存在返回 None"""
        return self.users.get(user_id)

    def get_user_or_default(self, user_id: int) -> str:
        """获取用户，不存在返回默认值"""
        return self.users.get(user_id, "Unknown")

    def get_all_users(self) -> List[str]:
        """获取所有用户"""
        return list(self.users.values())
```

### 2.2 Callable 类型

```python
from typing import Callable

# ========== 基本用法 ==========

# Callable[[参数类型...], 返回类型]
def apply_operation(
    x: int,
    operation: Callable[[int], int]
) -> int:
    """应用操作"""
    return operation(x)

def double(n: int) -> int:
    return n * 2

result = apply_operation(5, double)  # 10

# ========== 多个参数 ==========

def combine(
    a: int,
    b: int,
    func: Callable[[int, int], int]
) -> int:
    return func(a, b)

def add(x: int, y: int) -> int:
    return x + y

result = combine(3, 4, add)  # 7

# ========== 实际示例 ==========

from typing import Callable, List

def filter_list(
    items: List[int],
    predicate: Callable[[int], bool]
) -> List[int]:
    """过滤列表"""
    return [item for item in items if predicate(item)]

def is_even(n: int) -> bool:
    return n % 2 == 0

numbers = [1, 2, 3, 4, 5, 6]
evens = filter_list(numbers, is_even)  # [2, 4, 6]

# 使用 lambda
odds = filter_list(numbers, lambda x: x % 2 != 0)  # [1, 3, 5]

# ========== 装饰器类型 ==========

from typing import Callable, TypeVar

F = TypeVar('F', bound=Callable)

def logger(func: F) -> F:
    """日志装饰器"""
    def wrapper(*args, **kwargs):
        print(f"调用 {func.__name__}")
        return func(*args, **kwargs)
    return wrapper  # type: ignore

@logger
def greet(name: str) -> str:
    return f"Hello, {name}!"
```

### 2.3 TypeVar 和泛型

```python
from typing import TypeVar, Generic, List

# ========== TypeVar 基础 ==========

T = TypeVar('T')  # 任意类型
K = TypeVar('K')  # 键类型
V = TypeVar('V')  # 值类型

# 泛型函数
def first_element(items: List[T]) -> T:
    """获取列表第一个元素"""
    return items[0]

# 类型推断
result = first_element([1, 2, 3])      # T 推断为 int
result = first_element(["a", "b"])     # T 推断为 str

# ========== 约束 TypeVar ==========

# 限制为特定类型的子类型
from typing import TypeVar

Number = TypeVar('Number', int, float)  # 只能是 int 或 float

def add_numbers(a: Number, b: Number) -> Number:
    return a + b

add_numbers(1, 2)        # ✅ int
add_numbers(1.5, 2.5)    # ✅ float
# add_numbers(1, 2.5)    # ❌ 类型不匹配

# 有界 TypeVar
from typing import TypeVar, Protocol

S = TypeVar('S', bound=str)  # 必须是 str 或其子类

def process_string(s: S) -> S:
    return s.upper()

# ========== 泛型类 ==========

from typing import Generic, TypeVar, List, Optional

T = TypeVar('T')

class Stack(Generic[T]):
    """泛型栈"""

    def __init__(self) -> None:
        self.items: List[T] = []

    def push(self, item: T) -> None:
        self.items.append(item)

    def pop(self) -> Optional[T]:
        if self.items:
            return self.items.pop()
        return None

    def peek(self) -> Optional[T]:
        if self.items:
            return self.items[-1]
        return None

    def is_empty(self) -> bool:
        return len(self.items) == 0

    def __len__(self) -> int:
        return len(self.items)

# 使用
int_stack = Stack[int]()
int_stack.push(1)
int_stack.push(2)
print(int_stack.pop())  # 2

str_stack = Stack[str]()
str_stack.push("hello")
str_stack.push("world")
print(str_stack.pop())  # world
```

### 2.4 Protocol 结构化类型

```python
from typing import Protocol, runtime_checkable

# ========== 基础 Protocol ==========

class Drawable(Protocol):
    """可绘制协议"""

    def draw(self) -> None:
        ...

    def get_size(self) -> tuple:
        ...

# 任何实现了这些方法的类都符合 Drawable 协议
class Circle:
    def __init__(self, radius: float):
        self.radius = radius

    def draw(self) -> None:
        print(f"绘制圆形，半径={self.radius}")

    def get_size(self) -> tuple:
        return (self.radius * 2, self.radius * 2)

class Square:
    def __init__(self, side: float):
        self.side = side

    def draw(self) -> None:
        print(f"绘制正方形，边长={self.side}")

    def get_size(self) -> tuple:
        return (self.side, self.side)

# 函数接受任何符合协议的类
def render(shape: Drawable) -> None:
    shape.draw()
    print(f"尺寸：{shape.get_size()}")

render(Circle(5))   # ✅
render(Square(10))  # ✅

# ========== runtime_checkable ==========

@runtime_checkable
class Closable(Protocol):
    def close(self) -> None:
        ...

def safe_close(obj: object) -> None:
    if isinstance(obj, Closable):
        obj.close()

# ========== 泛型 Protocol ==========

from typing import Protocol, TypeVar, Callable

T = TypeVar('T')

class Comparable(Protocol):
    def __lt__(self, other: 'Comparable') -> bool:
        ...

def sort_items(items: list[Comparable]) -> list[Comparable]:
    return sorted(items)
```

### 3.1 None 和 Optional

```python
from typing import Optional, Union

# ========== None 的含义 ==========

# None 表示"无值"或"空值"
# 在类型提示中，None 通常需要显式声明

def get_user(user_id: int) -> Optional[str]:
    """
    查找用户，找到返回用户名，找不到返回 None

    Optional[str] 等价于 Union[str, None]
    表示可能是 str，也可能是 None
    """
    users = {1: "Alice", 2: "Bob"}
    return users.get(user_id)

# ========== 显式处理 None ==========

from typing import Optional

def process_name(name: Optional[str]) -> str:
    """处理可能为 None 的名字"""

    # 方式 1：检查 None
    if name is None:
        return "Unknown"
    return name.upper()

def process_name_v2(name: Optional[str]) -> str:
    """方式 2：使用 or"""
    return (name or "Unknown").upper()

# ========== Python 3.10+ 简化语法 ==========

# Python 3.10+ 可以使用 | 语法
def get_user_new(user_id: int) -> str | None:
    """等价于 Optional[str]"""
    users = {1: "Alice", 2: "Bob"}
    return users.get(user_id)

def process_name_new(name: str | None) -> str:
    """等价于 Optional[str]"""
    return (name or "Unknown").upper()

# ========== 返回值可能是 None 的常见场景 ==========

from typing import Optional, List, Dict

# 字典查找
def find_in_dict(key: str, data: Dict[str, int]) -> Optional[int]:
    return data.get(key)

# 列表查找
def find_first(items: List[int], target: int) -> Optional[int]:
    try:
        index = items.index(target)
        return index
    except ValueError:
        return None

# 条件返回
def safe_divide(a: int, b: int) -> Optional[float]:
    if b == 0:
        return None
    return a / b

# ========== 类型细化（Type Narrowing） ==========

def handle_user(user: Optional[dict]) -> None:
    """通过检查细化类型"""

    # 检查后，user 被细化为 dict
    if user is not None:
        print(user["name"])  # ✅ 类型检查器知道这里是 dict
    else:
        print("用户不存在")

def handle_user_v2(user: Optional[dict]) -> None:
    """使用 assert 细化类型"""
    assert user is not None, "用户不能为空"
    print(user["name"])  # ✅ 这里 user 被细化为 dict

# ========== 默认值为 None 的参数 ==========

from typing import Optional, List

# ⚠️ 错误：可变默认值
# def create_list(items: List[int] = []):  # ❌

# ✅ 正确做法
def create_list(items: Optional[List[int]] = None) -> List[int]:
    """使用 Optional 表示参数可能为 None"""
    if items is None:
        return []
    return items
```

### 3.2 Any 类型

```python
from typing import Any

# ========== Any 的含义 ==========

# Any 是"逃生舱口"类型
# 任何类型都与 Any 兼容：
# - 任何类型都可以赋值给 Any
# - Any 可以赋值给任何类型

def process_any(data: Any) -> Any:
    """接受并返回任意类型"""
    return data

# 这些调用都是合法的
process_any(1)              # ✅ int
process_any("hello")        # ✅ str
process_any([1, 2, 3])      # ✅ list
process_any({"key": "val"}) # ✅ dict

# Any 类型的值可以调用任何方法
def use_any(data: Any) -> None:
    result = data.some_method()  # ✅ 类型检查通过
    result.any_attr = 123        # ✅ 类型检查通过

# ========== Any 的常见使用场景 ==========

# 场景 1：未知结构的 JSON 数据
from typing import Any, Dict, List

def parse_json(json_str: str) -> Any:
    """解析 JSON，返回任意结构的数据"""
    import json
    return json.loads(json_str)

data = parse_json('{"name": "Alice", "age": 25}')
print(data["name"])  # 运行时无问题，但类型检查不报错

# 场景 2：处理混合类型列表
def sum_mixed(numbers: list[Any]) -> float:
    """处理混合类型列表"""
    total = 0.0
    for n in numbers:
        if isinstance(n, (int, float)):
            total += n
    return total

# 场景 3：装饰器泛型处理
from typing import Callable, TypeVar

F = TypeVar('F', bound=Callable[..., Any])

def generic_decorator(func: F) -> F:
    """通用装饰器，保留原函数签名"""
    def wrapper(*args: Any, **kwargs: Any) -> Any:
        print(f"调用 {func.__name__}")
        return func(*args, **kwargs)
    return wrapper  # type: ignore

# 场景 4：第三方库没有类型注解
from typing import Any

def use_external_lib() -> Any:
    """调用没有类型注解的第三方库"""
    import some_library  # 假设这个库没有类型信息
    return some_library.do_something()

# ========== Any vs object ==========

# object 是所有类型的基类
# Any 是特殊的"动态类型"

def with_object(obj: object) -> None:
    # ❌ object 只能调用所有对象共有的方法
    print(str(obj))  # ✅
    # print(obj.specific_method())  # ❌ 类型检查失败

def with_any(data: Any) -> None:
    # ✅ Any 可以调用任何方法（类型检查不报错）
    print(data.any_method())

# ========== 限制 Any 的使用 ==========

# 推荐：尽量使用具体类型
def process_list(items: list[int]) -> int:
    return sum(items)

# 如果必须使用 Any，添加注释说明原因
def legacy_function(data: Any) -> Any:
    """处理旧版数据格式"""
    # TODO: 迁移到具体类型
    return data

# ========== Any 类型检查行为 ==========

from typing import Any

x: Any = 42
x = "hello"      # ✅ Any 可以接受任何赋值
x = [1, 2, 3]    # ✅
x.method()       # ✅ 类型检查不报错

# 但运行时仍会出错
y: Any = 42
# y.append(1)    # ❌ 运行时错误：int 没有 append 方法
```

### 3.2 新特性（Python 3.10+）

```python
# ========== 类型联合（|） ==========

# Python 3.10+
def process(value: int | str | None) -> str:
    if value is None:
        return "none"
    return str(value)

# ========== 类型守卫（isinstance 细化） ==========

from typing import TypeGuard

def is_string_list(items: list) -> TypeGuard[list[str]]:
    """检查是否为字符串列表"""
    return all(isinstance(item, str) for item in items)

def process(items: list) -> None:
    if is_string_list(items):
        # items 被细化为 list[str]
        for item in items:
            print(item.upper())

# ========== ParamSpec（Python 3.10+） ==========

from typing import Callable, ParamSpec, TypeVar

P = ParamSpec('P')
R = TypeVar('R')

def with_logging(
    func: Callable[P, R]
) -> Callable[P, R]:
    """保留参数类型的装饰器"""
    def wrapper(*args: P.args, **kwargs: P.kwargs) -> R:
        print(f"调用 {func.__name__}")
        return func(*args, **kwargs)
    return wrapper

@with_logging
def add(a: int, b: int) -> int:
    return a + b

# ========== Concatenate（Python 3.10+） ==========

from typing import Concatenate, Callable, ParamSpec, TypeVar

P = ParamSpec('P')
R = TypeVar('R')

def add_callback(
    func: Callable[P, R],
    callback: Callable[[str], None]
) -> Callable[Concatenate[str, P], R]:
    def wrapper(msg: str, *args: P.args, **kwargs: P.kwargs) -> R:
        callback(msg)
        return func(*args, **kwargs)
    return wrapper
```

### 4.1 mypy 配置和使用

## 第五部分：最佳实践和模式

### 5.1 类型注解最佳实践

```python
# ========== DO: 推荐做法 ==========

# 1. 为公共 API 添加完整类型注解
def calculate_total(
    prices: list[float],
    tax_rate: float = 0.1
) -> float:
    return sum(prices) * (1 + tax_rate)

# 2. 使用 Optional 表示可能为 None
def find_user(user_id: int) -> User | None:
    ...

# 3. 为复杂返回类型使用 TypedDict
from typing import TypedDict

class ApiResponse(TypedDict):
    status: int
    data: dict
    message: str

def fetch_data() -> ApiResponse:
    ...

# 4. 使用 Literal 限制值范围
def set_level(level: Literal[1, 2, 3, 4, 5]) -> None:
    ...

# 5. 为回调使用 Callable
def process_items(
    items: list[int],
    transformer: Callable[[int], int]
) -> list[int]:
    ...

# ========== DON'T: 避免的做法 ==========

# 1. 不要过度使用 Any
# ❌
def process(data: Any) -> Any:
    ...

# ✅
def process(data: dict[str, int]) -> list[int]:
    ...

# 2. 不要注解所有局部变量（让推断工作）
# ❌
def func():
    x: int = 1
    y: str = "hello"

# ✅
def func():
    x = 1
    y = "hello"

# 3. 不要使用可变默认值
# ❌
def add_item(item: int, items: list[int] = []) -> list[int]:
    items.append(item)
    return items

# ✅
def add_item(item: int, items: list[int] | None = None) -> list[int]:
    if items is None:
        items = []
    items.append(item)
    return items

# 4. 不要注解 self 和 cls
# ❌
class MyClass:
    def method(self: "MyClass") -> None:
        ...

# ✅
class MyClass:
    def method(self) -> None:
        ...
```

### 练习 2：设计泛型类

```python
from typing import Generic, TypeVar, Optional, Callable

T = TypeVar('T')
R = TypeVar('R')

# 泛型链表
class Node(Generic[T]):
    def __init__(self, value: T):
        self.value = value
        self.next: Optional[Node[T]] = None

class LinkedList(Generic[T]):
    def __init__(self):
        self.head: Optional[Node[T]] = None

    def append(self, value: T) -> None:
        new_node = Node(value)
        if not self.head:
            self.head = new_node
            return

        current = self.head
        while current.next:
            current = current.next
        current.next = new_node

    def map(self, func: Callable[[T], R]) -> LinkedList[R]:
        result = LinkedList[R]()
        current = self.head
        while current:
            result.append(func(current.value))
            current = current.next
        return result

# 使用
int_list = LinkedList[int]()
int_list.append(1)
int_list.append(2)
int_list.append(3)

str_list = int_list.map(str)  # LinkedList[str]
```

## 总结

| 类别 | 类型注解 | 示例 |
|------|---------|------|
| 基本类型 | `int`, `str`, `float`, `bool` | `x: int = 1` |
| 空值 | `None`, `Optional[T]` | `name: Optional[str]` |
| 任意类型 | `Any` | `data: Any` |
| 容器 | `list[T]`, `dict[K,V]`, `set[T]`, `tuple[...]` | `nums: list[int]` |
| 可选 | `Optional[T]`, `T \| None` | `name: str \| None` |
| 联合 | `Union[A,B]`, `A \| B` | `value: int \| str` |
| 字面量 | `Literal[...]` | `status: Literal["ok"]` |
| 函数 | `Callable[[A], R]` | `func: Callable[[int], str]` |
| 泛型 | `TypeVar`, `Generic[T]` | `class Box(Generic[T])` |
| 协议 | `Protocol` | `class Drawable(Protocol)` |
| TypedDict | `TypedDict` | `class User(TypedDict)` |

类型提示让 Python 代码更可靠、更易维护！

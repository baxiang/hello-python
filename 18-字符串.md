### 1.2 字符串不可变性（Immutability）

**字符串不可变性**是 Python 字符串最重要的特性之一：**字符串创建后，其内容无法被修改**。

```python
text = "Hello"

# ========== 尝试修改单个字符 ==========

# ❌ TypeError: 'str' object does not support item assignment
# text[0] = "h"

# 必须创建新字符串
text = "h" + text[1:]  # 创建新字符串 "hello"
print(text)  # hello

# ========== 查看内存地址理解不可变性 ==========

text1 = "Hello"
print(id(text1))  # 例如：140234567890

# "修改"操作实际上是创建新对象
text1 = text1 + " World"
print(id(text1))  # 新的内存地址，不是原来的了

# ========== 不可变性的影响 ==========

# 1. 不能原地修改
text = "hello"
# text[0] = "H"  # ❌ 错误

# 2. 需要通过创建新字符串来"修改"
text = "H" + text[1:]  # 创建新字符串
print(text)  # Hello

# 3. 字符串方法返回新字符串，不修改原字符串
original = "  Hello  "
cleaned = original.strip()
print(original)  # "  Hello  " (原字符串不变)
print(cleaned)   # "Hello" (新字符串)
```

### 1.3 字符串驻留（String Interning）

**字符串驻留**是 Python 的内存优化机制：**相同的字符串在内存中只保存一份副本**。

```python
# ========== 驻留机制演示 ==========

# 相同的字符串字面量指向同一内存地址
a = "hello"
b = "hello"
print(a is b)        # True
print(id(a), id(b))  # 相同的内存地址

# ========== 驻留的规则 ==========

# ✅ 会被驻留的情况

# 1. 短字符串（通常是标识符友好的字符串）
x = "hello"
y = "hello"
print(x is y)  # True

# 2. 只包含字母、数字、下划线的字符串
var_name = "my_variable"
another = "my_variable"
print(var_name is another)  # True

# 3. 空字符串和单字符
empty1 = ""
empty2 = ""
print(empty1 is empty2)  # True

# ========== 不会被驻留的情况 ==========

# 1. 使用 str() 或 + 运行时创建的字符串
s1 = "hello"
s2 = str("he" + "llo")
print(s1 is s2)  # 可能为 False

# 2. 长字符串（超过一定长度）
long1 = "a" * 1000
long2 = "a" * 1000
print(long1 is long2)  # False

# 3. 包含特殊字符的字符串
special1 = "hello world!"
special2 = "hello world!"
print(special1 is special2)  # 可能为 False（取决于 Python 实现）

# 4. 通过函数调用返回的字符串
def get_string():
    return "hello"

s1 = get_string()
s2 = get_string()
print(s1 is s2)  # 可能为 False

# ========== 编译期优化 ==========

# 编译器会在编译时合并相同的字符串字面量
def test():
    s1 = "hello"
    s2 = "hello"
    return s1 is s2

print(test())  # True（同一函数内，编译器优化）

# ========== 手动控制驻留 ==========

import sys

# sys.intern() 强制驻留字符串
s1 = sys.intern("hello")
s2 = sys.intern("hello")
print(s1 is s2)  # True（强制驻留）

# 适用场景：大量重复字符串时节省内存
# 例如：处理大量重复的关键词、标签等

# ========== 驻留的性能优势 ==========

import timeit

# 使用驻留的字符串比较
def interning_test():
    a = "hello"
    b = "hello"
    return a is b  # 内存地址比较，O(1)

# 普通字符串比较
def normal_test():
    a = "hello"
    b = "hello"
    return a == b  # 内容比较，O(n)

print(f"驻留比较：{timeit.timeit(interning_test, number=1000000):.4f}秒")
print(f"普通比较：{timeit.timeit(normal_test, number=1000000):.4f}秒")

# ========== 实际应用 ==========

# 1. 配置项比较（使用 intern 加速）
import sys

CONFIG_OPTIONS = [sys.intern("debug"), sys.intern("release"), sys.intern("test")]

def check_config(option):
    # 使用 is 比较，速度更快
    return option in CONFIG_OPTIONS

# 2. 词法分析器/编译器中的标识符
keywords = {sys.intern(k): True for k in ["if", "else", "while", "for", "def"]}

def is_keyword(token):
    return token in keywords
```

### 1.4 `==` 和 `is` 的区别

```python
# ========== 核心区别 ==========

# ==  比较值（内容）是否相等
# is  比较身份（内存地址）是否相同

a = "hello"
b = "hello"
c = "".join(["h", "e", "l", "l", "o"])

print(a == b)  # True（内容相同）
print(a == c)  # True（内容相同）
print(a is b)  # True（可能驻留，同一对象）
print(a is c)  # False（c 是新创建的对象）

# ========== 使用建议 ==========

# ✅ 始终使用 == 比较字符串内容
if a == c:
    print("内容相同")

# ❌ 不要用 is 比较字符串内容
if a is c:  # 不可靠！
    print("可能不会执行")

# ✅ is 只用于与 None 比较
name = None
if name is None:  # 正确用法
    print("name is None")

# ========== 实验验证 ==========

def test_interning(s):
    """测试字符串是否会被驻留"""
    s1 = s
    s2 = s
    print(f"'{s}': s1 is s2 = {s1 is s2}")

# 测试不同字符串
test_interning("hello")           # True
test_interning("Hello World")     # 可能 True
test_interning("hello" * 100)     # False
test_interning("hello_123")       # True
test_interning("hello world!")    # 可能 False（含空格和标点）
```

### 2.1 索引访问

```python
text = "Python Programming"

# ========== 正向索引（从 0 开始）==========

print(text[0])    # P
print(text[1])    # y
print(text[6])    # P (Programming 的 P)
print(text[17])   # g (最后一个字符)

# ========== 反向索引（从 -1 开始）==========

print(text[-1])   # g (最后一个)
print(text[-2])   # n
print(text[-6])   # a
print(text[-18])  # P (第一个)

# ========== 索引越界 ==========

# print(text[20])  # ❌ IndexError: string index out of range

# 安全访问
def safe_index(s, index):
    if 0 <= index < len(s):
        return s[index]
    return None

print(safe_index(text, 0))    # P
print(safe_index(text, 100))  # None
```

### 2.2 字符串切片详解

```python
text = "Python Programming"

# ========== 基本切片 [start:end] ==========

print(text[0:6])    # Python (0-5，不包含 6)
print(text[7:18])   # Programming
print(text[0:6])    # Python (同上)

# ========== 省略 start 或 end ==========

print(text[:6])     # Python (从开头到 5)
print(text[7:])     # Programming (从 7 到结尾)
print(text[:])      # Python Programming (完整复制)

# ========== 负索引切片 ==========

print(text[-11:])   # Programming (最后 11 个)
print(text[:-12])   # Python (除了最后 11 个)
print(text[-18:-12]) # Python
print(text[-5:-2])  # mm

# ========== 步长切片 [start:end:step] ==========

numbers = "0123456789"

print(numbers[::2])     # 02468 (偶数位置)
print(numbers[1::2])    # 13579 (奇数位置)
print(numbers[::3])     # 0369 (每 3 个)
print(numbers[::-1])    # 9876543210 (反转字符串)
print(numbers[::-2])    # 97531 (反向每 2 个)
print(numbers[8:2:-1])  # 876543 (反向从 8 到 3)

# ========== 切片技巧 ==========

# 反转字符串
def reverse(s):
    return s[::-1]

print(reverse("Hello"))  # olleH

# 每隔 n 个字符
def every_nth(s, n):
    return s[::n]

print(every_nth("Hello World", 2))  # HloWrd

# 获取前 n 个字符
def first_n(s, n):
    return s[:n]

# 获取后 n 个字符
def last_n(s, n):
    return s[-n:] if n <= len(s) else s

# 安全切片（不会报错）
print(text[100:200])  # "" (空字符串，不会报错)
```

### 3.1 查找和检查方法

```python
text = "Hello, Python Programming!"

# ========== find / index - 查找子串 ==========

print(text.find("Python"))    # 7 (找到返回索引)
print(text.find("Java"))      # -1 (未找到返回 -1)

print(text.index("Python"))   # 7 (找到返回索引)
# text.index("Java")          # ❌ ValueError: substring not found

# rfind / rindex - 从右边查找
text2 = "abababa"
print(text2.find("aba"))      # 0 (从左找第一个)
print(text2.rfind("aba"))     # 4 (从右找第一个)

# 指定范围查找
print(text.find("o", 5, 15))  # 9 (在索引 5-15 之间查找)

# ========== startswith / endswith ==========

url = "https://www.example.com"
filename = "document.pdf"

print(url.startswith("https"))     # True
print(url.startswith("http"))      # True (https 包含 http)
print(url.endswith(".com"))        # True
print(filename.endswith(".pdf"))   # True

# 检查多个可能值
print(url.endswith((".com", ".org", ".net")))  # True

# 指定范围检查
print(url.startswith("www", 8))   # True (从索引 8 开始)

# ========== in - 成员检查 ==========

print("Python" in text)           # True
print("Java" not in text)         # True
print("o" in text)                # True (单个字符也可以)

# ========== count - 计数 ==========

text3 = "hello world, hello python"
print(text3.count("hello"))       # 2
print(text3.count("o"))           # 3
print(text3.count("hello", 6))    # 1 (从索引 6 开始)
print(text3.count("hello", 6, 20)) # 1 (范围查找)
```

### 3.2 大小写转换方法

```python
text = "Hello, PyThon ProGramming!"

# ========== lower / upper ==========

print(text.lower())   # hello, python programming!
print(text.upper())   # HELLO, PYTHON PROGRAMMING!

# 实用场景：不区分大小写比较
user_input = "HELLO"
correct = "hello"
print(user_input.lower() == correct.lower())  # True

# ========== capitalize / title ==========

sentence = "hello world, how are you?"
print(sentence.capitalize())  # Hello world, how are you? (首字母大写)

title_text = "hello world"
print(title_text.title())   # Hello World (每个单词首字母大写)

# ========== swapcase ==========

print(text.swapcase())  # hELLO, pYthON pROgRAMMING! (大小写互换)

# ========== casefold - 更激进的小写转换 ==========

# 德语特殊字符
german = "straße"
print(german.lower())     # straße
print(german.casefold())  # strasse (转换更彻底)

# 不区分大小写比较
print("Straße".lower() == "strasse".lower())  # False
print("Straße".casefold() == "strasse".casefold())  # True

# ========== is 系列方法 - 检查 ==========

tests = [
    "hello",      # 全小写
    "HELLO",      # 全大写
    "Hello",      # 首字母大写
    "123",        # 全数字
    "hello123",   # 字母数字
    " ",          # 空格
    "",           # 空字符串
    "你好",       # 中文
]

for t in tests:
    print(f"'{t}':")
    print(f"  islower(): {t.islower()}")      # 是否全小写
    print(f"  isupper(): {t.isupper()}")      # 是否全大写
    print(f"  istitle(): {t.istitle()}")      # 是否标题格式
    print(f"  isdigit(): {t.isdigit()}")      # 是否全数字
    print(f"  isalpha(): {t.isalpha()}")      # 是否全字母
    print(f"  isalnum(): {t.isalnum()}")      # 是否字母或数字
    print(f"  isspace(): {t.isspace()}")      # 是否空白
    print(f"  isprintable(): {t.isprintable()}") # 是否可打印
    print()
```

### 3.5 格式化方法

```python
# ========== format - 格式化 ==========

# 位置参数
print("{} {}".format("Hello", "World"))  # Hello World

# 格式化

    return f"Hi, {n}!"

# 日期格式化

print(f"{num:.2e}")      # '1.23e+04' (科学计数)

# 百分比

# ========== 旧式格式化（已过时但仍在使用）==========

# 多个值

# 字典
```

### 4.1 编码基础

```python
# ========== 编码和解码 ==========

text = "你好，世界！"

# 编码：字符串 -> 字节
utf8_bytes = text.encode('utf-8')
print(f"UTF-8 字节：{utf8_bytes}")  # b'\xe4\xbd\xa0\xe5\xa5\xbd...'

gbk_bytes = text.encode('gbk')
print(f"GBK 字节：{gbk_bytes}")

# 解码：字节 -> 字符串
print(utf8_bytes.decode('utf-8'))  # 你好，世界！
print(gbk_bytes.decode('gbk'))     # 你好，世界！

# ========== 编码错误处理 ==========

text = "Hello 你好 🌍"

# 尝试用 ASCII 编码（会失败）
try:
    text.encode('ascii')
except UnicodeEncodeError as e:
    print(f"编码错误：{e}")

# 错误处理方式
print(text.encode('ascii', errors='ignore'))    # b'Hello  ' (忽略)
print(text.encode('ascii', errors='replace'))   # b'Hello  ??' (替换为?)
print(text.encode('ascii', errors='backslashreplace'))  # b'Hello \u4f60\u597d \U0001f30d'

# ========== 常见编码 ==========

encodings = ['utf-8', 'gbk', 'gb2312', 'big5', 'latin-1', 'ascii']

for enc in encodings:
    try:
        encoded = "中文".encode(enc)
        decoded = encoded.decode(enc)
        print(f"{enc}: {encoded} -> {decoded}")
    except Exception as e:
        print(f"{enc}: 错误 - {e}")
```

```python
def snake_to_camel(name):
    """snake_case 转 camelCase"""
    parts = name.split("_")
    return parts[0] + "".join(word.capitalize() for word in parts[1:])

def camel_to_snake(name):
    """camelCase 转 snake_case"""
    import re
    return re.sub(r'(?<!^)(?=[A-Z])', '_', name).lower()

print(camel_to_snake("helloWorld"))       # hello_world

def truncate(text, length=50, suffix="..."):
    """截断文本到指定长度"""
    if len(text) <= length:
        return text
    return text[:length - len(suffix)] + suffix

print(truncate("这是一个很长的文本", 10))  # 这是一个很...

# ========== 填充和对齐 ==========

def pad_string(text, width, side='left', char=' '):
    """填充字符串"""
    if side == 'left':
        return text.rjust(width, char)
    elif side == 'right':
        return text.ljust(width, char)
    elif side == 'center':
        return text.center(width, char)

print(pad_string("42", 5, 'left', '0'))    # 00042
print(pad_string("Hi", 6, 'center', '-'))  # --Hi--

# ========== 移除前缀和后缀 ==========

def remove_prefix(text, prefix):
    """移除前缀"""
    if text.startswith(prefix):
        return text[len(prefix):]
    return text

def remove_suffix(text, suffix):
    """移除后缀"""
    if text.endswith(suffix):
        return text[:-len(suffix)]
    return text

# Python 3.9+ 可以使用内置方法
# text.removeprefix("prefix")
# text.removesuffix("suffix")
```

```python
def censor(text, words):
    for word in words:
        text = re.sub(r'\b' + re.escape(word) + r'\b', '*' * len(word), text, flags=re.IGNORECASE)
    return text

def is_strong_password(password):
    """至少 8 位，包含大小写字母、数字和特殊字符"""
    if len(password) < 8:
        return False
    if not re.search(r'[A-Z]', password):
        return False
    if not re.search(r'[a-z]', password):
        return False
    if not re.search(r'\d', password):
        return False
    if not re.search(r'[!@#$%^&*(),.?":{}|<>]', password):
        return False
    return True

print(is_strong_password("Pass123!"))  # True
print(is_strong_password("weak"))      # False
```

```python
def reverse_methods(text):
    """演示各种反转方法"""

    # 方法 1: 切片
    result1 = text[::-1]

    # 方法 2: reversed + join
    result2 = "".join(reversed(text))

    # 方法 3: 循环
    result3 = ""
    for char in text:
        result3 = char + result3

    # 方法 4: 列表反转
    chars = list(text)
    chars.reverse()
    result4 = "".join(chars)

    return result1, result2, result3, result4
```

```python
# 实现自己的字符串方法

def my_split(text, sep=None):
    """简易版 split"""
    if sep is None:
        sep = " "
    result = []
    current = ""
    for char in text:
        if char == sep:
            if current or sep != " ":  # 跳过连续分隔符
                result.append(current)
                current = ""
        else:
            current += char
    if current:
        result.append(current)
    return result

def my_join(iterable, sep=""):
    """简易版 join"""
    result = ""
    for i, item in enumerate(iterable):
        result += str(item)
        if i < len(iterable) - 1:
            result += sep
    return result

print(my_split("a,b,c", ","))  # ['a', 'b', 'c']
print(my_join(['a', 'b', 'c'], '-'))  # a-b-c
```

```python
def text_pipeline(text):
    """文本处理管道"""
    # 1. 转小写
    text = text.lower()

    # 2. 去除标点
    import string
    for char in string.punctuation:
        text = text.replace(char, " ")

    # 3. 标准化空格
    text = " ".join(text.split())

    # 4. 分词
    words = text.split()

    # 5. 去除停用词
    stopwords = {'the', 'a', 'an', 'is', 'are', 'in', 'on', 'at', 'to', 'of'}
    words = [w for w in words if w not in stopwords]

    # 6. 返回词频
    from collections import Counter
    return Counter(words)

text = "The quick brown fox jumps over the lazy dog. The dog was not amused!"
print(text_pipeline(text))
# Counter({'the': 2, 'dog': 2, 'quick': 1, 'brown': 1, 'fox': 1, 'jumps': 1, 'over': 1, 'lazy': 1, 'was': 1, 'not': 1, 'amused': 1})
```

## 15.1 项目一：天气查询工具

### 15.1.1 项目概述

本项目将构建一个命令行天气查询工具，用户可以通过城市名称查询实时天气信息。

**核心功能：**
- 调用第三方天气 API 获取数据
- 解析 JSON 响应并格式化输出
- 支持多个城市连续查询
- 友好的命令行交互界面

### 15.1.2 技术选型

| 组件 | 技术 | 说明 |
|------|------|------|
| HTTP 请求 | requests | 发送 API 请求 |
| 数据格式 | JSON | API 响应解析 |
| 时间处理 | datetime | 时间格式化 |
| 错误处理 | try-except | 网络异常处理 |

### 15.1.3 API 介绍

我们使用 OpenWeatherMap 提供的免费天气 API：

**请求格式：**
```
GET https://api.openweathermap.org/data/2.5/weather?q={城市名}&appid={API 密钥}&units=metric&lang=zh_cn
```

**参数说明：**
- `q`: 城市名称
- `appid`: API 密钥（需要在官网免费注册获取）
- `units=metric`: 使用摄氏度
- `lang=zh_cn`: 中文响应

**响应示例：**
```json
{
  "name": "Beijing",
  "main": {
    "temp": 15.5,
    "feels_like": 14.2,
    "humidity": 65
  },
  "weather": [
    {
      "description": "晴"
    }
  ],
  "wind": {
    "speed": 3.5
  }
}
```

### 15.1.4 代码实现

```python
import requests
from datetime import datetime


class WeatherChecker:
    """
    天气查询工具类

    封装了 OpenWeatherMap API 的调用逻辑，
    提供天气获取和格式化显示功能
    """

    # 类常量：API 基础 URL
    BASE_URL = "https://api.openweathermap.org/data/2.5/weather"

    def __init__(self, api_key):
        """
        初始化工具

        Args:
            api_key: OpenWeatherMap API 密钥
        """
        self.api_key = api_key

    def get_weather(self, city):
        """
        获取指定城市的天气信息

        Args:
            city: 城市名称（支持中文）

        Returns:
            包含天气信息的字典，如果失败返回 None
        """
        # 构建请求参数
        params = {
            'q': city,              # 城市名
            'appid': self.api_key,  # API 密钥
            'units': 'metric',      # 摄氏度单位
            'lang': 'zh_cn'         # 中文响应
        }

        try:
            # 发送 HTTP GET 请求
            response = requests.get(
                self.BASE_URL,
                params=params,
                timeout=10  # 设置超时时间
            )

            # 检查 HTTP 状态码，非 200 会抛出异常
            response.raise_for_status()

            # 解析 JSON 响应
            data = response.json()

            # 检查 API 是否返回错误（如城市不存在）
            if 'cod' in data and data['cod'] != 200:
                print(f"API 错误：{data.get('message', '未知错误')}")
                return None

            # 提取并整理需要的字段
            weather_info = {
                'city': data['name'],
                'country': data['sys']['country'],
                'temperature': data['main']['temp'],
                'feels_like': data['main']['feels_like'],
                'humidity': data['main']['humidity'],
                'pressure': data['main']['pressure'],
                'description': data['weather'][0]['description'],
                'wind_speed': data['wind']['speed'],
                'sunrise': data['sys']['sunrise'],
                'sunset': data['sys']['sunset']
            }

            return weather_info

        except requests.exceptions.Timeout:
            print("请求超时，请检查网络连接")
            return None
        except requests.exceptions.ConnectionError:
            print("连接失败，请检查网络或 API 地址")
            return None
        except requests.exceptions.RequestException as e:
            print(f"请求失败：{e}")
            return None
        except KeyError as e:
            print(f"数据解析错误，缺少字段：{e}")
            return None

    def format_time(self, timestamp):
        """将 Unix 时间戳转换为可读时间格式"""
        return datetime.fromtimestamp(timestamp).strftime('%H:%M:%S')

    def display_weather(self, weather_data):
        """
        格式化显示天气信息

        Args:
            weather_data: get_weather 返回的字典
        """
        if not weather_data:
            print("未获取到天气数据")
            return

        # 使用装饰线分隔显示
        print("\n" + "=" * 50)
        print(f"  {weather_data['city']}, {weather_data['country']} 实时天气")
        print("=" * 50)
        print(f"  天气状况 : {weather_data['description']}")
        print(f"  当前温度 : {weather_data['temperature']}°C")
        print(f"  体感温度 : {weather_data['feels_like']}°C")
        print(f"  相对湿度 : {weather_data['humidity']}%")
        print(f"  大气压强 : {weather_data['pressure']} hPa")
        print(f"  风速     : {weather_data['wind_speed']} m/s")
        if 'sunrise' in weather_data:
            print(f"  日出时间 : {self.format_time(weather_data['sunrise'])}")
            print(f"  日落时间 : {self.format_time(weather_data['sunset'])}")
        print("=" * 50 + "\n")

    def get_weather_by_coordinates(self, lat, lon):
        """
        通过经纬度获取天气

        Args:
            lat: 纬度
            lon: 经度
        """
        params = {
            'lat': lat,
            'lon': lon,
            'appid': self.api_key,
            'units': 'metric',
            'lang': 'zh_cn'
        }

        try:
            response = requests.get(self.BASE_URL, params=params, timeout=10)
            response.raise_for_status()
            return self._parse_response(response.json())
        except requests.exceptions.RequestException as e:
            print(f"请求失败：{e}")
            return None


def main():
    """主函数：提供交互式天气查询"""

    # TODO: 请替换为你自己的 API 密钥
    # 1. 访问 https://openweathermap.org/ 注册账号
    # 2. 进入 API Keys 页面创建密钥
    # 3. 将下面的占位符替换为你的实际密钥
    api_key = "your_api_key_here"

    # 检查 API 密钥是否被修改
    if api_key == "your_api_key_here":
        print("警告：请先申请 API 密钥并替换代码中的 'your_api_key_here'")
        print("申请地址：https://openweathermap.org/api")
        return

    # 创建天气查询工具实例
    weather_checker = WeatherChecker(api_key)

    print("=" * 50)
    print("       欢迎使用天气查询工具")
    print("=" * 50)
    print("输入城市名称查询天气，输入 'quit' 退出程序")
    print("支持中文城市名，如：北京、上海、Guangzhou")
    print("-" * 50)

    # 主循环：持续接收用户输入
    while True:
        try:
            # 获取用户输入
            city = input("\n请输入城市名称：").strip()

            # 退出条件判断
            if city.lower() in ('quit', 'exit', 'q'):
                print("感谢使用，再见！")
                break

            # 空输入处理
            if not city:
                print("城市名称不能为空")
                continue

            # 查询并显示天气
            print(f"正在查询 {city} 的天气...")
            weather_data = weather_checker.get_weather(city)
            weather_checker.display_weather(weather_data)

        except KeyboardInterrupt:
            # 处理 Ctrl+C 中断
            print("\n程序已中断")
            break
        except EOFError:
            # 处理输入流结束
            break


if __name__ == "__main__":
    main()
```

## 15.2 项目二：文件批量重命名工具

### 15.2.1 项目概述

批量重命名工具可以一次性处理大量文件，常用于照片整理、文档标准化等场景。

**核心功能：**
- 按扩展名筛选文件
- 添加前缀/后缀
- 序号重命名
- 文本替换
- 预览修改效果

### 15.2.2 技术要点

| 技术点 | 说明 |
|--------|------|
| pathlib.Path | 现代化路径处理 |
| glob 模式 | 文件匹配筛选 |
| 文件重命名 | os.rename 或 Path.rename |
| 字符串处理 | 前缀、后缀、替换操作 |

### 15.2.3 代码实现

```python
import os
from pathlib import Path
from datetime import datetime


class FileRenamer:
    """
    文件批量重命名工具

    支持多种重命名策略：前缀、后缀、序号、文本替换等
    """

    def __init__(self, directory):
        """
        初始化工具

        Args:
            directory: 目标目录路径
        """
        self.directory = Path(directory)

        # 验证目录有效性
        if not self.directory.exists():
            raise FileNotFoundError(f"目录不存在：{directory}")
        if not self.directory.is_dir():
            raise NotADirectoryError(f"路径不是目录：{directory}")

    def list_files(self, extension=None, include_hidden=False):
        """
        列出目录下的文件

        Args:
            extension: 文件扩展名过滤（如 'jpg'，不传则列出所有文件）
            include_hidden: 是否包含隐藏文件（以.开头）

        Returns:
            Path 对象列表
        """
        if extension:
            # 移除扩展名前的点（如果用户输入了.jpg）
            ext = extension.lstrip('.')
            files = list(self.directory.glob(f"*.{ext}"))
        else:
            # 获取所有文件（不包括子目录）
            files = [f for f in self.directory.iterdir() if f.is_file()]

        # 过滤隐藏文件
        if not include_hidden:
            files = [f for f in files if not f.name.startswith('.')]

        # 按文件名排序返回
        return sorted(files, key=lambda x: x.name)

    def preview_changes(self, new_names):
        """
        预览重命名效果

        Args:
            new_names: (旧名称，新名称) 元组列表
        """
        print(f"\n{'原文件名':<40} -> {'新文件名':<40}")
        print("-" * 85)
        for old_name, new_name in new_names:
            print(f"{old_name:<40} -> {new_name:<40}")
        print("-" * 85)
        print(f"共 {len(new_names)} 个文件\n")

    def add_prefix(self, prefix, extension=None, dry_run=True):
        """
        为文件添加前缀

        Args:
            prefix: 要添加的前缀
            extension: 文件扩展名过滤
            dry_run: True 仅预览，False 实际执行
        """
        files = self.list_files(extension)
        changes = []

        for file in files:
            new_name = f"{prefix}{file.name}"
            changes.append((file.name, new_name))

        if not changes:
            print("没有找到符合条件的文件")
            return

        self.preview_changes(changes)

        if not dry_run:
            for (old_name, new_name) in changes:
                old_path = self.directory / old_name
                new_path = self.directory / new_name
                old_path.rename(new_path)
                print(f"已重命名：{old_name} -> {new_name}")

    def add_suffix(self, suffix, extension=None, dry_run=True):
        """
        为文件添加后缀（添加在扩展名之前）

        Args:
            suffix: 要添加的后缀
            extension: 文件扩展名过滤
            dry_run: True 仅预览，False 实际执行
        """
        files = self.list_files(extension)
        changes = []

        for file in files:
            # 分离文件名和扩展名
            name = file.stem      # 不含扩展名的文件名
            ext = file.suffix     # 扩展名（包含点）
            new_name = f"{name}{suffix}{ext}"
            changes.append((file.name, new_name))

        if not changes:
            print("没有找到符合条件的文件")
            return

        self.preview_changes(changes)

        if not dry_run:
            for (old_name, new_name) in changes:
                old_path = self.directory / old_name
                new_path = self.directory / new_name
                old_path.rename(new_path)
                print(f"已重命名：{old_name} -> {new_name}")

    def rename_with_numbering(self, prefix="", start=1, extension=None, dry_run=True):
        """
        按序号重命名文件

        Args:
            prefix: 序号前的前缀
            start: 起始序号
            extension: 文件扩展名过滤
            dry_run: True 仅预览，False 实际执行
        """
        files = self.list_files(extension)
        changes = []

        for index, file in enumerate(files, start=start):
            ext = file.suffix
            # 使用 03d 格式化为 3 位数字（001, 002, ...）
            new_name = f"{prefix}{index:03d}{ext}"
            changes.append((file.name, new_name))

        if not changes:
            print("没有找到符合条件的文件")
            return

        self.preview_changes(changes)

        if not dry_run:
            for (old_name, new_name) in changes:
                old_path = self.directory / old_name
                new_path = self.directory / new_name
                old_path.rename(new_path)
                print(f"已重命名：{old_name} -> {new_name}")

    def replace_text(self, old_text, new_text, extension=None, dry_run=True):
        """
        替换文件名中的指定文本

        Args:
            old_text: 要被替换的文本
            new_text: 替换后的文本
            extension: 文件扩展名过滤
            dry_run: True 仅预览，False 实际执行
        """
        files = self.list_files(extension)
        changes = []

        for file in files:
            if old_text in file.name:
                new_name = file.name.replace(old_text, new_text)
                # 跳过新旧名称相同的情况
                if new_name != file.name:
                    changes.append((file.name, new_name))

        if not changes:
            print(f"没有找到包含 '{old_text}' 的文件")
            return

        self.preview_changes(changes)

        if not dry_run:
            for (old_name, new_name) in changes:
                old_path = self.directory / old_name
                new_path = self.directory / new_name
                old_path.rename(new_path)
                print(f"已重命名：{old_name} -> {new_name}")

    def rename_by_date(self, date_format="%Y%m%d", extension=None, dry_run=True):
        """
        按文件修改日期重命名

        Args:
            date_format: 日期格式化字符串
            extension: 文件扩展名过滤
            dry_run: True 仅预览，False 实际执行
        """
        files = self.list_files(extension)
        changes = []

        for file in files:
            # 获取文件修改时间
            mtime = datetime.fromtimestamp(file.stat().st_mtime)
            date_str = mtime.strftime(date_format)
            ext = file.suffix
            new_name = f"{date_str}{ext}"
            changes.append((file.name, new_name))

        self.preview_changes(changes)

        if not dry_run:
            for (old_name, new_name) in changes:
                old_path = self.directory / old_name
                new_path = self.directory / new_name
                old_path.rename(new_path)
                print(f"已重命名：{old_name} -> {new_name}")

    def undo_rename(self, backup_file):
        """
        从备份文件恢复（需要先创建备份）

        Args:
            backup_file: 备份文件路径
        """
        # 此功能需要先实现备份逻辑
        pass


def main():
    """交互式命令行界面"""

    print("=" * 60)
    print("          文件批量重命名工具")
    print("=" * 60)

    # 获取目标目录
    while True:
        dir_path = input("请输入要处理的目录路径（直接回车使用当前目录）：").strip()
        if not dir_path:
            dir_path = "."
            break
        if Path(dir_path).is_dir():
            break
        print("目录不存在，请重新输入")

    try:
        # 创建重命名工具实例
        renamer = FileRenamer(dir_path)
    except (FileNotFoundError, NotADirectoryError) as e:
        print(f"错误：{e}")
        return

    # 显示当前文件列表
    print("\n当前目录下的文件：")
    files = renamer.list_files()
    if not files:
        print("  （无文件）")
        return

    for i, f in enumerate(files, 1):
        size = f.stat().st_size
        size_str = f"{size / 1024:.1f}KB" if size > 1024 else f"{size}B"
        print(f"  {i:2}. {f.name:<35} ({size_str})")

    # 主菜单循环
    while True:
        print("\n--- 选择操作 ---")
        print("1. 添加前缀")
        print("2. 添加后缀")
        print("3. 序号重命名")
        print("4. 文本替换")
        print("5. 按日期重命名")
        print("6. 刷新文件列表")
        print("0. 退出")

        choice = input("请选择 (0-6): ").strip()

        if choice == '0':
            print("退出程序")

        elif choice == '1':
```

## 15.3 项目三：待办事项管理器

### 15.3.1 项目概述

待办事项（Todo）管理器是一个经典的 CRUD 应用，支持添加、查看、更新、删除待办事项，数据持久化保存到 JSON 文件。

**核心功能：**
- 添加待办事项（支持优先级）
- 查看所有/未完成的待办
- 标记事项为已完成
- 删除待办事项
- 数据持久化（JSON 文件）

### 15.3.2 数据模型

每个待办事项包含以下字段：

```json
{
  "id": 1,
  "title": "学习 Python",
  "priority": "high",
  "completed": false,
  "created_at": "2024-01-15 10:30:00",
  "completed_at": null,
  "tags": ["学习", "编程"]
}
```

### 15.3.3 代码实现

```python
import json
from datetime import datetime
from pathlib import Path
from typing import Optional, List, Dict, Any


class TodoItem:
    """待办事项数据类"""

    # 优先级选项
    PRIORITIES = ('low', 'medium', 'high')

    def __init__(self, todo_id: int, title: str, priority: str = 'medium',
                 completed: bool = False, created_at: Optional[str] = None,
                 completed_at: Optional[str] = None, tags: Optional[List[str]] = None):
        self.id = todo_id
        self.title = title
        self.priority = priority if priority in self.PRIORITIES else 'medium'
        self.completed = completed
        self.created_at = created_at or datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        self.completed_at = completed_at
        self.tags = tags or []

    def to_dict(self) -> Dict[str, Any]:
        """转换为字典（用于 JSON 序列化）"""
        return {
            'id': self.id,
            'title': self.title,
            'priority': self.priority,
            'completed': self.completed,
            'created_at': self.created_at,
            'completed_at': self.completed_at,
            'tags': self.tags
        }

    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> 'TodoItem':
        """从字典创建实例"""
        return cls(
            todo_id=data.get('id', 0),
            title=data.get('title', ''),
            priority=data.get('priority', 'medium'),
            completed=data.get('completed', False),
            created_at=data.get('created_at'),
            completed_at=data.get('completed_at'),
            tags=data.get('tags', [])
        )

    def __str__(self) -> str:
        status = "✓" if self.completed else "○"
        priority_icon = {'high': '🔴', 'medium': '🟡', 'low': '🟢'}.get(self.priority, '')
        return f"[{status}] {priority_icon} {self.title}"


class TodoManager:
    """
    待办事项管理器

    提供待办事项的增删改查功能，数据保存到 JSON 文件
    """

    def __init__(self, data_file: str = "todos.json"):
        """
        初始化管理器

        Args:
            data_file: 数据存储文件路径
        """
        self.data_file = Path(data_file)
        self.todos: List[TodoItem] = []
        self.next_id = 1  # 下一个待办的 ID
        self._load_todos()

    def _load_todos(self):
        """从文件加载待办事项（内部方法）"""
        if not self.data_file.exists():
            print(f"数据文件不存在，将创建新文件：{self.data_file}")
            return

        try:
            with open(self.data_file, 'r', encoding='utf-8') as f:
                data = json.load(f)
                self.todos = [TodoItem.from_dict(item) for item in data.get('todos', [])]
                self.next_id = data.get('next_id', 1)
            print(f"已加载 {len(self.todos)} 条待办事项")
        except json.JSONDecodeError as e:
            print(f"数据文件解析失败：{e}，将从空数据开始")
        except Exception as e:
            print(f"加载数据失败：{e}，将从空数据开始")

    def _save_todos(self):
        """保存待办事项到文件（内部方法）"""
        try:
            # 确保父目录存在
            self.data_file.parent.mkdir(parents=True, exist_ok=True)

            data = {
                'todos': [todo.to_dict() for todo in self.todos],
                'next_id': self.next_id
            }

            with open(self.data_file, 'w', encoding='utf-8') as f:
                json.dump(data, f, ensure_ascii=False, indent=2)

        except IOError as e:
            print(f"保存数据失败：{e}")

    def add(self, title: str, priority: str = 'medium', tags: Optional[List[str]] = None) -> TodoItem:
        """
        添加新的待办事项

        Args:
            title: 待办标题
            priority: 优先级 (low/medium/high)
            tags: 标签列表

        Returns:
            新创建的 TodoItem
        """
        if not title.strip():
            raise ValueError("标题不能为空")

        todo = TodoItem(
            todo_id=self.next_id,
            title=title.strip(),
            priority=priority,
            tags=tags
        )

        self.todos.append(todo)
        self.next_id += 1
        self._save_todos()

        print(f"✓ 已添加：{todo}")
        return todo

    def delete(self, todo_id: int) -> bool:
        """
        删除待办事项

        Args:
            todo_id: 待办 ID

        Returns:
            是否删除成功
        """
        for i, todo in enumerate(self.todos):
            if todo.id == todo_id:
                deleted = self.todos.pop(i)
                self._save_todos()
                print(f"✓ 已删除：{deleted.title}")
                return True

        print(f"✗ 未找到 ID 为 {todo_id} 的待办事项")
        return False

    def complete(self, todo_id: int) -> bool:
        """
        标记待办事项为已完成

        Args:
            todo_id: 待办 ID

        Returns:
            是否操作成功
        """
        for todo in self.todos:
            if todo.id == todo_id:
                if todo.completed:
                    print(f"该待办事项已完成：{todo.title}")
                    return False

                todo.completed = True
                todo.completed_at = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
                self._save_todos()
                print(f"✓ 已完成：{todo.title}")
                return True

        print(f"✗ 未找到 ID 为 {todo_id} 的待办事项")
        return False

    def uncomplete(self, todo_id: int) -> bool:
        """重置待办事项为未完成"""
        for todo in self.todos:
            if todo.id == todo_id:
                todo.completed = False
                todo.completed_at = None
                self._save_todos()
                print(f"✓ 已重置：{todo.title}")
                return True

        print(f"✗ 未找到 ID 为 {todo_id} 的待办事项")
        return False

    def list_all(self, show_completed: bool = True, priority_filter: Optional[str] = None,
                 search_text: Optional[str] = None) -> List[TodoItem]:
        """
        列出待办事项

        Args:
            show_completed: 是否显示已完成的
            priority_filter: 按优先级过滤
            search_text: 搜索标题包含的文本

        Returns:
            过滤后的待办列表
        """
        result = self.todos.copy()

        # 过滤已完成的
        if not show_completed:
            result = [t for t in result if not t.completed]

        # 按优先级过滤
        if priority_filter:
            result = [t for t in result if t.priority == priority_filter]

        # 搜索文本
        if search_text:
            result = [t for t in result if search_text.lower() in t.title.lower()]

        # 排序：未完成的在前，然后按优先级（高到低），最后按创建时间
        priority_order = {'high': 0, 'medium': 1, 'low': 2}
        result.sort(key=lambda t: (
            t.completed,  # False < True，未完成的在前
            priority_order.get(t.priority, 1),
            t.created_at
        ))

        return result

    def display(self, todos: Optional[List[TodoItem]] = None):
        """
        格式化显示待办列表

        Args:
            todos: 要显示的待办列表，None 则显示全部
        """
        if todos is None:
            todos = self.todos

        if not todos:
            print("暂无待办事项")
            return

        print(f"\n{'ID':<4} {'状态':<4} {'优先级':<8} {'标题':<30} {'创建时间':<20} {'标签'}")
        print("-" * 90)

        for todo in todos:
            status = "✓" if todo.completed else "○"
            priority_str = todo.priority.upper()
            tags_str = ", ".join(todo.tags) if todo.tags else "-"
            print(f"{todo.id:<4} {status:<4} {priority_str:<8} "
                  f"{todo.title[:28]:<30} {todo.created_at:<20} {tags_str}")

        print("-" * 90)
        print(f"共 {len(todos)} 项\n")

    def get_statistics(self) -> Dict[str, Any]:
        """获取统计信息"""
        total = len(self.todos)
        completed = sum(1 for t in self.todos if t.completed)
        pending = total - completed

        by_priority = {
            'high': sum(1 for t in self.todos if t.priority == 'high' and not t.completed),
            'medium': sum(1 for t in self.todos if t.priority == 'medium' and not t.completed),
            'low': sum(1 for t in self.todos if t.priority == 'low' and not t.completed)
        }

        return {
            'total': total,
            'completed': completed,
            'pending': pending,
            'by_priority': by_priority
        }

    def show_statistics(self):
        """显示统计信息"""
        stats = self.get_statistics()
        print("\n=== 待办统计 ===")
        print(f"总计：{stats['total']} 项")
        print(f"已完成：{stats['completed']} 项")
        print(f"待处理：{stats['pending']} 项")
        print(f"待处理按优先级:")
        print(f"  高优先级：{stats['by_priority']['high']}")
        print(f"  中优先级：{stats['by_priority']['medium']}")
        print(f"  低优先级：{stats['by_priority']['low']}")


def print_help():
    """打印帮助信息"""
    print("""
=== 帮助 ===
add [标题]     - 添加待办事项
done [ID]      - 标记为已完成
undo [ID]      - 重置为未完成
del [ID]       - 删除待办
list           - 列出所有待办
pending        - 列出未完成的
high           - 列出高优先级的
stats          - 显示统计
search [关键词] - 搜索待办
help           - 显示帮助
quit/exit      - 退出程序
""")


def main():
    """主函数：交互式命令行界面"""

    print("=" * 50)
    print("       待办事项管理器")
    print("=" * 50)
    print_help()

    manager = TodoManager()

    # 显示初始统计
    manager.show_statistics()
    manager.display()

    while True:
        try:
            # 获取用户输入
            user_input = input("\n待办 > ").strip()

            if not user_input:
                continue

            # 解析命令和参数
            parts = user_input.split(maxsplit=1)
            command = parts[0].lower()
            args = parts[1] if len(parts) > 1 else ""

            # 命令处理
            if command in ('quit', 'exit', 'q'):
                print("再见！")
                break

            elif command == 'help':
                print_help()

            elif command == 'add':
                if not args:
                    print("用法：add [标题]")
                    continue
                # 解析优先级
                priority = 'medium'
                if args.startswith('!'):
                    parts = args.split(maxsplit=1)
                    p = parts[0][1:].lower()
                    if p in TodoItem.PRIORITIES:
                        priority = p
                        args = parts[1] if len(parts) > 1 else ""
                manager.add(args, priority)

            elif command in ('done', 'complete'):
                try:
                    todo_id = int(args)
                    manager.complete(todo_id)
                except ValueError:
                    print("用法：done [ID]")

            elif command == 'undo':
                try:
                    todo_id = int(args)
                    manager.uncomplete(todo_id)
                except ValueError:
                    print("用法：undo [ID]")

            elif command in ('del', 'delete', 'remove'):
                try:
                    todo_id = int(args)
                    manager.delete(todo_id)
                except ValueError:
                    print("用法：del [ID]")

            elif command == 'list':
                manager.display()

            elif command == 'pending':
                todos = manager.list_all(show_completed=False)
                manager.display(todos)

            elif command == 'high':
                todos = manager.list_all(priority_filter='high')
                manager.display(todos)

            elif command == 'stats':
                manager.show_statistics()

            elif command == 'search':
                if not args:
                    print("用法：search [关键词]")
                    continue
                todos = manager.list_all(search_text=args)
                manager.display(todos)

            else:
                print(f"未知命令：{command}，输入 'help' 查看帮助")

        except KeyboardInterrupt:
            print("\n程序已中断")
            break
        except Exception as e:
            print(f"发生错误：{e}")


if __name__ == "__main__":
    main()
```

## 15.4 项目四：网页爬虫

### 15.4.1 项目概述

网页爬虫可以自动抓取网页内容，提取有用信息。本项目实现一个简易的通用爬虫，支持提取标题、链接和正文内容。

**核心功能：**
- 获取网页 HTML 内容
- 解析并提取关键信息
- 保存为 Markdown 格式
- 支持基础的反爬处理

### 15.4.2 技术栈

| 库 | 用途 |
|----|------|
| requests | HTTP 请求 |
| BeautifulSoup | HTML 解析 |
| urllib.parse | URL 处理 |

### 15.4.3 代码实现

```python
import requests
from bs4 import BeautifulSoup, Tag
from urllib.parse import urljoin, urlparse
from datetime import datetime
from pathlib import Path
import re
import time
import random


class WebScraper:
    """
    简易网页爬虫

    支持抓取网页内容、提取信息、保存为 Markdown
    """

    def __init__(self, base_url: str, delay: float = 1.0):
        """
        初始化爬虫

        Args:
            base_url: 基础 URL（起始页面）
            delay: 请求间隔时间（秒），用于礼貌爬取
        """
        self.base_url = base_url
        self.delay = delay
        self.last_request_time = 0

        # 创建 Session 复用连接
        self.session = requests.Session()

        # 设置请求头，伪装成浏览器
        self.session.headers.update({
            'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 '
                         '(KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36',
            'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,image/webp,*/*;q=0.8',
            'Accept-Language': 'zh-CN,zh;q=0.9,en;q=0.8',
        })

        # 已访问的 URL 集合（防止重复）
        self.visited_urls = set()

        # 抓取结果
        self.crawled_pages = []

    def _respect_delay(self):
        """遵守请求间隔，避免过快"""
        elapsed = time.time() - self.last_request_time
        if elapsed < self.delay:
            time.sleep(self.delay - elapsed)
        self.last_request_time = time.time()

    def fetch_page(self, url: str, timeout: int = 10) -> Optional[str]:
        """
        获取网页 HTML 内容

        Args:
            url: 目标 URL
            timeout: 超时时间（秒）

        Returns:
            HTML 文本，失败返回 None
        """
        # 检查 URL 是否已访问
        if url in self.visited_urls:
            return None

        # 遵守爬取间隔
        self._respect_delay()

        try:
            response = self.session.get(url, timeout=timeout)

            # 检查响应状态
            if response.status_code == 200:
                # 自动识别编码
                response.encoding = response.apparent_encoding
                self.visited_urls.add(url)
                return response.text
            elif response.status_code == 404:
                print(f"404 未找到：{url}")
                return None
            elif response.status_code == 403:
                print(f"403 禁止访问：{url}")
                return None
            else:
                print(f"HTTP {response.status_code}: {url}")
                return None

        except requests.exceptions.Timeout:
            print(f"请求超时：{url}")
            return None
        except requests.exceptions.ConnectionError:
            print(f"连接失败：{url}")
            return None
        except requests.exceptions.RequestException as e:
            print(f"请求异常：{e}")
            return None

    def extract_info(self, html: str, url: str) -> Dict[str, Any]:
        """
        从 HTML 中提取信息

        Args:
            html: HTML 内容
            url: 来源 URL

        Returns:
            包含提取信息的字典
        """
        soup = BeautifulSoup(html, 'html.parser')

        # 移除不需要的元素
        for element in soup.find_all(['script', 'style', 'noscript', 'iframe']):
            element.decompose()

        # 提取标题
        title = self._extract_title(soup)

        # 提取所有链接
        links = self._extract_links(soup, url)

        # 提取正文内容
        content = self._extract_content(soup)

        # 提取 Meta 描述
        description = self._extract_meta_description(soup)

        # 提取图片
        images = self._extract_images(soup, url)

        return {
            'url': url,
            'title': title,
            'description': description,
            'links': links,
            'content': content,
            'images': images,
            'crawled_at': datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        }

    def _extract_title(self, soup: BeautifulSoup) -> str:
        """提取网页标题"""
        # 优先使用 title 标签
        if soup.title and soup.title.string:
            return soup.title.string.strip()

        # 尝试 h1 标签
        h1 = soup.find('h1')
        if h1:
            return h1.get_text(strip=True)

        return "无标题"

    def _extract_links(self, soup: BeautifulSoup, base_url: str) -> List[Dict[str, str]]:
        """提取所有链接"""
        links = []
        seen = set()

        for a in soup.find_all('a', href=True):
            href = a['href'].strip()
            text = a.get_text(strip=True) or "[无文本]"

            # 跳过锚点和 javascript
            if href.startswith(('#', 'javascript:', 'mailto:')):
                continue

            # 转换为绝对 URL
            full_url = urljoin(base_url, href)

            # 去重
            if full_url not in seen:
                seen.add(full_url)
                links.append({
                    'text': text[:100],  # 限制文本长度
                    'url': full_url
                })

        return links

    def _extract_content(self, soup: BeautifulSoup, max_paragraphs: int = 20) -> List[str]:
        """提取正文内容"""
        content = []

        # 尝试查找主要内容区域
        main_area = soup.find('main') or soup.find('article') or soup.find('div', class_=re.compile(r'content|main|article'))

        if not main_area:
            main_area = soup

        # 提取段落和标题
        for tag in main_area.find_all(['p', 'h1', 'h2', 'h3', 'h4', 'li']):
            text = tag.get_text(strip=True)
            if text and len(text) > 10:  # 过滤太短的文本
                content.append(text)
                if len(content) >= max_paragraphs:
                    break

        return content

    def _extract_meta_description(self, soup: BeautifulSoup) -> str:
        """提取 Meta 描述"""
        meta = soup.find('meta', attrs={'name': 'description'}) or \
               soup.find('meta', attrs={'property': 'og:description'})
        if meta and meta.get('content'):
            return meta['content'].strip()
        return ""

    def _extract_images(self, soup: BeautifulSoup, base_url: str) -> List[str]:
        """提取图片 URL"""
        images = []
        seen = set()

        for img in soup.find_all('img', src=True):
            src = img['src'].strip()
            if src.startswith('data:'):  # 跳过 base64 图片
                continue

            full_url = urljoin(base_url, src)
            if full_url not in seen:
                seen.add(full_url)
                images.append(full_url)

        return images

    def save_to_markdown(self, info: Dict[str, Any], filename: Optional[str] = None) -> str:
        """
        保存为 Markdown 文件

        Args:
            info: extract_info 返回的字典
            filename: 输出文件名（可选）

        Returns:
            保存的文件路径
        """
        if not filename:
            # 根据标题生成文件名
            safe_title = re.sub(r'[^\w\u4e00-\u9fff\-]', '_', info['title'])[:50]
            filename = f"{safe_title}.md"

        with open(filename, 'w', encoding='utf-8') as f:
            # 标题
            f.write(f"# {info['title']}\n\n")

            # 元信息
            f.write(f"> 来源：{info['url']}\n")
            f.write(f"> 抓取时间：{info['crawled_at']}\n\n")

            # 描述
            if info['description']:
                f.write(f"**描述**: {info['description']}\n\n")

            # 分隔线
            f.write("---\n\n")

            # 内容
            f.write("## 内容\n\n")
            for para in info['content']:
                f.write(f"{para}\n\n")

            # 图片
            if info['images']:
                f.write("\n---\n\n")
                f.write("## 图片\n\n")
                for i, img_url in enumerate(info['images'][:10], 1):
                    f.write(f"{i}. ![]({img_url})\n")

            # 链接
            if info['links']:
                f.write("\n---\n\n")
                f.write("## 外链\n\n")
                for i, link in enumerate(info['links'][:30], 1):
                    f.write(f"{i}. [{link['text']}]({link['url']})\n")

        return filename

    def crawl(self, max_pages: int = 5) -> List[Dict[str, Any]]:
        """
        爬取网站（从 base_url 开始）

        Args:
            max_pages: 最大爬取页数

        Returns:
            爬取结果列表
        """
        print(f"开始爬取：{self.base_url}")
        print(f"最大页数：{max_pages}")

        to_visit = [self.base_url]
        results = []

        while to_visit and len(results) < max_pages:
            url = to_visit.pop(0)

            print(f"\n爬取中：{url}")
            html = self.fetch_page(url)

            if not html:
                continue

            info = self.extract_info(html, url)
            results.append(info)
            self.crawled_pages.append(info)

            # 将新链接加入待访问队列
            for link in info['links'][:10]:  # 限制每页提取的链接数
                if link['url'] not in self.visited_urls and link['url'] not in to_visit:
                    # 只爬取同域名
                    if urlparse(link['url']).netloc == urlparse(self.base_url).netloc:
                        to_visit.append(link['url'])

        print(f"\n爬取完成，共 {len(results)} 页")
        return results


def main():
    """使用示例"""

    print("=" * 60)
    print("          网页爬虫工具")
    print("=" * 60)

    # 默认爬取 Python 官网
    url = input("输入要爬取的 URL（直接回车爬取 Python 官网）：").strip()
    if not url:
        url = "https://www.python.org/"

    try:
        max_pages = int(input("最大爬取页数（默认 3）：") or "3")
    except ValueError:
        max_pages = 3

    # 创建爬虫
    scraper = WebScraper(url, delay=1.0)

    # 爬取
    results = scraper.crawl(max_pages)

    if results:
        # 保存第一页
        filename = scraper.save_to_markdown(results[0])
        print(f"\n✓ 已保存到：{filename}")
        print(f"  标题：{results[0]['title']}")
        print(f"  链接数：{len(results[0]['links'])}")
        print(f"  内容段落：{len(results[0]['content'])}")


if __name__ == "__main__":
    main()
```

## 15.5 项目五：数据分析工具

### 15.5.1 项目概述

CSV 数据分析工具可以读取 CSV 文件，进行基本统计分析，生成分析报告。

**核心功能：**
- 读取 CSV 文件
- 数值列统计分析
- 文本列频次分析
- 生成文本报告

### 15.5.2 代码实现

```python
import csv
from collections import Counter, defaultdict
from datetime import datetime
from pathlib import Path
from typing import List, Dict, Any, Optional


class DataAnalyzer:
    """
    CSV 数据分析工具

    读取 CSV 文件并进行统计分析
    """

    def __init__(self, filename: str):
        """
        初始化工具

        Args:
            filename: CSV 文件路径
        """
        self.filename = Path(filename)
        self.data: List[Dict[str, str]] = []
        self.headers: List[str] = []
        self._loaded = False

    def load_data(self, encoding: str = 'utf-8', max_rows: Optional[int] = None) -> int:
        """
        加载 CSV 数据

        Args:
            encoding: 文件编码
            max_rows: 最大加载行数（None 表示全部加载）

        Returns:
            加载的行数
        """
        if not self.filename.exists():
            raise FileNotFoundError(f"文件不存在：{self.filename}")

        with open(self.filename, 'r', encoding=encoding) as f:
            reader = csv.DictReader(f)
            self.headers = reader.fieldnames or []

            if max_rows:
                self.data = [next(reader) for _ in range(max_rows)]
            else:
                self.data = list(reader)

        self._loaded = True
        print(f"✓ 已加载 {len(self.data)} 条记录")
        print(f"  字段：{', '.join(self.headers)}")
        return len(self.data)

    def _is_numeric(self, values: List[str]) -> bool:
        """判断一组值是否为数值类型"""
        numeric_count = 0
        for v in values:
            if not v.strip():
                continue
            try:
                float(v)
                numeric_count += 1
            except ValueError:
                pass

        # 如果超过 80% 的非空值是数值，则认为是数值列
        non_empty = len([v for v in values if v.strip()])
        return numeric_count > 0 and numeric_count / non_empty > 0.8

    def describe_column(self, column: str) -> Dict[str, Any]:
        """
        描述某一列的统计信息

        Args:
            column: 列名

        Returns:
            统计信息字典
        """
        if column not in self.headers:
            raise ValueError(f"列不存在：{column}")

        # 提取该列所有值
        values = [row[column] for row in self.data]
        non_empty = [v for v in values if v.strip()]

        result = {
            'column': column,
            'total': len(values),
            'non_empty': len(non_empty),
            'empty': len(values) - len(non_empty),
        }

        # 判断类型并计算相应统计量
        if self._is_numeric(non_empty):
            numeric_values = [float(v) for v in non_empty]
            result['type'] = 'numeric'
            result['min'] = min(numeric_values)
            result['max'] = max(numeric_values)
            result['sum'] = sum(numeric_values)
            result['avg'] = sum(numeric_values) / len(numeric_values) if numeric_values else 0

            # 排序找中位数
            sorted_values = sorted(numeric_values)
            mid = len(sorted_values) // 2
            result['median'] = sorted_values[mid]

        else:
            # 文本类型统计
            counter = Counter(non_empty)
            result['type'] = 'text'
            result['unique'] = len(counter)
            result['top'] = counter.most_common(10)  # 前 10 常见值

        return result

    def generate_report(self, output_file: Optional[str] = None) -> str:
        """
        生成分析报告

        Args:
            output_file: 输出文件路径（可选）

        Returns:
            报告内容
        """
        if not self._loaded:
            self.load_data()

        lines = []
        lines.append("=" * 70)
        lines.append(f"数据分析报告")
        lines.append(f"文件：{self.filename}")
        lines.append(f"生成时间：{datetime.now().strftime('%Y-%m-%d %H:%M:%S')}")
        lines.append("=" * 70)
        lines.append("")
        lines.append(f"总记录数：{len(self.data)}")
        lines.append(f"字段数：{len(self.headers)}")
        lines.append("")
        lines.append("-" * 70)

        # 逐个字段分析
        for header in self.headers:
            stats = self.describe_column(header)
            lines.append("")
            lines.append(f"字段：【{header}】")
            lines.append(f"  总行数：{stats['total']}")
            lines.append(f"  非空行：{stats['non_empty']}")
            lines.append(f"  空值数：{stats['empty']}")

            if stats['type'] == 'numeric':
                lines.append(f"  类型：数值")
                lines.append(f"  最小值：{stats['min']}")
                lines.append(f"  最大值：{stats['max']}")
                lines.append(f"  总和：{stats['sum']:.2f}" if isinstance(stats['sum'], float) else f"  总和：{stats['sum']}")
                lines.append(f"  平均值：{stats['avg']:.2f}")
                lines.append(f"  中位数：{stats['median']}")
            else:
                lines.append(f"  类型：文本")
                lines.append(f"  唯一值数量：{stats['unique']}")
                if stats['top']:
                    lines.append(f"  前 {len(stats['top'])} 常见值:")
                    for value, count in stats['top']:
                        bar = '█' * min(count, 50)  # 简单的条形图
                        lines.append(f"    {value[:30]:<30} |{bar}| ({count})")

            lines.append("")

        lines.append("-" * 70)
        lines.append("报告结束")
        lines.append("=" * 70)

        report = "\n".join(lines)

        # 保存到文件
        if output_file:
            with open(output_file, 'w', encoding='utf-8') as f:
                f.write(report)
            print(f"✓ 报告已保存到：{output_file}")

        return report

    def filter_rows(self, condition) -> 'DataAnalyzer':
        """
        过滤行

        Args:
            condition: 过滤函数，接收一行数据返回布尔值

        Returns:
            新的 DataAnalyzer 实例
        """
        new_analyzer = DataAnalyzer.__new__(DataAnalyzer)
        new_analyzer.filename = self.filename
        new_analyzer.headers = self.headers
        new_analyzer.data = [row for row in self.data if condition(row)]
        new_analyzer._loaded = True
        return new_analyzer

    def group_by(self, column: str, agg_column: str, agg_func: str = 'count') -> Dict[str, Any]:
        """
        分组聚合

        Args:
            column: 分组列
            agg_column: 聚合列
            agg_func: 聚合函数（count/sum/avg）

        Returns:
            分组结果字典
        """
        groups = defaultdict(list)

        for row in self.data:
            key = row.get(column, '')
            value = row.get(agg_column, '0')
            groups[key].append(value)

        result = {}
        for key, values in groups.items():
            if agg_func == 'count':
                result[key] = len(values)
            elif agg_func == 'sum':
                numeric = [float(v) for v in values if v.strip()]
                result[key] = sum(numeric)
            elif agg_func == 'avg':
                numeric = [float(v) for v in values if v.strip()]
                result[key] = sum(numeric) / len(numeric) if numeric else 0

        return result


def create_sample_data():
    """创建示例数据"""
    sample_data = [
        ['name', 'age', 'city', 'score', 'department'],
        ['Alice', '25', 'Beijing', '90', 'Engineering'],
        ['Bob', '30', 'Shanghai', '85', 'Sales'],
        ['Charlie', '28', 'Beijing', '92', 'Engineering'],
        ['David', '35', 'Guangzhou', '78', 'HR'],
        ['Eve', '22', 'Shanghai', '88', 'Sales'],
        ['Frank', '29', 'Beijing', '95', 'Engineering'],
        ['Grace', '31', 'Guangzhou', '82', 'HR'],
        ['Henry', '27', 'Shanghai', '89', 'Sales'],
    ]

    filename = Path('sample.csv')
    with open(filename, 'w', encoding='utf-8', newline='') as f:
        writer = csv.writer(f)
        writer.writerows(sample_data)

    print(f"✓ 已创建示例数据：{filename}")
    return str(filename)


def main():
    """主函数"""

    print("=" * 60)
    print("          CSV 数据分析工具")
    print("=" * 60)

    # 创建示例数据或加载已有数据
    filename = create_sample_data()

    # 加载数据
    analyzer = DataAnalyzer(filename)
    analyzer.load_data()

    # 生成报告
    print("\n" + analyzer.generate_report())

    # 分组聚合示例
    print("\n=== 按部门分组统计平均分 ===")
    dept_scores = analyzer.group_by('department', 'score', 'avg')
    for dept, avg in sorted(dept_scores.items(), key=lambda x: -x[1]):
        print(f"  {dept:<15}: {avg:.1f}")


if __name__ == "__main__":
    main()
```

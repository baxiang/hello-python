> 综合运用 StateGraph 构建一个多步骤数据处理管道

## 实战项目：智能文章分析管道

我们将构建一个多步骤的文章分析系统：

```
┌──────────┐     ┌──────────┐     ┌──────────┐     ┌──────────┐
│  文章输入 │────→│  内容清理 │────→│  关键词   │────→│  摘要    │
│          │     │          │     │  提取     │     │  生成    │
└──────────┘     └──────────┘     └──────────┘     └──────────┘
                    │                                  │
                    ▼                                  ▼
               ┌──────────┐     ┌──────────┐     ┌──────────┐
               │  语言检测 │────→│  情感分析 │────→│  报告    │
               │          │     │          │     │  输出    │
               └──────────┘     └──────────┘     └──────────┘
```

## State 定义

```python
# app/state.py
from typing import Annotated, TypedDict
from operator import add
from langgraph.graph.message import add_messages


class AnalysisResult(TypedDict):
    """单次分析的结果"""
    type: str       # "keywords" | "summary" | "sentiment" | "language"
    content: str
    confidence: float


class ArticleState(TypedDict):
    """文章分析的完整状态"""
    # 输入
    raw_article: str
    
    # 中间结果
    cleaned_article: str
    detected_language: str
    keywords: list
    sentiment: str
    summary: str
    
    # 分析结果集合
    analyses: Annotated[list, add_messages]
    
    # 最终输出
    report: str
    
    # 执行追踪
    step_log: Annotated[list, add]
    error: str | None
```

## 节点实现

```python
# app/nodes.py
import re
from langchain.chat_models import init_chat_model
from langchain.messages import HumanMessage, AIMessage, SystemMessage
from app.state import ArticleState


def clean_article(state: ArticleState) -> dict:
    """清理文章：去除多余空白、标准化标点"""
    text = state["raw_article"]
    
    # 去除多余空白
    text = re.sub(r"\s+", " ", text).strip()
    
    # 去除不可见字符
    text = re.sub(r"[\x00-\x1f\x7f-\x9f]", "", text)
    
    return {
        "cleaned_article": text,
        "step_log": ["[clean] 文章已清理"],
    }


def detect_language(state: ArticleState) -> dict:
    """检测文章语言"""
    model = init_chat_model("moonshot:moonshot-v1-8k")
    
    response = model.invoke([
        SystemMessage("检测以下文本的语言。只返回语言名称，如：中文、英文、日文。"),
        HumanMessage(state["cleaned_article"][:200]),
    ])
    
    language = response.content.strip()
    
    return {
        "detected_language": language,
        "step_log": [f"[language] 检测语言: {language}"],
    }


def extract_keywords(state: ArticleState) -> dict:
    """提取关键词"""
    model = init_chat_model("moonshot:moonshot-v1-8k")
    
    text = state["cleaned_article"]
    if len(text) > 3000:
        text = text[:3000]  # 截断避免超长
    
    response = model.invoke([
        SystemMessage(
            "从以下文本中提取 5 个关键词。每行一个关键词，不要编号。"
        ),
        HumanMessage(text),
    ])
    
    keywords = [
        line.strip().strip("•-*")
        for line in response.content.split("\n")
        if line.strip()
    ][:5]
    
    return {
        "keywords": keywords,
        "step_log": [f"[keywords] 提取关键词: {', '.join(keywords)}"],
    }


def analyze_sentiment(state: ArticleState) -> dict:
    """情感分析"""
    model = init_chat_model("moonshot:moonshot-v1-8k")
    
    response = model.invoke([
        SystemMessage(
            "分析以下文本的情感。返回以下之一："
            "positive（正面）、negative（负面）、neutral（中性）。"
            "只返回关键词。"
        ),
        HumanMessage(state["cleaned_article"][:2000]),
    ])
    
    sentiment = response.content.strip().lower()
    # 标准化
    if "positive" in sentiment:
        sentiment = "positive"
    elif "negative" in sentiment:
        sentiment = "negative"
    else:
        sentiment = "neutral"
    
    return {
        "sentiment": sentiment,
        "step_log": [f"[sentiment] 情感分析: {sentiment}"],
    }


def generate_summary(state: ArticleState) -> dict:
    """生成摘要"""
    model = init_chat_model("moonshot:moonshot-v1-8k")
    
    text = state["cleaned_article"]
    if len(text) > 4000:
        text = text[:4000]
    
    response = model.invoke([
        SystemMessage("用 2-3 句话总结以下文本的核心内容："),
        HumanMessage(text),
    ])
    
    return {
        "summary": response.content.strip(),
        "step_log": ["[summary] 摘要已生成"],
    }


def generate_report(state: ArticleState) -> dict:
    """生成分析报告"""
    keywords_str = ", ".join(state["keywords"]) if state["keywords"] else "无"
    
    report = (
        f"{'='*50}\n"
        f"📄 文章分析报告\n"
        f"{'='*50}\n\n"
        f"📝 摘要:\n   {state['summary']}\n\n"
        f"🔤 语言: {state['detected_language']}\n\n"
        f"🔑 关键词: {keywords_str}\n\n"
        f"💭 情感: {state['sentiment']}\n\n"
        f"📊 分析步骤:\n"
    )
    
    for i, step in enumerate(state["step_log"], 1):
        report += f"   {i}. {step}\n"
    
    report += f"\n{'='*50}\n"
    
    return {
        "report": report,
        "step_log": ["[report] 报告已生成"],
    }
```

## 图构建

```python
# app/graph.py
from langgraph.graph import StateGraph, START, END
from app.state import ArticleState
from app.nodes import (
    clean_article,
    detect_language,
    extract_keywords,
    analyze_sentiment,
    generate_summary,
    generate_report,
)


def build_analyzer():
    """构建文章分析图"""
    builder = StateGraph(ArticleState)
    
    # 注册节点
    builder.add_node("clean", clean_article)
    builder.add_node("detect_language", detect_language)
    builder.add_node("extract_keywords", extract_keywords)
    builder.add_node("analyze_sentiment", analyze_sentiment)
    builder.add_node("generate_summary", generate_summary)
    builder.add_node("generate_report", generate_report)
    
    # 线性流程
    builder.add_edge(START, "clean")
    builder.add_edge("clean", "detect_language")
    builder.add_edge("detect_language", "extract_keywords")
    builder.add_edge("extract_keywords", "analyze_sentiment")
    builder.add_edge("analyze_sentiment", "generate_summary")
    builder.add_edge("generate_summary", "generate_report")
    builder.add_edge("generate_report", END)
    
    return builder.compile()
```

## 进阶：使用 stream 观察逐步执行

```python
def analyze_with_stream(article: str):
    """流式分析 — 每完成一步就输出"""
    graph = build_analyzer()
    
    for event in graph.stream(
        {
            "raw_article": article,
            "cleaned_article": "",
            "detected_language": "",
            "keywords": [],
            "sentiment": "",
            "summary": "",
            "report": "",
            "analyses": [],
            "step_log": [],
            "error": None,
        },
        stream_mode="updates",
    ):
        node_name = list(event.keys())[0]
        node_result = event[node_name]
        
        # 打印步骤日志
        if "step_log" in node_result:
            for log in node_result["step_log"]:
                print(f"  {log}")
        
        print()  # 空行分隔
```

```python
def safe_analyze(article: str) -> str:
    """带错误处理的分析"""
    graph = build_analyzer()
    
    try:
        result = graph.invoke({
            "raw_article": article,
            "cleaned_article": "",
            "detected_language": "",
            "keywords": [],
            "sentiment": "",
            "summary": "",
            "report": "",
            "analyses": [],
            "step_log": [],
            "error": None,
        })
        return result["report"]
    except Exception as e:
        # 获取错误信息
        error_report = (
            f"{'='*50}\n"
            f"❌ 分析失败\n"
            f"{'='*50}\n\n"
            f"错误信息: {str(e)}\n\n"
            f"请检查：\n"

            f"2. 网络连接是否正常\n"
            f"3. 输入文章是否为空\n"
        )
        return error_report
```

```python
# 修改图构建，让语言检测、关键词提取、情感分析并行执行
def build_parallel_analyzer():
    builder = StateGraph(ArticleState)
    
    builder.add_node("clean", clean_article)
    builder.add_node("detect_language", detect_language)
    builder.add_node("extract_keywords", extract_keywords)
    builder.add_node("analyze_sentiment", analyze_sentiment)
    builder.add_node("generate_summary", generate_summary)
    builder.add_node("generate_report", generate_report)
    
    # 清理后并行执行三个独立分析
    builder.add_edge(START, "clean")
    builder.add_edge("clean", "detect_language")
    builder.add_edge("clean", "extract_keywords")
    builder.add_edge("clean", "analyze_sentiment")
    
    # 全部完成后生成摘要和报告
    builder.add_edge("detect_language", "generate_summary")
    builder.add_edge("extract_keywords", "generate_summary")
    builder.add_edge("analyze_sentiment", "generate_summary")
    builder.add_edge("generate_summary", "generate_report")
    builder.add_edge("generate_report", END)
    
    return builder.compile()
```

## 进阶：使用 Command 改写路由

```python
from langgraph.types import Command

def analyze_with_command(state: ArticleState) -> Command:
    """使用 Command 的路由节点"""
    # 检查文章长度决定是否跳过分析
    if len(state["cleaned_article"]) < 50:
        return Command(
            update={
                "summary": "文章过短，无法分析",
                "step_log": ["[command] 文章过短，跳过分析"],
            },
            goto=END,
        )
    
    return Command(
        update={"step_log": ["[command] 开始完整分析"]},
        goto="detect_language",
    )
```

## 知识点回顾

本实战项目综合运用了之前学到的所有概念：

| 章节 | 应用 |
|------|------|
| 图基础概念 | StateGraph、compile、stream |
| State 与 Reducers | TypedDict、Annotated + add_messages/operator.add |
| Nodes | 6 个节点函数，每个接收 State 返回 dict |
| Edges | 线性边 + 并行边 |
| Send 与 Command | 进阶示例中的条件路由 |
| 运行时上下文 | config 传递、stream_mode |
| 可视化与调试 | stream 观察、get_graph 查看结构 |

## 扩展方向

1. **添加更多分析节点**：文本分类、命名实体识别、可读性评分
2. **使用检查点**：保存分析进度，支持恢复
3. **集成 LangSmith**：追踪每次分析的执行
4. **创建 Web 界面**：用 FastAPI + Gradio 构建交互式分析工具
5. **添加缓存**：对相同文章的重复分析使用缓存
6. **多语言支持**：根据检测到的语言使用不同的分析策略

## 小结

| 要点 | 说明 |
|------|------|
| State 设计 | 区分输入、中间结果、输出、追踪字段 |
| 节点职责 | 每个节点做一件事，返回更新 dict |
| 图构建 | 线性流程最简单，按需添加并行 |
| stream | 开发时用 stream 观察每步输出 |
| 错误处理 | try/except + 友好的错误信息 |
| 测试 | 用默认 State 工厂函数简化测试 |

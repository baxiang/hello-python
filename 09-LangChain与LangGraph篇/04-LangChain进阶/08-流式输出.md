# 09.04.08 - 流式输出

> Message 流式输出、工具调用流式输出、astream_events、SSE FastAPI 部署

## 为什么需要流式输出

```
流式输出场景：
┌──────────────────────────────────────┐
│  场景1：实时对话                      │
│  用户等待时逐字显示 AI 回复           │
│  减少感知延迟                        │
│                                       │
│  场景2：工具调用监控                  │
│  实时显示 Agent 思考过程              │
│  哪些工具被调用、参数是什么           │
│                                       │
│  场景3：SSE 部署                      │
│  FastAPI + Server-Sent Events         │
│  前端实时接收 token 流               │
│                                       │
│  场景4：进度反馈                      │
│  长任务执行中的进度更新               │
│  避免用户认为系统卡死                │
└──────────────────────────────────────┘
```

### 流式 vs 非流式

| 特性 | 非流式 | 流式 |
|------|--------|------|
| 响应延迟 | 等待全部生成完成 | 首个 token 立即返回 |
| 用户体验 | 长时间等待 | 实时看到输出 |
| 内存占用 | 缓存全部响应 | 逐块处理 |
| 适用场景 | 批量处理、API | 实时对话、UI |

## Message 流式输出

### 基本流式调用

```python
from langchain.chat_models import init_chat_model

model = init_chat_model("moonshot:moonshot-v1-8k")

# 流式输出
for chunk in model.stream("请介绍 Python 的优点"):
    print(chunk.text, end="", flush=True)
```

### Agent 流式输出

```python
from langchain.agents import create_agent

agent = create_agent(
    model="moonshot:moonshot-v1-8k",
    tools=[search_tool, calc_tool],
)

# 流式执行 Agent
for chunk in agent.stream(
    {"messages": [{"role": "user", "content": "Python 和 Java 有什么区别？"}]},
    stream_mode="messages",
):
    token, metadata = chunk
    print(token.text, end="", flush=True)
```

### stream_mode 选项

```
stream_mode 选项：
┌──────────────────────────────────────┐
│  messages    → 模型输出的 token 流    │
│  updates     → State 更新事件         │
│  custom      → 自定义事件             │
│  debug       → 调试信息               │
│  events      → 所有事件（包含元数据） │
└──────────────────────────────────────┘
```

```python
# 获取消息流
for chunk in agent.stream(inputs, stream_mode="messages"):
    token, metadata = chunk
    print(f"Token: {token.text}")

# 获取 State 更新
for chunk in agent.stream(inputs, stream_mode="updates"):
    node, update = chunk
    print(f"节点 {node} 更新: {update}")

# 获取所有事件
for chunk in agent.stream(inputs, stream_mode="events"):
    print(f"事件: {chunk}")
```

## 工具调用流式输出

### 监控工具调用

```python
from langchain.agents import create_agent

agent = create_agent(
    model="deepseek:deepseek-chat",
    tools=[search_tool, weather_tool],
)

# 流式执行，同时获取消息和更新
for chunk in agent.stream(
    {"messages": [{"role": "user", "content": "北京今天天气怎么样？"}]},
    stream_mode=["messages", "updates"],
    version="v2",
):
    if chunk["type"] == "messages":
        token, metadata = chunk["data"]
        
        # 判断是否来自工具调用
        if metadata.get("tool_call"):
            print(f"[工具调用] {metadata['tool_call']['name']}")
        else:
            print(token.text, end="", flush=True)
    
    elif chunk["type"] == "updates":
        node, update = chunk["data"]
        print(f"\n[节点更新] {node}: {update}")
```

### 子图流式输出

## astream_events

### astream_events v2

```python
from langchain.agents import create_agent

agent = create_agent(
    model="moonshot:moonshot-v1-8k",
    tools=[search_tool],
)

# 异步流式事件
async for event in agent.astream_events(
    {"messages": [{"role": "user", "content": "搜索 Python 教程"}]},
    version="v2",
):
    kind = event["event"]
    
    if kind == "on_chat_model_stream":
        # 模型流式输出
        token = event["data"]["chunk"].content
        print(token, end="", flush=True)
    
    elif kind == "on_tool_start":
        # 工具开始执行
        print(f"\n[工具开始] {event['name']}")
        print(f"参数: {event['data'].get('input')}")
    
    elif kind == "on_tool_end":
        # 工具执行完成
        print(f"\n[工具完成] {event['name']}")
        print(f"结果: {event['data'].get('output')[:100]}...")
    
    elif kind == "on_chain_start":
        # 链/节点开始执行
        print(f"\n[链开始] {event['name']}")
    
    elif kind == "on_chain_end":
        # 链/节点执行完成
        print(f"\n[链完成] {event['name']}")
```

### 事件类型总览

```
astream_events 事件类型：
┌──────────────────────────────────────┐
│  on_chat_model_start                 │
│    ↓                                 │
│  on_chat_model_stream (多次)          │
│    ↓                                 │
│  on_chat_model_end                   │
│    ↓                                 │
│  on_tool_start                       │
│    ↓                                 │
│  on_tool_stream (多次，如有)          │
│    ↓                                 │
│  on_tool_end                         │
│    ↓                                 │
│  on_chain_start                      │
│    ↓                                 │
│  on_chain_stream (多次)              │
│    ↓                                 │
│  on_chain_end                        │
└──────────────────────────────────────┘
```

### 过滤事件

```python
# 只关注模型流式输出
async for event in agent.astream_events(
    inputs,
    version="v2",
    include_names=["ChatOpenAI"],  # 只包含特定名称的事件
):
    if event["event"] == "on_chat_model_stream":

# 排除工具事件
async for event in agent.astream_events(
    inputs,
    version="v2",
    exclude_tags=["tool"],  # 排除工具相关事件
):
    print(event)
```

## FastAPI SSE 部署

### 基本 SSE 端点

```python
from fastapi import FastAPI
from fastapi.responses import StreamingResponse
from langchain.agents import create_agent
from pydantic import BaseModel

app = FastAPI()

# 创建 Agent
agent = create_agent(
    model="moonshot:moonshot-v1-8k",
    tools=[search_tool],
)

class ChatRequest(BaseModel):
    thread_id: str
    message: str

@app.post("/chat/stream")
async def stream_chat(request: ChatRequest):
    """SSE 流式聊天"""
    config = {"configurable": {"thread_id": request.thread_id}}
    inputs = {"messages": [{"role": "user", "content": request.message}]}
    
    async def generate():
        async for event in agent.astream_events(inputs, config=config, version="v2"):
            kind = event["event"]
            
            if kind == "on_chat_model_stream":
                token = event["data"]["chunk"].content
                if token:
                    # SSE 格式
                    yield f"data: {token}\n\n"
            
            elif kind == "on_tool_start":
                tool_name = event["name"]
                yield f"data: [TOOL_START] {tool_name}\n\n"
            
            elif kind == "on_tool_end":
                tool_name = event["name"]
                yield f"data: [TOOL_END] {tool_name}\n\n"
        
        # 结束标记
        yield "data: [DONE]\n\n"
    
    return StreamingResponse(
        generate(),
        media_type="text/event-stream",
        headers={
            "Cache-Control": "no-cache",
            "Connection": "keep-alive",
        },
    )
```

### 完整 SSE 端点

```python
from fastapi import FastAPI
from fastapi.responses import StreamingResponse
from langchain.agents import create_agent
from langgraph.checkpoint.memory import MemorySaver
from pydantic import BaseModel
import json
import asyncio

app = FastAPI()

# 创建带记忆的 Agent
checkpointer = MemorySaver()
agent = create_agent(
    model="moonshot:moonshot-v1-8k",
    tools=[search_tool, calc_tool],
    checkpointer=checkpointer,
)

class ChatRequest(BaseModel):
    thread_id: str
    message: str
    include_thoughts: bool = False  # 是否包含思考过程

@app.post("/v1/chat/completions")
async def chat_completions(request: ChatRequest):
    """OpenAI 兼容的流式聊天端点"""
    config = {"configurable": {"thread_id": request.thread_id}}
    inputs = {"messages": [{"role": "user", "content": request.message}]}
    
    async def generate():
        # 发送创建事件
        yield f"data: {json.dumps({'type': 'create'})}\n\n"
        
        async for event in agent.astream_events(inputs, config=config, version="v2"):
            kind = event["event"]
            
            if kind == "on_chat_model_stream":
                token = event["data"]["chunk"].content
                if token:
                    data = {
                        "type": "content",
                        "content": token,
                    }
                    yield f"data: {json.dumps(data)}\n\n"
            
            elif kind == "on_tool_start" and request.include_thoughts:
                data = {
                    "type": "tool_start",
                    "tool": event["name"],
                    "args": event["data"].get("input"),
                }
                yield f"data: {json.dumps(data)}\n\n"
            
            elif kind == "on_tool_end" and request.include_thoughts:
                data = {
                    "type": "tool_end",
                    "tool": event["name"],
                    "output": str(event["data"].get("output", ""))[:200],
                }
                yield f"data: {json.dumps(data)}\n\n"
        
        # 发送完成事件
        yield f"data: {json.dumps({'type': 'done'})}\n\n"
        yield "data: [DONE]\n\n"
    
    return StreamingResponse(
        generate(),
        media_type="text/event-stream",
        headers={
            "Cache-Control": "no-cache",
            "Connection": "keep-alive",
            "X-Accel-Buffering": "no",  # 禁用 Nginx 缓冲
        },
    )
```

### 前端集成

```html
<!DOCTYPE html>
<html>
<head>
    <title>AI 助手</title>
</head>
<body>
    <div id="messages"></div>
    <input type="text" id="input" placeholder="输入消息..." />
    <button onclick="sendMessage()">发送</button>
    
    <script>
        const messagesDiv = document.getElementById('messages');
        const input = document.getElementById('input');
        let currentResponse = null;
        
        async function sendMessage() {
            const message = input.value;
            input.value = '';
            
            // 显示用户消息
            appendMessage('user', message);
            
            // 创建 AI 回复容器
            currentResponse = appendMessage('ai', '');
            
            // 发起 SSE 请求
            const response = await fetch('/v1/chat/completions', {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({
                    thread_id: 'session-1',
                    message: message,
                    include_thoughts: true,
                }),
            });
            
            const reader = response.body.getReader();
            const decoder = new TextDecoder();
            
            while (true) {
                const {value, done} = await reader.read();
                if (done) break;
                
                const text = decoder.decode(value);
                const lines = text.split('\n');
                
                for (const line of lines) {
                    if (line.startsWith('data: ')) {
                        const data = line.slice(6);
                        
                        if (data === '[DONE]') continue;
                        
                        try {
                            const parsed = JSON.parse(data);
                            
                            if (parsed.type === 'content') {
                                currentResponse.textContent += parsed.content;
                            } else if (parsed.type === 'tool_start') {
                                currentResponse.textContent += `\n[工具调用: ${parsed.tool}]\n`;
                            } else if (parsed.type === 'tool_end') {
                                currentResponse.textContent += `\n[工具完成]\n`;
                            }
                        } catch (e) {
                            // 纯文本
                            currentResponse.textContent += data;
                        }
                    }
                }
            }
        }
        
        function appendMessage(role, content) {
            const div = document.createElement('div');
            div.className = role;
            div.textContent = content;
            messagesDiv.appendChild(div);
            return div;
        }
        
        input.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') sendMessage();
        });
    </script>
</body>
</html>
```

## 流式输出最佳实践

### 1. 错误处理

```python
async def safe_stream(agent, inputs, config):
    """安全的流式输出"""
    try:
        async for event in agent.astream_events(inputs, config=config, version="v2"):
            if event["event"] == "on_chat_model_stream":
                token = event["data"]["chunk"].content
                if token:
                    yield f"data: {token}\n\n"
    except Exception as e:
        yield f"data: [ERROR] {str(e)}\n\n"
    finally:
        yield "data: [DONE]\n\n"
```

### 2. 超时控制

```python
import asyncio

async def stream_with_timeout(agent, inputs, config, timeout=60):
    """带超时的流式输出"""
    async def generate():
        async for event in agent.astream_events(inputs, config=config, version="v2"):
            if event["event"] == "on_chat_model_stream":
                token = event["data"]["chunk"].content
                if token:
                    yield f"data: {token}\n\n"
        
        yield "data: [DONE]\n\n"
    
    try:
        return await asyncio.wait_for(
            StreamingResponse(generate(), media_type="text/event-stream"),
            timeout=timeout,
        )
    except asyncio.TimeoutError:
        return {"error": "响应超时"}
```

### 3. 心跳包

```python
async def stream_with_heartbeat(agent, inputs, config):
    """带心跳包的流式输出"""
    last_event_time = asyncio.get_event_loop().time()
    
    async def heartbeat():
        """定期发送心跳"""
        while True:
            await asyncio.sleep(15)  # 每 15 秒
            if asyncio.get_event_loop().time() - last_event_time > 15:
                yield "data: [HEARTBEAT]\n\n"
    
    async def generate():
        nonlocal last_event_time
        
        async for event in agent.astream_events(inputs, config=config, version="v2"):
            last_event_time = asyncio.get_event_loop().time()
            
            if event["event"] == "on_chat_model_stream":
                token = event["data"]["chunk"].content
                if token:
                    yield f"data: {token}\n\n"
        
        yield "data: [DONE]\n\n"
    
    # 合并生成器和心跳
    return StreamingResponse(
        generate(),
        media_type="text/event-stream",
    )
```

### 完整事件日志

```python
async def debug_stream(agent, inputs, config):
    """调试：记录所有事件"""
    events_log = []
    
    async for event in agent.astream_events(inputs, config=config, version="v2"):
        events_log.append({
            "event": event["event"],
            "name": event.get("name"),
            "tags": event.get("tags"),
            "metadata": event.get("metadata"),
        })
        
        # 实时输出
        if event["event"] == "on_chat_model_stream":
            token = event["data"]["chunk"].content
            print(token, end="", flush=True)
    
    # 打印事件序列
    print("\n\n事件序列:")
    for i, e in enumerate(events_log):
        print(f"{i}: {e['event']} - {e['name']}")
    
    return events_log
```

## 小结

| 要点 | 说明 |
|------|------|
| stream | 同步流式输出，逐 token 返回 |
| astream_events v2 | 异步事件流，包含丰富元数据 |
| stream_mode | messages / updates / custom / debug / events |
| SSE 部署 | FastAPI + StreamingResponse + text/event-stream |
| 前端集成 | EventSource 或 fetch + ReadableStream |
| 最佳实践 | 错误处理、超时控制、心跳包 |

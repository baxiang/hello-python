# 09.04.06 - 动态工具选择

> 根据上下文动态过滤、注册和切换工具，实现灵活的 Agent 能力

## 为什么需要动态工具选择

```
工具选择场景：
┌──────────────────────────────────────┐
│  场景1：权限控制                      │
│  用户: 游客                          │
│  可用工具: 搜索、天气                │
│  禁用工具: 邮件、删除、配置          │
│                                       │
│  场景2：按需加载                      │
│  查询: "今天天气如何"                 │
│  加载工具: weather_query             │
│  不加载: search, email, calc         │
│                                       │
│  场景3：上下文相关                    │
│  对话阶段: 代码编写                   │
│  加载工具: code_exec, debug          │
│  不加载: email, file_manage          │
│                                       │
│  场景4：MCP 集成                      │
│  动态连接外部工具服务器               │
│  按需发现和使用工具                   │
└──────────────────────────────────────┘
```

### 工具选择 vs 静态工具

| 特性 | 静态工具 | 动态工具 |
|------|---------|---------|
| 工具列表 | 创建时固定 | 运行时可变 |
| Token 消耗 | 所有工具描述都发送 | 只发送相关工具 |
| 权限控制 | 无 | 支持 |
| 扩展性 | 需要重新创建 Agent | 运行时添加 |
| 适用场景 | 简单 Agent | 复杂应用 |

## 过滤预注册工具

### 方案1：按工具名称过滤

```python
from langchain.agents.middleware import wrap_model_call, ModelRequest, ModelResponse

@wrap_model_call
def name_based_filter(request: ModelRequest, handler) -> ModelResponse:
    """按工具名称过滤"""
    all_tools = request.tools
    
    # 只保留 public_ 开头的工具
    public_tools = [t for t in all_tools if t.name.startswith("public_")]
    
    print(f"总工具数: {len(all_tools)}")
    print(f"过滤后: {[t.name for t in public_tools]}")
    
    return handler(request.override(tools=public_tools))
```

### 方案2：按工具类别过滤

```python
from langchain.agents.middleware import wrap_model_call, ModelRequest, ModelResponse

# 工具分类
TOOL_CATEGORIES = {
    "search": ["web_search", "image_search", "doc_search"],
    "communication": ["send_email", "send_sms", "send_notification"],
    "calculation": ["calculator", "unit_converter", "currency_converter"],
    "file": ["read_file", "write_file", "delete_file"],
    "admin": ["create_user", "delete_user", "system_config"],
}

@wrap_model_call
def category_based_filter(request: ModelRequest, handler) -> ModelResponse:
    """按工具类别过滤"""
    all_tools = request.tools
    allowed_categories = request.context.allowed_categories
    
    # 收集允许类别的所有工具
    allowed_tool_names = set()
    for category in allowed_categories:
        allowed_tool_names.update(TOOL_CATEGORIES.get(category, []))
    
    # 过滤
    filtered_tools = [t for t in all_tools if t.name in allowed_tool_names]
    
    print(f"允许类别: {allowed_categories}")
    print(f"过滤后工具: {[t.name for t in filtered_tools]}")
    
    return handler(request.override(tools=filtered_tools))
```

### 方案3：按查询内容过滤

```python
from langchain.agents.middleware import wrap_model_call, ModelRequest, ModelResponse

# 工具关键词映射
TOOL_KEYWORDS = {
    "web_search": ["搜索", "查找", "查询", "search"],
    "weather_query": ["天气", "temperature", "weather"],
    "calculator": ["计算", "多少", "calc"],
    "send_email": ["邮件", "email", "发送"],
    "read_file": ["读取", "文件", "file", "read"],
}

@wrap_model_call
def query_based_filter(request: ModelRequest, handler) -> ModelResponse:
    """按查询内容过滤工具"""
    last_msg = request.state["messages"][-1]
    query = last_msg.content.lower()
    
    # 匹配相关工具
    relevant_tools = []
    for tool_name, keywords in TOOL_KEYWORDS.items():
        if any(kw in query for kw in keywords):
            relevant_tools.append(tool_name)
    
    # 过滤工具
    if relevant_tools:
        filtered_tools = [t for t in request.tools if t.name in relevant_tools]
        print(f"查询匹配工具: {relevant_tools}")
    else:
        # 没有匹配到，保留所有工具
        filtered_tools = request.tools
        print("无匹配工具，保留所有")
    
    return handler(request.override(tools=filtered_tools))
```

## 运行时工具注册

### 方案1：通过 AgentMiddleware 注册

```python
from langchain.agents.middleware import AgentMiddleware, ModelRequest, ModelResponse, ToolCallRequest
from langchain.tools import tool

@tool
def get_time() -> str:
    """获取当前时间"""
    import datetime
    return datetime.datetime.now().isoformat()

@tool
def calculate(expression: str) -> str:
    """计算表达式"""
    try:
        result = eval(expression)
        return str(result)
    except Exception as e:
        return f"计算错误: {e}"

class DynamicToolMiddleware(AgentMiddleware):
    """动态工具中间件"""
    
    def wrap_model_call(self, request: ModelRequest, handler) -> ModelResponse:
        # 根据上下文添加工具
        tools = list(request.tools)
        
        # 如果查询涉及时间，添加 get_time
        last_msg = request.state["messages"][-1]
        if "时间" in last_msg.content or "几点" in last_msg.content:
            if not any(t.name == "get_time" for t in tools):
                tools.append(get_time)
        
        # 如果查询涉及计算，添加 calculate
        if "计算" in last_msg.content or "多少" in last_msg.content:
            if not any(t.name == "calculate" for t in tools):
                tools.append(calculate)
        
        print(f"动态工具: {[t.name for t in tools]}")
        
        return handler(request.override(tools=tools))
    
    def wrap_tool_call(self, request: ToolCallRequest, handler):
        # 处理动态注册的工具调用
        if request.tool_call["name"] == "get_time":
            return get_time.invoke({})
        elif request.tool_call["name"] == "calculate":
            return calculate.invoke(request.tool_call["args"])
        
        return handler(request)
```

### 方案2：基于 State 的工具注册

```python
from langchain.agents.middleware import wrap_model_call, ModelRequest, ModelResponse
from langchain.tools import tool

@tool
def search_knowledge_base(query: str) -> str:
    """搜索知识库"""
    return f"知识库搜索结果: 关于 {query} 的信息"

@tool
def search_web(query: str) -> str:
    """搜索网页"""
    return f"网页搜索结果: 关于 {query} 的信息"

@wrap_model_call
def state_based_tool_selection(request: ModelRequest, handler) -> ModelResponse:
    """根据 State 选择工具"""
    tools = list(request.tools)
    
    # 根据对话阶段选择工具
    stage = request.state.get("conversation_stage", "initial")
    
    if stage == "initial":
        # 初始阶段只提供基础工具
        tools = [t for t in tools if t.name in ["search_web"]]
    elif stage == "research":
        # 研究阶段提供所有搜索工具
        if not any(t.name == "search_knowledge_base" for t in tools):
            tools.append(search_knowledge_base)
    elif stage == "analysis":
        # 分析阶段提供分析工具
        pass
    
    print(f"对话阶段: {stage}")
    print(f"选中工具: {[t.name for t in tools]}")
    
    return handler(request.override(tools=tools))
```

### 方案3：基于 Store 的工具注册

```python
from langchain.agents.middleware import wrap_model_call, ModelRequest, ModelResponse
from langchain.tools import tool

@tool
def use_personal_tool(query: str) -> str:
    """使用个人工具"""
    return f"个人工具结果: {query}"

@wrap_model_call
def store_based_tool_selection(request: ModelRequest, handler) -> ModelResponse:
    """基于 Store 选择工具"""
    user_id = request.context.user_id
    
    # 从 Store 加载用户自定义工具配置
    config = request.runtime.store.get(
        ("user", user_id),
        "tool_config"
    )
    
    tools = list(request.tools)
    
    if config:
        enabled_tools = config.value.get("enabled_tools", [])
        
        # 只启用用户配置的工具
        tools = [t for t in tools if t.name in enabled_tools]
        
        # 加载用户自定义工具
        if config.value.get("use_personal", False):
            if not any(t.name == "use_personal_tool" for t in tools):
                tools.append(use_personal_tool)
    
    print(f"用户工具配置: {tools}")
    
    return handler(request.override(tools=tools))
```

## MCP 集成

### MCP 概述

MCP（Model Context Protocol）允许 Agent 连接外部工具服务器：

```
MCP 架构：
┌──────────────────────────────────────┐
│  Agent                               │
│    ↓                                 │
│  MCP Client                          │
│    ↓                                 │
│  MCP Server                          │
│    ├─ 工具1: 文件操作                │
│    ├─ 工具2: 数据库查询              │
│    └─ 工具3: API 调用                │
└──────────────────────────────────────┘
```

### MCP 工具集成

```python
from langchain.tools import tool
from langchain.agents.middleware import wrap_model_call, ModelRequest, ModelResponse

# 模拟 MCP 工具
@tool
def mcp_file_read(path: str) -> str:
    """通过 MCP 读取文件"""
    return f"MCP 读取文件 {path}: 文件内容..."

@tool
def mcp_file_write(path: str, content: str) -> str:
    """通过 MCP 写入文件"""
    return f"MCP 写入文件 {path}: 成功"

@tool
def mcp_db_query(sql: str) -> str:
    """通过 MCP 查询数据库"""
    return f"MCP 查询结果: [{sql}] 返回 3 条记录"

@wrap_model_call
def mcp_tool_integration(request: ModelRequest, handler) -> ModelResponse:
    """集成 MCP 工具"""
    tools = list(request.tools)
    
    # 根据用户权限决定是否添加 MCP 工具
    user_role = request.context.user_role
    
    if user_role in ["admin", "developer"]:
        mcp_tools = [mcp_file_read, mcp_file_write, mcp_db_query]
        
        # 避免重复添加
        existing_names = {t.name for t in tools}
        for mcp_tool in mcp_tools:
            if mcp_tool.name not in existing_names:
                tools.append(mcp_tool)
        
        print(f"MCP 工具已启用: {[t.name for t in mcp_tools]}")
    
    return handler(request.override(tools=tools))
```

### MCP 动态发现

```python

# 模拟 MCP 服务器工具发现

    return [
        {"name": "mcp_file_read", "description": "读取文件"},
        {"name": "mcp_file_write", "description": "写入文件"},
        {"name": "mcp_db_query", "description": "查询数据库"},

@wrap_model_call
def mcp_dynamic_discovery(request: ModelRequest, handler) -> ModelResponse:
    """动态发现 MCP 工具"""
    # 检查是否需要连接 MCP 服务器
    if request.context.need_mcp:
        discovered = discover_mcp_tools(request.context.mcp_server_url)
        
        # 将发现的工具添加到工具列表
        # 实际实现中会创建对应的 langchain 工具对象
        print(f"发现 MCP 工具: {discovered}")
    
    return handler(request)
```

```python
from langchain.agents.middleware import wrap_tool_call, ToolCallRequest
from langchain.messages import ToolMessage

# 工具结果缓存
tool_cache = {}

@wrap_tool_call
def caching_tool_call(request: ToolCallRequest, handler):
    """缓存工具调用结果"""
    tool_name = request.tool_call["name"]
    tool_args = str(request.tool_call["args"])
    
    cache_key = f"{tool_name}:{tool_args}"
    
    if cache_key in tool_cache:
        print(f"工具缓存命中: {cache_key}")
        return tool_cache[cache_key]
    
    result = handler(request)
    tool_cache[cache_key] = result
    
    return result
```

## 完整示例：智能工具选择器

```python
from langchain.agents.middleware import AgentMiddleware, ModelRequest, ModelResponse, ToolCallRequest
from langchain.tools import tool
from langchain.messages import ToolMessage
from langchain.chat_models import init_chat_model

# 工具定义
@tool
def web_search(query: str) -> str:
    """搜索互联网"""
    return f"搜索结果: {query}"

@tool
def weather_query(city: str) -> str:
    """查询天气"""
    return f"{city} 天气: 晴，25°C"

@tool
def calculator(expression: str) -> str:
    """计算表达式"""
    try:
        return str(eval(expression))
    except:
        return "计算错误"

@tool
def send_email(to: str, subject: str, body: str) -> str:
    """发送邮件"""
    return f"邮件已发送给 {to}"

# 工具分类
TOOL_CATEGORIES = {
    "search": [web_search],
    "weather": [weather_query],
    "calculation": [calculator],
    "communication": [send_email],
}

class SmartToolSelector(AgentMiddleware):
    """智能工具选择器"""
    
    def wrap_model_call(self, request: ModelRequest, handler) -> ModelResponse:
        """根据查询智能选择工具"""
        last_msg = request.state["messages"][-1]
        query = last_msg.content
        
        # 分析查询意图
        needed_categories = self._analyze_query(query)
        
        # 选择相关工具
        selected_tools = []
        for category in needed_categories:
            selected_tools.extend(TOOL_CATEGORIES.get(category, []))
        
        # 如果没匹配到，使用所有工具
        if not selected_tools:
            selected_tools = request.tools
        
        print(f"查询意图: {needed_categories}")
        print(f"选择工具: {[t.name for t in selected_tools]}")
        
        return handler(request.override(tools=selected_tools))
    
    def _analyze_query(self, query: str) -> list[str]:
        """分析查询意图"""
        categories = []
        
        # 搜索意图
        if any(w in query for w in ["搜索", "查找", "查询"]):
            categories.append("search")
        
        # 天气意图
        if any(w in query for w in ["天气", "temperature", "weather"]):
            categories.append("weather")
        
        # 计算意图
        if any(w in query for w in ["计算", "多少", "+", "-", "*", "/"]):
            categories.append("calculation")
        
        # 通信意图
        if any(w in query for w in ["邮件", "email", "发送", "通知"]):
            categories.append("communication")
        
        return categories
```

## 小结

| 要点 | 说明 |
|------|------|
| 过滤方式 | 按名称、类别、查询内容、用户权限 |
| 运行时注册 | 通过 `@wrap_model_call` 和 `request.override(tools=...)` |
| AgentMiddleware | 组合多个钩子，可定义 tools |
| MCP 集成 | 动态发现和连接外部工具服务器 |
| 工具缓存 | 避免重复调用相同参数的工具 |
| 最佳实践 | 优化工具描述、按功能分组、智能意图分析 |

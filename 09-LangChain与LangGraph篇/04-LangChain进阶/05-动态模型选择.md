# 09.04.05 - 动态模型选择

> 根据查询复杂度、用户角色、上下文状态动态选择最合适的模型

## 为什么需要动态模型选择

```
模型选择场景：
┌──────────────────────────────────────┐
│  场景1：简单问候                      │
│  查询: "你好"                        │
│  推荐: moonshot-v1-8k（轻量、快速）  │
│                                       │
│  场景2：复杂分析                      │
│  查询: "比较 Python 和 Rust 的内存管理│
│       策略，分析各自优缺点"           │
│  推荐: deepseek-chat（强大、准确）   │
│                                       │
│  场景3：代码生成                      │
│  查询: "写一个快速排序算法"           │
│  推荐: deepseek-chat（代码能力强）   │
│                                       │
│  场景4：长上下文                      │
│  查询: 基于 50 条历史消息回答         │
│  推荐: moonshot-v1-32k（大上下文）   │
└──────────────────────────────────────┘
```

### 成本 vs 质量权衡

| 模型 | 单次成本 | 质量 | 速度 | 适用场景 |
|------|---------|------|------|---------|
| moonshot-v1-8k | 低 | 中 | 快 | 简单对话、问候 |
| moonshot-v1-32k | 中 | 中 | 中 | 长上下文对话 |
| deepseek-chat | 中 | 高 | 中 | 复杂分析、代码 |
| gpt-4o | 高 | 最高 | 慢 | 关键任务 |

## @wrap_model_call 实现动态选择

### 基础实现

```python
from langchain.agents.middleware import wrap_model_call, ModelRequest, ModelResponse
from langchain.chat_models import init_chat_model

# 预定义模型
SIMPLE_MODEL = init_chat_model("moonshot:moonshot-v1-8k")
ADVANCED_MODEL = init_chat_model("deepseek:deepseek-chat")

@wrap_model_call
def basic_model_selector(request: ModelRequest, handler) -> ModelResponse:
    """基础模型选择器"""
    last_msg = request.state["messages"][-1]
    query = last_msg.content
    
    # 简单判断
    if len(query) < 20:
        model = SIMPLE_MODEL
    else:
        model = ADVANCED_MODEL
    
    print(f"选择模型: {model}")
    return handler(request.override(model=model))
```

## 基于 State 的模型选择

### 方案1：根据对话轮次选择

```python
from langchain.agents.middleware import wrap_model_call, ModelRequest, ModelResponse
from langchain.chat_models import init_chat_model

# 模型池
MODEL_POOL = {
    "initial": init_chat_model("moonshot:moonshot-v1-8k"),
    "active": init_chat_model("deepseek:deepseek-chat"),
    "extended": init_chat_model("deepseek:deepseek-chat"),
}

@wrap_model_call
def stage_based_model_selector(request: ModelRequest, handler) -> ModelResponse:
    """根据对话阶段选择模型"""
    message_count = len(request.state["messages"])
    
    # 根据消息数量判断阶段
    if message_count <= 3:
        stage = "initial"
    elif message_count <= 10:
        stage = "active"
    else:
        stage = "extended"
    
    model = MODEL_POOL[stage]
    print(f"对话阶段: {stage} (消息数: {message_count})")
    print(f"选择模型: {model}")
    
    return handler(request.override(model=model))
```

### 方案4：根据 State 中的上下文长度选择

```python
from langchain.agents.middleware import wrap_model_call, ModelRequest, ModelResponse
from langchain.chat_models import init_chat_model

CONTEXT_MODELS = {
    "short": init_chat_model("moonshot:moonshot-v1-8k"),    # < 4K tokens
    "medium": init_chat_model("moonshot:moonshot-v1-32k"),  # 4K-16K tokens
    "long": init_chat_model("deepseek:deepseek-chat"),      # > 16K tokens
}

@wrap_model_call
def context_length_selector(request: ModelRequest, handler) -> ModelResponse:
    """根据上下文长度选择模型"""
    messages = request.state["messages"]
    
    # 估算总 token 数（粗略估算：每字符约 0.5 token）
    total_chars = sum(len(m.content) for m in messages)
    estimated_tokens = total_chars * 0.5
    
    if estimated_tokens < 4000:
        context_level = "short"
    elif estimated_tokens < 16000:
        context_level = "medium"
    else:
        context_level = "long"
    
    model = CONTEXT_MODELS[context_level]
    print(f"估算 Token: {estimated_tokens}")
    print(f"上下文级别: {context_level}")
    print(f"选择模型: {model}")
    
    return handler(request.override(model=model))
```

## 基于 Store 的模型选择

### 方案1：根据用户历史偏好选择

```python
from langchain.agents.middleware import wrap_model_call, ModelRequest, ModelResponse
from langchain.chat_models import init_chat_model

# 模型池
MODEL_POOL = {
    "fast": init_chat_model("moonshot:moonshot-v1-8k"),
    "balanced": init_chat_model("deepseek:deepseek-chat"),
    "quality": init_chat_model("openai:gpt-4o"),
}

@wrap_model_call
def preference_based_selector(request: ModelRequest, handler) -> ModelResponse:
    """根据用户偏好选择模型"""
    user_id = request.context.user_id
    
    # 从 Store 加载用户偏好
    prefs = request.runtime.store.get(
        ("user", user_id),
        "model_preference"
    )
    
    if prefs:
        preference = prefs.value.get("preference", "balanced")
        model = MODEL_POOL.get(preference, MODEL_POOL["balanced"])
        print(f"用户偏好: {preference}")
    else:
        # 默认使用平衡模型
        model = MODEL_POOL["balanced"]
        print("使用默认模型 balanced")
    
    return handler(request.override(model=model))
```

### 方案2：根据用户历史表现选择

```python
from langchain.agents.middleware import wrap_model_call, ModelRequest, ModelResponse
from langchain.chat_models import init_chat_model

@wrap_model_call
def history_based_selector(request: ModelRequest, handler) -> ModelResponse:
    """根据用户历史交互选择模型"""
    user_id = request.context.user_id
    
    # 从 Store 加载用户交互历史
    interactions = request.runtime.store.search(
        ("user", user_id, "interactions"),
        query="",
        limit=10,
    )
    
    # 分析历史交互复杂度
    if interactions:
        avg_query_length = sum(
            len(i.value.get("query_length", 0))
            for i in interactions
        ) / len(interactions)
        
        if avg_query_length > 100:
            model = init_chat_model("deepseek:deepseek-chat")
            print(f"历史平均查询长度 {avg_query_length}，使用高级模型")
        else:
            model = init_chat_model("moonshot:moonshot-v1-8k")
            print(f"历史平均查询长度 {avg_query_length}，使用轻量模型")
    else:
        model = init_chat_model("moonshot:moonshot-v1-8k")
        print("新用户，使用轻量模型")
    
    return handler(request.override(model=model))
```

## 综合动态模型选择

### 多因素决策

```python
from langchain.agents.middleware import wrap_model_call, ModelRequest, ModelResponse
from langchain.chat_models import init_chat_model

# 模型池
MODEL_POOL = {
    "moonshot-8k": init_chat_model("moonshot:moonshot-v1-8k"),
    "moonshot-32k": init_chat_model("moonshot:moonshot-v1-32k"),
    "deepseek": init_chat_model("deepseek:deepseek-chat"),
    "gpt4o": init_chat_model("openai:gpt-4o"),
}

def calculate_model_score(query: str, state: dict, context) -> dict[str, int]:
    """计算每个模型的得分"""
    scores = {name: 0 for name in MODEL_POOL}
    
    # 因素1：查询长度
    if len(query) < 20:
        scores["moonshot-8k"] += 3
    elif len(query) < 100:
        scores["deepseek"] += 2
    else:
        scores["deepseek"] += 3
        scores["gpt4o"] += 2
    
    # 因素2：查询复杂度
    complex_keywords = ["分析", "比较", "优化", "架构", "设计", "实现"]
    if any(kw in query for kw in complex_keywords):
        scores["deepseek"] += 3
        scores["gpt4o"] += 2
    
    # 因素3：上下文长度
    msg_count = len(state.get("messages", []))
    if msg_count > 20:
        scores["moonshot-32k"] += 3
        scores["deepseek"] += 2
    
    # 因素4：用户等级
    user_tier = state.get("user_tier", "free")
    if user_tier == "enterprise":
        scores["gpt4o"] += 3
    elif user_tier == "premium":
        scores["deepseek"] += 2
    
    # 因素5：优先级
    if hasattr(context, 'priority'):
        if context.priority == "high":
            scores["gpt4o"] += 3
        elif context.priority == "low":
            scores["moonshot-8k"] += 3
    
    return scores

@wrap_model_call
def comprehensive_model_selector(request: ModelRequest, handler) -> ModelResponse:
    """综合多因素选择模型"""
    last_msg = request.state["messages"][-1]
    query = last_msg.content
    
    # 计算得分
    scores = calculate_model_score(query, request.state, request.context)
    
    # 选择最高分模型
    best_model = max(scores, key=scores.get)
    
    print(f"查询: {query[:50]}...")
    print(f"模型得分: {scores}")
    print(f"选择: {best_model}")
    
    return handler(request.override(model=MODEL_POOL[best_model]))
```

## 模型降级策略

```python
from langchain.agents.middleware import wrap_model_call, ModelRequest, ModelResponse
from langchain.chat_models import init_chat_model

# 降级链
FALLBACK_CHAIN = [
    init_chat_model("deepseek:deepseek-chat"),
    init_chat_model("moonshot:moonshot-v1-32k"),
    init_chat_model("moonshot:moonshot-v1-8k"),
]

@wrap_model_call
def fallback_model_selector(request: ModelRequest, handler) -> ModelResponse:
    """带降级的模型选择"""
    last_error = None
    
    for model in FALLBACK_CHAIN:
        try:
            return handler(request.override(model=model))
        except Exception as e:
            last_error = e
            print(f"模型 {model} 失败: {e}")
            continue
    
    # 所有模型都失败
    raise last_error
```

## 模型选择监控

```python
from langchain.agents.middleware import wrap_model_call, ModelRequest, ModelResponse
from collections import defaultdict
import time

# 模型使用统计
model_stats = defaultdict(lambda: {"count": 0, "total_time": 0, "errors": 0})

@wrap_model_call
def monitored_model_selector(request: ModelRequest, handler) -> ModelResponse:
    """带监控的模型选择"""
    model = request.model
    model_name = str(model)
    
    start_time = time.time()
    
    try:
        response = handler(request)
        
        duration = time.time() - start_time
        model_stats[model_name]["count"] += 1
        model_stats[model_name]["total_time"] += duration
        
        print(f"[模型监控] {model_name}")
        print(f"  调用次数: {model_stats[model_name]['count']}")
        print(f"  平均耗时: {model_stats[model_name]['total_time'] / model_stats[model_name]['count']:.2f}s")
        
        return response
    except Exception as e:
        model_stats[model_name]["errors"] += 1
        print(f"[模型监控] {model_name} 错误: {e}")
        raise
```

## 小结

| 要点 | 说明 |
|------|------|
| 选择依据 | State（消息数、角色、任务）、Store（偏好、历史）、Context（优先级、功能） |
| 实现方式 | `@wrap_model_call` + `handler(request.override(model=...))` |
| 模型池 | 预定义不同能力的模型 |
| 决策策略 | 规则判断、打分系统、学习算法 |
| 降级策略 | 多级 fallback，确保可用性 |
| 监控 | 统计调用次数、耗时、错误率 |

# 09.04.02 - 自定义中间件

> 深入理解每个装饰器的用法，创建功能完整的自定义中间件

## 自定义中间件的方式

LangChain 提供两种方式创建中间件：

```
创建方式：
┌──────────────────────────────────────┐
│  方式1：装饰器（推荐）                │
│  ├─ @before_agent                    │
│  ├─ @after_agent                     │
│  ├─ @wrap_model_call                 │
│  └─ @wrap_tool_call                  │
│                                       │
│  方式2：AgentMiddleware 类           │
│  ├─ 继承 AgentMiddleware             │
│  ├─ 实现多个钩子方法                  │
│  └─ 可定义 state_schema 和 tools     │
└──────────────────────────────────────┘
```

## @before_agent 详解

### 基本用法

```python
from langchain.agents.middleware import before_agent

@before_agent
def simple_logger(state, runtime):
    """最简单的中间件：记录输入"""
    last_msg = state["messages"][-1]
    print(f"[{last_msg.role}] {last_msg.content[:50]}...")
    return None  # 不修改 State
```

### 修改 State

```python
from langchain.agents.middleware import before_agent

@before_agent
def inject_timestamp(state, runtime):
    """注入时间戳到 State"""
    import datetime
    now = datetime.datetime.now().isoformat()
    
    return {
        "metadata": {
            **state.get("metadata", {}),
            "last_updated": now,
            "message_count": len(state["messages"]),
        }
    }
```

### 提前终止

```python
from langchain.agents.middleware import before_agent
from langchain.messages import AIMessage

@before_agent
def block_sensitive_input(state, runtime):
    """检测敏感输入并提前终止"""
    last_msg = state["messages"][-1]
    sensitive_words = ["密码", "信用卡", "身份证号"]
    
    for word in sensitive_words:
        if word in last_msg.content:
            return {
                "messages": [
                    AIMessage(content=f"检测到敏感词'{word}'，请勿输入敏感信息。")
                ]
            }
    
    return None  # 继续正常执行
```

### 完整示例：输入验证中间件

```python
from langchain.agents.middleware import before_agent
from langchain.messages import AIMessage
import re

@before_agent
def input_validator(state, runtime):
    """全面的输入验证中间件"""
    last_msg = state["messages"][-1]
    content = last_msg.content
    
    # 1. 检查空输入
    if not content or not content.strip():
        return {
            "messages": [AIMessage("请输入有效内容，不能为空。")]
        }
    
    # 2. 检查长度限制
    if len(content) > 5000:
        return {
            "messages": [AIMessage("输入内容过长，请限制在 5000 字以内。")]
        }
    
    # 3. 检查 SQL 注入
    sql_patterns = [
            r"(\b(SELECT|INSERT|UPDATE|DELETE|DROP|UNION)\b.*\b(FROM|INTO|TABLE|WHERE)\b)",
        ]
    for pattern in sql_patterns:
        if re.search(pattern, content, re.IGNORECASE):
            return {
                "messages": [AIMessage("检测到可疑输入，请重新输入。")]
            }
    
    # 4. 检查 PII（个人敏感信息）
    pii_patterns = [
        r'\b\d{16}\b',           # 信用卡号
        r'\b\d{3}-\d{2}-\d{4}\b',  # SSN
        r'\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b',  # 邮箱
    ]
    
    detected_pii = []
    for pattern in pii_patterns:
        if re.search(pattern, content):
            detected_pii.append(pattern)
    
    if detected_pii:
        return {
            "messages": [
                AIMessage("检测到敏感信息，请去除后重新输入。")
            ]
        }
    
    # 5. 通过验证
    return None
```

### 跳转到特定节点

```python
from langchain.agents.middleware import after_agent

@after_agent(can_jump_to=["retry", "escalate", "end"])
def quality_check(state, runtime):
    """输出质量检查"""
    last_msg = state["messages"][-1]
    content = last_msg.content
    
    # 检查回复质量

        # 回复太短，可能不完整

    if "我不知道" in content or "无法回答" in content:
        # 模型无法回答，需要升级处理
        return {"quality_check": "escalate"}
    
    return {"quality_check": "pass"}
```

### 完整示例：输出格式化中间件

```python
from langchain.agents.middleware import after_agent
import re

@after_agent
def format_output(state, runtime):
    """格式化 AI 输出"""
    last_msg = state["messages"][-1]
    content = last_msg.content
    
    # 1. 确保代码块有语言标记
    content = re.sub(
        r'```\n',
        '```python\n',
        content
    )
    
    # 2. 修复多余的换行
    content = re.sub(r'\n{3,}', '\n\n', content)
    
    # 3. 确保段落间有空行
    content = re.sub(r'([。！？])\n([A-Za-z\u4e00-\u9fff])', r'\1\n\n\2', content)
    
    last_msg.content = content
    return None
```

## @wrap_model_call 详解

### ModelRequest 字段

```
ModelRequest 结构：
┌──────────────────────────────────────┐
│  state: dict                         │
│    当前 Agent State                   │
│                                       │
│  tools: list[BaseTool]               │
│    当前可用工具列表                    │
│                                       │
│  model: BaseChatModel                │
│    当前使用的模型                     │
│                                       │
│  runtime: Runtime                    │
│    运行时上下文对象                    │
│                                       │
│  response_format: type | None        │
│    期望的响应格式（Pydantic 类型）    │
│                                       │
│  方法：                               │
│  override(model=, tools=, ...)       │
│    创建修改后的请求                   │
└──────────────────────────────────────┘
```

### 基本用法：日志记录

```python
from langchain.agents.middleware import wrap_model_call, ModelRequest, ModelResponse

@wrap_model_call
def logging_middleware(request: ModelRequest, handler) -> ModelResponse:
    """记录模型调用详情"""
    # 调用前
    print("=" * 50)
    print(f"模型调用开始")
    print(f"  模型: {request.model}")
    print(f"  消息数: {len(request.state['messages'])}")
    print(f"  工具数: {len(request.tools)}")
    
    # 调用模型
    response = handler(request)
    
    # 调用后
    print(f"  响应长度: {len(response.content)}")
    print(f"  响应前100字: {response.content[:100]}...")
    print("=" * 50)
    
    return response
```

### 动态模型选择

```python
from langchain.agents.middleware import wrap_model_call, ModelRequest, ModelResponse
from langchain.chat_models import init_chat_model

# 预定义不同级别的模型
simple_model = init_chat_model("moonshot:moonshot-v1-8k")
advanced_model = init_chat_model("deepseek:deepseek-chat")

@wrap_model_call
def smart_model_selector(request: ModelRequest, handler) -> ModelResponse:
    """根据查询复杂度选择模型"""
    last_msg = request.state["messages"][-1]
    query = last_msg.content
    
    # 简单问题用轻量模型
    simple_indicators = ["是什么", "怎么办", "为什么", "如何"]
    is_simple = any(indicator in query for indicator in simple_indicators)
    
    if is_simple and len(query) < 50:
        model = simple_model
        print(f"使用轻量模型: moonshot-v1-8k")
    else:
        model = advanced_model
        print(f"使用高级模型: deepseek-chat")
    
    # 使用选定的模型处理请求
    return handler(request.override(model=model))
```

### 动态工具过滤

```python
from langchain.agents.middleware import wrap_model_call, ModelRequest, ModelResponse

@wrap_model_call
def tool_filter(request: ModelRequest, handler) -> ModelResponse:
    """根据用户角色过滤工具"""
    user_role = request.runtime.context.get("user_role", "guest")
    
    # 定义工具访问权限
    tool_permissions = {
        "guest": ["public_search", "weather_query"],
        "user": ["public_search", "weather_query", "email_send"],
        "admin": ["public_search", "weather_query", "email_send", "file_delete", "system_config"],
    }
    
    allowed_tools = tool_permissions.get(user_role, [])
    
    # 过滤工具列表
    filtered_tools = [t for t in request.tools if t.name in allowed_tools]
    
    print(f"用户角色: {user_role}")
    print(f"允许工具: {allowed_tools}")
    print(f"过滤后工具: {[t.name for t in filtered_tools]}")
    
    return handler(request.override(tools=filtered_tools))
```

### 动态 Response Format

```python
from langchain.agents.middleware import wrap_model_call, ModelRequest, ModelResponse
from pydantic import BaseModel, Field

class StructuredResponse(BaseModel):
    answer: str = Field(description="回答内容")
    confidence: float = Field(description="置信度 0-1")
    sources: list[str] = Field(description="引用来源", default_factory=list)

@wrap_model_call
def enforce_structure(request: ModelRequest, handler) -> ModelResponse:
    """强制结构化输出"""
    # 检查是否需要结构化输出
    if request.runtime.context.get("need_structured", False):
        return handler(request.override(response_format=StructuredResponse))
    
    return handler(request)
```

### 请求缓存

```python
from langchain.agents.middleware import wrap_model_call, ModelRequest, ModelResponse
import hashlib

# 简单内存缓存
cache = {}

@wrap_model_call
def caching_middleware(request: ModelRequest, handler) -> ModelResponse:
    """缓存模型响应"""
    # 生成缓存键
    last_msg = request.state["messages"][-1]
    cache_key = hashlib.md5(last_msg.content.encode()).hexdigest()
    
    # 检查缓存
    if cache_key in cache:
        print(f"缓存命中: {cache_key[:8]}")
        cached_response = cache[cache_key]
        return cached_response
    
    # 调用模型
    response = handler(request)
    
    # 存入缓存
    cache[cache_key] = response
    print(f"缓存存入: {cache_key[:8]}")
    
    return response
```

### 基本用法：错误处理

```python
from langchain.agents.middleware import wrap_tool_call, ToolCallRequest
from langchain.messages import ToolMessage

@wrap_tool_call
def error_handler(request: ToolCallRequest, handler):
    """统一错误处理"""
    tool_name = request.tool_call["name"]
    
    try:
        return handler(request)
    except ConnectionError as e:
        return ToolMessage(
            content=f"工具 {tool_name} 网络连接失败，请稍后重试。",
            tool_call_id=request.tool_call["id"],
        )
    except ValueError as e:
        return ToolMessage(
            content=f"工具 {tool_name} 参数错误: {str(e)}",
            tool_call_id=request.tool_call["id"],
        )
    except Exception as e:
        return ToolMessage(
            content=f"工具 {tool_name} 调用失败: 未知错误",
            tool_call_id=request.tool_call["id"],
        )
```

### 工具调用限流

```python
from langchain.agents.middleware import wrap_tool_call, ToolCallRequest
from langchain.messages import ToolMessage
from collections import defaultdict
import time

# 记录每个工具的调用时间
tool_call_times = defaultdict(list)

@wrap_tool_call
def rate_limiter(request: ToolCallRequest, handler):
    """工具调用限流"""
    tool_name = request.tool_call["name"]
    now = time.time()
    
    # 清理 60 秒前的记录
    tool_call_times[tool_name] = [
        t for t in tool_call_times[tool_name] if now - t < 60
    ]
    
    # 检查频率限制（每分钟最多 10 次）
    if len(tool_call_times[tool_name]) >= 10:
        return ToolMessage(
            content=f"工具 {tool_name} 调用过于频繁，请稍后再试。",
            tool_call_id=request.tool_call["id"],
        )
    
    # 记录调用时间
    tool_call_times[tool_name].append(now)
    
    return handler(request)
```

## AgentMiddleware 类详解

当需要组合多个钩子并共享状态时，使用类方式更合适：

### 基本结构

```python
from langchain.agents.middleware import AgentMiddleware, ModelRequest, ModelResponse, ToolCallRequest

class LoggingMiddleware(AgentMiddleware):
    """日志中间件：记录所有操作"""
    
    def before_agent(self, state, runtime):
        print(f"[Before] 消息数: {len(state['messages'])}")
        return None
    
    def after_agent(self, state, runtime):
        last_msg = state["messages"][-1]
        print(f"[After] 回复: {last_msg.content[:50]}...")
        return None
    
    def wrap_model_call(self, request: ModelRequest, handler) -> ModelResponse:
        print(f"[Model] 调用: {request.model}")
        response = handler(request)
        print(f"[Model] 响应: {len(response.content)} chars")
        return response
    
    def wrap_tool_call(self, request: ToolCallRequest, handler):
        print(f"[Tool] 调用: {request.tool_call['name']}")
        result = handler(request)
        print(f"[Tool] 结果: {str(result)[:50]}...")
        return result
```

### 定义 State Schema

```python
from langchain.agents.middleware import AgentMiddleware
from pydantic import BaseModel

class AuditState(BaseModel):
    """审计状态"""
    audit_log: list[dict] = []

class AuditMiddleware(AgentMiddleware):
    """审计中间件：记录所有操作到 State"""
    
    state_schema = AuditState
    
    def before_agent(self, state, runtime):
        log_entry = {
            "type": "before_agent",
            "message_count": len(state["messages"]),
            "timestamp": time.time(),
        }
        
        audit_log = state.get("audit_log", [])
        return {"audit_log": [*audit_log, log_entry]}
    
    def wrap_model_call(self, request: ModelRequest, handler) -> ModelResponse:
        log_entry = {
            "type": "model_call",
            "model": str(request.model),
            "timestamp": time.time(),
        }
        
        audit_log = request.state.get("audit_log", [])
        # 注意：这里无法直接修改 request.state
        # 需要通过其他方式传递日志
        
        return handler(request)
```

### 完整示例：性能监控中间件

```python
from langchain.agents.middleware import AgentMiddleware, ModelRequest, ModelResponse, ToolCallRequest
import time
from dataclasses import dataclass, field

@dataclass
class PerformanceMetrics:
    """性能指标"""
    model_calls: int = 0
    tool_calls: int = 0
    total_model_time: float = 0.0
    total_tool_time: float = 0.0
    iteration_count: int = 0

class PerformanceMiddleware(AgentMiddleware):
    """性能监控中间件"""
    
    def __init__(self):
        self.metrics = PerformanceMetrics()
    
    def before_agent(self, state, runtime):
        self.metrics.iteration_count += 1
        self.metrics.iteration_start = time.time()
        return None
    
    def after_agent(self, state, runtime):
        iteration_time = time.time() - self.metrics.iteration_start
        print(f"[Performance] 迭代 #{self.metrics.iteration_count}: {iteration_time:.2f}s")
        return None
    
    def wrap_model_call(self, request: ModelRequest, handler) -> ModelResponse:
        start = time.time()
        response = handler(request)
        duration = time.time() - start
        
        self.metrics.model_calls += 1
        self.metrics.total_model_time += duration
        
        print(f"[Performance] 模型调用: {duration:.2f}s")
        print(f"[Performance] 累计模型时间: {self.metrics.total_model_time:.2f}s")
        
        return response
    
    def wrap_tool_call(self, request: ToolCallRequest, handler):
        start = time.time()
        result = handler(request)
        duration = time.time() - start
        
        self.metrics.tool_calls += 1
        self.metrics.total_tool_time += duration
        
        print(f"[Performance] 工具 {request.tool_call['name']}: {duration:.2f}s")
        
        return result
    
    def get_summary(self) -> dict:
        """获取性能摘要"""
        return {
            "iterations": self.metrics.iteration_count,
            "model_calls": self.metrics.model_calls,
            "tool_calls": self.metrics.tool_calls,
            "avg_model_time": self.metrics.total_model_time / max(self.metrics.model_calls, 1),
            "avg_tool_time": self.metrics.total_tool_time / max(self.metrics.tool_calls, 1),
            "total_time": self.metrics.total_model_time + self.metrics.total_tool_time,
        }
```

@before_agent
def load_user_prefs(state, runtime):

    user_id = runtime.context.get("user_id")

from langchain.agents.middleware import wrap_model_call, ModelRequest, ModelResponse

# 模型池
MODEL_POOL = {
    "simple": init_chat_model("moonshot:moonshot-v1-8k"),
    "standard": init_chat_model("deepseek:deepseek-chat"),
    "advanced": init_chat_model("openai:gpt-4o"),
}

@wrap_model_call
def dynamic_model_router(request: ModelRequest, handler) -> ModelResponse:
    """动态模型路由器"""
    last_msg = request.state["messages"][-1]
    query = last_msg.content
    
    # 判断查询复杂度
    complexity = estimate_complexity(query)
    
    if complexity == "simple":
        model = MODEL_POOL["simple"]
    elif complexity == "standard":
        model = MODEL_POOL["standard"]
    else:
        model = MODEL_POOL["advanced"]
    
    print(f"查询复杂度: {complexity}")
    print(f"选择模型: {model}")
    
    return handler(request.override(model=model))

def estimate_complexity(query: str) -> str:
    """估算查询复杂度"""
    # 简单指标
    if len(query) < 20 and any(w in query for w in ["是什么", "怎么办", "多少"]):
        return "simple"
    
    # 复杂指标
    if len(query) > 100 or any(w in query for w in ["分析", "比较", "优化", "架构"]):
        return "advanced"
    
    return "standard"

## 动态工具选择实战

```python
from langchain.agents.middleware import wrap_model_call, ModelRequest, ModelResponse
from langchain.tools import tool

# 工具注册表
TOOL_REGISTRY = {
    "search": [web_search, image_search],
    "calculation": [calculator, unit_converter],
    "communication": [send_email, send_sms],
    "file": [read_file, write_file, delete_file],
}

@wrap_model_call
def context_aware_tools(request: ModelRequest, handler) -> ModelResponse:
    """根据上下文选择工具子集"""
    last_msg = request.state["messages"][-1]
    query = last_msg.content
    
    # 分析查询需要的工具类别
    needed_categories = analyze_query(query)
    
    # 收集相关工具
    selected_tools = []
    for category in needed_categories:
        selected_tools.extend(TOOL_REGISTRY.get(category, []))
    
    # 如果没匹配到，使用所有工具
    if not selected_tools:
        selected_tools = request.tools
    
    print(f"查询类别: {needed_categories}")
    print(f"选中工具: {[t.name for t in selected_tools]}")
    
    return handler(request.override(tools=selected_tools))

def analyze_query(query: str) -> list[str]:
    """分析查询需要的工具类别"""
    categories = []
    
    if any(w in query for w in ["搜索", "查找", "查询"]):
        categories.append("search")
    if any(w in query for w in ["计算", "转换", "多少"]):
        categories.append("calculation")
    if any(w in query for w in ["发送", "通知", "邮件"]):
        categories.append("communication")
    if any(w in query for w in ["文件", "保存", "读取"]):
        categories.append("file")
    
    return categories
```

## 动态 Response Format 实战

```python
from langchain.agents.middleware import wrap_model_call, ModelRequest, ModelResponse
from pydantic import BaseModel, Field

class SimpleAnswer(BaseModel):
    """简单回答格式"""
    answer: str

class DetailedAnswer(BaseModel):
    """详细回答格式"""
    summary: str = Field(description="摘要")
    details: list[str] = Field(description="详细说明")
    references: list[str] = Field(description="参考来源", default_factory=list)

class CodeAnswer(BaseModel):
    """代码回答格式"""
    explanation: str
    code: str
    language: str

@wrap_model_call
def adaptive_response_format(request: ModelRequest, handler) -> ModelResponse:
    """根据查询类型选择响应格式"""
    last_msg = request.state["messages"][-1]
    query = last_msg.content
    
    # 判断查询类型
    if "代码" in query or "实现" in query or "怎么写" in query:
        response_format = CodeAnswer
    elif len(query) > 50 or "详细" in query:
        response_format = DetailedAnswer
    else:
        response_format = SimpleAnswer
    
    print(f"响应格式: {response_format.__name__}")
    
    return handler(request.override(response_format=response_format))
```

## 小结

| 要点 | 说明 |
|------|------|
| 四个装饰器 | `@before_agent`, `@after_agent`, `@wrap_model_call`, `@wrap_tool_call` |
| AgentMiddleware 类 | 组合多个钩子，可定义 state_schema 和 tools |
| State 更新 | 返回 dict 合并到 State，返回 None 不修改 |
| 动态模型 | `handler(request.override(model=new_model))` |
| 动态工具 | `handler(request.override(tools=filtered_tools))` |
| 动态格式 | `handler(request.override(response_format=PydanticClass))` |
| 错误处理 | `@wrap_tool_call` 捕获异常返回 ToolMessage |

# 09.04.09 - 进阶实战

> 结合中间件、RAG、动态选择、结构化输出的完整项目

## 项目概述

构建一个智能客服 Agent，具备以下能力：

```
智能客服架构：
┌──────────────────────────────────────┐
│  用户输入                             │
│    ↓                                 │
│  ┌────────────────────────────────┐  │
│  │ 中间件链                        │  │
│  │  1. 输入验证                   │  │
│  │  2. RAG 上下文注入             │  │
│  │  3. 动态模型选择               │  │
│  │  4. 动态工具选择               │  │
│  └────────────────────────────────┘  │
│    ↓                                 │
│  LLM (结构化输出)                    │
│    ↓                                 │
│  ┌────────────────────────────────┐  │
│  │ 输出处理                        │  │
│  │  1. 输出验证                   │  │
│  │  2. 日志记录                   │  │
│  │  3. 流式响应                   │  │
│  └────────────────────────────────┘  │
│    ↓                                 │
│  用户收到回复                        │
└──────────────────────────────────────┘
```

### 功能清单

| 功能 | 技术 | 说明 |
|------|------|------|
| 输入验证 | @before_agent | 敏感词检测、长度限制 |
| RAG 上下文 | Retriever | 从知识库检索相关文档 |
| 动态模型选择 | @wrap_model_call | 根据查询复杂度选择模型 |
| 动态工具选择 | @wrap_model_call | 按意图选择工具子集 |
| 结构化输出 | Pydantic + ToolStrategy | 确保输出格式一致 |
| 流式输出 | astream_events | 实时返回 token |
| 日志追踪 | Middleware | 记录所有操作 |
| 权限控制 | Context | 基于角色的工具过滤 |

```python
# models.py
from pydantic import BaseModel, Field
from typing import Literal

class ChatResponse(BaseModel):
    """客服回复"""
    answer: str = Field(description="回答内容")
    confidence: float = Field(description="置信度 0-1", ge=0, le=1)
    category: Literal["greeting", "answer", "clarification", "escalation"] = Field(
        description="回复类型"
    )
    follow_up_questions: list[str] = Field(
        description="建议的后续问题",
        default_factory=list,
    )

class OrderInfo(BaseModel):
    """订单信息"""
    order_id: str = Field(description="订单 ID")
    status: Literal["pending", "shipped", "delivered", "cancelled"] = Field(
        description="订单状态"
    )
    estimated_delivery: str = Field(description="预计送达时间")

class ServiceResponse(BaseModel):
    """服务响应（Union 类型）"""
    response: ChatResponse | OrderInfo | str = Field(
        description="响应内容"
    )
```

```python
def search_knowledge_base(query: str) -> str:

    knowledge_base = {

    }
    
    for keyword, answer in knowledge_base.items():
        if keyword in query:
            return answer
    
    return "未找到相关信息，请尝试其他关键词。"
```

```python
# rag.py
from langchain_community.vectorstores import FAISS
from langchain.embeddings import init_embeddings
from langchain_core.documents import Document
from langchain_text_splitters import RecursiveCharacterSplitter

def load_knowledge_base(file_path: str = "data/knowledge_base.txt"):
    """加载知识库"""
    # 模拟文档
    docs = [
        Document(
            page_content="我们支持 7 天无理由退货。退货条件：商品完好、包装完整、附带发票。退货流程：登录账户 → 订单详情 → 申请退货 → 寄回商品 → 退款。",
            metadata={"category": "退货"},
        ),
        Document(
            page_content="运费政策：订单满 99 元免运费。不满 99 元运费 10 元。偏远地区运费可能更高。",
            metadata={"category": "运费"},
        ),
        Document(
            page_content="会员权益：黄金会员享受 95 折优惠。积分可用于兑换商品或抵扣现金。1 元 = 1 积分。",
            metadata={"category": "会员"},
        ),
        Document(
            page_content="发票说明：所有订单默认开具电子发票。如需纸质发票，请联系客服。发票内容默认为商品名称。",
            metadata={"category": "发票"},
        ),
    ]
    
    # 分割文档
    splitter = RecursiveCharacterSplitter(
        chunk_size=300,
        chunk_overlap=50,
    )
    
    chunks = splitter.split_documents(docs)
    
    # 创建向量存储
    embeddings = init_embeddings("zhipuai:embedding-2")
    vectorstore = FAISS.from_documents(chunks, embeddings)
    
    return vectorstore.as_retriever(search_kwargs={"k": 2})
```

## 5. 中间件定义

```python
# middleware.py
from langchain.agents.middleware import (
    before_agent,
    after_agent,
    wrap_model_call,
    wrap_tool_call,
    ModelRequest,
    ModelResponse,
    ToolCallRequest,
)
from langchain.messages import AIMessage
from config import MODEL_POOL, TOOL_CATEGORIES, USER_PERMISSIONS
import time

# 输入验证中间件
@before_agent
def input_validator(state, runtime):
    """验证用户输入"""
    last_msg = state["messages"][-1]
    content = last_msg.content
    
    # 空输入检查
    if not content or not content.strip():
        return {
            "messages": [AIMessage("请输入有效内容。")]
        }
    
    # 长度限制
    if len(content) > 2000:
        return {
            "messages": [AIMessage("输入内容过长，请限制在 2000 字以内。")]
        }
    
    # 敏感词检查
    sensitive_words = ["密码", "信用卡", "验证码"]
    for word in sensitive_words:
        if word in content:
            return {
                "messages": [
                    AIMessage(f"请勿输入敏感信息（包含'{word}'）。")
                ]
            }
    
    return None

# RAG 上下文注入
@before_agent
def rag_context_injector(state, runtime):
    """注入 RAG 检索结果"""
    last_msg = state["messages"][-1]
    
    # 从 context 获取 retriever
    retriever = runtime.context.retriever
    docs = retriever.invoke(last_msg.content)
    
    if docs:
        context = "参考资料:\n" + "\n\n".join(
            f"[{i+1}] {doc.page_content}"
            for i, doc in enumerate(docs)
        )
        return {"context": context}
    
    return None

# 动态模型选择
@wrap_model_call
def smart_model_selector(request: ModelRequest, handler) -> ModelResponse:
    """智能模型选择"""
    last_msg = request.state["messages"][-1]
    query = last_msg.content
    
    # 简单问候用轻量模型
    if len(query) < 10 and any(w in query for w in ["你好", "嗨", "谢谢"]):
        model = MODEL_POOL["simple"]
    # 订单查询用标准模型
    elif any(w in query for w in ["订单", "物流", "退款"]):
        model = MODEL_POOL["standard"]
    # 复杂问题用高级模型
    else:
        model = MODEL_POOL["advanced"]
    
    return handler(request.override(model=model))

# 动态工具选择
@wrap_model_call
def smart_tool_selector(request: ModelRequest, handler) -> ModelResponse:
    """智能工具选择"""
    last_msg = request.state["messages"][-1]
    query = last_msg.content
    
    # 分析查询意图
    needed_categories = []
    
    if any(w in query for w in ["搜索", "查询", "什么", "怎么"]):
        needed_categories.append("search")
    if any(w in query for w in ["订单", "物流", "退款", "发货"]):
        needed_categories.append("order")
    if any(w in query for w in ["账户", "会员", "积分"]):
        needed_categories.append("account")
    if any(w in query for w in ["投诉", "工单", "客服"]):
        needed_categories.append("support")
    
    # 根据用户权限过滤
    user_role = request.context.user_role
    allowed_categories = USER_PERMISSIONS.get(user_role, [])
    
    # 收集允许的工具
    allowed_tool_names = set()
    for category in needed_categories:
        if category in allowed_categories:
            allowed_tool_names.update(TOOL_CATEGORIES.get(category, []))
    
    # 过滤工具
    if allowed_tool_names:
        filtered_tools = [t for t in request.tools if t.name in allowed_tool_names]
    else:
        filtered_tools = request.tools
    
    return handler(request.override(tools=filtered_tools))

# 工具调用日志
@wrap_tool_call
def tool_logger(request: ToolCallRequest, handler):
    """记录工具调用"""
    tool_name = request.tool_call["name"]
    start_time = time.time()
    
    print(f"[工具] {tool_name} 开始")
    print(f"  参数: {request.tool_call['args']}")
    
    result = handler(request)
    
    duration = time.time() - start_time
    print(f"[工具] {tool_name} 完成 ({duration:.2f}s)")
    print(f"  结果: {str(result)[:100]}...")
    
    return result

# 输出日志
@after_agent
def output_logger(state, runtime):
    """记录输出"""
    last_msg = state["messages"][-1]
    print(f"[输出] {last_msg.content[:100]}...")
    return None
```

## 6. Agent 构建

```python
# agent.py
from langchain.agents import create_agent
from langgraph.checkpoint.memory import MemorySaver
from rag import load_knowledge_base
from tools import (
    search_knowledge_base,
    search_web,
    query_order,
    create_order,
    query_account,
    create_ticket,
)
from middleware import (
    input_validator,
    rag_context_injector,
    smart_model_selector,
    smart_tool_selector,
    tool_logger,
    output_logger,
)
from models import ChatResponse
from config import ServiceContext

def create_service_agent():
    """创建客服 Agent"""
    # 加载知识库
    retriever = load_knowledge_base()
    
    # 创建 Agent
    agent = create_agent(
        model="moonshot:moonshot-v1-8k",
        tools=[
            search_knowledge_base,
            search_web,
            query_order,
            create_order,
            query_account,
            create_ticket,
        ],
        middleware=[
            input_validator,
            rag_context_injector,
            smart_model_selector,
            smart_tool_selector,
            tool_logger,
            output_logger,
        ],
        checkpointer=MemorySaver(),
        context_schema=ServiceContext,
    )
    
    return agent, retriever
```

## 7. FastAPI 部署

```python
# main.py
from fastapi import FastAPI
from fastapi.responses import StreamingResponse
from pydantic import BaseModel
import json

from agent import create_service_agent
from config import ServiceContext

app = FastAPI(title="智能客服系统")

# 初始化 Agent
agent, retriever = create_service_agent()

class ChatRequest(BaseModel):
    user_id: str
    message: str
    user_role: str = "guest"

@app.post("/chat")
async def chat(request: ChatRequest):
    """非流式聊天"""
    config = {"configurable": {"thread_id": f"user-{request.user_id}"}}
    context = ServiceContext(
        user_id=request.user_id,
        user_role=request.user_role,
    )
    
    # 注入 retriever 到 context
    class ContextWithRetriever(ServiceContext):
        retriever = retriever
    
    context_with_retriever = ContextWithRetriever(
        user_id=request.user_id,
        user_role=request.user_role,
    )
    context_with_retriever.retriever = retriever
    
    result = agent.invoke(
        {"messages": [{"role": "user", "content": request.message}]},
        config=config,
        context=context_with_retriever,
    )
    
    return {
        "answer": result["messages"][-1].content,
        "thread_id": f"user-{request.user_id}",
    }

@app.post("/chat/stream")
async def stream_chat(request: ChatRequest):
    """流式聊天"""
    config = {"configurable": {"thread_id": f"user-{request.user_id}"}}
    
    class ContextWithRetriever(ServiceContext):
        retriever = retriever
    
    context = ContextWithRetriever(
        user_id=request.user_id,
        user_role=request.user_role,
    )
    
    inputs = {"messages": [{"role": "user", "content": request.message}]}
    
    async def generate():
        async for event in agent.astream_events(
            inputs,
            config=config,
            context=context,
            version="v2",
        ):
            kind = event["event"]
            
            if kind == "on_chat_model_stream":
                token = event["data"]["chunk"].content
                if token:
                    yield f"data: {token}\n\n"
            
            elif kind == "on_tool_start":
                yield f"data: [TOOL] {event['name']}\n\n"
            
            elif kind == "on_tool_end":
                yield f"data: [DONE_TOOL] {event['name']}\n\n"
        
        yield "data: [DONE]\n\n"
    
    return StreamingResponse(
        generate(),
        media_type="text/event-stream",
        headers={
            "Cache-Control": "no-cache",
            "Connection": "keep-alive",
        },
    )

@app.get("/health")
async def health():
    """健康检查"""
    return {"status": "healthy", "version": "1.0.0"}
```

## 小结

| 要点 | 说明 |
|------|------|
| 中间件链 | 输入验证 → RAG → 模型选择 → 工具选择 → 日志 |
| RAG 注入 | before_agent 中检索文档并注入 State |
| 动态模型 | 根据查询内容选择不同能力的模型 |
| 动态工具 | 根据意图和用户权限过滤工具 |
| 结构化输出 | Pydantic 定义响应格式 |
| 流式部署 | FastAPI + astream_events + SSE |
| 生产就绪 | 健康检查、错误处理、日志记录 |

## 第三步：多轮对话

```python
from langchain.chat_models import init_chat_model
from langchain.messages import SystemMessage, HumanMessage, AIMessage

class Conversation:
    """多轮对话管理器"""

    def __init__(self, model, system_prompt: str):
        self.model = model
        self.messages = [SystemMessage(system_prompt)]

    def send(self, query: str, stream: bool = True) -> str:
        """发送消息并获取回复"""
        self.messages.append(HumanMessage(query))

        if stream:
            full_response = None
            for chunk in self.model.stream(self.messages):
                if full_response is None:
                    full_response = chunk
                else:
                    full_response += chunk
                print(chunk.content, end="", flush=True)
            print()
            response = full_response
        else:
            response = self.model.invoke(self.messages)

        self.messages.append(response)
        return response.content

# 使用
model = init_chat_model("deepseek:deepseek-chat")
conv = Conversation(model, "你是一个 Python 编程老师，用简单的方式教学。")

# 多轮对话
conv.send("什么是装饰器？")
conv.send("能给我一个实际的例子吗？")
conv.send("那类方法也可以用装饰器吗？")
```

```python
from pydantic import BaseModel, Field
from langchain.chat_models import init_chat_model

class CodeExplanation(BaseModel):
    """代码解释"""
    summary: str = Field(description="一句话总结代码功能")
    key_concepts: list[str] = Field(description="涉及的关键概念")
    line_by_line: list[str] = Field(description="逐行解释")
    complexity: str = Field(description="时间复杂度和空间复杂度")

model = init_chat_model("moonshot:moonshot-v1-8k")
explainer = model.with_structured_output(CodeExplanation)

code = """
def fibonacci(n):
    if n <= 1:
        return n
    a, b = 0, 1
    for _ in range(2, n + 1):
        a, b = b, a + b
    return b
"""

result = explainer.invoke(f"解释以下代码:\n```python\n{code}\n```")

print(f"功能: {result.summary}")
print(f"概念: {result.key_concepts}")
print(f"复杂度: {result.complexity}")
```

## 第五步：工具集成

```python
from langchain.tools import tool
from langchain.chat_models import init_chat_model
from langchain.messages import HumanMessage, ToolMessage

# 定义工具
@tool
def search_docs(query: str) -> str:
    """搜索 Python 官方文档"""
    docs = {
        "decorator": "装饰器使用 @ 语法修改函数行为。详见: https://docs.python.org/3/glossary.html#term-decorator",
        "list": "列表是有序可变集合。详见: https://docs.python.org/3/tutorial/datastructures.html",
        "class": "类是创建对象的蓝图。详见: https://docs.python.org/3/tutorial/classes.html",
    }
    for key, doc in docs.items():
        if key in query.lower():
            return doc
    return f"未找到关于 '{query}' 的官方文档，但可以尝试 https://docs.python.org/3/"

@tool
def run_code(code: str) -> str:
    """执行 Python 代码并返回结果"""
    import io
    import sys

    old_stdout = sys.stdout
    sys.stdout = io.StringIO()

    try:
        exec(code)
        output = sys.stdout.getvalue()
        return f"输出: {output}" if output else "执行成功，无输出"
    except Exception as e:
        return f"错误: {type(e).__name__}: {e}"
    finally:
        sys.stdout = old_stdout

# 初始化模型
model = init_chat_model("moonshot:moonshot-v1-8k")
model_with_tools = model.bind_tools([search_docs, run_code])
tool_map = {t.name: t for t in [search_docs, run_code]}

def chat_with_tools(query: str, max_iterations: int = 3):
    """带工具调用的对话"""
    messages = [HumanMessage(query)]

    for _ in range(max_iterations):
        response = model_with_tools.invoke(messages)
        messages.append(response)

        if not response.tool_calls:
            return response.content

        for tc in response.tool_calls:
            tool_name = tc["name"]
            selected_tool = tool_map.get(tool_name)
            if selected_tool:
                result = selected_tool.invoke(tc["args"])
            else:
                result = f"错误: 未知工具 {tool_name}"

            messages.append(ToolMessage(content=result, tool_call_id=tc["id"]))
            print(f"🔧 {tool_name}({tc['args']}) → {result[:100]}")

    return response.content

# 测试
print(chat_with_tools("装饰器的官方文档在哪里？"))
print(chat_with_tools("执行这段代码: print('Hello, World!')"))
```

## 第六步：完整应用

```python
"""
智能代码助手 - 完整版
"""

from langchain.chat_models import init_chat_model
from langchain.messages import SystemMessage, HumanMessage, AIMessage, ToolMessage
from langchain.tools import tool
from langchain_core.prompts import ChatPromptTemplate
from pydantic import BaseModel, Field
import sys
import io

# ─── 工具定义 ───────────────────────────────────────────────

@tool
def search_docs(query: str) -> str:
    """搜索 Python 官方文档"""
    docs = {
        "decorator": "装饰器使用 @ 语法修改函数行为。",
        "list": "列表是有序可变集合。",
        "generator": "生成器是使用 yield 的函数，返回迭代器。",
        "context manager": "上下文管理器使用 with 语句管理资源。",
    }
    for key, doc in docs.items():
        if key in query.lower():
            return doc
    return f"未找到 '{query}' 的具体文档，请访问 https://docs.python.org/3/"

@tool
def run_code(code: str) -> str:
    """安全执行 Python 代码（仅用于学习目的）"""
    old_stdout = sys.stdout
    sys.stdout = io.StringIO()

    try:
        exec(code, {"__builtins__": {"print": print, "range": range, "len": len}})
        output = sys.stdout.getvalue()
        return f"执行成功\n输出: {output}" if output else "执行成功，无输出"
    except Exception as e:
        return f"执行失败: {type(e).__name__}: {e}"
    finally:
        sys.stdout = old_stdout

# ─── 结构化输出定义 ────────────────────────────────────────

class CodeReviewResult(BaseModel):
    """代码审查结果"""
    score: int = Field(description="代码质量评分，1-10")
    issues: list[str] = Field(description="发现的问题")
    suggestions: list[str] = Field(description="改进建议")
    summary: str = Field(description="总体评价")

# ─── 提示词模板 ─────────────────────────────────────────────

SYSTEM_PROMPT = """你是一个智能代码助手，拥有以下能力：

1. 解释代码的逻辑和原理
2. 审查代码质量并给出建议
3. 根据需求生成代码示例
4. 调试和修复 Bug
5. 搜索相关文档

你的回答应该：
- 准确、简洁
- 包含代码示例
- 指出常见错误
- 使用中文回答"""

CODE_REVIEW_PROMPT = ChatPromptTemplate.from_messages([
    ("system", "你是一个严格的代码审查员。"),
    ("human", "审查以下代码:\n```python\n{code}\n```"),
])

# ─── 助手类 ─────────────────────────────────────────────────

class CodeAssistant:
    """智能代码助手"""

    def __init__(self, model_provider: str = "moonshot:moonshot-v1-8k"):
        self.model = init_chat_model(model_provider)
        self.model_with_tools = self.model.bind_tools([search_docs, run_code])
        self.tool_map = {t.name: t for t in [search_docs, run_code]}
        self.review_model = self.model.with_structured_output(CodeReviewResult)
        self.history: list = [SystemMessage(SYSTEM_PROMPT)]

    def explain(self, code: str) -> str:
        """解释代码"""
        prompt = f"请详细解释以下代码的功能、原理和可能的陷阱:\n```python\n{code}\n```"
        return self._invoke(prompt)

    def review(self, code: str) -> CodeReviewResult:
        """审查代码"""
        messages = CODE_REVIEW_PROMPT.format_messages(code=code)
        return self.review_model.invoke(messages)

    def generate(self, requirement: str) -> str:
        """根据需求生成代码"""
        prompt = f"根据以下需求编写 Python 代码，包含详细注释:\n{requirement}"
        return self._invoke(prompt)

    def debug(self, error: str, code: str = "") -> str:
        """调试错误"""
        prompt = f"分析以下错误并提供解决方案:\n"
        if code:
            prompt += f"代码:\n```python\n{code}\n```\n"
        prompt += f"错误信息:\n{error}"
        return self._invoke(prompt)

    def chat(self, query: str, stream: bool = True) -> str:
        """通用对话（支持工具调用）"""
        messages = self.history + [HumanMessage(query)]
        max_iterations = 3

        for _ in range(max_iterations):
            response = self.model_with_tools.invoke(messages)

            if stream and not response.tool_calls:
                full = None
                for chunk in self.model_with_tools.stream(messages):
                    if full is None:
                        full = chunk
                    else:
                        full += chunk
                    print(chunk.content, end="", flush=True)
                print()
                response = full

            messages.append(response)

            if not response.tool_calls:
                self.history.extend([HumanMessage(query), response])
                return response.content

            for tc in response.tool_calls:
                selected_tool = self.tool_map.get(tc["name"])
                if selected_tool:
                    result = selected_tool.invoke(tc["args"])
                else:
                    result = f"错误: 未知工具 {tc['name']}"

                messages.append(ToolMessage(content=result, tool_call_id=tc["id"]))
                print(f"\n🔧 {tc['name']} → {result[:80]}...")

        self.history.extend([HumanMessage(query), response])
        return response.content

    def _invoke(self, prompt: str) -> str:
        """内部调用方法"""
        response = self.model.invoke([
            SystemMessage(SYSTEM_PROMPT),
            HumanMessage(prompt),
        ])
        return response.content

    def clear_history(self):
        """清空对话历史"""
        self.history = [SystemMessage(SYSTEM_PROMPT)]

# ─── 交互式命令行 ───────────────────────────────────────────

def main():
    """主程序"""
    assistant = CodeAssistant("moonshot:moonshot-v1-8k")

    print("=" * 60)
    print("  智能代码助手")
    print("  支持命令: explain, review, generate, debug, chat")
    print("  输入 'quit' 退出, 'clear' 清空历史")
    print("=" * 60)

    while True:
        try:
            cmd = input("\n> ").strip()

            if cmd.lower() in ("quit", "exit", "q"):
                print("再见！")
                break

            if cmd.lower() == "clear":
                assistant.clear_history()
                print("历史已清空")
                continue

            if cmd.startswith("explain "):
                code = cmd[8:]
                print(f"\n📖 代码解释:\n")
                print(assistant.explain(code))

            elif cmd.startswith("review "):
                code = cmd[7:]
                result = assistant.review(code)
                print(f"\n🔍 代码审查:")
                print(f"  评分: {result.score}/10")
                print(f"  问题: {result.issues}")
                print(f"  建议: {result.suggestions}")
                print(f"  总结: {result.summary}")

            elif cmd.startswith("generate "):
                req = cmd[9:]
                print(f"\n✨ 生成代码:\n")
                print(assistant.generate(req))

            elif cmd.startswith("debug "):
                error = cmd[6:]
                print(f"\n🐛 调试分析:\n")
                print(assistant.debug(error))

            elif cmd.startswith("chat "):
                query = cmd[5:]
                assistant.chat(query)

            elif cmd:
                print("未知命令。支持: explain, review, generate, debug, chat")

        except KeyboardInterrupt:
            print("\n\n再见！")
            break
        except Exception as e:
            print(f"错误: {e}")

if __name__ == "__main__":
    main()
```

# 09.01.04 - 工具调用

> 工具调用让 LLM 能够与外部世界交互，是构建 Agent 的核心能力

## 什么是工具调用

LLM 本身无法执行代码、搜索网络或查询数据库。工具调用（Tool Calling）让 LLM 可以：

```
┌────────────────────────────────────────────────────────────┐
│                    工具调用工作流程                          │
│                                                            │
│   用户提问                                                  │
│     │                                                      │
│     ▼                                                      │
│  ┌──────────┐    工具调用请求    ┌──────────────┐           │
│  │          │  ─────────────→    │              │           │
│  │   LLM    │                    │  调用工具     │           │
│  │          │  ←─────────────    │              │           │
│  └──────────┘    工具执行结果    └──────────────┘           │
│     │                                                      │
│     ▼                                                      │
│  生成最终回复（基于工具结果）                                │
└────────────────────────────────────────────────────────────┘
```

## 定义工具

### 方式1：@tool 装饰器（推荐）

```python
from langchain.tools import tool

@tool
def add(a: int, b: int) -> int:
    """计算两个数字的和"""
    return a + b

@tool
def search(query: str) -> str:
    """搜索互联网获取信息"""
    return f"搜索结果: 关于 '{query}' 的信息"

# 查看工具的 OpenAI 格式描述
print(add.schema())
# {
#     'description': '计算两个数字的和',
#     'properties': {
#         'a': {'title': 'A', 'type': 'integer'},
#         'b': {'title': 'B', 'type': 'integer'},
#     },
#     'required': ['a', 'b'],
#     'title': 'add',
#     'type': 'object',
# }
```

**参数类型映射**：

| Python 类型 | JSON Schema 类型 | 说明 |
|------------|-----------------|------|
| `int` | `integer` | 整数 |
| `float` | `number` | 浮点数 |
| `str` | `string` | 字符串 |
| `bool` | `boolean` | 布尔值 |
| `list[str]` | `array` of strings | 字符串列表 |
| `dict` | `object` | 字典 |
| `Optional[str]` | `string` (nullable) | 可选参数 |

### 方式2：带参数描述

```python
@tool
def get_weather(city: str, unit: str = "celsius") -> str:
    """获取指定城市的天气信息

    Args:
        city: 城市名称，如 "北京"、"Shanghai"
        unit: 温度单位，"celsius"（摄氏度）或 "fahrenheit"（华氏度）
    """
    weather_db = {
        "北京": {"celsius": 25, "fahrenheit": 77},
        "上海": {"celsius": 22, "fahrenheit": 72},
    }
    data = weather_db.get(city, {})
    temp = data.get(unit, "未知")
    return f"{city} 当前温度: {temp}°{'C' if unit == 'celsius' else 'F'}"

print(get_weather.schema())
```

### 方式3：直接定义 JSON Schema

```python
from langchain_core.tools import StructuredTool

def calculator(expression: str) -> str:
    """计算数学表达式"""
    try:
        return str(eval(expression))
    except Exception as e:
        return f"计算错误: {e}"

tool = StructuredTool.from_function(
    func=calculator,
    name="calculator",
    description="计算数学表达式，支持加减乘除",
)
```

### 方式4：手动创建工具

```python
from langchain_core.tools import BaseTool

class WeatherTool(BaseTool):
    name: str = "get_weather"
    description: str = "获取指定城市的天气信息"

    def _run(self, city: str) -> str:
        """同步执行"""
        weather = {"北京": "晴 25°C", "上海": "多云 22°C"}
        return weather.get(city, f"无法获取 {city} 的天气")

    async def _arun(self, city: str) -> str:
        """异步执行（可选）"""
        return self._run(city)

tool = WeatherTool()
```

## 绑定工具到模型

### bind_tools 基础

```python
from langchain.chat_models import init_chat_model
from langchain.tools import tool
from langchain.messages import HumanMessage

@tool
def search(query: str) -> str:
    """搜索信息"""
    return f"搜索结果: {query}"

@tool
def calculator(expression: str) -> str:
    """计算表达式"""
    return f"计算结果: {eval(expression)}"

# 初始化模型
model = init_chat_model("moonshot:moonshot-v1-8k")

# 绑定多个工具
model_with_tools = model.bind_tools([search, calculator])

# 调用
response = model_with_tools.invoke(HumanMessage("2的10次方是多少？"))
print(f"内容: {response.content}")
print(f"工具调用: {response.tool_calls}")
```

### 检查是否有工具调用

```python
response = model_with_tools.invoke(HumanMessage("今天北京天气如何？"))

if response.tool_calls:
    print("AI 决定调用工具:")
    for tc in response.tool_calls:
        print(f"  工具名: {tc['name']}")
        print(f"  参数: {tc['args']}")
        print(f"  调用ID: {tc['id']}")
else:
    print("AI 直接回复，没有调用工具")
    print(f"回复内容: {response.content}")
```

## 工具调用示例

### 示例1: 单工具调用

```python
from langchain.chat_models import init_chat_model
from langchain.tools import tool
from langchain.messages import HumanMessage, ToolMessage

@tool
def get_current_time() -> str:
    """获取当前时间"""
    from datetime import datetime
    return datetime.now().strftime("%Y-%m-%d %H:%M:%S")

model = init_chat_model("moonshot:moonshot-v1-8k")
model_with_tools = model.bind_tools([get_current_time])

# 用户提问
messages = [HumanMessage("现在几点了？")]

# AI 可能调用工具
response = model_with_tools.invoke(messages)
messages.append(response)

# 执行工具调用
for tool_call in response.tool_calls:
    selected_tool = {"get_current_time": get_current_time}[tool_call["name"]]
    tool_output = selected_tool.invoke(tool_call["args"])
    messages.append(ToolMessage(content=tool_output, tool_call_id=tool_call["id"]))

# AI 根据工具结果生成最终回复
final_response = model_with_tools.invoke(messages)
print(final_response.content)
```

## 并行工具调用

LLM 可以一次调用多个工具（取决于模型能力）：

```python
@tool
def get_temperature(city: str) -> str:
    """获取城市温度"""
    temps = {"北京": 25, "上海": 22, "广州": 30}
    return f"{city}: {temps.get(city, '未知')}°C"

model = init_chat_model("moonshot:moonshot-v1-8k")
model_with_tools = model.bind_tools([get_temperature])

# 一次查询多个城市
response = model_with_tools.invoke(HumanMessage(
    "北京、上海和广州的温度分别是多少？"
))

if response.tool_calls:
    print(f"并行调用 {len(response.tool_calls)} 个工具:")
    for tc in response.tool_calls:
        print(f"  {tc['name']}(city='{tc['args']['city']}')")
```

  get_temperature(city='上海')
  get_temperature(city='广州')

## 强制工具调用

### 强制调用特定工具

```python
from langchain.chat_models import init_chat_model
from langchain.tools import tool
from langchain.messages import HumanMessage

@tool
def search(query: str) -> str:
    """搜索信息"""
    return f"关于 {query} 的搜索结果"

model = init_chat_model("moonshot:moonshot-v1-8k")

# 强制调用 search 工具
model_with_forced_tool = model.bind_tools(
    [search],
    tool_choice="search",  # 强制调用这个工具
)

response = model_with_forced_tool.invoke(HumanMessage("你好"))
print(f"工具调用: {response.tool_calls}")  # 一定会调用 search
```

### tool_choice 选项

| 值 | 说明 |
|----|------|
| `None` (默认) | 模型自行决定是否调用工具 |
| `"auto"` | 同 None，模型自行决定 |
| `"any"` | 必须调用某个工具 |
| `"工具名"` | 强制调用指定名称的工具 |

```python
# 必须调用任意工具
model = model.bind_tools([tool1, tool2], tool_choice="any")

# 强制调用特定工具
model = model.bind_tools([tool1, tool2], tool_choice="tool1")
```

## 完整工具执行循环

### 基础循环

```python
from langchain.chat_models import init_chat_model
from langchain.tools import tool
from langchain.messages import HumanMessage, ToolMessage, AIMessage

@tool
def search(query: str) -> str:
    """搜索信息"""
    knowledge_base = {
        "Python": "Python 是一种高级编程语言，由 Guido van Rossum 于 1991 年发布。",
        "GIL": "GIL（全局解释器锁）是 CPython 中防止多线程同时执行 Python 字节码的机制。",
        "装饰器": "装饰器是 Python 中修改函数行为的语法，用 @ 语法糖使用。",
    }
    for key, value in knowledge_base.items():
        if key.lower() in query.lower():
            return value
    return f"未找到关于 '{query}' 的信息"

@tool
def get_code_example(topic: str) -> str:
    """获取代码示例"""
    examples = {
        "装饰器": """
def my_decorator(func):
    def wrapper():
        print("执行前")
        func()
        print("执行后")
    return wrapper

@my_decorator
def say_hello():
    print("Hello!")
""",
    }
    return examples.get(topic, f"没有找到 {topic} 的代码示例")

# 创建工具映射
tools = [search, get_code_example]
tool_map = {t.name: t for t in tools}

# 初始化模型
model = init_chat_model("moonshot:moonshot-v1-8k")
model_with_tools = model.bind_tools(tools)

def chat_with_tools(query: str, max_iterations: int = 5) -> str:
    """完整的工具调用对话循环"""
    messages = [HumanMessage(query)]

    for i in range(max_iterations):
        # 调用模型
        response = model_with_tools.invoke(messages)
        messages.append(response)

        # 如果没有工具调用，说明完成了对话
        if not response.tool_calls:
            return response.content

        # 执行所有工具调用
        for tool_call in response.tool_calls:
            tool_name = tool_call["name"]
            tool_args = tool_call["args"]
            tool_id = tool_call["id"]

            selected_tool = tool_map.get(tool_name)
            if selected_tool:
                result = selected_tool.invoke(tool_args)
            else:
                result = f"错误: 未知工具 {tool_name}"

            messages.append(ToolMessage(content=result, tool_call_id=tool_id))

    return "达到最大迭代次数"

# 使用
result = chat_with_tools("什么是 GIL？能给我看一下代码示例吗？")
print(result)
```

### 带日志的循环

```python
def chat_with_tools_verbose(query: str, max_iterations: int = 5):
    """带详细日志的工具调用循环"""
    messages = [HumanMessage(query)]
    print(f"用户: {query}\n")

    for i in range(max_iterations):
        print(f"--- 第 {i+1} 轮 ---")

        response = model_with_tools.invoke(messages)
        messages.append(response)

        if not response.tool_calls:
            print(f"AI: {response.content}")
            return response.content

        # 显示工具调用信息
        for tc in response.tool_calls:
            print(f"🔧 调用工具: {tc['name']}")
            print(f"   参数: {tc['args']}")

        # 执行工具
        for tc in response.tool_calls:
            selected_tool = tool_map.get(tc["name"])
            if selected_tool:
                result = selected_tool.invoke(tc["args"])
            else:
                result = f"错误: 未知工具 {tc['name']}"

            messages.append(ToolMessage(content=result, tool_call_id=tc["id"]))
            print(f"   结果: {result[:100]}...")

        print()

    return "达到最大迭代次数"
```

## 错误处理

### 工具执行失败

```python
@tool
def divide(a: float, b: float) -> float:
    """除法计算"""
    return a / b

def safe_tool_execution(tool_call, tool_map):
    """安全执行工具，捕获异常"""
    tool_name = tool_call["name"]
    tool_args = tool_call["args"]
    tool_id = tool_call["id"]

    try:
        selected_tool = tool_map.get(tool_name)
        if not selected_tool:
            return ToolMessage(
                content=f"错误: 工具 {tool_name} 不存在",
                tool_call_id=tool_id,
                status="error",
            )

        result = selected_tool.invoke(tool_args)
        return ToolMessage(
            content=str(result),
            tool_call_id=tool_id,
        )
    except Exception as e:
        return ToolMessage(
            content=f"执行失败: {type(e).__name__}: {e}",
            tool_call_id=tool_id,
            status="error",
        )

# 使用
tool_call = {"name": "divide", "args": {"a": 10, "b": 0}, "id": "call_1"}
msg = safe_tool_execution(tool_call, {"divide": divide})
print(msg.content)  # 执行失败: ZeroDivisionError: division by zero
```

### 超时处理

```python
import asyncio
from langchain.tools import tool

@tool
def slow_search(query: str) -> str:
    """可能很慢的搜索"""
    import time
    time.sleep(10)  # 模拟慢查询
    return f"结果: {query}"

async def execute_with_timeout(tool_call, timeout: int = 5):
    """带超时的工具执行"""
    try:
        result = await asyncio.wait_for(
            asyncio.to_thread(tool_map[tool_call["name"]].invoke, tool_call["args"]),
            timeout=timeout,
        )
        return ToolMessage(content=str(result), tool_call_id=tool_call["id"])
    except asyncio.TimeoutError:
        return ToolMessage(
            content=f"工具执行超时（{timeout}秒）",
            tool_call_id=tool_call["id"],
            status="error",
        )
```

## 动态工具选择

```python
from langchain.chat_models import init_chat_model
from langchain.tools import tool

# 不同领域的工具
math_tools = [add, multiply, divide]
search_tools = [search, get_code_example]

model = init_chat_model("moonshot:moonshot-v1-8k")

def get_tools_for_query(query: str):
    """根据查询内容动态选择工具"""
    math_keywords = ["加", "减", "乘", "除", "计算", "多少"]
    if any(kw in query for kw in math_keywords):
        return math_tools
    return search_tools

# 动态绑定工具
query = "计算 3 + 5"
dynamic_tools = get_tools_for_query(query)
dynamic_model = model.bind_tools(dynamic_tools)

response = dynamic_model.invoke(HumanMessage(query))
```

## 工具最佳实践

### 1. 工具描述要清晰

```python
# 好的描述
@tool
def search_web(query: str) -> str:
    """搜索互联网获取最新信息。当需要实时信息或超出训练数据范围的知识时使用。"""
    ...

# 不好的描述（太模糊）
@tool
def search(query: str) -> str:
    """搜索"""
    ...
```

### 2. 参数类型要精确

```python
# 使用明确的类型和默认值
@tool
def get_weather(city: str, days: int = 1) -> str:
    """获取天气预报

    Args:
        city: 城市名称，如 "北京"
        days: 预报天数，默认 1 天，最大 7 天
    """
    ...
```

### 3. 工具数量不宜过多

| 工具数量 | 效果 |
|---------|------|
| 1-5 个 | 模型能很好地选择 |
| 5-15 个 | 模型通常能正确选择 |
| 15-30 个 | 选择准确率下降 |
| 30+ 个 | 建议分组或使用工具路由 |

### 4. 错误信息要有用

```python
# 好的错误信息
def divide(a: float, b: float) -> float:
    if b == 0:
        raise ValueError("除数不能为零")
    return a / b

# 工具返回错误时，LLM 可以据此调整
# ToolMessage: "执行失败: ValueError: 除数不能为零"
# → LLM 可以告诉用户除数不能为零
```

### 5. 使用工具链

```python
# 一个工具的输出作为另一个工具的输入
@tool
def extract_keywords(text: str) -> list[str]:
    """从文本中提取关键词"""
    return text.split()

@tool
def search_by_keywords(keywords: list[str]) -> str:
    """根据关键词搜索"""
    return f"搜索关键词: {keywords}"

# LLM 会自行编排工具调用顺序
model_with_tools = model.bind_tools([extract_keywords, search_by_keywords])
response = model_with_tools.invoke(HumanMessage(
    "从这段文字中提取关键词然后搜索相关信息: Python 是一种流行的编程语言"
))
```

## 小结

| 要点 | 说明 |
|------|------|
| 定义工具 | `@tool` 装饰器最简单，支持类型注解自动生成 Schema |
| 绑定工具 | `model.bind_tools([tool1, tool2])` |
| 强制调用 | `tool_choice="工具名"` 或 `tool_choice="any"` |
| 并行调用 | LLM 可一次输出多个 tool_calls |
| 执行循环 | 循环：调用 → 检查 → 执行工具 → 添加结果 → 再次调用 |
| 错误处理 | 工具异常转为 ToolMessage，status="error" |
| 动态选择 | 根据查询内容动态选择工具集 |

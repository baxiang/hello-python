## 图迁移概述

当应用需求变化时，需要迁移图结构而不丢失数据：

```
迁移场景：
┌──────────────────────────────────────┐
│  场景1：添加新节点                    │
│  v1: A → B → C                       │
│  v2: A → B → D → C                   │
│                                       │
│  场景2：修改节点逻辑                  │
│  v1: process_data (简单处理)          │
│  v2: process_data (复杂处理)          │
│                                       │
│  场景3：状态模式变更                  │
│  v1: State = {name: str}             │
│  v2: State = {name: str, age: int}   │
└──────────────────────────────────────┘
```

### 在图中使用缓存

```python
from langgraph.graph import StateGraph, START, END
from langgraph.cache import InMemoryCache
from typing import TypedDict

class State(TypedDict):
    query: str
    result: str

# 创建带缓存的图
cache = InMemoryCache()

def cached_node(state: State):
    """缓存节点"""
    query = state["query"]
    
    # 检查缓存
    cached = cache.get(query)
    if cached:
        print(f"缓存命中: {query}")
        return {"result": cached}
    
    # 计算结果
    print(f"缓存未命中: {query}")
    result = f"处理结果: {query}"
    
    # 存入缓存
    cache.put(query, result)
    
    return {"result": result}

builder = StateGraph(State)
builder.add_node("process", cached_node)
builder.add_edge(START, "process")
builder.add_edge("process", END)

graph = builder.compile()

# 第一次调用
result1 = graph.invoke({"query": "测试查询"})
print(result1["result"])

# 第二次调用（缓存命中）
result2 = graph.invoke({"query": "测试查询"})
print(result2["result"])
```

### 缓存键生成

```python
import hashlib

def generate_cache_key(state: dict, node_name: str) -> str:
    """生成缓存键"""
    # 基于节点名和状态生成键
    key_data = f"{node_name}:{state.get('query', '')}"
    return hashlib.md5(key_data.encode()).hexdigest()
```

## 迁移策略

### 自动迁移

```python
def migrate_state(state: dict, target_version: str) -> dict:
    """自动状态迁移"""
    current_version = state.get("_version", "v1")
    
    migrations = {
        ("v1", "v2"): migrate_v1_to_v2,
        ("v2", "v3"): migrate_v2_to_v3,
    }
    
    # 逐步迁移
    while current_version != target_version:
        key = (current_version, target_version)
        if key not in migrations:
            raise ValueError(f"无法从 {current_version} 迁移到 {target_version}")
        
        state = migrations[key](state)
        current_version = target_version
    
    return state
```

### 迁移检查点

```python
def migrate_checkpoint(checkpoint: dict, target_version: str) -> dict:
    """迁移 checkpoint"""
    # 迁移 state
    checkpoint["values"] = migrate_state(
        checkpoint["values"],
        target_version,
    )
    
    # 更新版本标记
    checkpoint["values"]["_version"] = target_version
    
    return checkpoint
```

## 缓存策略

### TTL 缓存

```python
from langgraph.cache import CachePolicy
import time

class TTLCache:
    """TTL 缓存实现"""
    
    def __init__(self, ttl: int = 3600):
        self.ttl = ttl
        self.cache = {}
    
    def get(self, key: str):
        if key in self.cache:
            value, timestamp = self.cache[key]
            if time.time() - timestamp < self.ttl:
                return value
            else:
                del self.cache[key]
        return None
    
    def put(self, key: str, value):
        self.cache[key] = (value, time.time())
    
    def clear(self):
        self.cache.clear()
```

### LRU 缓存

```python
from collections import OrderedDict

class LRUCache:
    """LRU 缓存"""
    
    def __init__(self, max_size: int = 1000):
        self.max_size = max_size
        self.cache = OrderedDict()
    
    def get(self, key: str):
        if key in self.cache:
            self.cache.move_to_end(key)
            return self.cache[key]
        return None
    
    def put(self, key: str, value):
        if key in self.cache:
            self.cache.move_to_end(key)
        self.cache[key] = value
        
        if len(self.cache) > self.max_size:
            self.cache.popitem(last=False)
```

## 缓存与持久化

### 缓存 + Checkpointer

```python
from langgraph.checkpoint.memory import MemorySaver
from langgraph.cache import InMemoryCache

# 缓存层（快）
cache = InMemoryCache()

# 持久化层（慢但可靠）
checkpointer = MemorySaver()

def process_with_cache_and_persist(state: State):
    query = state["query"]
    
    # 1. 检查缓存
    cached = cache.get(query)
    if cached:
        return {"result": cached}
    
    # 2. 检查持久化
    config = {"configurable": {"thread_id": query}}
    saved_state = graph.get_state(config)
    if saved_state and saved_state.values.get("result"):
        result = saved_state.values["result"]
        cache.put(query, result)  # 回填缓存
        return {"result": result}
    
    # 3. 计算
    result = f"处理: {query}"
    
    # 4. 存入缓存和持久化
    cache.put(query, result)
    
    return {"result": result}
```

```python
def invalidate_cache(key: str, reason: str):
    """缓存失效"""
    print(f"缓存失效: {key} (原因: {reason})")
    cache.delete(key)
```

### 2. 缓存监控

```python
cache_stats = {"hits": 0, "misses": 0}

def monitored_cache_get(key: str):
    value = cache.get(key)
    if value:
        cache_stats["hits"] += 1
    else:
        cache_stats["misses"] += 1
    return value

def get_cache_stats():
    total = cache_stats["hits"] + cache_stats["misses"]
    hit_rate = cache_stats["hits"] / total if total > 0 else 0
    return {
        "hits": cache_stats["hits"],
        "misses": cache_stats["misses"],
        "hit_rate": f"{hit_rate:.1%}",
    }
```

> 并行中断、多中断 ID、验证循环、工具中断

## 多中断场景

当图中有多个节点同时调用 `interrupt()` 时，需要处理多个中断：

```
多中断场景：
┌──────────────────────────────────────┐
│  场景1：并行节点                      │
│  START → node_a (interrupt)          │
│       → node_b (interrupt)           │
│  两个节点同时中断                     │
│                                       │
│  场景2：串行节点                      │
│  node_a (interrupt) → node_b (interrupt)│
│  顺序中断，逐个处理                   │
│                                       │
│  场景3：工具中断                      │
│  tool_a (interrupt) + tool_b (interrupt)│
│  多个工具同时需要审核                 │
└──────────────────────────────────────┘
```

## 并行节点中断

### 并行中断示例

```python
from typing import Annotated, TypedDict
from operator import add
from langgraph.graph import StateGraph, START, END
from langgraph.types import interrupt, Command
from langgraph.checkpoint.memory import MemorySaver

class State(TypedDict):
    vals: Annotated[list[str], add]

def node_a(state: State):
    """节点 A"""
    answer = interrupt("question_a")
    return {"vals": [f"a:{answer}"]}

def node_b(state: State):
    """节点 B"""
    answer = interrupt("question_b")
    return {"vals": [f"b:{answer}"]}

# 构建图
builder = StateGraph(State)
builder.add_node("a", node_a)
builder.add_node("b", node_b)
builder.add_edge(START, "a")
builder.add_edge(START, "b")  # 并行执行
builder.add_edge("a", END)
builder.add_edge("b", END)

graph = builder.compile(checkpointer=MemorySaver())
config = {"configurable": {"thread_id": "1"}}

# 调用 - 两个节点同时中断
result = graph.invoke({"vals": []}, config)

# 检查中断
print(result["__interrupt__"])
# 输出: (Interrupt(value='question_a', id='uuid-a'), Interrupt(value='question_b', id='uuid-b'))
```

### 处理多个中断

```python
# 方法1：按 ID 映射恢复
resume_map = {
    i.id: f"answer for {i.value}"
    for i in result["__interrupt__"]
}

# 恢复执行
result = graph.invoke(Command(resume=resume_map), config)
print(result["vals"])  # ['a:answer for question_a', 'b:answer for question_b']
```

## 串行中断

### 逐个处理

```python
from typing import TypedDict
from langgraph.graph import StateGraph, START, END
from langgraph.types import interrupt, Command

class State(TypedDict):
    step1_result: str
    step2_result: str

def step1_node(state: State):
    """步骤 1"""
    answer = interrupt("步骤 1 输入")
    return {"step1_result": f"step1:{answer}"}

def step2_node(state: State):
    """步骤 2"""
    answer = interrupt("步骤 2 输入")
    return {"step2_result": f"step2:{answer}"}

builder = StateGraph(State)
builder.add_node("step1", step1_node)
builder.add_node("step2", step2_node)
builder.add_edge(START, "step1")
builder.add_edge("step1", "step2")
builder.add_edge("step2", END)

graph = builder.compile(checkpointer=MemorySaver())
config = {"configurable": {"thread_id": "serial"}}

# 第一次中断（step1）
result1 = graph.invoke({"step1_result": ""}, config)
print(f"中断: {result1['__interrupt__']}")

# 恢复 step1
result2 = graph.invoke(Command(resume="input1"), config)
print(f"中断: {result2['__interrupt__']}")

# 恢复 step2
result3 = graph.invoke(Command(resume="input2"), config)
print(f"结果: {result3}")
```

### 验证循环

```python
from typing import TypedDict
from langgraph.graph import StateGraph, START, END
from langgraph.types import interrupt

class State(TypedDict):
    age: int

def get_age_node(state: State):
    """获取年龄（带验证循环）"""
    prompt = "请输入年龄（正整数）"
    
    while True:
        answer = interrupt(prompt)
        
        try:
            age = int(answer)
            if age > 0:
                return {"age": age}
            prompt = f"'{answer}' 不是正整数，请重新输入"
        except (ValueError, TypeError):
            prompt = f"'{answer}' 不是有效数字，请重新输入"
```

## 工具中断

### 多个工具同时中断

```python
from langchain.tools import tool
from langgraph.types import interrupt

@tool
def send_email(to: str, subject: str, body: str) -> str:
    """发送邮件（需要审核）"""
    response = interrupt({
        "action": "send_email",
        "to": to,
        "subject": subject,
        "body": body,
    })
    
    if response.get("approve"):
        return f"邮件已发送给 {response.get('to', to)}"
    return "邮件已取消"

@tool
def delete_file(file_path: str) -> str:
    """删除文件（需要审核）"""
    response = interrupt({
        "action": "delete_file",
        "file_path": file_path,
        "message": "确认删除？",
    })
    
    if response.get("approve"):
        return f"文件 {file_path} 已删除"
    return "删除已取消"
```

## 多中断 ID 管理

### ID 追踪

```python
def handle_multiple_interrupts(graph, initial_input, config):
    """处理多个中断"""
    current_input = initial_input
    
    while True:
        result = graph.invoke(current_input, config)
        
        if "__interrupt__" not in result:
            # 完成
            return result
        
        interrupts = result["__interrupt__"]
        print(f"收到 {len(interrupts)} 个中断")
        
        # 构建恢复映射
        resume_map = {}
        for interrupt_obj in interrupts:
            print(f"中断 ID: {interrupt_obj.id}")
            print(f"中断值: {interrupt_obj.value}")
            
            # 获取用户响应
            user_response = get_user_response(interrupt_obj.value)
            resume_map[interrupt_obj.id] = user_response
        
        # 恢复执行
        current_input = Command(resume=resume_map)
```

### 带超时的中断

```python
import asyncio

async def handle_interrupt_with_timeout(graph, initial_input, config, timeout=300):
    """带超时的中断处理"""
    current_input = initial_input
    
    while True:
        result = graph.invoke(current_input, config)
        
        if "__interrupt__" not in result:
            return result
        
        # 等待用户响应，带超时
        try:
            user_response = await asyncio.wait_for(
                get_user_response_async(result["__interrupt__"]),
                timeout=timeout,
            )
        except asyncio.TimeoutError:
            # 超时，取消操作
            return {"error": "操作超时"}
        
        current_input = Command(resume=user_response)
```

## 并行中断最佳实践

### 1. 独立处理每个中断

```python
def process_interrupts(interrupts):
    """独立处理每个中断"""
    resume_map = {}
    
    for interrupt_obj in interrupts:
        value = interrupt_obj.value
        interrupt_id = interrupt_obj.id
        
        # 根据中断值类型处理
        if isinstance(value, dict):
            if value.get("type") == "approval":
                response = handle_approval(value)
            elif value.get("type") == "input":
                response = handle_input(value)
            else:
                response = "unknown"
        else:
            response = handle_simple(value)
        
        resume_map[interrupt_id] = response
    
    return resume_map
```

### 2. 中断优先级

```python
def prioritize_interrupts(interrupts):
    """按优先级处理中断"""
    priority_order = {
        "security": 0,   # 安全相关最高
        "approval": 1,   # 审批次之
        "input": 2,      # 输入再次
        "info": 3,       # 信息最低
    }
    
    # 排序
    sorted_interrupts = sorted(
        interrupts,
        key=lambda x: priority_order.get(
            x.value.get("type", "info") if isinstance(x.value, dict) else "info",
            3,
        ),
    )
    
    return sorted_interrupts
```

### 打印中断信息

```python
def debug_interrupts(result):
    """调试中断"""
    if "__interrupt__" not in result:
        print("无中断")
        return
    
    interrupts = result["__interrupt__"]
    print(f"中断数量: {len(interrupts)}")
    
    for i, interrupt_obj in enumerate(interrupts):
        print(f"\n中断 {i + 1}:")
        print(f"  ID: {interrupt_obj.id}")
        print(f"  值: {interrupt_obj.value}")
        print(f"  类型: {type(interrupt_obj.value)}")
```

## 小结

| 要点 | 说明 |
|------|------|
| 并行中断 | 多个节点同时中断，用 ID 映射恢复 |
| 串行中断 | 逐个处理，顺序恢复 |
| 工具中断 | 工具内调用 interrupt，恢复时可覆盖参数 |
| 验证循环 | while True + interrupt 直到输入有效 |
| ID 管理 | 每个中断有唯一 ID，Command(resume={id: value}) |
| 最佳实践 | 独立处理、优先级排序、批量 UI |
| 调试 | 打印中断信息、查看中断历史 |

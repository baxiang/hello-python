# 09.05.09 - 进阶实战

> 结合持久化、中断、子图的完整项目：智能审批工作流

## 项目概述

构建一个智能审批工作流系统，具备以下能力：

```
审批工作流架构：
┌──────────────────────────────────────┐
│  提交申请                             │
│    ↓                                 │
│  自动审核（子图）                     │
│  ├─ 格式检查                         │
│  ├─ 规则验证                         │
│  └─ 风险评估                         │
│    ↓                                 │
│  人工审核（中断）                     │
│    ↓                                 │
│  执行审批                             │
│    ↓                                 │
│  通知结果                             │
│    ↓                                 │
│  持久化（PostgreSQL）                 │
└──────────────────────────────────────┘
```

### 功能清单

| 功能 | 技术 | 说明 |
|------|------|------|
| 自动审核 | 子图 | 格式检查、规则验证、风险评估 |
| 人工审核 | 中断 | interrupt() 等待人工决策 |
| 持久化 | PostgresSaver | 审批记录持久化 |
| 长期记忆 | PostgresStore | 用户历史、审批模板 |
| 流式输出 | astream_events | 实时审批进度 |
| 状态追踪 | Checkpointer | 审批流程状态管理 |

from typing import TypedDict, Literal

class Application(TypedDict):
    """申请"""
    applicant_id: str
    applicant_name: str
    application_type: Literal["leave", "expense", "purchase"]
    amount: float
    reason: str
    attachments: list[str]

class AutoReviewResult(TypedDict):
    """自动审核结果"""
    format_valid: bool
    rules_passed: bool
    risk_level: Literal["low", "medium", "high"]
    suggestions: list[str]

class ApprovalState(TypedDict):
    """审批状态"""
    application: Application
    auto_review: AutoReviewResult
    human_decision: Literal["approved", "rejected", "needs_revision"]
    human_comment: str
    final_status: Literal["pending", "approved", "rejected"]
    notification_sent: bool

## 2. 自动审核子图

```python
# subgraphs.py
from typing import TypedDict
from langgraph.graph import StateGraph, START, END

class ReviewState(TypedDict):
    application: dict
    format_valid: bool
    rules_passed: bool
    risk_level: str
    suggestions: list[str]

def format_check(state: ReviewState):
    """格式检查"""
    app = state["application"]
    suggestions = []
    
    # 检查必填字段
    required = ["applicant_id", "applicant_name", "application_type", "reason"]
    missing = [f for f in required if not app.get(f)]
    
    if missing:
        suggestions.append(f"缺少字段: {', '.join(missing)}")
    
    # 检查金额
    if app.get("application_type") == "expense" and app.get("amount", 0) <= 0:
        suggestions.append("报销金额必须大于 0")
    
    format_valid = len(suggestions) == 0
    
    return {
        "format_valid": format_valid,
        "suggestions": suggestions,
    }

def rule_validation(state: ReviewState):
    """规则验证"""
    app = state["application"]
    suggestions = list(state.get("suggestions", []))
    
    # 报销规则
    if app.get("application_type") == "expense":
        amount = app.get("amount", 0)
        if amount > 10000:
            suggestions.append("超过 10000 元需要额外审批")
        if amount > 50000:
            suggestions.append("超过 50000 元需要总经理审批")
    
    # 请假规则
    if app.get("application_type") == "leave":
        if "天数" not in app.get("reason", ""):
            suggestions.append("请假申请请注明天数")
    
    rules_passed = not any("需要" in s or "请注明" in s for s in suggestions)
    
    return {
        "rules_passed": rules_passed,
        "suggestions": suggestions,
    }

def risk_assessment(state: ReviewState):
    """风险评估"""
    app = state["application"]
    amount = app.get("amount", 0)
    
    if amount > 50000:
        risk_level = "high"
    elif amount > 10000:
        risk_level = "medium"
    else:
        risk_level = "low"
    
    return {"risk_level": risk_level}

# 构建审核子图
review_builder = StateGraph(ReviewState)
review_builder.add_node("format_check", format_check)
review_builder.add_node("rule_validation", rule_validation)
review_builder.add_node("risk_assessment", risk_assessment)

review_builder.add_edge(START, "format_check")
review_builder.add_edge("format_check", "rule_validation")
review_builder.add_edge("rule_validation", "risk_assessment")
review_builder.add_edge("risk_assessment", END)

review_graph = review_builder.compile()
```

## 3. 主图定义

```python
# graph.py
from langgraph.graph import StateGraph, START, END
from langgraph.types import interrupt, Command
from langgraph.checkpoint.postgres import PostgresSaver
from langgraph.store.postgres import PostgresStore
from typing import Literal
from subgraphs import review_graph

class ApprovalState(TypedDict):
    application: dict
    auto_review: dict
    human_decision: str
    human_comment: str
    final_status: str
    notification_sent: bool

def run_auto_review(state: ApprovalState):
    """运行自动审核子图"""
    review_input = {
        "application": state["application"],
        "format_valid": False,
        "rules_passed": False,
        "risk_level": "low",
        "suggestions": [],
    }
    
    result = review_graph.invoke(review_input)
    
    return {
        "auto_review": {
            "format_valid": result["format_valid"],
            "rules_passed": result["rules_passed"],
            "risk_level": result["risk_level"],
            "suggestions": result["suggestions"],
        }
    }

def human_review(state: ApprovalState) -> Command[Literal["approve", "reject", END]]:
    """人工审核"""
    auto_review = state["auto_review"]
    
    # 显示审核信息
    review_info = {
        "application": state["application"],
        "auto_review": auto_review,
        "question": "请审核此申请",
        "options": ["approve", "reject", "needs_revision"],
    }
    
    # 中断等待人工决策
    decision = interrupt(review_info)
    
    if decision == "approve":
        return Command(goto="approve")
    elif decision == "reject":
        return Command(goto="reject")
    else:
        return Command(goto="approve")  # 默认批准

def approve(state: ApprovalState):
    """批准"""
    return {
        "human_decision": "approved",
        "final_status": "approved",
    }

def reject(state: ApprovalState):
    """拒绝"""
    return {
        "human_decision": "rejected",
        "final_status": "rejected",
    }

def send_notification(state: ApprovalState):
    """发送通知"""
    status = state["final_status"]
    applicant = state["application"]["applicant_name"]
    
    print(f"通知 {applicant}: 申请已{status}")
    
    return {"notification_sent": True}

# 构建主图
builder = StateGraph(ApprovalState)
builder.add_node("auto_review", run_auto_review)
builder.add_node("human_review", human_review)
builder.add_node("approve", approve)
builder.add_node("reject", reject)
builder.add_node("notify", send_notification)

builder.add_edge(START, "auto_review")
builder.add_edge("auto_review", "human_review")
builder.add_edge("approve", "notify")
builder.add_edge("reject", "notify")
builder.add_edge("notify", END)

# 编译
checkpointer = PostgresSaver.from_conn_string("postgresql://...")
checkpointer.setup()

store = PostgresStore.from_conn_string("postgresql://...")
store.setup()

graph = builder.compile(checkpointer=checkpointer, store=store)
```

## 4. FastAPI 部署

```python
# main.py
from fastapi import FastAPI
from fastapi.responses import StreamingResponse
from pydantic import BaseModel
import json

from graph import graph

app = FastAPI(title="智能审批系统")

class ApplicationRequest(BaseModel):
    applicant_id: str
    applicant_name: str
    application_type: str
    amount: float = 0
    reason: str

@app.post("/apply")
async def submit_application(request: ApplicationRequest):
    """提交申请"""
    import uuid
    thread_id = f"app-{uuid.uuid4().hex[:8]}"
    
    config = {"configurable": {"thread_id": thread_id}}
    
    application = {
        "applicant_id": request.applicant_id,
        "applicant_name": request.applicant_name,
        "application_type": request.application_type,
        "amount": request.amount,
        "reason": request.reason,
        "attachments": [],
    }
    
    # 运行到中断点
    result = graph.invoke(
        {
            "application": application,
            "auto_review": {},
            "human_decision": "",
            "human_comment": "",
            "final_status": "pending",
            "notification_sent": False,
        },
        config=config,
    )
    
    if "__interrupt__" in result:
        return {
            "thread_id": thread_id,
            "status": "pending_review",
            "review_info": result["__interrupt__"][0].value,
        }
    
    return {
        "thread_id": thread_id,
        "status": result["final_status"],
    }

@app.post("/review/{thread_id}")
async def review_application(thread_id: str, decision: str):
    """审核申请"""
    config = {"configurable": {"thread_id": thread_id}}
    
    result = graph.invoke(Command(resume=decision), config=config)
    
    return {
        "thread_id": thread_id,
        "status": result["final_status"],
        "notification_sent": result["notification_sent"],
    }

@app.get("/status/{thread_id}")
async def get_status(thread_id: str):
    """查询状态"""
    config = {"configurable": {"thread_id": thread_id}}
    state = graph.get_state(config)
    
    return {
        "thread_id": thread_id,
        "status": state.values.get("final_status", "unknown"),
        "next": state.next,
    }

@app.post("/apply/stream")
async def stream_application(request: ApplicationRequest):
    """流式提交申请"""
    import uuid
    thread_id = f"app-{uuid.uuid4().hex[:8]}"
    config = {"configurable": {"thread_id": thread_id}}
    
    application = {
        "applicant_id": request.applicant_id,
        "applicant_name": request.applicant_name,
        "application_type": request.application_type,
        "amount": request.amount,
        "reason": request.reason,
        "attachments": [],
    }
    
    inputs = {
        "application": application,
        "auto_review": {},
        "human_decision": "",
        "human_comment": "",
        "final_status": "pending",
        "notification_sent": False,
    }
    
    async def generate():
        async for event in graph.astream_events(inputs, config=config, version="v2"):
            kind = event["event"]
            
            if kind == "on_chain_start":
                yield f"data: [START] {event['name']}\n\n"
            elif kind == "on_chain_end":
                yield f"data: [END] {event['name']}\n\n"
            elif kind == "on_interrupt":
                yield f"data: [INTERRUPT] {json.dumps(event['data'])}\n\n"
        
        yield "data: [DONE]\n\n"
    
    return StreamingResponse(
        generate(),
        media_type="text/event-stream",
    )
```

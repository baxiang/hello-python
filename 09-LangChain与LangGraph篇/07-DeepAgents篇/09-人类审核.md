# 09.07.09 - 人类审核

> 配置敏感工具操作的人类审批流程

## 什么是人类审核

人类审核（Human-in-the-Loop）让 Agent 在执行敏感工具时暂停，等待人类审批后再继续。

```
Agent 执行
  ↓
调用工具
  ↓
需要审批？
├─ 否 → 执行工具 → 继续
└─ 是 → 暂停执行
       ↓
    人类审查
       ↓
    ┌─ approve → 用原始参数执行
    ├─ edit    → 修改参数后执行
    └─ reject  → 跳过执行
       ↓
    继续执行
```

## 基本配置

`interrupt_on` 参数为每个工具配置审核行为：

| 值 | 说明 |
|---|------|
| `True` | 启用审核，允许 approve / edit / reject |
| `False` | 不启用审核 |
| `{"allowed_decisions": [...]}` | 自定义允许的决策类型 |

```python
from langchain.tools import tool
from deepagents import create_deep_agent
from langgraph.checkpoint.memory import MemorySaver

@tool
def delete_file(path: str) -> str:
    """删除文件"""
    return f"已删除 {path}"

@tool
def read_file(path: str) -> str:
    """读取文件"""
    return f"{path} 的内容..."

@tool
def send_email(to: str, subject: str, body: str) -> str:
    """发送邮件"""
    return f"邮件已发送至 {to}"

# Checkpointer 是必需的
checkpointer = MemorySaver()

agent = create_deep_agent(
    model="moonshot:moonshot-v1-8k",
    tools=[delete_file, read_file, send_email],
    interrupt_on={
        "delete_file": True,  # 默认：approve, edit, reject
        "read_file": False,   # 不需要审核
        "send_email": {"allowed_decisions": ["approve", "reject"]},  # 不允许编辑
    },
    checkpointer=checkpointer,
)
```

## 决策类型

| 决策 | 说明 |
|------|------|
| `approve` | 使用 Agent 原始参数执行 |
| `edit` | 修改参数后执行 |
| `reject` | 跳过执行 |

```python
interrupt_on = {
    # 高风险：所有选项
    "delete_file": {"allowed_decisions": ["approve", "edit", "reject"]},
    # 中风险：不允许编辑
    "write_file": {"allowed_decisions": ["approve", "reject"]},
    # 必须批准（不可拒绝）
    "critical_operation": {"allowed_decisions": ["approve"]},
}
```

## 处理审核

### 单工具调用

```python
from langchain_core.utils.uuid import uuid7
from langgraph.types import Command

config = {"configurable": {"thread_id": str(uuid7())}}

# 第一次调用
result = agent.invoke(
    {"messages": [{"role": "user", "content": "删除 temp.txt"}]},
    config=config,
    version="v2",
)

# 检查是否中断
if result.interrupts:
    interrupt_value = result.interrupts[0].value
    action_requests = interrupt_value["action_requests"]
    review_configs = interrupt_value["review_configs"]

    # 创建工具名到审核配置的映射
    config_map = {cfg["action_name"]: cfg for cfg in review_configs}

    # 显示待审操作
    for action in action_requests:
        review_config = config_map[action["name"]]
        print(f"工具: {action['name']}")
        print(f"参数: {action['args']}")
        print(f"允许决策: {review_config['allowed_decisions']}")

    # 用户决策
    decisions = [
        {"type": "approve"}  # 批准删除
    ]

    # 恢复执行
    result = agent.invoke(
        Command(resume={"decisions": decisions}),
        config=config,  # 必须用相同的 config
        version="v2",
    )

print(result.value["messages"][-1].content)
```

### 多工具调用

当 Agent 同时调用多个需要审批的工具时，所有审核合并到一个中断：

```python
config = {"configurable": {"thread_id": str(uuid7())}}

result = agent.invoke(
    {"messages": [{
        "role": "user",
        "content": "删除 temp.txt 并发送邮件给 admin@example.com"
    }]},
    config=config,
    version="v2",
)

if result.interrupts:
    interrupt_value = result.interrupts[0].value
    action_requests = interrupt_value["action_requests"]

    # 两个工具需要审批
    assert len(action_requests) == 2

    # 按顺序提供决策
    decisions = [
        {"type": "approve"},   # delete_file
        {"type": "reject"},    # send_email
    ]

    result = agent.invoke(
        Command(resume={"decisions": decisions}),
        config=config,
        version="v2",
    )
```

### 编辑参数

```python
if result.interrupts:
    interrupt_value = result.interrupts[0].value
    action_request = interrupt_value["action_requests"][0]

    # Agent 原始参数
    print(action_request["args"])  # {"to": "everyone@company.com", ...}

    # 用户决定编辑收件人
    decisions = [{
        "type": "edit",
        "edited_action": {
            "name": action_request["name"],  # 必须包含工具名
            "args": {"to": "team@company.com", "subject": "...", "body": "..."}
        }
    }]

    result = agent.invoke(
        Command(resume={"decisions": decisions}),
        config=config,
        version="v2",
    )
```

## 子代理审核

### 子代理工具调用审核

每个子代理可单独配置：

```python
agent = create_deep_agent(
    model="moonshot:moonshot-v1-8k",
    tools=[delete_file, read_file],
    interrupt_on={
        "delete_file": True,
        "read_file": False,
    },
    subagents=[{
        "name": "file-manager",
        "description": "管理文件操作",
        "system_prompt": "你是文件管理助手。",
        "tools": [delete_file, read_file],
        "interrupt_on": {
            "delete_file": True,
            "read_file": True,  # 子代理中读也要审核
        }
    }],
    checkpointer=checkpointer,
)
```

### 工具内中断

子代理工具可直接调用 `interrupt()`：

```python
from langchain.agents import create_agent
from langchain.tools import tool
from langgraph.types import interrupt, Command
from deepagents import create_deep_agent, CompiledSubAgent

@tool(description="在执行操作前请求人类审批")
def request_approval(action_description: str) -> str:
    """使用 interrupt() 暂停并等待审批"""
    approval = interrupt({
        "type": "approval_request",
        "action": action_description,
        "message": f"请审批: {action_description}",
    })

    if approval.get("approved"):
        return f"操作 '{action_description}' 已批准"
    else:
        return f"操作 '{action_description}' 被拒绝"

# 创建子代理
compiled_subagent = create_agent(
    model="moonshot:moonshot-v1-8k",
    tools=[request_approval],
    name="approval-agent",
)

# 父 Agent
parent_agent = create_deep_agent(
    model="moonshot:moonshot-v1-8k",
    checkpointer=MemorySaver(),
    subagents=[
        CompiledSubAgent(
            name="approval-agent",
            description="可以请求审批的代理",
            runnable=compiled_subagent,
        ),
    ],
)

thread_id = "test-interrupt"
config = {"configurable": {"thread_id": thread_id}}

result = parent_agent.invoke(
    {"messages": [{"role": "user", "content": "使用审批代理请求批准'部署到生产'"}]},
    config=config,
    version="v2",
)

if result.interrupts:
    interrupt_value = result.interrupts[0].value
    print(f"类型: {interrupt_value.get('type')}")
    print(f"操作: {interrupt_value.get('action')}")

    # 审批通过
    result2 = parent_agent.invoke(
        Command(resume={"approved": True}),
        config=config,
        version="v2",
    )
```

## 最佳实践

### 始终使用 Checkpointer

```python
from langgraph.checkpoint.memory import MemorySaver

checkpointer = MemorySaver()
agent = create_deep_agent(
    model="moonshot:moonshot-v1-8k",
    tools=[...],
    interrupt_on={...},
    checkpointer=checkpointer,  # 必需
)
```

### 使用相同的 thread_id

```python
config = {"configurable": {"thread_id": "my-thread"}}
result = agent.invoke(input, config=config, version="v2")

# 恢复（必须用相同的 config）
result = agent.invoke(Command(resume={...}), config=config, version="v2")
```

### 按风险配置审核

```python
interrupt_on = {
    # 高风险：完整控制
    "delete_file": {"allowed_decisions": ["approve", "edit", "reject"]},
    "send_email": {"allowed_decisions": ["approve", "edit", "reject"]},
    # 中风险：不允许编辑
    "write_file": {"allowed_decisions": ["approve", "reject"]},
    # 低风险：不审核
    "read_file": False,
    "list_files": False,
}
```

## 小结

| 要点 | 说明 |
|------|------|
| interrupt_on | 为每个工具配置审核行为 |
| 决策类型 | approve、edit、reject |
| 多工具中断 | 合并到一个中断，按顺序提供决策 |
| 子代理审核 | 可单独配置，支持工具内 interrupt() |
| Checkpointer | 必需，用于持久化中断状态 |
| 恢复执行 | Command(resume={...}) + 相同 thread_id |

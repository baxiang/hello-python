# 09.06.05 - 流式部署

> FastAPI SSE 流式部署、前端集成、实时 UI、WebSocket 备选

## 流式部署概述

将 Agent 的流式输出通过 SSE 或 WebSocket 推送给前端：

```
流式部署架构：
┌──────────────────────────────────────┐
│  前端                                 │
│  ├─ EventSource (SSE)                │
│  └─ WebSocket (备选)                 │
│    ↓                                 │
│  FastAPI                             │
│  ├─ /chat/stream (SSE)              │
│  └─ /ws/chat (WebSocket)            │
│    ↓                                 │
│  Agent                               │
│  ├─ astream_events                   │
│  └─ stream_mode                      │
└──────────────────────────────────────┘
```

---

## SSE 基础

### SSE vs WebSocket

| 特性 | SSE | WebSocket |
|------|-----|-----------|
| 方向 | 单向（服务器→客户端） | 双向 |
| 协议 | HTTP | ws/wss |
| 重连 | 自动 | 需手动 |
| 复杂度 | 简单 | 较复杂 |
| 适用 | 流式输出 | 实时交互 |

## FastAPI SSE 端点

### 基本 SSE

```python
from fastapi import FastAPI
from fastapi.responses import StreamingResponse
from langchain.agents import create_agent
from pydantic import BaseModel
import json

app = FastAPI()

agent = create_agent(
    model="moonshot:moonshot-v1-8k",
    tools=[search_tool],
)

class ChatRequest(BaseModel):
    thread_id: str
    message: str

@app.post("/chat/stream")
async def stream_chat(request: ChatRequest):
    """SSE 流式聊天"""
    config = {"configurable": {"thread_id": request.thread_id}}
    inputs = {"messages": [{"role": "user", "content": request.message}]}
    
    async def generate():
        async for event in agent.astream_events(inputs, config=config, version="v2"):
            kind = event["event"]
            
            if kind == "on_chat_model_stream":
                token = event["data"]["chunk"].content
                if token:
                    yield f"data: {json.dumps({'type': 'token', 'content': token})}\n\n"
            
            elif kind == "on_tool_start":
                yield f"data: {json.dumps({'type': 'tool_start', 'name': event['name']})}\n\n"
            
            elif kind == "on_tool_end":
                yield f"data: {json.dumps({'type': 'tool_end', 'name': event['name']})}\n\n"
        
        yield f"data: {json.dumps({'type': 'done'})}\n\n"
        yield "data: [DONE]\n\n"
    
    return StreamingResponse(
        generate(),
        media_type="text/event-stream",
        headers={
            "Cache-Control": "no-cache",
            "Connection": "keep-alive",
            "X-Accel-Buffering": "no",
        },
    )
```

### 完整 SSE 端点

```python
from fastapi import FastAPI
from fastapi.responses import StreamingResponse
from langchain.agents import create_agent
from langgraph.checkpoint.postgres import PostgresSaver
from pydantic import BaseModel
import json
import time

app = FastAPI(title="流式 Agent API")

# 创建 Agent
checkpointer = PostgresSaver.from_conn_string("postgresql://...")
checkpointer.setup()

agent = create_agent(
    model="moonshot:moonshot-v1-8k",
    tools=[search_tool, calc_tool],
    checkpointer=checkpointer,
)

class ChatRequest(BaseModel):
    thread_id: str
    message: str
    include_thoughts: bool = False
    stream_format: str = "sse"  # sse or json

@app.post("/v1/chat/stream")
async def stream_chat_v1(request: ChatRequest):
    """OpenAI 兼容流式接口"""
    config = {"configurable": {"thread_id": request.thread_id}}
    inputs = {"messages": [{"role": "user", "content": request.message}]}
    
    start_time = time.time()
    
    async def generate():
        # 创建事件
        yield f"data: {json.dumps({'id': 'chatcmpl-123', 'object': 'chat.completion.chunk', 'created': int(time.time()), 'model': 'agent', 'choices': [{'index': 0, 'delta': {'role': 'assistant'}, 'finish_reason': None}]})}\n\n"
        
        async for event in agent.astream_events(inputs, config=config, version="v2"):
            kind = event["event"]
            
            if kind == "on_chat_model_stream":
                token = event["data"]["chunk"].content
                if token:
                    chunk = {
                        "id": "chatcmpl-123",
                        "object": "chat.completion.chunk",
                        "created": int(time.time()),
                        "model": "agent",
                        "choices": [{
                            "index": 0,
                            "delta": {"content": token},
                            "finish_reason": None,
                        }],
                    }
                    yield f"data: {json.dumps(chunk)}\n\n"
            
            elif kind == "on_tool_start" and request.include_thoughts:
                chunk = {
                    "id": "chatcmpl-123",
                    "object": "chat.completion.chunk",
                    "created": int(time.time()),
                    "model": "agent",
                    "choices": [{
                        "index": 0,
                        "delta": {
                            "content": None,
                            "tool_calls": [{
                                "index": 0,
                                "id": event.get("run_id", ""),
                                "type": "function",
                                "function": {
                                    "name": event["name"],
                                    "arguments": json.dumps(event["data"].get("input", {})),
                                },
                            }],
                        },
                        "finish_reason": None,
                    }],
                }
                yield f"data: {json.dumps(chunk)}\n\n"
            
            elif kind == "on_chain_end":
                # 完成
                duration = time.time() - start_time
                chunk = {
                    "id": "chatcmpl-123",
                    "object": "chat.completion.chunk",
                    "created": int(time.time()),
                    "model": "agent",
                    "choices": [{
                        "index": 0,
                        "delta": {},
                        "finish_reason": "stop",
                    }],
                    "usage": {
                        "prompt_tokens": 0,
                        "completion_tokens": 0,
                        "total_tokens": 0,
                        "duration_ms": round(duration * 1000, 2),
                    },
                }
                yield f"data: {json.dumps(chunk)}\n\n"
        
        yield "data: [DONE]\n\n"
    
    return StreamingResponse(
        generate(),
        media_type="text/event-stream",
        headers={
            "Cache-Control": "no-cache",
            "Connection": "keep-alive",
            "X-Accel-Buffering": "no",
        },
    )
```

## WebSocket 备选

### WebSocket 端点

```python
from fastapi import FastAPI, WebSocket
from langchain.agents import create_agent

app = FastAPI()

agent = create_agent(
    model="moonshot:moonshot-v1-8k",
    tools=[search_tool],
)

@app.websocket("/ws/chat")
async def websocket_chat(websocket: WebSocket):
    """WebSocket 聊天"""
    await websocket.accept()
    
    thread_id = "ws-session"
    config = {"configurable": {"thread_id": thread_id}}
    
    try:
        while True:
            # 接收消息
            data = await websocket.receive_text()
            message = data
            
            # 流式发送
            inputs = {"messages": [{"role": "user", "content": message}]}
            
            async for event in agent.astream_events(inputs, config=config, version="v2"):
                if event["event"] == "on_chat_model_stream":
                    token = event["data"]["chunk"].content
                    if token:
                        await websocket.send_text(token)
            
            # 发送结束标记
            await websocket.send_text("[DONE]")
    
    except Exception as e:
        await websocket.close()
```

## 前端集成

### SSE 前端

```html
<!DOCTYPE html>
<html>
<head>
    <title>流式 AI 助手</title>
    <style>
        #messages { height: 400px; overflow-y: auto; border: 1px solid #ccc; padding: 10px; }
        .user { color: blue; }
        .ai { color: green; }
        .tool { color: orange; }
    </style>
</head>
<body>
    <div id="messages"></div>
    <input type="text" id="input" placeholder="输入消息..." style="width: 80%;" />
    <button onclick="sendMessage()">发送</button>
    
    <script>
        const messagesDiv = document.getElementById('messages');
        const input = document.getElementById('input');
        let currentResponse = null;
        
        async function sendMessage() {
            const message = input.value;
            if (!message) return;
            input.value = '';
            
            appendMessage('user', message);
            currentResponse = appendMessage('ai', '');
            
            const response = await fetch('/v1/chat/stream', {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({
                    thread_id: 'session-1',
                    message: message,
                    include_thoughts: true,
                }),
            });
            
            const reader = response.body.getReader();
            const decoder = new TextDecoder();
            let buffer = '';
            
            while (true) {
                const {value, done} = await reader.read();
                if (done) break;
                
                buffer += decoder.decode(value, {stream: true});
                const lines = buffer.split('\n');
                buffer = lines.pop();
                
                for (const line of lines) {
                    if (line.startsWith('data: ')) {
                        const data = line.slice(6);
                        
                        if (data === '[DONE]') continue;
                        
                        try {
                            const parsed = JSON.parse(data);
                            
                            if (parsed.choices?.[0]?.delta?.content) {
                                currentResponse.textContent += parsed.choices[0].delta.content;
                            }
                        } catch (e) {
                            // 忽略解析错误
                        }
                    }
                }
            }
            
            messagesDiv.scrollTop = messagesDiv.scrollHeight;
        }
        
        function appendMessage(role, content) {
            const div = document.createElement('div');
            div.className = role;
            div.textContent = content;
            messagesDiv.appendChild(div);
            messagesDiv.scrollTop = messagesDiv.scrollHeight;
            return div;
        }
        
        input.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') sendMessage();
        });
    </script>
</body>
</html>
```

### React 集成

```tsx
import { useState, useRef, useEffect } from 'react';

function ChatComponent() {
    const [messages, setMessages] = useState([]);
    const [input, setInput] = useState('');
    const [isLoading, setIsLoading] = useState(false);
    const messagesEndRef = useRef(null);
    
    const sendMessage = async () => {
        if (!input.trim() || isLoading) return;
        
        const userMessage = { role: 'user', content: input };
        setMessages(prev => [...prev, userMessage]);
        setInput('');
        setIsLoading(true);
        
        // 创建 AI 消息占位
        setMessages(prev => [...prev, { role: 'assistant', content: '' }]);
        
        try {
            const response = await fetch('/v1/chat/stream', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    thread_id: 'session-1',
                    message: input,
                }),
            });
            
            const reader = response.body.getReader();
            const decoder = new TextDecoder();
            let buffer = '';
            let aiContent = '';
            
            while (true) {
                const { value, done } = await reader.read();
                if (done) break;
                
                buffer += decoder.decode(value, { stream: true });
                const lines = buffer.split('\n');
                buffer = lines.pop();
                
                for (const line of lines) {
                    if (line.startsWith('data: ') && line !== 'data: [DONE]') {
                        try {
                            const data = JSON.parse(line.slice(6));
                            if (data.choices?.[0]?.delta?.content) {
                                aiContent += data.choices[0].delta.content;
                                setMessages(prev => [
                                    ...prev.slice(0, -1),
                                    { role: 'assistant', content: aiContent },
                                ]);
                            }
                        } catch (e) {
                            // 忽略
                        }
                    }
                }
            }
        } catch (error) {
            console.error('Stream error:', error);
        } finally {
            setIsLoading(false);
        }
    };
    
    useEffect(() => {
        messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });
    }, [messages]);
    
    return (
        <div>
            <div className="messages">
                {messages.map((msg, i) => (
                    <div key={i} className={msg.role}>
                        {msg.content}
                    </div>
                ))}
                <div ref={messagesEndRef} />
            </div>
            <input
                value={input}
                onChange={(e) => setInput(e.target.value)}
                onKeyPress={(e) => e.key === 'Enter' && sendMessage()}
                placeholder="输入消息..."
                disabled={isLoading}
            />
            <button onClick={sendMessage} disabled={isLoading}>
                {isLoading ? '发送中...' : '发送'}
            </button>
        </div>
    );
}
```

## 部署配置

### Nginx 配置

```nginx
server {
    listen 80;
    
    location /v1/chat/stream {
        proxy_pass http://localhost:8000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        
        # SSE 配置
        proxy_buffering off;
        proxy_cache off;
        proxy_read_timeout 300s;
    }
}
```

### Gunicorn 配置

```bash
gunicorn app.main:app \
    --workers 4 \
    --worker-class uvicorn.workers.UvicornWorker \
    --bind 0.0.0.0:8000 \
    --timeout 300
```

## 小结

| 要点 | 说明 |
|------|------|
| SSE | 单向流式，自动重连，简单 |
| WebSocket | 双向流式，需手动重连 |
| FastAPI | StreamingResponse + text/event-stream |
| 前端 | EventSource 或 fetch + ReadableStream |
| OpenAI 兼容 | 标准 chunk 格式 |
| 部署 | Nginx 关闭缓冲，Gunicorn UvicornWorker |
| 调试 | 记录所有事件 |

## Evaluation（评估）

### 定义评估器

```python
from langsmith.schemas import Run, Example

def correctness_evaluator(run: Run, example: Example) -> dict:
    """正确性评估"""
    output = run.outputs.get("output", "")
    expected = example.outputs.get("expected", "")
    
    # 简单字符串匹配
    score = 1.0 if output.strip() == expected.strip() else 0.0
    
    return {
        "key": "correctness",
        "score": score,
    }

def relevance_evaluator(run: Run, example: Example) -> dict:
    """相关性评估"""
    output = run.outputs.get("output", "")
    input_text = example.inputs.get("input", "")
    
    # 检查输出是否包含输入中的关键词
    keywords = input_text.split()
    matched = sum(1 for kw in keywords if kw in output)
    score = matched / len(keywords) if keywords else 0
    
    return {
        "key": "relevance",
        "score": score,
    }
```

### 运行评估

```python
from langsmith import Client

client = Client()

# 定义测试数据集
dataset_name = "agent-test-dataset"

# 运行评估
results = client.evaluate(
    lambda inputs: agent.invoke({
        "messages": [{"role": "user", "content": inputs["input"]}]
    }),
    data=dataset_name,
    evaluators=[correctness_evaluator, relevance_evaluator],
)

# 查看结果
for result in results:
    print(f"Input: {result.run.inputs['input']}")
    print(f"Output: {result.run.outputs}")
    print(f"Scores: {result.evaluation_results}")
```

## Debugging（调试）

### 调试 Agent 行为

```python
from langsmith import Client

client = Client()

# 获取特定追踪
run = client.read_run("run-id")

print(f"输入: {run.inputs}")
print(f"输出: {run.outputs}")
print(f"状态: {run.status}")
print(f"错误: {run.error}")

# 获取子追踪
for child in run.child_runs:
    print(f"子追踪: {child.name}")
    print(f"  输入: {child.inputs}")
    print(f"  输出: {child.outputs}")
```

### 比较追踪

```python
# 比较两次运行
```

## 自定义监控

### 指标采集

```python
from prometheus_client import Counter, Histogram

# 定义指标
AGENT_CALLS = Counter('agent_calls_total', 'Total agent calls')
AGENT_ERRORS = Counter('agent_errors_total', 'Total agent errors')
AGENT_LATENCY = Histogram('agent_latency_seconds', 'Agent latency')
TOKEN_USAGE = Counter('agent_tokens_total', 'Total tokens', ['type'])

def monitor_agent_call(func):
    """监控 Agent 调用"""
    def wrapper(*args, **kwargs):
        start_time = time.time()
        AGENT_CALLS.inc()
        
        try:
            result = func(*args, **kwargs)
            
            duration = time.time() - start_time
            AGENT_LATENCY.observe(duration)
            
            # Token 使用
            if hasattr(result, 'response_metadata'):
                usage = result.response_metadata.get('token_usage', {})
                TOKEN_USAGE.labels(type='prompt').inc(usage.get('prompt_tokens', 0))
                TOKEN_USAGE.labels(type='completion').inc(usage.get('completion_tokens', 0))
            
            return result
        except Exception as e:
            AGENT_ERRORS.inc()
            raise
    
    return wrapper
```

### 自定义 Dashboard

```python
from fastapi import FastAPI
from prometheus_client import generate_latest, CONTENT_TYPE_LATEST

app = FastAPI()

@app.get("/metrics")
async def metrics():
    """Prometheus 指标端点"""
    return Response(generate_latest(), media_type=CONTENT_TYPE_LATEST)

@app.get("/dashboard")
async def dashboard():
    """简单 Dashboard"""
    return {
        "total_calls": AGENT_CALLS._value.get(),
        "total_errors": AGENT_ERRORS._value.get(),
        "avg_latency": AGENT_LATENCY._sum.get() / max(AGENT_LATENCY._count.get(), 1),
        "total_tokens": TOKEN_USAGE._value.get(),
    }
```

```
┌──────────────────────────────────────┐

│  ├─ 任务状态                         │
│  ├─ 临时变量                         │
│  └─ Checkpointer 持久化              │
│                                       │
│  Store（长期）                        │
│  ├─ 用户画像                         │
│  ├─ 偏好设置                         │
│  ├─ 交互历史                         │

│  ├─ 用户 ID                          │
```

```python
def update_topic(state: ConversationState):
    """更新话题"""
    # 分析当前话题
    last_msg = state["messages"][-1]
    
    # 简单话题检测
    if "Python" in last_msg.content:
        topic = "Python"
    elif "JavaScript" in last_msg.content:
        topic = "JavaScript"
    else:
        topic = state.get("topic", "general")
    
    return {"topic": topic}
```

## Store 实战模式

### 用户画像

```python
from langgraph.store.memory import InMemoryStore

class UserProfile:
    """用户画像管理"""
    
    def __init__(self, store: InMemoryStore):
        self.store = store
    
    def get_profile(self, user_id: str) -> dict:
        """获取用户画像"""
        result = self.store.get(("user", user_id), "profile")
        return result.value if result else {}
    
    def update_profile(self, user_id: str, updates: dict):
        """更新用户画像"""
        current = self.get_profile(user_id)
        merged = {**current, **updates}
        self.store.put(("user", user_id), "profile", merged)
    
    def add_preference(self, user_id: str, category: str, value: str):
        """添加偏好"""
        prefs = self.store.get(("user", user_id), "preferences")
        
        if prefs:
            current_prefs = prefs.value
            current_prefs.setdefault(category, []).append(value)
        else:
            current_prefs = {category: [value]}
        
        self.store.put(("user", user_id), "preferences", current_prefs)
    
    def get_preferences(self, user_id: str) -> dict:
        """获取偏好"""
        result = self.store.get(("user", user_id), "preferences")
        return result.value if result else {}
```

### 交互历史

```python
class InteractionHistory:
    """交互历史管理"""
    
    def __init__(self, store: InMemoryStore):
        self.store = store
    
    def add_interaction(self, user_id: str, interaction: dict):
        """添加交互记录"""
        import uuid
        
        history_namespace = ("user", user_id, "history")
        interaction_id = str(uuid.uuid4())
        
        self.store.put(history_namespace, interaction_id, {
            "timestamp": interaction.get("timestamp"),
            "input": interaction.get("input"),
            "output": interaction.get("output"),
            "metadata": interaction.get("metadata", {}),
        })
    
    def get_recent(self, user_id: str, limit: int = 10) -> list[dict]:
        """获取最近交互"""
        history_namespace = ("user", user_id, "history")
        results = self.store.search(history_namespace, limit=limit)
        
        return [r.value for r in results]
    
    def search_by_topic(self, user_id: str, topic: str) -> list[dict]:
        """按话题搜索"""
        history_namespace = ("user", user_id, "history")
        results = self.store.search(
            history_namespace,
            query=topic,
            limit=10,
        )
        
        return [r.value for r in results]
```

### 使用 Store 中间件

```python
from langchain.agents.middleware import before_agent

@before_agent
def load_user_context(state, runtime):
    """加载用户上下文"""
    user_id = runtime.context.user_id
    
    # 加载画像
    profile = runtime.store.get(("user", user_id), "profile")
    
    # 加载偏好
    prefs = runtime.store.get(("user", user_id), "preferences")
    
    # 加载最近交互
    history = runtime.store.search(
        ("user", user_id, "history"),
        limit=5,
    )
    
    # 构建上下文
    context_parts = []
    
    if profile:
        context_parts.append(f"用户信息: {profile.value}")
    
    if prefs:
        context_parts.append(f"用户偏好: {prefs.value}")
    
    if history:
        context_parts.append("最近交互:")
        for h in history:
            context_parts.append(f"- {h.value.get('input', '')[:50]}")
    
    if context_parts:
        return {"user_context": "\n".join(context_parts)}
    
    return None

@after_agent
def save_interaction(state, runtime):
    """保存交互"""
    user_id = runtime.context.user_id
    
    last_user_msg = next(
        (m for m in reversed(state["messages"]) if m.type == "human"),
        None
    )
    last_ai_msg = next(
        (m for m in reversed(state["messages"]) if m.type == "ai"),
        None
    )
    
    if last_user_msg and last_ai_msg:
        import uuid
        runtime.store.put(
            ("user", user_id, "history"),
            str(uuid.uuid4()),
            {
                "input": last_user_msg.content,
                "output": last_ai_msg.content[:500],
                "timestamp": "2024-01-01T00:00:00",
            }
        )
```

## 综合模式

### 完整上下文管理

```python
from langchain.agents.middleware import AgentMiddleware, ModelRequest, ModelResponse
from langgraph.store.memory import InMemoryStore

class ContextManagerMiddleware(AgentMiddleware):
    """上下文管理中间件"""
    
    def __init__(self, store: InMemoryStore):
        self.store = store
    
    def before_agent(self, state, runtime):
        """加载所有上下文"""
        user_id = runtime.context.user_id
        
        # 1. State 上下文
        message_count = len(state.get("messages", []))
        
        # 2. Store 上下文
        profile = self.store.get(("user", user_id), "profile")
        prefs = self.store.get(("user", user_id), "preferences")
        
        # 3. Context 上下文
        user_role = runtime.context.user_role
        
        # 构建完整上下文
        full_context = {
            "message_count": message_count,
            "user_profile": profile.value if profile else {},
            "user_preferences": prefs.value if prefs else {},
            "user_role": user_role,
        }
        
        return {"full_context": full_context}
    
    def after_agent(self, state, runtime):
        """保存上下文"""
        user_id = runtime.context.user_id
        
        # 保存交互
        last_user_msg = next(
            (m for m in reversed(state["messages"]) if m.type == "human"),
            None
        )
        last_ai_msg = next(
            (m for m in reversed(state["messages"]) if m.type == "ai"),
            None
        )
        
        if last_user_msg and last_ai_msg:
            import uuid
            self.store.put(
                ("user", user_id, "history"),
                str(uuid.uuid4()),
                {
                    "input": last_user_msg.content,
                    "output": last_ai_msg.content[:500],
                }
            )
        
        return None
```

## 最佳实践

### 1. 上下文分层

```python
# 第一层：系统级（不变）
SYSTEM_CONTEXT = "你是一个专业的 AI 助手。"

# 第二层：用户级（持久）
def load_user_context(user_id, store):
    profile = store.get(("user", user_id), "profile")
    return profile.value if profile else {}

# 第三层：会话级（短期）
def load_session_context(thread_id, checkpointer):
    state = checkpointer.get(thread_id)
    return state.values if state else {}

# 第四层：消息级（当前）
def load_message_context(messages):
    return messages[-1].content if messages else ""
```

### 2. 上下文压缩

```python
def compress_context(context: dict, max_length: int = 1000) -> str:
    """压缩上下文"""
    parts = []
    total_length = 0
    
    for key, value in context.items():
        part = f"{key}: {value}"
        if total_length + len(part) > max_length:
            break
        parts.append(part)
        total_length += len(part)
    
    return "\n".join(parts)
```

### 3. 上下文版本控制

```python
def migrate_context(context: dict, version: str) -> dict:
    """上下文版本迁移"""
    if version == "v1":
        # V1 → V2
        context.setdefault("message_count", 0)
        context.setdefault("topic", "general")
        context["_version"] = "v2"
    
    return context
```

## 小结

| 要点 | 说明 |
|------|------|
| State | 短期状态，对话消息、任务状态 |
| Store | 长期记忆，用户画像、偏好、历史 |
| Context | 运行时信息，用户 ID、角色、配置 |
| 加载 | before_agent 中加载所有上下文 |
| 保存 | after_agent 中保存交互 |
| 权限 | Context 控制工具访问权限 |
| 调试 | 打印完整上下文 |
| 最佳实践 | 分层、压缩、版本控制 |

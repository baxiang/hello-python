```
生产中间件架构：

│    ↓                                 │
│  1. RateLimitMiddleware              │

```

## 日志中间件

### 结构化日志

```python
from langchain.agents.middleware import AgentMiddleware, ModelRequest, ModelResponse, ToolCallRequest
import json
import time
from datetime import datetime

class StructuredLoggingMiddleware(AgentMiddleware):
    """结构化日志中间件"""
    
    def __init__(self, log_file: str = "agent.log"):
        self.log_file = log_file
    
    def _log(self, level: str, event: str, data: dict):
        """写入结构化日志"""
        log_entry = {
            "timestamp": datetime.now().isoformat(),
            "level": level,
            "event": event,
            **data,
        }
        
        with open(self.log_file, "a") as f:
            f.write(json.dumps(log_entry, ensure_ascii=False) + "\n")
    
    def before_agent(self, state, runtime):
        self._log("INFO", "before_agent", {
            "message_count": len(state.get("messages", [])),
            "user_id": runtime.context.get("user_id", "unknown"),
        })
        return None
    
    def after_agent(self, state, runtime):
        last_msg = state["messages"][-1] if state.get("messages") else None
        self._log("INFO", "after_agent", {
            "response_length": len(last_msg.content) if last_msg else 0,
        })
        return None
    
    def wrap_model_call(self, request: ModelRequest, handler) -> ModelResponse:
        start_time = time.time()
        
        self._log("INFO", "model_call_start", {
            "model": str(request.model),
            "tools_count": len(request.tools),
        })
        
        try:
            response = handler(request)
            
            duration = time.time() - start_time
            self._log("INFO", "model_call_end", {
                "duration_ms": round(duration * 1000, 2),
                "response_length": len(response.content),
            })
            
            return response
        except Exception as e:
            duration = time.time() - start_time
            self._log("ERROR", "model_call_error", {
                "duration_ms": round(duration * 1000, 2),
                "error": str(e),
            })
            raise
    
    def wrap_tool_call(self, request: ToolCallRequest, handler):
        start_time = time.time()
        tool_name = request.tool_call["name"]
        
        self._log("INFO", "tool_call_start", {
            "tool": tool_name,
            "args": request.tool_call["args"],
        })
        
        try:
            result = handler(request)
            
            duration = time.time() - start_time
            self._log("INFO", "tool_call_end", {
                "tool": tool_name,
                "duration_ms": round(duration * 1000, 2),
                "result_length": len(str(result)),
            })
            
            return result
        except Exception as e:
            duration = time.time() - start_time
            self._log("ERROR", "tool_call_error", {
                "tool": tool_name,
                "duration_ms": round(duration * 1000, 2),
                "error": str(e),
            })
            raise
```

## 限流中间件

### Token Bucket 限流

```python
from langchain.agents.middleware import before_agent
from langchain.messages import AIMessage
import time
from collections import defaultdict

class TokenBucket:
    """Token Bucket 限流器"""
    
    def __init__(self, rate: float, capacity: float):
        self.rate = rate           # 每秒生成 token 数
        self.capacity = capacity   # 桶容量
        self.tokens = capacity     # 当前 token 数
        self.last_refill = time.time()
    
    def consume(self, tokens: float = 1) -> bool:
        """消费 token"""
        now = time.time()
        
        # 补充 token
        elapsed = now - self.last_refill
        self.tokens = min(self.capacity, self.tokens + elapsed * self.rate)
        self.last_refill = now
        
        # 消费
        if self.tokens >= tokens:
            self.tokens -= tokens
            return True
        
        return False

# 用户限流器
user_limiters = defaultdict(lambda: TokenBucket(rate=1, capacity=5))

@before_agent
def rate_limit_middleware(state, runtime):
    """限流中间件"""
    user_id = runtime.context.get("user_id", "anonymous")
    limiter = user_limiters[user_id]
    
    if not limiter.consume():
        return {
            "messages": [
                AIMessage("请求过于频繁，请稍后再试。")
            ]
        }
    
    return None
```

### 全局限流

```python
from langchain.agents.middleware import wrap_model_call, ModelRequest, ModelResponse
import time

class GlobalRateLimiter:
    """全局限流器"""
    
    def __init__(self, max_requests: int, window: float = 60):
        self.max_requests = max_requests
        self.window = window
        self.requests = []
    
    def is_allowed(self) -> bool:
        now = time.time()
        
        # 清理过期请求
        self.requests = [t for t in self.requests if now - t < self.window]
        
        if len(self.requests) >= self.max_requests:
            return False
        
        self.requests.append(now)
        return True

global_limiter = GlobalRateLimiter(max_requests=100, window=60)

@wrap_model_call
def global_rate_limit(request: ModelRequest, handler) -> ModelResponse:
    """全局限流"""
    if not global_limiter.is_allowed():
        raise Exception("全局请求超限，请稍后再试。")
    
    return handler(request)
```

## PII 检测中间件

### 输入 PII 检测

```python
from langchain.agents.middleware import before_agent
from langchain.messages import AIMessage
import re

# PII 模式
PII_PATTERNS = {
    "credit_card": r'\b\d{4}[\s-]?\d{4}[\s-]?\d{4}[\s-]?\d{4}\b',
    "id_card": r'\b\d{17}[\dXx]\b',
    "phone": r'\b1[3-9]\d{9}\b',
    "email": r'\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b',
}

def detect_pii(text: str) -> list[str]:
    """检测 PII"""
    detected = []
    
    for name, pattern in PII_PATTERNS.items():
        if re.search(pattern, text):
            detected.append(name)
    
    return detected

def redact_pii(text: str) -> str:
    """脱敏 PII"""
    result = text
    
    for pattern in PII_PATTERNS.values():
        result = re.sub(pattern, "[REDACTED]", result)
    
    return result

@before_agent
def pii_detection(state, runtime):
    """PII 检测中间件"""
    last_msg = state["messages"][-1]
    content = last_msg.content
    
    detected = detect_pii(content)
    
    if detected:
        # 记录审计日志
        print(f"PII 检测: {detected}")
        
        # 脱敏
        redacted = redact_pii(content)
        
        # 警告用户
        return {
            "messages": [
                AIMessage(f"检测到敏感信息 ({', '.join(detected)})，已自动脱敏。请重新输入。")
            ],
            "redacted_input": redacted,
        }
    
    return None
```

### 输出 PII 过滤

```python
from langchain.agents.middleware import after_agent

@after_agent
def output_pii_filter(state, runtime):
    """输出 PII 过滤"""
    last_msg = state["messages"][-1]
    
    if last_msg.type == "ai":
        detected = detect_pii(last_msg.content)
        
        if detected:
            # 脱敏输出
            last_msg.content = redact_pii(last_msg.content)
            print(f"输出 PII 过滤: {detected}")
    
    return None
```

## 审计中间件

### 完整审计日志

```python
from langchain.agents.middleware import AgentMiddleware, ModelRequest, ModelResponse, ToolCallRequest
import uuid
import time
from datetime import datetime

class AuditMiddleware(AgentMiddleware):
    """审计中间件"""
    
    def __init__(self, audit_store=None):
        self.audit_store = audit_store  # 可以是数据库或文件
    
    def _create_audit_record(self, event_type: str, data: dict):
        """创建审计记录"""
        return {
            "id": str(uuid.uuid4()),
            "timestamp": datetime.now().isoformat(),
            "event_type": event_type,
            **data,
        }
    
    def _save_audit(self, record: dict):
        """保存审计记录"""
        if self.audit_store:
            self.audit_store.save(record)
        else:
            print(f"[AUDIT] {json.dumps(record, ensure_ascii=False)}")
    
    def before_agent(self, state, runtime):
        record = self._create_audit_record("before_agent", {
            "user_id": runtime.context.get("user_id"),
            "message_count": len(state.get("messages", [])),
        })
        self._save_audit(record)
        return None
    
    def after_agent(self, state, runtime):
        record = self._create_audit_record("after_agent", {
            "user_id": runtime.context.get("user_id"),
            "response_truncated": state["messages"][-1].content[:100],
        })
        self._save_audit(record)
        return None
    
    def wrap_model_call(self, request: ModelRequest, handler) -> ModelResponse:
        start_time = time.time()
        
        record = self._create_audit_record("model_call", {
            "model": str(request.model),
            "tools": [t.name for t in request.tools],
        })
        
        try:
            response = handler(request)
            record["status"] = "success"
            record["duration_ms"] = round((time.time() - start_time) * 1000, 2)
            self._save_audit(record)
            return response
        except Exception as e:
            record["status"] = "error"
            record["error"] = str(e)
            record["duration_ms"] = round((time.time() - start_time) * 1000, 2)
            self._save_audit(record)
            raise
    
    def wrap_tool_call(self, request: ToolCallRequest, handler):
        start_time = time.time()
        
        record = self._create_audit_record("tool_call", {
            "tool": request.tool_call["name"],
            "args": request.tool_call["args"],
        })
        
        try:
            result = handler(request)
            record["status"] = "success"
            record["duration_ms"] = round((time.time() - start_time) * 1000, 2)
            self._save_audit(record)
            return result
        except Exception as e:
            record["status"] = "error"
            record["error"] = str(e)
            record["duration_ms"] = round((time.time() - start_time) * 1000, 2)
            self._save_audit(record)
            raise
```

## 缓存中间件

### 响应缓存

```python
from langchain.agents.middleware import wrap_model_call, ModelRequest, ModelResponse
import hashlib

class ResponseCache:
    """响应缓存"""
    
    def __init__(self, max_size: int = 1000):
        self.max_size = max_size
        self.cache = {}
    
    def _generate_key(self, messages: list) -> str:
        """生成缓存键"""
        content = "".join(m.content for m in messages)
        return hashlib.md5(content.encode()).hexdigest()
    
    def get(self, key: str):
        return self.cache.get(key)
    
    def put(self, key: str, value):
        if len(self.cache) >= self.max_size:
            # 删除最旧的
            oldest_key = next(iter(self.cache))
            del self.cache[oldest_key]
        
        self.cache[key] = value

response_cache = ResponseCache()

@wrap_model_call
def cache_middleware(request: ModelRequest, handler) -> ModelResponse:
    """缓存中间件"""
    key = response_cache._generate_key(request.state["messages"])
    
    # 检查缓存
    cached = response_cache.get(key)
    if cached:
        print(f"缓存命中: {key[:8]}")
        return cached
    
    # 调用模型
    response = handler(request)
    
    # 存入缓存
    response_cache.put(key, response)
    print(f"缓存存入: {key[:8]}")
    
    return response
```

## 小结

| 要点 | 说明 |
|------|------|
| 日志 | StructuredLoggingMiddleware 记录所有操作 |
| 限流 | TokenBucket / GlobalRateLimiter 控制频率 |
| PII | 检测和脱敏敏感信息（信用卡、身份证、手机、邮箱） |
| 审计 | AuditMiddleware 完整操作记录 |
| 缓存 | ResponseCache 避免重复调用 |
| 组合 | 按顺序排列中间件链 |

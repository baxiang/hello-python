## 项目1：对话机器人

### 功能

- 多轮对话记忆
- 流式输出
- 人工审核（HITL）
- 用户偏好记忆

### 完整代码

```python
# chatbot/app/main.py
from fastapi import FastAPI
from fastapi.responses import StreamingResponse
from langchain.agents import create_agent
from langgraph.checkpoint.postgres import PostgresSaver
from langgraph.store.postgres import PostgresStore
from langchain.agents.middleware import before_agent, after_agent
from pydantic import BaseModel
from dataclasses import dataclass
import json

app = FastAPI(title="对话机器人")

# 配置
DATABASE_URL = "postgresql://user:pass@localhost:5432/chatbot"

@dataclass
class ChatContext:
    user_id: str

# 中间件
@before_agent
def load_user_prefs(state, runtime):
    """加载用户偏好"""
    user_id = runtime.context.user_id
    prefs = runtime.store.get(("user", user_id), "preferences")
    
    if prefs:
        return {"user_preferences": prefs.value}
    return None

@after_agent
def save_interaction(state, runtime):
    """保存交互"""
    user_id = runtime.context.user_id
    last_user = next((m for m in reversed(state["messages"]) if m.type == "human"), None)
    last_ai = next((m for m in reversed(state["messages"]) if m.type == "ai"), None)
    
    if last_user and last_ai:
        import uuid
        runtime.store.put(
            ("user", user_id, "history"),
            str(uuid.uuid4()),
            {"input": last_user.content, "output": last_ai.content[:500]},
        )

# 创建 Agent
checkpointer = PostgresSaver.from_conn_string(DATABASE_URL)
checkpointer.setup()

store = PostgresStore.from_conn_string(DATABASE_URL)
store.setup()

agent = create_agent(
    model="moonshot:moonshot-v1-8k",
    middleware=[load_user_prefs, save_interaction],
    checkpointer=checkpointer,
    store=store,
    context_schema=ChatContext,
)

class ChatRequest(BaseModel):
    user_id: str
    message: str

@app.post("/chat")
async def chat(request: ChatRequest):
    """对话"""
    config = {"configurable": {"thread_id": f"user-{request.user_id}"}}
    context = ChatContext(user_id=request.user_id)
    
    result = agent.invoke(
        {"messages": [{"role": "user", "content": request.message}]},
        config=config,
        context=context,
    )
    
    return {"response": result["messages"][-1].content}

@app.post("/chat/stream")
async def stream_chat(request: ChatRequest):
    """流式对话"""
    config = {"configurable": {"thread_id": f"user-{request.user_id}"}}
    context = ChatContext(user_id=request.user_id)
    inputs = {"messages": [{"role": "user", "content": request.message}]}
    
    async def generate():
        async for event in agent.astream_events(inputs, config=config, context=context, version="v2"):
            if event["event"] == "on_chat_model_stream":
                token = event["data"]["chunk"].content
                if token:
                    yield f"data: {json.dumps({'token': token})}\n\n"
        yield "data: [DONE]\n\n"
    
    return StreamingResponse(generate(), media_type="text/event-stream")
```

## 项目2：文档 QA

### 功能

- PDF/文本加载
- 向量检索
- RAG 上下文注入
- 引用来源

### 完整代码

```python
# doc_qa/app/main.py
from fastapi import FastAPI
from langchain.agents import create_agent
from langchain.agents.middleware import before_agent
from langchain_community.document_loaders import PyPDFLoader, TextLoader
from langchain_text_splitters import RecursiveCharacterSplitter
from langchain_community.vectorstores import FAISS
from langchain.embeddings import init_embeddings
from pydantic import BaseModel
from dataclasses import field

app = FastAPI(title="文档 QA")

# RAG 配置
@dataclass
class RAGContext:
    retriever: object = field(default=None)

# 加载文档
def load_documents(paths: list[str]):
    """加载和分割文档"""
    all_docs = []
    
    for path in paths:
        if path.endswith(".pdf"):
            loader = PyPDFLoader(path)
        else:
            loader = TextLoader(path)
        
        docs = loader.load()
        
        splitter = RecursiveCharacterSplitter(
            chunk_size=500,
            chunk_overlap=50,
        )
        
        all_docs.extend(splitter.split_documents(docs))
    
    print(f"加载了 {len(all_docs)} 个文档块")
    return all_docs

# 创建向量存储
def create_retriever(doc_paths: list[str]):
    """创建 Retriever"""
    docs = load_documents(doc_paths)
    
    embeddings = init_embeddings("zhipuai:embedding-2")
    vectorstore = FAISS.from_documents(docs, embeddings)
    
    return vectorstore.as_retriever(search_kwargs={"k": 3})

# RAG 中间件
@before_agent
def inject_rag_context(state, runtime):
    """注入 RAG 上下文"""
    last_msg = state["messages"][-1]
    
    docs = runtime.context.retriever.invoke(last_msg.content)
    
    if docs:
        context = "根据以下文档回答问题:\n" + "\n\n".join(
            f"[{i+1}] {doc.page_content}"
            for i, doc in enumerate(docs)
        )
        return {"context": context}
    
    return None

# 创建 Agent
retriever = create_retriever(["data/doc1.pdf", "data/doc2.txt"])

agent = create_agent(
    model="deepseek:deepseek-chat",
    middleware=[inject_rag_context],
    context_schema=RAGContext,
)

class QARequest(BaseModel):
    question: str

@app.post("/qa")
async def answer_question(request: QARequest):
    """文档问答"""
    context = RAGContext(retriever=retriever)
    
    result = agent.invoke(
        {"messages": [{"role": "user", "content": request.question}]},
        context=context,
    )
    
    return {"answer": result["messages"][-1].content}
```

## 项目3：Agent 助手

### 功能

- 多工具调用
- 动态工具选择
- 输入验证
- 输出格式化

### 完整代码

```python
# agent_assistant/app/main.py
from fastapi import FastAPI
from langchain.agents import create_agent
from langchain.agents.middleware import before_agent, wrap_model_call, ModelRequest, ModelResponse
from langchain.tools import tool
from langchain.messages import AIMessage
from pydantic import BaseModel
import re

app = FastAPI(title="Agent 助手")

# 工具
@tool
def search_web(query: str) -> str:
    """搜索网页"""
    return f"搜索结果: {query}"

@tool
def calculate(expression: str) -> str:
    """计算表达式"""
    try:
        return str(eval(expression, {"__builtins__": {}}, {}))
    except Exception as e:
        return f"计算错误: {e}"

@tool
def get_weather(city: str) -> str:
    """查询天气"""
    return f"{city}: 晴，25°C"

@tool
def send_email(to: str, subject: str, body: str) -> str:
    """发送邮件"""
    return f"邮件已发送给 {to}"

# 中间件
@before_agent
def validate_input(state, runtime):
    """输入验证"""
    content = state["messages"][-1].content
    
    if not content.strip():
        return {"messages": [AIMessage("请输入有效内容。")]}
    
    if len(content) > 2000:
        return {"messages": [AIMessage("输入过长。")]}
    
    return None

@wrap_model_call
def filter_tools(request: ModelRequest, handler) -> ModelResponse:
    """工具过滤"""
    query = request.state["messages"][-1].content
    
    # 根据查询选择工具
    needed = []
    if any(w in query for w in ["搜索", "查找"]):
        needed.append("search_web")
    if any(w in query for w in ["计算", "多少"]):
        needed.append("calculate")
    if any(w in query for w in ["天气"]):
        needed.append("get_weather")
    if any(w in query for w in ["邮件", "发送"]):
        needed.append("send_email")
    
    if needed:
        filtered = [t for t in request.tools if t.name in needed]
        return handler(request.override(tools=filtered))
    
    return handler(request)

# 创建 Agent
agent = create_agent(
    model="moonshot:moonshot-v1-8k",
    tools=[search_web, calculate, get_weather, send_email],
    middleware=[validate_input, filter_tools],
)

class AssistantRequest(BaseModel):
    query: str

@app.post("/assistant")
async def assistant(request: AssistantRequest):
    """Agent 助手"""
    result = agent.invoke({
        "messages": [{"role": "user", "content": request.query}]
    })
    
    return {"response": result["messages"][-1].content}
```

## 项目4：多 Agent 系统

### 功能

- 研究 Agent + 写作 Agent
- 监督者 Agent
- 子图架构
- 结果聚合

### 完整代码

```python
# multi_agent_system/app/main.py
from fastapi import FastAPI
from langchain.agents import create_agent
from langchain.tools import tool
from typing import TypedDict, Literal
from langgraph.graph import StateGraph, START, END
from langgraph.types import Command

app = FastAPI(title="多 Agent 系统")

# 工具
@tool
def search_research(query: str) -> str:
    """研究搜索"""
    return f"研究结果: {query}"

@tool
def verify_fact(fact: str) -> str:
    """事实验证"""
    return f"验证结果: {fact} - 可信"

# 研究 Agent
research_agent = create_agent(
    model="deepseek:deepseek-chat",
    tools=[search_research, verify_fact],
    name="research_agent",
)

# 写作 Agent
writer_agent = create_agent(
    model="moonshot:moonshot-v1-8k",
    tools=[],
    name="writer_agent",
)

# 状态
class MultiAgentState(TypedDict):
    task: str
    research_result: str
    final_output: str

# 监督者
def supervisor(state: MultiAgentState) -> Command[Literal["research", "write", END]]:
    """监督者"""
    if not state.get("research_result"):
        return Command(goto="research")
    elif not state.get("final_output"):
        return Command(goto="write")
    return Command(goto=END)

# 研究步骤
def research_step(state: MultiAgentState):
    """研究"""
    result = research_agent.invoke({
        "messages": [{"role": "user", "content": f"研究: {state['task']}"}]
    })
    return {"research_result": result["messages"][-1].content}

# 写作步骤
def write_step(state: MultiAgentState):
    """写作"""
    prompt = f"基于以下研究结果撰写报告:\n\n{state['research_result']}"
    
    result = writer_agent.invoke({
        "messages": [{"role": "user", "content": prompt}]
    })
    return {"final_output": result["messages"][-1].content}

# 构建图
builder = StateGraph(MultiAgentState)
builder.add_node("supervisor", supervisor)
builder.add_node("research", research_step)
builder.add_node("write", write_step)

builder.add_edge(START, "supervisor")
builder.add_edge("research", "supervisor")
builder.add_edge("write", "supervisor")
builder.add_edge("supervisor", END)

graph = builder.compile()

class TaskRequest(BaseModel):
    task: str

@app.post("/multi-agent")
async def run_multi_agent(request: TaskRequest):
    """多 Agent 执行"""
    result = graph.invoke({"task": request.task})
    
    return {
        "research": result.get("research_result"),
        "output": result.get("final_output"),
    }
```

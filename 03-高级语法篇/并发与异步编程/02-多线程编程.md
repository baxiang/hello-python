# 多线程编程

本章讲解 Python 多线程编程，包括线程创建、线程参数、线程池等内容。

---

## 创建线程

### 基本用法

```python
import threading
import time

def worker(name, delay):
    """工作函数"""
    for i in range(3):
        time.sleep(delay)
        print(f"{name} 执行第 {i+1} 次")

# 创建线程对象
t1 = threading.Thread(target=worker, args=("线程 1", 1))
t2 = threading.Thread(target=worker, args=("线程 2", 1.5))

# 启动线程
t1.start()
t2.start()

# 等待线程完成
t1.join()
t2.join()

print("所有线程完成")
```

### 线程参数

```python
import threading

def greet(name, greeting="Hello", times=1):
    for _ in range(times):
        print(f"{greeting}, {name}!")

# args - 位置参数（注意：单个参数也要加逗号）
t1 = threading.Thread(target=greet, args=("Alice",))

# kwargs - 关键字参数
t2 = threading.Thread(target=greet, kwargs={
    "name": "Bob",
    "greeting": "Hi",
    "times": 3
})

# 混合使用
t3 = threading.Thread(
    target=greet,
    args=("Carol",),           # 位置参数
    kwargs={"greeting": "Hey"}  # 关键字参数
)

t1.start()
t2.start()
t3.start()
```

## Thread 类详解

### 常用属性和方法

```python
import threading
import time

def worker():
    time.sleep(2)
    print("工作完成")

t = threading.Thread(target=worker)

# 属性
print(t.name)      # 线程名
print(t.daemon)    # 是否为守护线程
print(t.ident)     # 线程标识符（启动前为 None）

# 启动线程
t.start()

# 方法
print(t.is_alive())  # 是否存活
t.join(timeout=1)    # 等待线程结束（最多等待 1 秒）
```

### 守护线程

```python
import threading
import time

def background_task():
    while True:
        print("后台任务运行中...")
        time.sleep(1)

# 创建守护线程
t = threading.Thread(target=background_task)
t.daemon = True  # 设置为守护线程
t.start()

# 主线程结束时，守护线程会自动终止
time.sleep(3)
print("主线程结束")
# 守护线程随之终止
```

```
┌─────────────────────────────────────────────────────────────┐
│              守护线程 vs 普通线程                            │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│   普通线程：                                                 │
│   • 主线程会等待所有普通线程结束                             │
│   • 程序在所有普通线程结束后才退出                           │
│                                                             │
│   守护线程：                                                 │
│   • 主线程结束时自动终止                                     │
│   • 适合后台任务（日志、监控等）                             │
│   • 不能阻塞程序退出                                         │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

## 继承 Thread 类

### 自定义线程类

```python
import threading
import time

class WorkerThread(threading.Thread):
    """自定义线程类"""
    
    def __init__(self, name, count):
        super().__init__()
        self.name = name
        self.count = count
        self.result = None
    
    def run(self):
        """线程执行的方法"""
        for i in range(self.count):
            time.sleep(0.5)
            print(f"{self.name}: 执行 {i+1}/{self.count}")
        self.result = f"{self.name} 完成"

# 使用
threads = []
for i in range(3):
    t = WorkerThread(f"Worker-{i}", 3)
    threads.append(t)
    t.start()

for t in threads:
    t.join()
    print(f"结果: {t.result}")
```

## 线程池

### 使用 ThreadPoolExecutor

```python
from concurrent.futures import ThreadPoolExecutor
import time

def task(n):
    time.sleep(1)
    return n * n

# 创建线程池
with ThreadPoolExecutor(max_workers=3) as executor:
    # 提交单个任务
    future = executor.submit(task, 5)
    print(future.result())  # 25
    
    # 批量提交任务
    results = executor.map(task, range(1, 6))
    print(list(results))  # [1, 4, 9, 16, 25]
```

### Future 对象

```python
from concurrent.futures import ThreadPoolExecutor
import time

def task(n):
    time.sleep(1)
    return n * 2

with ThreadPoolExecutor(max_workers=3) as executor:
    # 提交多个任务
    futures = [executor.submit(task, i) for i in range(5)]
    
    # 获取结果
    for future in futures:
        print(future.result())
    
    # 或使用 as_completed 按完成顺序获取
    from concurrent.futures import as_completed
    futures = [executor.submit(task, i) for i in range(5)]
    for future in as_completed(futures):
        print(future.result())
```

### 回调函数

```python
from concurrent.futures import ThreadPoolExecutor

def task(n):
    return n * 2

def on_complete(future):
    """任务完成回调"""
    print(f"任务完成，结果: {future.result()}")

with ThreadPoolExecutor(max_workers=2) as executor:
    future = executor.submit(task, 10)
    future.add_done_callback(on_complete)
```

## threading 模块常用函数

```python
import threading

# 当前线程对象
current = threading.current_thread()
print(current.name)  # MainThread

# 当前活跃线程数
print(threading.active_count())

# 所有线程对象
for t in threading.enumerate():
    print(t.name)

# 主线程对象
main = threading.main_thread()
print(main.name)  # MainThread

# 线程本地存储
local_data = threading.local()
local_data.value = 10
```

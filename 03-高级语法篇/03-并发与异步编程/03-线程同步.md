# 线程同步

本章讲解多线程编程中的同步机制，包括 Lock、RLock、Semaphore、Event 等。

---

## 为什么需要同步

### 竞态条件

```python
import threading

# 共享变量
counter = 0

def increment():
    global counter
    for _ in range(100000):
        counter += 1  # 不是原子操作！

# 创建多个线程
threads = []
for _ in range(10):
    t = threading.Thread(target=increment)
    threads.append(t)
    t.start()

for t in threads:
    t.join()

print(f"预期: 1000000, 实际: {counter}")
# 实际结果可能小于 1000000，因为竞态条件
```

```
┌─────────────────────────────────────────────────────────────┐
│              竞态条件示意                                    │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│   counter += 1 的执行过程：                                 │
│   1. 读取 counter 的值                                      │
│   2. 加 1                                                   │
│   3. 写回 counter                                           │
│                                                             │
│   两个线程同时执行：                                         │
│   ┌─────────────────────────────────────────────────────┐   │
│   │ 时间 │ 线程 A              │ 线程 B              │   │   │
│   ├──────┼─────────────────────┼─────────────────────┤   │   │
│   │ T1   │ 读取 counter = 0    │                     │   │   │
│   │ T2   │                     │ 读取 counter = 0    │   │   │
│   │ T3   │ 计算 0 + 1 = 1      │                     │   │   │
│   │ T4   │                     │ 计算 0 + 1 = 1      │   │   │
│   │ T5   │ 写回 counter = 1    │                     │   │   │
│   │ T6   │                     │ 写回 counter = 1    │   │   │
│   └─────────────────────────────────────────────────────┘   │
│                                                             │
│   结果：两次加 1，但 counter 只增加了 1                      │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

## Lock 锁

### 基本用法

```python
import threading

counter = 0
lock = threading.Lock()

def increment():
    global counter
    for _ in range(100000):
        with lock:  # 获取锁，自动释放
            counter += 1

threads = []
for _ in range(10):
    t = threading.Thread(target=increment)
    threads.append(t)
    t.start()

for t in threads:
    t.join()

print(f"结果: {counter}")  # 正确：1000000
```

### 手动获取和释放

```python
import threading

lock = threading.Lock()

# 方式 1：with 语句（推荐）
with lock:
    # 临界区代码
    pass

# 方式 2：手动获取释放
lock.acquire()
try:
    # 临界区代码
    pass
finally:
    lock.release()

# 方式 3：非阻塞获取
if lock.acquire(blocking=False):
    try:
        # 临界区代码
        pass
    finally:
        lock.release()
else:
    print("获取锁失败")
```

## RLock 可重入锁

### 概念说明

RLock 允许同一个线程多次获取锁，但必须释放相同次数。

```python
import threading

rlock = threading.RLock()

def outer():
    with rlock:
        print("outer 获取锁")
        inner()

def inner():
    with rlock:  # 同一线程可以再次获取
        print("inner 获取锁")

outer()  # 正常工作

# 如果用 Lock 会死锁
lock = threading.Lock()
def outer_deadlock():
    with lock:
        inner_deadlock()  # 死锁！

def inner_deadlock():
    with lock:  # 等待自己释放锁 → 死锁
        pass
```

```
┌─────────────────────────────────────────────────────────────┐
│              Lock vs RLock                                   │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│   Lock：                                                     │
│   • 同一线程不能重复获取                                    │
│   • 重复获取会死锁                                          │
│   • 性能略高                                                │
│                                                             │
│   RLock：                                                    │
│   • 同一线程可重复获取                                      │
│   • 需要释放相同次数                                        │
│   • 适合递归调用                                            │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

## Semaphore 信号量

### 概念说明

信号量控制同时访问资源的线程数量。

```python
import threading
import time

# 最多允许 3 个线程同时访问
semaphore = threading.Semaphore(3)

def limited_resource(name):
    with semaphore:
        print(f"{name} 获取资源")
        time.sleep(1)
        print(f"{name} 释放资源")

threads = []
for i in range(6):
    t = threading.Thread(target=limited_resource, args=(f"Thread-{i}",))
    threads.append(t)
    t.start()

for t in threads:
    t.join()

# 输出：每次最多 3 个线程同时执行
```

### 应用场景

```python
import threading
import time

# 限制数据库连接数
db_semaphore = threading.Semaphore(5)

def query_database(query_id):
    with db_semaphore:
        print(f"查询 {query_id} 开始")
        time.sleep(0.5)  # 模拟查询
        print(f"查询 {query_id} 完成")

# 限制并发请求数
request_semaphore = threading.Semaphore(10)

def make_request(url):
    with request_semaphore:
        # 发送请求
        pass
```

## Event 事件

### 基本用法

Event 用于线程间通信，一个线程发出信号，其他线程等待。

```python
import threading
import time

event = threading.Event()

def waiter(name):
    print(f"{name} 等待事件...")
    event.wait()  # 阻塞，直到事件被设置
    print(f"{name} 收到事件，继续执行")

def setter():
    time.sleep(2)
    print("设置事件")
    event.set()  # 设置事件，唤醒所有等待的线程

# 创建线程
threads = [
    threading.Thread(target=waiter, args=("A",)),
    threading.Thread(target=waiter, args=("B",)),
    threading.Thread(target=setter)
]

for t in threads:
    t.start()

for t in threads:
    t.join()
```

### Event 方法

```python
import threading

event = threading.Event()

# 设置事件
event.set()

# 清除事件
event.clear()

# 等待事件（阻塞）
event.wait()

# 等待事件（带超时）
event.wait(timeout=5)  # 最多等待 5 秒

# 检查事件是否被设置
print(event.is_set())
```

## Condition 条件变量

### 基本用法

Condition 用于复杂的线程同步，结合了 Lock 和 Event 的功能。

```python
import threading
import time
import random

# 生产者-消费者模型
buffer = []
buffer_size = 5
condition = threading.Condition()

def producer():
    for i in range(10):
        with condition:
            while len(buffer) >= buffer_size:
                print("缓冲区满，生产者等待")
                condition.wait()  # 等待消费者消费
            
            item = f"商品-{i}"
            buffer.append(item)
            print(f"生产: {item}")
            
            condition.notify_all()  # 通知消费者
        time.sleep(random.random())

def consumer():
    for _ in range(10):
        with condition:
            while not buffer:
                print("缓冲区空，消费者等待")
                condition.wait()  # 等待生产者生产
            
            item = buffer.pop(0)
            print(f"消费: {item}")
            
            condition.notify_all()  # 通知生产者
        time.sleep(random.random())

# 启动线程
t1 = threading.Thread(target=producer)
t2 = threading.Thread(target=consumer)
t1.start()
t2.start()
t1.join()
t2.join()
```

## Barrier 屏障

### 基本用法

Barrier 让多个线程在某个点同步，等待所有线程到达后再继续。

```python
import threading
import time
import random

def worker(barrier, name):
    print(f"{name} 开始工作")
    time.sleep(random.random())
    print(f"{name} 到达屏障，等待其他线程")
    
    barrier.wait()  # 等待所有线程到达
    
    print(f"{name} 继续执行")

# 创建屏障，等待 3 个线程
barrier = threading.Barrier(3)

threads = []
for i in range(3):
    t = threading.Thread(target=worker, args=(barrier, f"Thread-{i}"))
    threads.append(t)
    t.start()

for t in threads:
    t.join()
```

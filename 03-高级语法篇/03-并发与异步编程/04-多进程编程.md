# 多进程编程

本章讲解 Python 多进程编程，包括进程创建、进程间通信、进程池等内容。

---

## 进程基础

### 创建进程

```python
import multiprocessing
import time

def worker(name, delay):
    """工作函数"""
    for i in range(3):
        time.sleep(delay)
        print(f"{name} 执行第 {i+1} 次, PID: {multiprocessing.current_process().pid}")

# 创建进程对象
p1 = multiprocessing.Process(target=worker, args=("进程 1", 1))
p2 = multiprocessing.Process(target=worker, args=("进程 2", 1.5))

# 启动进程
p1.start()
p2.start()

# 等待进程完成
p1.join()
p2.join()

print("所有进程完成")
```

### 守护进程

```python
import multiprocessing
import time

def background_task():
    while True:
        print("后台任务运行中...")
        time.sleep(1)

if __name__ == '__main__':
    # 创建守护进程
    p = multiprocessing.Process(target=background_task)
    p.daemon = True
    p.start()
    
    time.sleep(3)
    print("主进程结束")
    # 守护进程随之终止
```

## 进程间通信

### Queue 队列

```python
import multiprocessing

def producer(queue):
    for i in range(5):
        queue.put(f"消息-{i}")
        print(f"生产: 消息-{i}")

def consumer(queue):
    while True:
        item = queue.get()
        if item is None:  # 结束信号
            break
        print(f"消费: {item}")

if __name__ == '__main__':
    queue = multiprocessing.Queue()
    
    p1 = multiprocessing.Process(target=producer, args=(queue,))
    p2 = multiprocessing.Process(target=consumer, args=(queue,))
    
    p1.start()
    p2.start()
    
    p1.join()
    queue.put(None)  # 发送结束信号
    p2.join()
```

### Pipe 管道

```python
import multiprocessing

def sender(conn):
    conn.send("Hello from sender!")
    conn.close()

def receiver(conn):
    msg = conn.recv()
    print(f"收到: {msg}")
    conn.close()

if __name__ == '__main__':
    # 创建管道，返回两个连接对象
    parent_conn, child_conn = multiprocessing.Pipe()
    
    p1 = multiprocessing.Process(target=sender, args=(parent_conn,))
    p2 = multiprocessing.Process(target=receiver, args=(child_conn,))
    
    p1.start()
    p2.start()
    
    p1.join()
    p2.join()
```

### 共享内存

```python
import multiprocessing

def worker(value, array):
    value.value = 100
    for i in range(len(array)):
        array[i] = i * 2

if __name__ == '__main__':
    # 共享值
    value = multiprocessing.Value('i', 0)  # 整数
    
    # 共享数组
    array = multiprocessing.Array('i', [0] * 5)  # 整数数组
    
    p = multiprocessing.Process(target=worker, args=(value, array))
    p.start()
    p.join()
    
    print(f"Value: {value.value}")    # 100
    print(f"Array: {list(array)}")    # [0, 2, 4, 6, 8]
```

### Manager

```python
import multiprocessing

def worker(d, l):
    d['key'] = 'value'
    l.append(1)

if __name__ == '__main__':
    with multiprocessing.Manager() as manager:
        # 共享字典
        shared_dict = manager.dict()
        # 共享列表
        shared_list = manager.list()
        
        p = multiprocessing.Process(target=worker, args=(shared_dict, shared_list))
        p.start()
        p.join()
        
        print(f"Dict: {shared_dict}")  # {'key': 'value'}
        print(f"List: {shared_list}")  # [1]
```

## 进程池

### ProcessPoolExecutor

```python
from concurrent.futures import ProcessPoolExecutor
import time

def task(n):
    time.sleep(1)
    return n * n

if __name__ == '__main__':
    with ProcessPoolExecutor(max_workers=4) as executor:
        # 提交单个任务
        future = executor.submit(task, 5)
        print(future.result())  # 25
        
        # 批量提交
        results = executor.map(task, range(1, 6))
        print(list(results))  # [1, 4, 9, 16, 25]
```

### multiprocessing.Pool

```python
import multiprocessing
import time

def task(n):
    time.sleep(1)
    return n * n

if __name__ == '__main__':
    with multiprocessing.Pool(processes=4) as pool:
        # apply_async - 异步执行
        result = pool.apply_async(task, (5,))
        print(result.get())  # 25
        
        # map - 批量执行
        results = pool.map(task, range(1, 6))
        print(results)  # [1, 4, 9, 16, 25]
        
        # map_async - 异步批量执行
        result = pool.map_async(task, range(1, 6))
        print(result.get())  # [1, 4, 9, 16, 25]
        
        # starmap - 多参数
        def add(a, b):
            return a + b
        results = pool.starmap(add, [(1, 2), (3, 4), (5, 6)])
        print(results)  # [3, 7, 11]
```

## CPU 密集型任务示例

```python
import multiprocessing
import time

def cpu_bound_task(n):
    """CPU 密集型任务：计算 n 以内的素数个数"""
    count = 0
    for i in range(2, n):
        is_prime = True
        for j in range(2, int(i ** 0.5) + 1):
            if i % j == 0:
                is_prime = False
                break
        if is_prime:
            count += 1
    return count

if __name__ == '__main__':
    numbers = [100000, 100000, 100000, 100000]
    
    # 单进程
    start = time.time()
    results = [cpu_bound_task(n) for n in numbers]
    print(f"单进程: {time.time() - start:.2f}s")
    
    # 多进程
    start = time.time()
    with multiprocessing.Pool() as pool:
        results = pool.map(cpu_bound_task, numbers)
    print(f"多进程: {time.time() - start:.2f}s")
    
    # 多进程可以利用多核，速度提升明显
```

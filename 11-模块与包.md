## 第一部分：模块基础

### 7.1 什么是模块

#### 概念说明

**模块（Module）** 是一个包含 Python 代码的文件（`.py` 后缀），可以包含：
- 函数（functions）
- 类（classes）
- 变量（variables）
- 常量（constants）

```
┌─────────────────────────────────────────┐
│            模块的概念图                 │
├─────────────────────────────────────────┤
│                                         │
│   大型程序                              │
│  ┌─────────────────────────────────┐    │
│  │  main.py                        │    │
│  │                                 │    │
│  │  import math      ← 导入模块    │    │
│  │  import utils     ← 导入模块    │    │
│  │  from config import API_KEY     │    │
│  └─────────────────────────────────┘    │
│           │         │         │         │
│           ▼         ▼         ▼         │
│  ┌───────────┐ ┌─────────┐ ┌─────────┐ │
│  │ math.py   │ │utils.py │ │config.py│ │
│  │ (标准库)  │ │(自定义) │ │(自定义) │ │
│  └───────────┘ └─────────┘ └─────────┘ │
│                                         │
│  好处：代码复用、便于维护、分工合作     │
│                                         │
└─────────────────────────────────────────┘
```

#### 为什么使用模块


### 7.2 导入模块的四种方式

#### 方式一：导入整个模块

```python
# 语法：import 模块名

import math

# 使用时需要加上模块名前缀
print(math.sqrt(16))    # 4.0
print(math.pi)          # 3.141592653589793
print(math.ceil(4.2))   # 5（向上取整）
print(math.floor(4.8))  # 4（向下取整）
```

**图解：**
```
import math

┌─────────────────────────────────┐
│           math 模块              │
├─────────────────────────────────│
│  sqrt()  ← math.sqrt() 调用它   │
│  pi      ← math.pi 访问它       │
│  ceil()                         │
│  floor()                        │
│  sin(), cos(), tan()            │
│  log(), exp()                   │
│  ...                            │
└─────────────────────────────────┘
      ↑
      │
   需要加前缀
```

#### 方式二：导入特定内容

```python
# 语法：from 模块名 import 内容 1, 内容 2

from math import sqrt, pi

# 使用时不需要加模块名前缀
print(sqrt(16))    # 4.0
print(pi)          # 3.141592653589793

# ❌ 以下会报错，因为没有导入
# print(math.ceil(4.2))  # NameError!
```

**图解：**
```
from math import sqrt, pi

┌─────────────────────────────────┐
│           math 模块              │
├─────────────────────────────────│
│  sqrt()  ──────────────→ 直接导入│
│  pi      ──────────────→ 直接导入│
│  ceil()  (未导入，不可用)        │
│  floor() (未导入，不可用)        │
└─────────────────────────────────┘

直接使用：sqrt(16) ✓
```

#### 方式三：导入所有内容

```python
# 语法：from 模块名 import *

from math import *

# 所有公开内容都可以直接使用
print(sqrt(16))
print(pi)
print(ceil(4.2))

# ⚠️ 不推荐使用！原因：
# 1. 不知道有哪些名字被导入
# 2. 可能覆盖自己定义的名字
# 3. 代码可读性差
```

**警告：**
```
┌─────────────────────────────────────────┐
│          ⚠️ 不推荐使用 import *          │
├─────────────────────────────────────────┤
│                                         │
│  问题 1：名字冲突                         │
│  ─────────────────────                  │
│  # 你自己定义了 sum 函数                 │
│  def sum(a, b):                         │
│      return a + b                       │
│                                         │
│  # 从某模块导入所有内容                  │
│  from some_module import *              │
│                                         │
│  # ❌ 你的 sum 函数被覆盖了！             │
│                                         │
│  ─────────────────────────────────      │
│                                         │
│  问题 2：不知道有哪些名字                │
│  阅读代码时，不知道哪些名字可用          │
│                                         │
│  ─────────────────────────────────      │
│                                         │
│  ✅ 推荐做法：明确导入需要的内容         │
│  from math import sqrt, pi              │
│                                         │
└─────────────────────────────────────────┘
```

#### 方式四：使用别名

```python
# 给模块起别名
import math as m

print(m.sqrt(16))    # 4.0
print(m.pi)          # 3.141592653589793

# 给导入的内容起别名
from math import sqrt as square_root

print(square_root(16))  # 4.0

# 常见用例：导入第三方库
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
```

### 7.3 math 模块 - 数学运算

```python
import math

# ─────────────────────────────────────
# 基础运算
# ─────────────────────────────────────

# 平方根
print(math.sqrt(16))        # 4.0
print(math.sqrt(2))         # 1.4142135623730951

# 幂运算（x 的 y 次方）
print(math.pow(2, 3))       # 8.0（2³ = 8）
print(math.pow(10, 2))      # 100.0（10² = 100）

# 绝对值（fabs 返回浮点数）
print(math.fabs(-5.5))      # 5.5

# ─────────────────────────────────────
# 取整函数
# ─────────────────────────────────────

# 向上取整（天花板）
print(math.ceil(4.1))       # 5
print(math.ceil(4.9))       # 5

# 向下取整（地板）
print(math.floor(4.1))      # 4
print(math.floor(4.9))      # 4

# 截断取整（向 0 取整）
print(math.trunc(4.9))      # 4
print(math.trunc(-4.9))     # -4

# ─────────────────────────────────────
# 取整函数图解
# ─────────────────────────────────────
"""
数轴上的取整函数：

         floor(4.1) = 4
              │
         ┌────┴────┐
─────┴────┼────┴────┼────┴────┼────>
         4    4.1   5         6
              │
         ┌────┴────┐
              │
         ceil(4.1) = 5

         trunc(-4.9) = -4（向 0 取整）
              │
         ┌────┴────┐
─────┴────┼────┴────┼────┴────┼────>
        -5    -4.9  -4        0
              │
         ┌────┴────┐
              │
         floor(-4.9) = -5（向下取整）
"""

# ─────────────────────────────────────
# 三角函数（弧度制）
# ─────────────────────────────────────

print(math.sin(math.pi / 2))   # 1.0（sin(90°) = 1）
print(math.cos(0))             # 1.0（cos(0°) = 1）
print(math.tan(math.pi / 4))   # 1.0（tan(45°) = 1）

# 角度和弧度的转换
print(math.radians(180))       # 3.14159...（180° = π 弧度）
print(math.degrees(math.pi))   # 180.0（π 弧度 = 180°）

# ─────────────────────────────────────
# 对数函数
# ─────────────────────────────────────

# 自然对数（以 e 为底）
print(math.log(math.e))        # 1.0
print(math.log(10))            # 2.302585...

# 以 2 为底的对数
print(math.log2(8))            # 3.0（2³ = 8）

# 以 10 为底的对数
print(math.log10(100))         # 2.0（10² = 100）

# ─────────────────────────────────────
# 常量
# ─────────────────────────────────────

print(math.pi)                 # 3.141592653589793
print(math.e)                  # 2.718281828459045（自然常数）
print(math.tau)                # 6.283185...（2π）
print(math.inf)                # 无穷大
print(math.nan)                # 非数字
```

**常用数学公式实现：**
```python
import math

# 计算圆的面积
def circle_area(radius):
    return math.pi * radius ** 2

print(circle_area(5))  # 78.53981633974483

# 计算两点间距离
def distance(x1, y1, x2, y2):
    return math.sqrt((x2 - x1) ** 2 + (y2 - y1) ** 2)

print(distance(0, 0, 3, 4))  # 5.0（勾股定理）

# 计算三角形面积（海伦公式）
def triangle_area(a, b, c):
    s = (a + b + c) / 2  # 半周长
    return math.sqrt(s * (s - a) * (s - b) * (s - c))

print(triangle_area(3, 4, 5))  # 6.0
```

### 7.4 random 模块 - 随机数

```python
import random

# ─────────────────────────────────────
# 随机浮点数
# ─────────────────────────────────────

# 0.0 到 1.0 之间的随机浮点数（不包含 1.0）
print(random.random())         # 例如：0.847329...

# 指定范围的随机浮点数
print(random.uniform(1.5, 3.5))  # 例如：2.384729...

# ─────────────────────────────────────
# 随机整数
# ─────────────────────────────────────

# 指定范围的随机整数（包含两端）
print(random.randint(1, 6))    # 模拟掷骰子：1-6

# 指定步长的随机整数
print(random.randrange(0, 10, 2))  # 0, 2, 4, 6, 8 中的一个

# ─────────────────────────────────────
# 随机选择
# ─────────────────────────────────────

# 从序列中随机选择一个元素
colors = ['红', '黄', '蓝', '绿']
print(random.choice(colors))   # 随机选一个

# 从序列中随机选择多个元素（不重复）
print(random.sample(colors, 2))  # 随机选 2 个，不重复

# 从序列中随机选择多个元素（可重复）
print(random.choices(colors, k=3))  # 随机选 3 个，可重复

# ─────────────────────────────────────
# 随机打乱
# ─────────────────────────────────────

# 原地打乱列表（修改原列表）
cards = [1, 2, 3, 4, 5]
random.shuffle(cards)
print(cards)  # 例如：[3, 1, 5, 2, 4]

# ─────────────────────────────────────
# 设置随机种子（用于复现结果）
# ─────────────────────────────────────

random.seed(42)
print(random.random())  # 每次都得到相同结果

random.seed(42)
print(random.random())  # 与上面相同的结果

# ─────────────────────────────────────
# 实际应用示例
# ─────────────────────────────────────

# 生成随机验证码
def generate_code(length=6):
    chars = '0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ'
    return ''.join(random.choice(chars) for _ in range(length))

print(generate_code())  # 例如：aB3xY9

# 生成随机密码（包含特殊字符）
def generate_password(length=12):
    chars = '0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ!@#$%^&*'
    return ''.join(random.choice(chars) for _ in range(length))

print(generate_password())  # 例如：aB3$xY9@mN2!

# 抽奖程序
def lucky_draw(participants, winners=3):
    return random.sample(participants, winners)

names = ['张三', '李四', '王五', '赵六', '钱七']
print(lucky_draw(names))  # 随机选 3 个获奖者
```

### 7.5 datetime 模块 - 日期时间

```python
from datetime import datetime, date, time, timedelta

# ─────────────────────────────────────
# 获取当前时间
# ─────────────────────────────────────

# 当前日期和时间
now = datetime.now()
print(now)  # 2024-01-15 14:30:45.123456

# 当前日期
today = date.today()
print(today)  # 2024-01-15

# 当前时间
now_time = datetime.now().time()
print(now_time)  # 14:30:45.123456

# ─────────────────────────────────────
# 创建特定日期时间
# ─────────────────────────────────────

# 创建日期时间对象
birthday = datetime(2000, 1, 1, 10, 30, 0)
print(birthday)  # 2000-01-01 10:30:00

# 创建日期对象
new_year = date(2024, 1, 1)
print(new_year)  # 2024-01-01

# 创建时间对象
lunch_time = time(12, 0, 0)
print(lunch_time)  # 12:00:00

# ─────────────────────────────────────
# 日期格式化（datetime → 字符串）
# ─────────────────────────────────────

now = datetime.now()

# 常用格式
print(now.strftime("%Y-%m-%d"))           # 2024-01-15
print(now.strftime("%Y/%m/%d"))           # 2024/01/15
print(now.strftime("%Y 年%m 月%d 日"))     # 2024 年 01 月 15 日
print(now.strftime("%H:%M:%S"))           # 14:30:45
print(now.strftime("%Y-%m-%d %H:%M:%S"))  # 2024-01-15 14:30:45

# 其他格式
print(now.strftime("%A"))                 # Monday（星期几）
print(now.strftime("%B"))                 # January（月份）
print(now.strftime("%j"))                 # 015（一年中的第几天）

# 常用格式代码表
"""
%Y - 四位年份（2024）
%y - 两位年份（24）
%m - 月份（01-12）
%d - 日期（01-31）
%H - 小时（00-23）
%M - 分钟（00-59）
%S - 秒（00-59）
%A - 星期几全称（Monday）
%a - 星期几简称（Mon）
%B - 月份全称（January）
%b - 月份简称（Jan）
%j - 一年中的第几天（001-366）
%U - 一年中的第几周（00-53）
"""

# ─────────────────────────────────────
# 解析字符串（字符串 → datetime）
# ─────────────────────────────────────

date_str = "2024-01-15"
parsed = datetime.strptime(date_str, "%Y-%m-%d")
print(parsed)  # 2024-01-15 00:00:00

date_str = "15/01/2024"
parsed = datetime.strptime(date_str, "%d/%m/%Y")
print(parsed)  # 2024-01-15 00:00:00

# ─────────────────────────────────────
# 日期计算（timedelta）
# ─────────────────────────────────────

now = datetime.now()

# 明天
tomorrow = now + timedelta(days=1)
print(tomorrow.strftime("%Y-%m-%d"))

# 昨天
yesterday = now - timedelta(days=1)
print(yesterday.strftime("%Y-%m-%d"))

# 一周后
next_week = now + timedelta(weeks=1)
print(next_week.strftime("%Y-%m-%d"))

# 100 天后
hundred_days = now + timedelta(days=100)
print(hundred_days.strftime("%Y-%m-%d"))

# ─────────────────────────────────────
# 计算时间差
# ─────────────────────────────────────

start = datetime(2024, 1, 1)
end = datetime(2024, 12, 31)

diff = end - start
print(diff)           # 365 days, 0:00:00
print(diff.days)      # 365（天数）
print(diff.seconds)   # 0（秒数）

# 计算倒计时
target = datetime(2024, 12, 31, 23, 59, 59)
now = datetime.now()
countdown = target - now
print(f"距离目标还有 {countdown.days} 天")

# ─────────────────────────────────────
# 实际应用示例
# ─────────────────────────────────────

# 计算年龄
def calculate_age(birth_date):
    today = date.today()
    age = today.year - birth_date.year
    # 如果今年生日还没过，减 1 岁
    if (today.month, today.day) < (birth_date.month, birth_date.day):
        age -= 1
    return age

birth = date(2000, 5, 20)
print(calculate_age(birth))  # 例如：23

# 判断闰年
def is_leap_year(year):
    # 能被 4 整除但不能被 100 整除，或能被 400 整除
    return year % 4 == 0 and (year % 100 != 0 or year % 400 == 0)

print(is_leap_year(2024))  # True
print(is_leap_year(2100))  # False

# 工作日计算
def work_days(start_date, days):
    current = start_date
    work_day_count = 0
    result_days = 0

    while work_day_count < days:
        current += timedelta(days=1)
        # 周一到周五（weekday() 返回 0-4 是周一到周五）
        if current.weekday() < 5:
            work_day_count += 1
            result_days += 1

    return current

start = date(2024, 1, 1)
print(work_days(start, 10))  # 10 个工作日后的日期
```

### 7.6 os 模块 - 操作系统接口

```python
import os

# ─────────────────────────────────────
# 目录操作
# ─────────────────────────────────────

# 获取当前工作目录
cwd = os.getcwd()
print(cwd)  # /Users/username/project

# 切换工作目录
os.chdir("/path/to/directory")

# 获取当前目录的父目录

def batch_rename(folder, old_prefix, new_prefix):

        if filename.startswith(old_prefix):
            old_path = os.path.join(folder, filename)
            new_filename = filename.replace(old_prefix, new_prefix)
            new_path = os.path.join(folder, new_filename)
            os.rename(old_path, new_path)
            print(f"{filename} → {new_filename}")

# 清理临时文件
def clean_temp_files(folder, extension=".tmp"):
    for filename in os.listdir(folder):
        if filename.endswith(extension):
            file_path = os.path.join(folder, filename)
            os.remove(file_path)
            print(f"已删除：{filename}")
```

### 7.7 json 模块 - JSON 处理

```python
import json

# ─────────────────────────────────────
# JSON 简介
# ─────────────────────────────────────

"""
JSON (JavaScript Object Notation) 是一种轻量级的数据交换格式。

JSON 格式：
{
    "name": "Alice",
    "age": 25,
    "city": "Beijing",
    "hobbies": ["reading", "coding"],
    "married": true,
    "salary": null
}

Python 与 JSON 类型对应：
┌──────────────┬──────────────┐
│    Python    │     JSON     │
├──────────────┼──────────────┤
│    dict      │    object    │
│    list      │    array     │
│    str       │    string    │
│    int/float │    number    │
│    bool      │    boolean   │
│    None      │    null      │
└──────────────┴──────────────┘
"""

# ─────────────────────────────────────
# 序列化（Python → JSON）
# ─────────────────────────────────────

# Python 字典
data = {
    "name": "Alice",
    "age": 25,
    "city": "Beijing",
    "hobbies": ["reading", "coding"]
}

# 转为 JSON 字符串
json_str = json.dumps(data)
print(json_str)
# {"name": "Alice", "age": 25, "city": "Beijing", "hobbies": ["reading", "coding"]}

# 美化输出（缩进）
json_str_pretty = json.dumps(data, indent=2, ensure_ascii=False)
print(json_str_pretty)
"""
{
  "name": "Alice",
  "age": 25,
  "city": "Beijing",
  "hobbies": [
    "reading",
    "coding"
  ]
}
"""

# 排序键
json_str_sorted = json.dumps(data, sort_keys=True)
print(json_str_sorted)

# ─────────────────────────────────────
# 反序列化（JSON → Python）
# ─────────────────────────────────────

# JSON 字符串
json_str = '{"name": "Alice", "age": 25, "city": "Beijing"}'

# 转为 Python 字典
data = json.loads(json_str)
print(data)  # {'name': 'Alice', 'age': 25, 'city': 'Beijing'}
print(data["name"])  # Alice
print(type(data))    # <class 'dict'>

# ─────────────────────────────────────
# 读写 JSON 文件
# ─────────────────────────────────────

# 写入 JSON 文件
data = {
    "students": [
        {"name": "Alice", "age": 20},
        {"name": "Bob", "age": 21}
    ]
}

with open("students.json", "w", encoding="utf-8") as f:
    json.dump(data, f, indent=2, ensure_ascii=False)

# 读取 JSON 文件
with open("students.json", "r", encoding="utf-8") as f:
    data = json.load(f)

print(data)
print(data["students"][0]["name"])  # Alice

# ─────────────────────────────────────
# 实际应用示例
# ─────────────────────────────────────

# 保存程序配置
def save_config(config, filename="config.json"):
    with open(filename, "w", encoding="utf-8") as f:
        json.dump(config, f, indent=2, ensure_ascii=False)

def load_config(filename="config.json"):
    try:
        with open(filename, "r", encoding="utf-8") as f:
            return json.load(f)
    except FileNotFoundError:
        return {}  # 返回默认配置

# 使用示例
config = {
    "theme": "dark",
    "language": "zh-CN",
    "font_size": 14
}
save_config(config)
loaded = load_config()
print(loaded)

# 数据导出功能
def export_to_json(data, filename):
    with open(filename, "w", encoding="utf-8") as f:
        json.dump(data, f, indent=2, ensure_ascii=False)
    print(f"数据已导出到 {filename}")

# 数据导入功能
def import_from_json(filename):
    with open(filename, "r", encoding="utf-8") as f:
        return json.load(f)
```

```python
# 使用 random 和 string 模块生成安全密码

import random
import string

def generate_password(length=12, use_special=True):
    """生成随机密码"""
    # 基础字符
    chars = string.ascii_letters + string.digits

    # 添加特殊字符
    if use_special:
        chars += string.punctuation

    # 生成密码
    password = ''.join(random.choice(chars) for _ in range(length))
    return password

# 测试
```

### 练习 3：日期计算器

```python
# 使用 datetime 模块计算日期

from datetime import datetime, timedelta

def days_between(date1_str, date2_str, date_format="%Y-%m-%d"):
    """计算两个日期之间相差的天数"""
    date1 = datetime.strptime(date1_str, date_format)
    date2 = datetime.strptime(date2_str, date_format)
    return abs((date2 - date1).days)

def add_days(date_str, days, date_format="%Y-%m-%d"):
    """在给定日期上增加/减少天数"""
    date = datetime.strptime(date_str, date_format)
    new_date = date + timedelta(days=days)
    return new_date.strftime(date_format)

def is_weekend(date_str, date_format="%Y-%m-%d"):
    """判断是否是周末"""
    date = datetime.strptime(date_str, date_format)
    return date.weekday() >= 5  # 5=周六，6=周日

# 测试
print(days_between("2024-01-01", "2024-12-31"))  # 365
print(add_days("2024-01-01", 100))               # 2024-04-10
print(is_weekend("2024-01-13"))                  # True（周六）
```

```python
def reverse_string(s):
    """反转字符串"""
    return s[::-1]

def to_camel_case(s):
    """转驼峰命名"""
    parts = s.split("_")
    return parts[0] + "".join(p.capitalize() for p in parts[1:])
```

```python
# my_tools/number_tools.py
def is_prime(n):
    """判断是否是质数"""
    if n < 2:
        return False
    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False
    return True

def factorial(n):
    """计算阶乘"""
    result = 1
    for i in range(2, n + 1):
        result *= i
    return result
```

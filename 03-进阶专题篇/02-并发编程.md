### 14.1 并发与并行

#### 概念辨析

```
┌─────────────────────────────────────────┐
│        并发 vs 并行 概念图              │
├─────────────────────────────────────────┤
│                                         │
│  并发（Concurrency）                    │
│  ─────────────────                      │
│  多个任务在同一时间段内交替执行         │
│                                         │
│  类比：一个人照顾多个孩子               │
│  ────────────────────                  │
│  喂老大吃饭 → 陪老二玩 → 给老三洗澡    │
│  → 再回来喂老大 → ...                  │
│                                         │
│  宏观上同时，微观上交替                 │
│                                         │
│  ─────────────────────────────────      │
│                                         │
│  并行（Parallelism）                    │
│  ─────────────────                      │
│  多个任务在同一时刻同时执行             │
│                                         │
│  类比：多个保姆各照顾一个孩子           │
│  ──────────────────────                │
│  保姆 A 喂老大，保姆 B 陪老二，          │
│  保姆 C 给老三洗澡                      │
│                                         │
│  宏观和微观上都同时                     │
│                                         │
└─────────────────────────────────────────┘
```

#### 图解并发与并行

```
并发（单核 CPU）：
─────────────────────────────────> 时间
│ 任务 A │ 任务 B │ 任务 C │ 任务 A │
│       │       │       │       │
└───────┴───────┴───────┴───────┘
        交替执行，快速切换

并行（多核 CPU）：
核心 1: │ 任务 A │───────────────> 时间
核心 2: │ 任务 B │───────────────> 时间
核心 3: │ 任务 C │───────────────> 时间
        同时执行
```

### 14.3 创建线程

#### 基础用法

```python
import threading
import time

# ─────────────────────────────────────
# 方式 1：使用 Thread 类
# ─────────────────────────────────────

def worker(name, delay):
    """工作函数"""
    for i in range(3):
        time.sleep(delay)
        print(f"{name} 执行第 {i+1} 次")

# 创建线程对象
t1 = threading.Thread(target=worker, args=("线程 1", 1))
t2 = threading.Thread(target=worker, args=("线程 2", 1.5))

# 启动线程
t1.start()
t2.start()

# 等待线程完成
t1.join()
t2.join()

print("所有线程完成")
```

**线程生命周期图解：**
```
主线程
  │
  ├── 创建 Thread 对象
  │      │
  │      ▼
  │   线程（未启动）
  │
  ├── start()
  │      │
  │      ▼
  │   线程（运行中）← 操作系统调度
  │
  └── join()
         │
         ▼
      等待线程结束
         │
         ▼
      线程（已结束）
```

#### 线程参数详解

### 14.4 守护线程

```python
import threading
import time

def background_worker():
    """后台工作线程"""
    while True:
        print("后台任务运行中...")
        time.sleep(1)

# ─────────────────────────────────────
# 非守护线程（默认）
# ─────────────────────────────────────
t1 = threading.Thread(target=background_worker)
# t1.daemon = False  # 默认值
t1.start()

# 主线程结束时，会等待 t1 完成

# ─────────────────────────────────────
# 守护线程
# ─────────────────────────────────────
t2 = threading.Thread(target=background_worker, daemon=True)
# 或者
# t2.daemon = True
t2.start()

# 主线程结束时，t2 会被强制终止

# ─────────────────────────────────────
# 实际应用场景：定期保存进度
# ─────────────────────────────────────
def auto_save():
    while True:
        time.sleep(60)
        print("自动保存进度...")

# 设置为守护线程，主程序退出时自动停止
save_thread = threading.Thread(target=auto_save, daemon=True)
save_thread.start()
```

### 14.6 继承 Thread 类

```python
import threading
import time

class MyThread(threading.Thread):
    """自定义线程类"""

    def __init__(self, name, delay):
        # 调用父类构造函数
        super().__init__()
        self.name = name
        self.delay = delay
        self.result = None

    def run(self):
        """线程执行的代码（必须叫 run）"""
        print(f"{self.name} 开始执行")
        result = 0
        for i in range(5):
            time.sleep(self.delay)
            result += i
            print(f"{self.name} 第 {i+1} 次")
        self.result = result
        print(f"{self.name} 执行完成")

# 使用
t1 = MyThread("线程 A", 0.5)
t2 = MyThread("线程 B", 0.7)

t1.start()
t2.start()

t1.join()
t2.join()

# 获取结果（需要自己保存）
print(f"线程 A 结果：{t1.result}")
print(f"线程 B 结果：{t2.result}")
```

### 14.7 线程安全问题

#### 竞态条件

```python
import threading
import time

# ─────────────────────────────────────
# ❌ 线程不安全的代码
# ─────────────────────────────────────

counter = 0

def increment():
    global counter
    for _ in range(100000):
        # 这行代码实际包含三个操作：
        # 1. 读取 counter
        # 2. 加 1
        # 3. 写回 counter
        counter += 1

# 创建 5 个线程
threads = []
for _ in range(5):
    t = threading.Thread(target=increment)
    threads.append(t)
    t.start()

for t in threads:
    t.join()

# 期望结果：500000
# 实际结果：可能小于 500000（因为竞态条件）
print(f"最终计数：{counter}")
```

**竞态条件图解：**
```
线程 A                      线程 B
  │                          │
  │ 读取 counter = 10        │
  │                          │ 读取 counter = 10
  │ 计算 10 + 1 = 11        │
  │                          │ 计算 10 + 1 = 11
  │ 写回 counter = 11       │
  │                          │ 写回 counter = 11
  │                          │
  │ 期望：12                 │ 实际：11！
```

### 14.8 Lock - 互斥锁

#### 基本用法

```python
import threading

# ─────────────────────────────────────
# 创建锁
# ─────────────────────────────────────
lock = threading.Lock()

counter = 0

def increment():
    global counter
    for _ in range(100000):
        # ──────────────────────────────
        # 方式 1：手动加锁和解锁
        # ──────────────────────────────
        lock.acquire()
        try:
            counter += 1
        finally:
            lock.release()

        # ──────────────────────────────
        # 方式 2：使用 with（推荐）
        # ──────────────────────────────
        # with lock:
        #     counter += 1

# 测试
threads = []
for _ in range(5):
    t = threading.Thread(target=increment)
    threads.append(t)
    t.start()

for t in threads:
    t.join()

print(f"最终计数：{counter}")  # 500000（正确！）
```

**锁的工作原理：**
```
线程 A                      锁          线程 B
  │                        │             │
  │  acquire() ───────────►│             │
  │   获得锁               │             │  acquire()
  │   进入临界区           │             │  等待...
  │   counter += 1         │             │  等待...
  │  release() ───────────►│             │
  │                        │  释放锁     │
  │                        │◄────────────┤
  │                        │             │  获得锁
  │                        │             │  进入临界区
```

### 14.9 RLock - 可重入锁

```python
import threading

# ─────────────────────────────────────
# Lock vs RLock
# ─────────────────────────────────────

# Lock：同一线程重复获取会死锁
lock = threading.Lock()

def recursive_with_lock():
    lock.acquire()
    try:
        print("第一层")
        # lock.acquire()  # ❌ 会死锁！
    finally:
        lock.release()

# RLock：允许同一线程多次获取
rlock = threading.RLock()

def recursive_with_rlock(n):
    rlock.acquire()
    try:
        print(f"第 {n} 层")
        if n > 1:
            recursive_with_rlock(n - 1)  # ✅ 可以递归
    finally:
        rlock.release()

# 测试
recursive_with_rlock(3)
# 第 3 层
# 第 2 层
# 第 1 层

# ─────────────────────────────────────
# 实际应用场景：类中多个方法互相调用
# ─────────────────────────────────────

class BankAccount:
    def __init__(self):
        self.balance = 1000
        self.lock = threading.RLock()

    def deposit(self, amount):
        with self.lock:
            self.balance += amount
            self._log("存款", amount)

    def withdraw(self, amount):
        with self.lock:
            if self.balance >= amount:
                self.balance -= amount
                self._log("取款", amount)
                return True
            return False

    def _log(self, action, amount):
        # 内部方法也需要锁，但已被外层方法持有
        # 用 RLock 可以避免死锁
        with self.lock:
            print(f"{action}{amount}元，余额：{self.balance}")
```

### 14.10 Semaphore - 信号量

```python
import threading
import time

# ─────────────────────────────────────
# 信号量：限制同时访问的线程数
# ─────────────────────────────────────

# 创建信号量，最多 3 个线程同时访问
semaphore = threading.Semaphore(3)

def access_resource(thread_name):
    with semaphore:
        print(f"{thread_name} 开始访问资源")
        time.sleep(1)
        print(f"{thread_name} 结束访问")

# 创建 10 个线程
threads = []
for i in range(10):
    t = threading.Thread(target=access_resource, args=(f"线程-{i}",))
    threads.append(t)
    t.start()

for t in threads:
    t.join()

# 输出示例：
# 线程 -0 开始访问
# 线程 -1 开始访问
# 线程 -2 开始访问
# （最多 3 个同时访问）
# ... 等待的线程 ...
```

**信号量图解：**
```
信号量（3 个许可）
  │
  ├── 线程 1：获得许可 → 剩余 2
  ├── 线程 2：获得许可 → 剩余 1
  ├── 线程 3：获得许可 → 剩余 0
  ├── 线程 4：等待...
  ├── 线程 5：等待...
  │
  │ 线程 1 完成，释放许可 → 剩余 1
  │ 线程 4：获得许可 → 剩余 0
  │
  │ ...
```

### 14.12 Process 池

#### Pool 基础

```python
from multiprocessing import Pool
import time

def square(x):
    """计算平方"""
    return x * x

if __name__ == "__main__":
    # ─────────────────────────────────
    # 创建进程池（4 个工作进程）
    # ─────────────────────────────────
    with Pool(processes=4) as pool:
        # ──────────────────────────────
        # map - 顺序处理，返回结果列表
        # ──────────────────────────────
        results = pool.map(square, range(10))
        print(results)
        # [0, 1, 4, 9, 16, 25, 36, 49, 64, 81]

        # ──────────────────────────────
        # map_async - 异步处理
        # ──────────────────────────────
        async_result = pool.map_async(square, range(10))

        # 可以做其他事情...

        # 获取结果（会等待完成）
        results = async_result.get()
        print(results)

        # ──────────────────────────────
        # apply - 同步应用单个参数
        # ──────────────────────────────
        result = pool.apply(square, args=(5,))
        print(result)  # 25

        # ──────────────────────────────
        # apply_async - 异步应用
        # ──────────────────────────────
        async = pool.apply_async(square, args=(6,))
        result = async.get()
        print(result)  # 36
```

#### 实际应用场景

```python
from multiprocessing import Pool
import time

def process_image(filename):
    """处理单张图片（CPU 密集型）"""
    print(f"处理 {filename}")
    time.sleep(0.5)  # 模拟处理
    return f"{filename}_processed"

if __name__ == "__main__":
    images = [f"image_{i}.jpg" for i in range(20)]

    # 串行处理（慢）
    start = time.time()
    results = [process_image(img) for img in images]
    print(f"串行：{time.time() - start:.2f}秒")

    # 并行处理（快）
    start = time.time()
    with Pool(processes=4) as pool:
        results = pool.map(process_image, images)
    print(f"并行：{time.time() - start:.2f}秒")
    # 速度提升约 4 倍！
```

## 第五部分：进程间通信

### 14.13 Queue - 队列

```python
from multiprocessing import Process, Queue
import time

def producer(queue, name):
    """生产者：生产数据放入队列"""
    for i in range(5):
        item = f"{name} 生产的消息 {i}"
        queue.put(item)
        print(f"{name} 放入：{item}")
        time.sleep(0.1)
    queue.put(None)  # 结束标记

def consumer(queue, name):
    """消费者：从队列取出数据处理"""
    while True:
        item = queue.get()
        if item is None:
            queue.put(None)  # 传递给其他消费者
            break
        print(f"{name} 消费：{item}")
        time.sleep(0.2)

if __name__ == "__main__":
    # 创建队列
    q = Queue()

    # 创建生产者
    p1 = Process(target=producer, args=(q, "生产者 A"))
    p2 = Process(target=producer, args=(q, "生产者 B"))

    # 创建消费者
    c1 = Process(target=consumer, args=(q, "消费者 X"))
    c2 = Process(target=consumer, args=(q, "消费者 Y"))

    # 启动
    p1.start()
    p2.start()
    c1.start()
    c2.start()

    # 等待完成
    p1.join()
    p2.join()
    c1.join()
    c2.join()
```

### 14.14 Pipe - 管道

```python
from multiprocessing import Process, Pipe
import time

def sender(conn, name):
    """发送者"""
    for i in range(5):
        msg = f"{name} 发送消息 {i}"
        conn.send(msg)
        print(f"发送：{msg}")
        time.sleep(0.1)
    conn.close()

def receiver(conn):
    """接收者"""
    while True:
        try:
            msg = conn.recv()
            print(f"接收：{msg}")
        except EOFError:
            print("管道已关闭")
            break
    conn.close()

if __name__ == "__main__":
    # 创建管道（返回两端的连接对象）
    parent_conn, child_conn = Pipe()

    # 创建进程
    p1 = Process(target=sender, args=(parent_conn, "发送者"))
    p2 = Process(target=receiver, args=(child_conn,))

    # 启动
    p1.start()
    p2.start()

    # 等待
    p1.join()
    p2.join()
```

**Pipe vs Queue：**
```
┌─────────────────────────────────────────┐
│          Pipe vs Queue                  │
├─────────────────────────────────────────┤
│                                         │
│  Pipe（管道）                           │
│  • 用于两个进程间的通信                │
│  • 性能更高                             │
│  • 支持双向通信                         │
│                                         │
```

### 14.15 Manager - 共享对象

```python
from multiprocessing import Process, Manager

def worker(shared_list, shared_dict, index):
    """工作进程：修改共享对象"""
    shared_list.append(index)
    shared_dict[f"process_{index}"] = index * 10

if __name__ == "__main__":
    with Manager() as manager:
        # ──────────────────────────────
        # 创建共享列表
        # ──────────────────────────────
        shared_list = manager.list()

        # ──────────────────────────────
        # 创建共享字典
        # ──────────────────────────────
        shared_dict = manager.dict()

        # ──────────────────────────────
        # 创建共享命名空间
        # ──────────────────────────────
        shared_ns = manager.Namespace()
        shared_ns.counter = 0

        # 创建多个进程
        processes = []
        for i in range(5):
            p = Process(target=worker, args=(shared_list, shared_dict, i))
            processes.append(p)
            p.start()

        for p in processes:
            p.join()

        # 查看结果
        print(f"共享列表：{shared_list}")
        # [0, 1, 2, 3, 4]

        print(f"共享字典：{dict(shared_dict)}")
        # {'process_0': 0, 'process_1': 10, ...}

        print(f"计数器：{shared_ns.counter}")
```

### 14.16 协程（Coroutine）详解

#### 什么是协程？

```
┌─────────────────────────────────────────┐
│              协程概念图解               │
├─────────────────────────────────────────┤
│                                         │
│  协程（Coroutine）= 用户态的轻量级线程 │

│                                         │

│  ──────────                             │
│  • 线程 = 操作系统调度的执行单元       │
```

#### 协程的发展历史

```python
# ─────────────────────────────────────
# Python 协程的三代演进
# ─────────────────────────────────────

# 第一代：Python 3.3 之前（yield 模拟协程）
# 使用 yield 实现简单的协程，语法复杂

        x = yield
        print(f"收到：{x}")

# 第二代：Python 3.3-3.9（async/await 诞生）

    print(f"收到：{x}")

# ─────────────────────────────────────
# 第三代：Python 3.10+（类型注解增强）
# 添加了更严格的类型检查
async def typed_coroutine() -> str:
    x: str = await some_async_func()
    return x
```

### 14.17 深入理解 await

#### 核心困惑：await 到底阻塞了什么？

很多初学者看到这句话会困惑：

> "await 不会阻塞整个程序，只阻塞当前协程"

这听起来很矛盾：**阻塞不就是卡住不动吗？怎么会"阻塞了但又没完全阻塞"？**

让我用一个生活化的比喻来解释：

```
┌─────────────────────────────────────────────────────────┐
│              生活化比喻：餐厅等外卖                     │
├─────────────────────────────────────────────────────────┤
│                                                         │
│  场景：一个厨师（事件循环）要处理 3 桌客人的订单        │
│                                                         │
│  ─────────────────────────────────────────────────      │
│                                                         │
│  ❌ 同步方式（time.sleep）- 笨厨师                      │
│  ─────────────────────────────                          │
│  1. 厨师给 1 桌炒菜，发现需要等外卖食材                 │
│  2. 厨师站在门口等外卖！其他什么都不干！                │
│  3. 2 桌、3 桌的客人饿着肚子等...                       │
│  4. 外卖到了，继续炒 1 桌的菜                           │
│  5. 然后才开始处理 2 桌...                              │
│                                                         │
│  结果：1 桌等外卖时，整个餐厅都停摆了！                 │
│                                                         │
│  ─────────────────────────────────────────────────      │
│                                                         │
│  ✅ 异步方式（await）- 聪明厨师                         │
│  ─────────────────────────────                          │
│  1. 厨师给 1 桌炒菜，发现需要等外卖食材                 │
│  2. 厨师说："好的，我等外卖"（await）                   │
│  3. 然后转身去给 2 桌炒菜、给 3 桌打包...               │
│  4. 外卖到了（await 完成），回来继续炒 1 桌的菜         │
│                                                         │
│  结果：1 桌在等外卖，但 2 桌、3 桌的菜也在做！          │
│                                                         │
└─────────────────────────────────────────────────────────┘
```

**关键点**：
- `await` = "我先记下这件事等着，但不闲着，去干别的"
- `time.sleep` = "我傻等着，什么都不干"

#### 同步 vs 异步 的直观对比

```python

# ❌ 同步版本 - 使用 time.sleep
# ─────────────────────────────────────────

import time

def sync_task(name, delay):
    """同步任务"""
    print(f"[{name}] 开始工作")
    time.sleep(delay)  # 阻塞！整个程序都停在这里
    print(f"[{name}] 完成工作")
    return f"{name} 的结果"

def sync_main():
    start = time.time()

    # 顺序执行：一个接一个
    sync_task("任务 1", 1)   # 1 秒
    sync_task("任务 2", 2)   # 2 秒
    sync_task("任务 3", 1.5) # 1.5 秒

    print(f"总耗时：{time.time() - start:.2f}秒")
    # 输出：总耗时：4.50 秒

sync_main()
```

```
      │←──────── 总共 4.5 秒 ────────→│
```

```python
# ─────────────────────────────────────────
# ✅ 异步版本 - 使用 await（顺序执行）
# ─────────────────────────────────────────

import asyncio
import time

async def async_task(name, delay):
    """异步任务"""
    print(f"[{name}] 开始工作")
    await asyncio.sleep(delay)  # 非阻塞！让出控制权
    print(f"[{name}] 完成工作")
    return f"{name} 的结果"

async def async_main():
    start = time.time()

    # 注意：如果像这样直接 await，还是顺序执行
    await async_task("任务 1", 1)   # 1 秒
    await async_task("任务 2", 2)   # 2 秒
    await async_task("任务 3", 1.5) # 1.5 秒

    print(f"总耗时：{time.time() - start:.2f}秒")
    # 输出：总耗时：4.50 秒（和同步一样！）

asyncio.run(async_main())
```

**重点**：单独的 `await` 确实会让当前协程"暂停"，但关键是**在暂停期间，其他协程可以运行**！

### 深入理解："其他协程可以运行"的前提

很多初学者看到这里会有疑问：

> **"为什么顺序执行的 await 也是 4.5 秒？不是说 await 不阻塞吗？"**

这是一个非常好的问题！让我详细解释。

#### 核心要点

```
┌─────────────────────────────────────────────────────────┐
│                                                         │
│  "await 不会阻塞整个程序，只阻塞当前协程"              │
│   ↓                                                     │
│  意思是：await 暂停时，事件循环可以去运行             │
│         其他已经启动的协程                             │
│   ↓                                                     │
│  但前提是：你必须先创建并启动其他协程！               │
│         （用 gather 或 create_task）                   │
│                                                         │
└─────────────────────────────────────────────────────────┘
```

#### 关键区别：什么时候"其他协程"能运行？

**情况 1：顺序执行（`await` 一个接一个）**

```python
# 异步版本 - 顺序执行（4.5 秒）
async def async_main():
    await async_task("任务 1", 1)   # 1 秒
    await async_task("任务 2", 2)   # 2 秒
    await async_task("任务 3", 1.5) # 1.5 秒
```

```
时间线：
0s    1s    2s    3s    4s    5s
│     │     │     │     │     │
任务 1: ████┘
任务 2:         ████████┘
任务 3:                 ██████┘

发生什么：
┌──────────────────────────────────────┐
│ 0s: 启动任务 1                        │
│ 1s: 任务 1 完成，才启动任务 2          │
│ 3s: 任务 2 完成，才启动任务 3          │
│ 4.5s: 任务 3 完成                     │
└──────────────────────────────────────┘
```

**问题**：任务 1 在 `await` 时，任务 2 和任务 3 **还没被创建**！

```python
# 代码执行顺序：
await async_task("任务 1", 1)   # ← 这里：任务 2 的代码还没执行！
                                #   所以没有"其他协程"可以运行
await async_task("任务 2", 2)   # ← 任务 1 完成后，才执行到这里
await async_task("任务 3", 1.5) # ← 任务 2 完成后，才执行到这里
```

**情况 2：并发执行（`gather` 一起）**


```
└──────────────────────────────────────────────┘
```

#### await 的"暂停"到底是什么？

```python
# ─────────────────────────────────────────
# 深入理解 await 的"暂停"
# ─────────────────────────────────────────

async def demo_await():
    print("1. 开始")

    # await 在这里做了什么？
    await asyncio.sleep(1)
    # ↑ 这行代码实际发生了：
    #
    # 1. 告诉事件循环："我要暂停了"
    # 2. 事件循环记录下："这个任务 1 秒后需要恢复"
    # 3. 事件循环去运行其他任务
    # 4. 1 秒后，事件循环回来继续执行下一行
    #
    # 关键：await 期间，CPU 没有浪费在"等待"上！

    print("2. 结束")
```

**对比两种等待：**

```
┌─────────────────────────────────────────────────────────┐
│         time.sleep() vs await asyncio.sleep()          │
├─────────────────────────────────────────────────────────┤
│                                                         │
│  time.sleep(1)  - 同步等待                             │
│  ──────────────────────                                │
│  CPU: ████████████████ 占用（空转等待）               │
│  程序：完全停止                                         │
│  其他任务：无法执行                                     │
│                                                         │
│  ─────────────────────────────────────────────────      │
│                                                         │
```

### 14.20 Task 对象详解

#### 创建和管理任务

```python
import asyncio

# ─────────────────────────────────────────
# create_task vs gather
# ─────────────────────────────────────────

async def worker(name, delay):
    for i in range(3):
        await asyncio.sleep(delay)
        print(f"{name}: 第 {i+1} 次")
    return f"{name} 完成"

async def main():
    # ──────────────────────────────────
    # 方式 1：create_task - 立即执行
    # ──────────────────────────────────
    task1 = asyncio.create_task(worker("任务 A", 0.5))
    task2 = asyncio.create_task(worker("任务 B", 0.7))

    # 等待完成
    result1 = await task1
    result2 = await task2


    # ──────────────────────────────────
    # 方式 2：gather - 一起等待
    # ──────────────────────────────────
    results = await asyncio.gather(
        worker("任务 1", 0.5),
        worker("任务 2", 0.7)
    )


    # ──────────────────────────────────
    # 方式 3：wait - 更灵活的控制
    # ──────────────────────────────────
    tasks = [
        asyncio.create_task(worker(f"任务-{i}", 0.3))
        for i in range(5)
    ]

    # 等待第一个完成
    done, pending = await asyncio.wait(
        tasks,
        return_when=asyncio.FIRST_COMPLETED
    )
    print(f"已完成：{len(done)}, 未完成：{len(pending)}")

    # 取消未完成的任务
    for t in pending:
        t.cancel()

asyncio.run(main())
```

### 14.28 await 工作原理图解

#### 事件循环如何调度任务

```

│            await 工作流程详细图解                       │
├─────────────────────────────────────────────────────────┤
│                                                         │
│  代码：                                                │
│  ───────                                               │
│  async def task1():                                    │

│  async def task2():                                    │
│      print("C")                                        │

│                                                         │
│  async def main():                                     │
│      await asyncio.gather(task1(), task2())            │
│                                                         │
│  ─────────────────────────────────────────────────      │
│                                                         │

│  ────────────────────                                  │
│                                                         │
│  第 1 步：运行 task1()                                  │

│  └─ 遇到 await asyncio.sleep(1)                        │

│  └─ 事件循环等待（不 CPU，不耗能）                     │
│                                                         │
│  第 4 步：1 秒到了！                                     │
│  ├─ 事件循环唤醒 task1                                 │
│  ├─ task1 从 await 处继续执行                          │
│  └─ 打印 "B"，task1 完成                               │

│                                                         │
│  第 6 步：所有任务完成                                  │
│  └─ gather 返回结果                                    │
│                                                         │
│  输出顺序：A → C → (等待) → B → D                      │
```

```python
import asyncio

async def worker(name, delay):
    """工作协程"""

        await asyncio.sleep(delay)

    print("任务已创建，正在运行...")

    # 等待任务完成
    result1 = await task1
    result2 = await task2

    print(result1)
    print(result2)

    # ─────────────────────────────────
    # 获取任务状态
    # ─────────────────────────────────
    print(f"任务 1 是否完成：{task1.done()}")
    print(f"任务 2 是否完成：{task2.done()}")

    # ─────────────────────────────────
    # 取消任务
    # ─────────────────────────────────
    task3 = asyncio.create_task(worker("任务 3", 1))
    await asyncio.sleep(0.1)
    task3.cancel()
    try:
        await task3
    except asyncio.CancelledError:
        print("任务 3 已取消")

asyncio.run(main())
```

### 14.33 超时处理

```python
import asyncio

async def long_task():
    """长时间任务"""
    await asyncio.sleep(5)
    return "任务完成"

async def main():
    # ─────────────────────────────────
    # wait_for - 带超时的等待
    # ─────────────────────────────────
    try:
        result = await asyncio.wait_for(
            long_task(),
            timeout=2  # 2 秒超时
        )
        print(result)
    except asyncio.TimeoutError:
        print("任务超时！")

    # ─────────────────────────────────
    # wait - 等待多个任务
    # ─────────────────────────────────
    tasks = [
        asyncio.create_task(asyncio.sleep(1)),
        asyncio.create_task(asyncio.sleep(2)),
        asyncio.create_task(asyncio.sleep(3)),
    ]

    # 等待所有完成
    done, pending = await asyncio.wait(tasks)
    print(f"完成：{len(done)}, 未完成：{len(pending)}")

    # 等待第一个完成
    done, pending = await asyncio.wait(tasks, return_when=asyncio.FIRST_COMPLETED)
    print(f"第一个完成：{done}")

asyncio.run(main())
```

### 14.35 asyncio.Queue - 异步队列

#### 为什么需要异步队列

```
┌─────────────────────────────────────────┐
│     生产者 - 消费者模型的异步版本       │
├─────────────────────────────────────────┤
│                                         │
│  问题：标准 queue.Queue 是同步的       │
│  ─────────────────────────────          │
│  • put() 会阻塞整个事件循环            │
│  • get() 会阻塞整个事件循环            │
│  • 导致其他协程无法运行                  │
│                                         │
│  解决：使用 asyncio.Queue              │
│  ─────────────────────────────          │
│  • put() 是协程，await 等待             │
│  • get() 是协程，await 等待             │
│  • 不会阻塞，等待时其他协程可运行      │
│                                         │
└─────────────────────────────────────────┘
```

#### 基本使用

```python
import asyncio

async def producer(queue, name, count):
    """生产者：生产数据放入队列"""
    for i in range(count):
        item = f"{name} 生产的消息 {i}"
        # put 是协程，队列满时会等待
        await queue.put(item)
        print(f"放入：{item}")
        await asyncio.sleep(0.1)  # 模拟生产时间
    # 放入结束标记
    await queue.put(None)

async def consumer(queue, name):
    """消费者：从队列取出数据处理"""
    while True:
        # get 是协程，队列空时会等待
        item = await queue.get()

        # 检查结束标记
        if item is None:
            # 重新放入，让其他消费者也能退出
            await queue.put(None)
            break

        print(f"{name} 消费：{item}")
        await asyncio.sleep(0.2)  # 模拟消费时间
        # 标记任务完成
        queue.task_done()

async def main():
    # 创建队列（maxsize=5 表示最多容纳 5 个元素）
    queue = asyncio.Queue(maxsize=5)

    # 创建生产者
    producer_task = asyncio.create_task(
        producer(queue, "生产者", 10)
    )

    # 创建多个消费者
    consumer_tasks = [
        asyncio.create_task(consumer(queue, f"消费者-{i}"))
        for i in range(3)
    ]

    # 等待生产者完成
    await producer_task

    # 等待队列中所有任务处理完成
    await queue.join()

    # 取消消费者（因为 None 标记可能没被所有消费者处理）
    for task in consumer_tasks:
        task.cancel()

    await asyncio.gather(*consumer_tasks, return_exceptions=True)

# asyncio.run(main())
```

```python
import multiprocessing
import asyncio
import time

procs = [multiprocessing.Process(target=cpu_task) for _ in range(4)]
```

#### 示例 3：异步爬虫

```python
import asyncio
import aiohttp

async def fetch(session, url):
    """获取单个 URL"""
    try:
        async with session.get(url) as response:
            html = await response.text()
            return {"url": url, "status": response.status, "size": len(html)}
    except Exception as e:
        return {"url": url, "error": str(e)}

async def crawl(urls):
    """并发爬取多个 URL"""
    async with aiohttp.ClientSession() as session:
        # 创建所有任务
        tasks = [fetch(session, url) for url in urls]
        # 并发执行
        results = await asyncio.gather(*tasks)
        return results

if __name__ == "__main__":
    urls = [
        "https://www.python.org",
        "https://www.github.com",
        "https://www.stackoverflow.com",
        # ... 更多 URL
    ]

    results = asyncio.run(crawl(urls))
    for result in results:
        if "error" in result:
            print(f"错误：{result['url']} - {result['error']}")
        else:
            print(f"成功：{result['url']} - 状态{result['status']} - {result['size']}字节")
```

### 练习 1：线程安全计数器

```python
import threading

class ThreadSafeCounter:
    """线程安全的计数器"""

    def __init__(self, initial=0):
        self.value = initial
        self.lock = threading.Lock()

    def increment(self, amount=1):
        with self.lock:
            self.value += amount
            return self.value

    def decrement(self, amount=1):
        with self.lock:
            self.value -= amount
            return self.value

    def get(self):
        with self.lock:
            return self.value

# 测试
counter = ThreadSafeCounter()
threads = []
for _ in range(10):
    t = threading.Thread(target=lambda: [counter.increment() for _ in range(1000)])
    threads.append(t)
    t.start()

for t in threads:
    t.join()

print(f"最终值：{counter.get()}")  # 应为 10000
```

### 练习 2：生产者 - 消费者模型

```python
from multiprocessing import Process, Queue
import random
import time

def producer(queue, name):
    """生产者"""
    for i in range(10):
        item = random.randint(1, 100)
        queue.put(item)
        print(f"{name} 生产：{item}")
        time.sleep(random.random() * 0.5)
    queue.put(None)  # 结束标记

def consumer(queue, name):
    """消费者"""
    total = 0
    count = 0
    while True:
        item = queue.get()
        if item is None:
            queue.put(None)  # 传递给其他消费者
            break
        total += item
        count += 1
        print(f"{name} 消费：{item}")
        time.sleep(random.random() * 0.3)
    print(f"{name} 总计处理 {count} 个，总和 {total}")

if __name__ == "__main__":
    q = Queue()

    # 2 个生产者
    p1 = Process(target=producer, args=(q, "生产者 A"))
    p2 = Process(target=producer, args=(q, "生产者 B"))

    # 3 个消费者
    c1 = Process(target=consumer, args=(q, "消费者 X"))
    c2 = Process(target=consumer, args=(q, "消费者 Y"))
    c3 = Process(target=consumer, args=(q, "消费者 Z"))

    for p in [p1, p2, c1, c2, c3]:
        p.start()

    for p in [p1, p2, c1, c2, c3]:
        p.join()
```

### 练习 3：并发计算 PI

```python
from multiprocessing import Pool
import random

def estimate_pi_chunk(n):
    """用蒙特卡洛方法估算 PI（一部分）"""
    inside = 0
    for _ in range(n):
        x, y = random.random(), random.random()
        if x*x + y*y <= 1:
            inside += 1
    return inside

if __name__ == "__main__":
    total_points = 1000000
    num_processes = 4
    points_per_process = total_points // num_processes

    with Pool(processes=num_processes) as pool:
        results = pool.map(estimate_pi_chunk, [points_per_process] * num_processes)

    total_inside = sum(results)
    pi_estimate = 4 * total_inside / total_points
    print(f"估算的 PI 值：{pi_estimate}")
```

### 12.2 打开和关闭文件

#### open() 函数详解

```python
# 基本语法
file = open("filename.txt", mode="r", encoding="utf-8")

# 参数说明：
# file     - 文件路径（字符串或 Path 对象）
# mode     - 打开模式（r/w/a/x/b/+）
# encoding - 文件编码（推荐总是指定 utf-8）
# buffering- 缓冲策略（通常不用设置）
```

#### 两种打开方式对比

```python
# ─────────────────────────────────────
# 方式 1：手动打开和关闭（不推荐）
# ─────────────────────────────────────

file = open("example.txt", "r", encoding="utf-8")
content = file.read()
# ... 处理内容 ...
file.close()  # 必须记得关闭！

# ⚠️ 问题：如果中间发生异常，file.close() 可能不会执行
# 导致资源泄露！

# ─────────────────────────────────────
# 方式 2：使用 with 语句（强烈推荐）
# ─────────────────────────────────────

with open("example.txt", "r", encoding="utf-8") as file:
    content = file.read()
    # ... 处理内容 ...
# 离开 with 块时，文件自动关闭！

# ✅ 优点：即使发生异常，文件也会被正确关闭
```

**with 语句工作原理图解：**
```
with open("file.txt") as f:
    # 进入 with 块时
    # ─────────────────
    # 1. 执行 open() 打开文件
    # 2. 将文件对象赋值给 f
    # 3. 执行块内代码

    content = f.read()

    # 离开 with 块时（无论是否正常离开）
    # ─────────────────
    # 1. 自动调用 f.close()
    # 2. 释放文件资源

# 文件已关闭 ✓
```

#### 完整的文件操作流程

```python
# 标准文件操作流程（推荐模板）

filename = "example.txt"

try:
    with open(filename, "r", encoding="utf-8") as f:
        content = f.read()
        print(content)
except FileNotFoundError:
    print(f"错误：文件 {filename} 不存在")
except PermissionError:
    print(f"错误：没有权限读取 {filename}")
except UnicodeDecodeError:
    print(f"错误：无法用 utf-8 解码 {filename}")
except Exception as e:
    print(f"发生未知错误：{e}")
```

## 第二部分：文件打开模式

### 12.3 文本模式详解

```
┌─────────────────────────────────────────┐
│          文件打开模式一览               │
├─────────────────────────────────────────┤
│                                         │
│  基础模式（4 种）：                      │
│  ───────────────────                    │
│  'r' - 只读（Read）                     │
│      文件必须存在，指针在开头           │
│                                         │
│  'w' - 写入（Write）                    │
│      文件不存在则创建，存在则清空       │
│      指针在开头                         │
│                                         │
│  'a' - 追加（Append）                   │
│      文件不存在则创建                   │
│      文件存在时，指针在末尾              │
│                                         │
│  'x' - 独占创建（eXclusive）            │
│      文件必须不存在，否则报错           │
│      创建新文件，指针在开头             │
│                                         │
│  ─────────────────────────────────      │
│                                         │
│  组合模式：                             │
│  ───────────────────                    │
│  'b' - 二进制模式（Binary）             │
│      与 r/w/a/x 组合：rb, wb, ab, xb    │
│                                         │
│  '+' - 更新模式（读写）                 │
│      与 r/w/a 组合：r+, w+, a+          │
│                                         │
└─────────────────────────────────────────┘
```

#### 模式详解与对比

```python
# ─────────────────────────────────────
# 'r' - 只读模式
# ─────────────────────────────────────

# 文件必须存在，否则报错 FileNotFoundError
with open("example.txt", "r", encoding="utf-8") as f:
    content = f.read()  # 可以读取
    # f.write("x")      # ❌ 报错：不支持写入

# ─────────────────────────────────────
# 'w' - 写入模式（覆盖）
# ─────────────────────────────────────

# 文件不存在：创建
# 文件已存在：清空原内容！
with open("output.txt", "w", encoding="utf-8") as f:
    f.write("第一行\n")
    f.write("第二行\n")
    # 原文件内容已被清空！

# ─────────────────────────────────────
# 'a' - 追加模式
# ─────────────────────────────────────

# 文件不存在：创建
# 文件已存在：在末尾追加，保留原内容
with open("log.txt", "a", encoding="utf-8") as f:
    f.write("新的日志\n")
    # 内容添加到文件末尾

# ─────────────────────────────────────
# 'x' - 独占创建模式
# ─────────────────────────────────────

# 文件必须不存在，否则报 FileExistsError
try:
    with open("new.txt", "x", encoding="utf-8") as f:
        f.write("新文件")
except FileExistsError:
    print("文件已存在，创建失败")

# 用途：确保不会意外覆盖已有文件
```

### 12.4 二进制模式

#### 文本模式 vs 二进制模式

```
┌─────────────────────────────────────────┐
│       文本模式与二进制模式对比          │
├─────────────────────────────────────────┤
│                                         │
│  文本模式（默认）                       │
│  ──────────────────                    │
│  • 处理字符串（str）                    │
│  • 自动处理换行符转换                   │
│    Windows: \r\n  ←→  \n               │
│    Mac/Linux: \n                        │
│  • 需要指定编码（utf-8）               │
│  • 用于：文本文件、代码、配置文件       │
│                                         │
│  ─────────────────────────────────      │
│                                         │
│  二进制模式（'b'）                      │
│  ──────────────────                    │
│  • 处理字节（bytes）                    │
│  • 不转换换行符，原样读取               │
│  • 不需要编码参数                       │
│  • 用于：图片、音频、视频、压缩包       │
│                                         │
└─────────────────────────────────────────┘
```

#### 二进制文件操作示例

```python
# ─────────────────────────────────────
# 复制图片文件
# ─────────────────────────────────────

# 方式 1：简单复制
with open("source.jpg", "rb") as src:
    with open("copy.jpg", "wb") as dst:
        dst.write(src.read())

# 方式 2：分块复制（大文件推荐）
def copy_file(src_path, dst_path, chunk_size=1024*1024):
    """复制文件，每次读取 1MB"""
    with open(src_path, "rb") as src:
        with open(dst_path, "wb") as dst:
            while True:
                chunk = src.read(chunk_size)
                if not chunk:  # 读取完毕
                    break
                dst.write(chunk)

copy_file("large_video.mp4", "backup.mp4")

# ─────────────────────────────────────
# 读写二进制数据
# ─────────────────────────────────────

# PNG 文件头（前 8 字节）
png_header = b'\x89PNG\r\n\x1a\n'

with open("test.png", "wb") as f:
    f.write(png_header)
    # 可以继续写入其他数据...

# 读取并检查文件类型
def check_file_type(filename):
    with open(filename, "rb") as f:
        header = f.read(8)

    if header.startswith(b'\x89PNG'):
        return "PNG 图片"
    elif header.startswith(b'\xFF\xD8'):
        return "JPEG 图片"
    elif header.startswith(b'GIF8'):
        return "GIF 图片"
    else:
        return "未知类型"

print(check_file_type("test.png"))  # PNG 图片
```

#### bytes 对象基础

```python
# bytes 字面量
data = b"Hello"  # 字节字符串
print(data)       # b'Hello'
print(type(data)) # <class 'bytes'>

# bytes 与字符串转换
text = "你好"

```

```python
with open("example.txt", "r", encoding="utf-8") as f:
```

#### readline() - 读取一行

```python
with open("example.txt", "r", encoding="utf-8") as f:
    line1 = f.readline()
    print(repr(line1))  # 'Hello, World!\n'

    line2 = f.readline()
    print(repr(line2))  # '第二行内容\n'

    line3 = f.readline()
    print(repr(line3))  # '第三行内容'（最后一行可能无换行）

# 逐行读取直到文件末尾
with open("example.txt", "r", encoding="utf-8") as f:
    while True:
        line = f.readline()
        if not line:  # 空字符串表示已到文件末尾
            break
        print(line.strip())  # strip() 去除换行符
```

#### readlines() - 读取所有行

```python
with open("example.txt", "r", encoding="utf-8") as f:
    lines = f.readlines()
    print(lines)
    # ['Hello, World!\n', '第二行内容\n', '第三行内容']

# 遍历行
with open("example.txt", "r", encoding="utf-8") as f:
    for i, line in enumerate(f.readlines(), 1):
        print(f"第{i}行：{line.strip()}")
```

#### 迭代文件对象（推荐）

```python
# 最推荐的方式：直接迭代文件对象
with open("example.txt", "r", encoding="utf-8") as f:
    for line in f:
        print(line.strip())

# 优点：
# 1. 内存效率高：一次只读一行
# 2. 代码简洁
# 3. 适合大文件

# 带行号的遍历
with open("example.txt", "r", encoding="utf-8") as f:
    for i, line in enumerate(f, 1):
        print(f"{i}: {line.strip()}")
```

### 12.6 文件指针操作

#### tell() 和 seek()

```python
with open("example.txt", "r", encoding="utf-8") as f:
    # 初始位置
    print(f.tell())  # 0

    # 读取 5 个字符
    f.read(5)
    print(f.tell())  # 5

    # 读取 10 个字符
    f.read(10)
    print(f.tell())  # 15

# seek() 移动读取指针
with open("example.txt", "r", encoding="utf-8") as f:
    # 移动到文件开头
    f.seek(0)
    print(f.tell())  # 0

    # 移动到文件第 10 个字符处
    f.seek(10)
    print(f.read(5))  # 读取第 10-14 个字符

# seek() 的第二个参数（参考点）
with open("example.txt", "rb") as f:  # 二进制模式才支持
    # 从文件开头移动（默认）
    f.seek(10, 0)  # 或 f.seek(10)

    # 从当前位置移动
    f.seek(5, 1)   # 向后移动 5 字节

    # 从文件末尾移动
    f.seek(-10, 2) # 从末尾向前 10 字节
```

```
               │
```

### 12.7 写入方法详解

#### write() - 写入字符串

```python
# 写入模式（覆盖）
with open("output.txt", "w", encoding="utf-8") as f:

    # 内容添加到文件末尾

with open("output.txt", "w", encoding="utf-8") as f:

```

#### writelines() - 写入多行

```python
lines = [
    "第一行\n",
    "第二行\n",
    "第三行\n"
]

with open("output.txt", "w", encoding="utf-8") as f:
    f.writelines(lines)
    # 注意：不会自动添加换行符，需要自己加！

# ❌ 常见错误
lines = ["第一行", "第二行", "第三行"]  # 没有换行符
with open("output.txt", "w", encoding="utf-8") as f:
    f.writelines(lines)
# 结果：第一行第二行第三行（全在一行）

# ✅ 正确做法
lines = ["第一行\n", "第二行\n", "第三行\n"]
with open("output.txt", "w", encoding="utf-8") as f:
    f.writelines(lines)
```

### 12.8 写入时的缓冲

#### 刷新缓冲区

```python
# 默认情况下，写入的内容会先缓存
with open("output.txt", "w", encoding="utf-8") as f:
    f.write("第一行\n")  # 可能在缓冲区
    f.write("第二行\n")  # 可能在缓冲区

    # 强制写入磁盘
    f.flush()
    # 现在内容已经写入磁盘

# 使用刷新模式（每行都刷新）
with open("log.txt", "a", encoding="utf-8", buffering=1) as f:
    f.write("这行会立即刷新\n")

# 无缓冲模式（二进制模式专属）
with open("data.bin", "wb", buffering=0) as f:
    f.write(b"立即写入磁盘")
```

#### 实时日志示例

```python
import time

def write_log(message, filename="realtime.log"):
    """写入实时日志"""
    with open(filename, "a", encoding="utf-8") as f:
        timestamp = time.strftime("%Y-%m-%d %H:%M:%S")
        f.write(f"[{timestamp}] {message}\n")
        f.flush()  # 确保立即写入
        print(f"日志已写入：{message}")

# 使用
write_log("程序启动")
time.sleep(1)
write_log("处理数据中...")
time.sleep(1)
write_log("程序结束")
```

### 12.9 pathlib 模块详解

#### Path 基础

```python
from pathlib import Path

# 创建路径对象
p = Path("data/file.txt")

# 路径拼接（推荐方式）
data_dir = Path("data")
file_path = data_dir / "subfolder" / "file.txt"
print(file_path)  # data/subfolder/file.txt

# 与字符串拼接对比
# ❌ 字符串拼接（不推荐）
path = "data" + "/" + "subfolder" + "/" + "file.txt"

# ✅ Path 拼接（推荐）
path = Path("data") / "subfolder" / "file.txt"
```

#### 路径属性和方法

```python
from pathlib import Path

p = Path("/home/user/documents/report.pdf")

# ─────────────────────────────────────
# 获取路径各部分
# ─────────────────────────────────────

print(p.name)        # report.pdf（文件名）
print(p.stem)        # report（不含扩展名）
print(p.suffix)      # .pdf（扩展名）
print(p.suffixes)    # ['.pdf']（所有扩展名）
print(p.parent)      # /home/user/documents（父目录）
print(p.parents)     # 所有父目录的序列
print(p.parts)       # ('/', 'home', 'user', 'documents', 'report.pdf')

# ─────────────────────────────────────
# 路径转换
# ─────────────────────────────────────

print(p.absolute())  # 绝对路径
print(p.resolve())   # 解析符号链接后的绝对路径

# ─────────────────────────────────────
# 路径判断
# ─────────────────────────────────────

print(p.exists())    # 是否存在
print(p.is_file())   # 是否是文件
print(p.is_dir())    # 是否是目录
print(p.is_absolute())  # 是否是绝对路径
```

#### else 和 finally 子句

```python
def process_file(filename):
    """处理文件的完整示例"""
    f = None
    try:
        print("尝试打开文件...")
        f = open(filename, "r", encoding="utf-8")
        content = f.read()
        result = int(content) / 10

    except FileNotFoundError:
        print("文件不存在")
        return None

    except ValueError:
        print("内容不是有效的数字")
        return None

    except ZeroDivisionError:
        print("除零错误")
        return None

    else:
        # 没有异常时执行
        print(f"处理成功，结果：{result}")
        return result

    finally:
        # 总是执行，无论是否有异常
        print("清理资源...")
        if f:
            f.close()

# 测试
process_file("data.txt")
```

```python
from datetime import datetime

def backup_file(filename):
    """为文件创建带时间戳的备份"""
    src = Path(filename)

    if not src.exists():
        print(f"错误：文件 '{filename}' 不存在")
        return None

    dst = src.parent / backup_name

    shutil.copy2(src, dst)  # copy2 保留元数据

    print(f"已创建备份：{backup_name}")
    return dst
```

### 12.14 简单日志模块

```python
from pathlib import Path
from datetime import datetime

class SimpleLogger:
    """简单的日志记录器"""

    def __init__(self, log_file="app.log"):
        self.log_file = Path(log_file)

    def _get_timestamp(self):
        """获取当前时间戳"""
        return datetime.now().strftime("%Y-%m-%d %H:%M:%S")

    def _write(self, level, message):
        """写入日志"""
        timestamp = self._get_timestamp()
        log_entry = f"[{timestamp}] [{level}] {message}\n"

        with open(self.log_file, "a", encoding="utf-8") as f:
            f.write(log_entry)

    def info(self, message):
        """信息级别日志"""
        self._write("INFO", message)

    def warning(self, message):
        """警告级别日志"""
        self._write("WARNING", message)

    def error(self, message):
        """错误级别日志"""
        self._write("ERROR", message)

    def debug(self, message):
        """调试级别日志"""
        self._write("DEBUG", message)

# 使用
logger = SimpleLogger()
logger.info("程序启动")
logger.debug("初始化配置")
logger.warning("内存使用率较高")
```

### 12.15 批量重命名工具

```python
from pathlib import Path

def batch_rename(folder, pattern, replacement):
    """
    批量重命名文件

    参数：
        folder: 目录路径
        pattern: 要替换的模式
        replacement: 替换为新字符串
    """
    folder = Path(folder)

    if not folder.exists():
        print(f"错误：目录 '{folder}' 不存在")
        return

    if not folder.is_dir():
        print(f"错误：'{folder}' 不是目录")
        return

    count = 0
    for file_path in folder.iterdir():
        if file_path.is_file() and pattern in file_path.name:
            new_name = file_path.name.replace(pattern, replacement)
            new_path = file_path.parent / new_name

            try:
                file_path.rename(new_path)
                print(f"{file_path.name} → {new_name}")
                count += 1
            except Exception as e:
                print(f"重命名失败：{file_path.name} - {e}")

    print(f"\n共重命名 {count} 个文件")

# 使用
# 将所有.txt 文件中的"old"替换为"new"
batch_rename("./data", "old", "new")

# 批量添加前缀
def add_prefix(folder, prefix, extension=None):
    """为文件添加前缀"""
    folder = Path(folder)

    for i, file_path in enumerate(folder.iterdir(), 1):
        if file_path.is_file():
            if extension and file_path.suffix != extension:
                continue

            new_name = f"{prefix}_{i:03d}{file_path.suffix}"
            new_path = file_path.parent / new_name
            file_path.rename(new_path)
            print(f"{file_path.name} → {new_name}")

# 将所有图片重命名为 img_001.jpg, img_002.jpg...
add_prefix("./photos", "img")
```

### 12.16 CSV 文件处理

```python
# 不使用 csv 模块，手动解析 CSV

def read_csv(filename):
    """读取 CSV 文件，返回字典列表"""
    result = []

    with open(filename, "r", encoding="utf-8") as f:
        lines = f.readlines()

    if not lines:
        return result

    # 第一行是表头
    headers = [h.strip() for h in lines[0].split(",")]

    # 处理数据行
    for line in lines[1:]:
        values = [v.strip() for v in line.split(",")]
        row = dict(zip(headers, values))
        result.append(row)

    return result

def write_csv(filename, data, headers=None):
    """将字典列表写入 CSV 文件"""
    if not data:
        return

    # 如果没有指定表头，使用第一个字典的键
    if headers is None:
        headers = list(data[0].keys())

    with open(filename, "w", encoding="utf-8") as f:
        # 写入表头
        f.write(",".join(headers) + "\n")

        # 写入数据
        for row in data:
            values = [str(row.get(h, "")) for h in headers]
            f.write(",".join(values) + "\n")

# 使用示例
students = [
    {"name": "Alice", "age": "20", "score": "90"},
    {"name": "Bob", "age": "21", "score": "85"},
    {"name": "Charlie", "age": "19", "score": "95"}
]

# 写入 CSV
write_csv("students.csv", students)

# 读取 CSV
data = read_csv("students.csv")
for row in data:
    print(row)

# 计算平均分
def calculate_average(filename, column):
    """计算 CSV 某一列的平均值"""
    data = read_csv(filename)
    values = [float(row[column]) for row in data if column in row]

    if not values:
        return 0

    return sum(values) / len(values)

avg_score = calculate_average("students.csv", "score")
print(f"平均分：{avg_score:.2f}")
```

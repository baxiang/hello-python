## 第一部分 - Python logging 模块

### 1.1 基础用法

```python
import logging

# 配置日志
logging.basicConfig(
    level=logging.DEBUG,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
    datefmt='%Y-%m-%d %H:%M:%S',
    handlers=[
        logging.FileHandler('app.log'),
        logging.StreamHandler()
    ]
)

logger = logging.getLogger(__name__)

# 不同级别的日志
logger.debug("调试信息 - 详细信息")
logger.info("信息 - 正常运行的消息")
logger.warning("警告 - 潜在问题")
logger.error("错误 - 严重问题")
logger.critical("严重错误 - 系统可能无法继续运行")
```

### 1.2 日志级别

| 级别 | 数值 | 使用场景 |
|------|------|---------|
| DEBUG | 10 | 调试信息，详细的技术细节 |
| INFO | 20 | 一般信息，系统正常运行消息 |
| WARNING | 30 | 警告信息，不影响运行但需要注意 |
| ERROR | 40 | 错误信息，功能执行失败 |
| CRITICAL | 50 | 严重错误，系统可能崩溃 |

### 1.3 日志器层次结构

```python
import logging

# 创建日志器
logger = logging.getLogger('myapp')
logger.setLevel(logging.DEBUG)

# 创建子日志器
db_logger = logging.getLogger('myapp.database')
api_logger = logging.getLogger('myapp.api')

# 子日志器继承父日志器的配置
# myapp.database 的日志会传递到 myapp

# 禁止传递
db_logger.propagate = False
```

### 2.2 日志轮转

```python
import logging
from logging.handlers import (
    RotatingFileHandler,
    TimedRotatingFileHandler
)

logger = logging.getLogger(__name__)

# 按大小轮转
rotating_handler = RotatingFileHandler(
    'logs/app.log',
    maxBytes=10*1024*1024,  # 10MB
    backupCount=5
)

# 按时间轮转
timed_handler = TimedRotatingFileHandler(
    'logs/app.log',
    when='midnight',      # 每天午夜轮转
    interval=1,           # 每天
    backupCount=30,       # 保留 30 天
    encoding='utf8'
)

logger.addHandler(rotating_handler)
logger.addHandler(timed_handler)
```

```python
import logging
from pythonjsonlogger import jsonlogger

logger = logging.getLogger(__name__)

# 配置 JSON 格式
log_handler = logging.StreamHandler()
formatter = jsonlogger.JsonFormatter(
    fmt='%(asctime)s %(name)s %(levelname)s %(message)s',
    datefmt='%Y-%m-%dT%H:%M:%S'
)
log_handler.setFormatter(formatter)
logger.addHandler(log_handler)

# 使用
logger.info("User login", extra={
    "user_id": 123,
    "username": "alice",
    "ip_address": "192.168.1.1"
})
# 输出：{"asctime": "...", "name": "...", "levelname": "INFO", "message": "User login", "user_id": 123, ...}
```

### 3.2 上下文日志

```python
import logging
from contextlib import contextmanager

class ContextLogger:
    def __init__(self, name: str):
        self.logger = logging.getLogger(name)
        self.context = {}
    
    @contextmanager
    def bind_context(self, **kwargs):
        """绑定上下文"""
        old_context = self.context.copy()
        self.context.update(kwargs)
        try:
            yield
        finally:
            self.context = old_context
    
    def info(self, message: str, **extra):
        self.logger.info(message, extra={**self.context, **extra})
    
    def error(self, message: str, **extra):
        self.logger.error(message, extra={**self.context, **extra})

# 使用
logger = ContextLogger('myapp')

with logger.bind_context(user_id=123, request_id='abc'):
    logger.info("Processing request")
    logger.info("Request completed")
```

## 第四部分 - 性能监控

### 4.1 装饰器记录执行时间

```python
import logging
import time
from functools import wraps

logger = logging.getLogger(__name__)

def timing_decorator(func):
    """记录函数执行时间"""
    @wraps(func)
    def wrapper(*args, **kwargs):
        start = time.time()
        try:
            result = func(*args, **kwargs)
            logger.info(f"{func.__name__} executed in {time.time() - start:.3f}s")
            return result
        except Exception as e:
            logger.error(f"{func.__name__} failed after {time.time() - start:.3f}s: {e}")
            raise
    return wrapper

# 使用
@timing_decorator
def slow_function():
    time.sleep(1)
    return "done"
```

### 4.2 性能分析

```python
import cProfile
import pstats
import logging
from io import StringIO

logger = logging.getLogger(__name__)

def profile_function(func):
    """性能分析装饰器"""
    @wraps(func)
    def wrapper(*args, **kwargs):
        profiler = cProfile.Profile()
        profiler.enable()
        
        try:
            result = func(*args, **kwargs)
        finally:
            profiler.disable()
            
            # 输出统计
            stream = StringIO()
            stats = pstats.Stats(profiler, stream=stream)
            stats.sort_stats('cumulative')
            stats.print_stats(10)
            
            logger.info(f"Profile for {func.__name__}:\n{stream.getvalue()}")
        
        return result
    return wrapper
```

## 第五部分 - 应用监控

### 5.1 健康检查端点

```python
from flask import Flask, jsonify
import logging
import psutil

app = Flask(__name__)
logger = logging.getLogger(__name__)

@app.route('/health')
def health_check():
    """健康检查端点"""
    try:
        # 检查系统资源
        cpu_percent = psutil.cpu_percent()
        memory = psutil.virtual_memory()
        disk = psutil.disk_usage('/')
        
        status = "healthy"
        if cpu_percent > 90 or memory.percent > 90:
            status = "degraded"
        
        logger.info(f"Health check: {status}")
        
        return jsonify({
            "status": status,
            "metrics": {
                "cpu_percent": cpu_percent,
                "memory_percent": memory.percent,
                "disk_percent": disk.percent,
            }
        }), 200 if status == "healthy" else 503
    
    except Exception as e:
        logger.error(f"Health check failed: {e}")
        return jsonify({"status": "unhealthy", "error": str(e)}), 500
```

### 5.2 指标收集

```python
import time
from collections import defaultdict
from dataclasses import dataclass, field
from typing import Dict, List
import logging

logger = logging.getLogger(__name__)

@dataclass
class Metric:
    name: str
    value: float
    labels: Dict[str, str] = field(default_factory=dict)
    timestamp: float = field(default_factory=time.time)

class MetricsCollector:
    def __init__(self):
        self.metrics: List[Metric] = []
    
    def record(self, name: str, value: float, **labels):
        """记录指标"""
        metric = Metric(name=name, value=value, labels=labels)
        self.metrics.append(metric)
        logger.debug(f"Metric recorded: {name}={value}")
    
    def record_latency(self, operation: str, latency_ms: float):
        """记录延迟"""
        self.record('latency', latency_ms, operation=operation)
    
    def record_error(self, operation: str, error_type: str):
        """记录错误"""
        self.record('error_count', 1, operation=operation, error_type=error_type)
    
    def get_summary(self) -> Dict:
        """获取指标摘要"""
        summary = defaultdict(list)
        for metric in self.metrics[-1000:]:  # 最近 1000 条
            key = f"{metric.name}:{tuple(sorted(metric.labels.items()))}"
            summary[key].append(metric.value)
        
        return {
            name: {
                'count': len(values),
                'avg': sum(values) / len(values),
                'min': min(values),
                'max': max(values),
            }
            for name, values in summary.items()
        }

# 使用
metrics = MetricsCollector()

def process_request():
    start = time.time()
    try:
        # 处理请求
        pass
        latency = (time.time() - start) * 1000
        metrics.record_latency('process_request', latency)
    except Exception as e:
        metrics.record_error('process_request', type(e).__name__)
        raise
```

## 最佳实践清单

### ✅ 必须遵守

1. 生产环境日志级别至少为 INFO
2. 错误日志必须包含堆栈跟踪
3. 敏感信息不得记录到日志

### ✅ 推荐做法

1. 使用结构化日志（JSON 格式）
2. 实现日志轮转，避免磁盘占满
3. 添加请求 ID 便于追踪
4. 监控关键性能指标
5. 设置告警阈值

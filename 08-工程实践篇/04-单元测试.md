# 第 04 章 - 单元测试

> 测试是保证代码质量的重要手段，本章介绍 Python 测试的最佳实践。

## 第一部分 - pytest 基础

### 1.1 安装与使用

```bash
# 安装
uv add --dev pytest pytest-cov

# 运行所有测试
pytest

# 运行特定测试文件
pytest tests/test_module.py

# 运行特定测试函数
pytest tests/test_module.py::test_function

# 显示覆盖率
pytest --cov=src/my_package --cov-report=html

# 显示详细输出
pytest -v
```

### 1.2 编写测试

```python
# tests/test_calculator.py
import pytest
from src.my_package.calculator import add, divide

def test_add():
    """测试加法"""
    assert add(2, 3) == 5
    assert add(-1, 1) == 0

def test_divide():
    """测试除法"""
    assert divide(10, 2) == 5
    assert divide(7, 2) == 3.5

def test_divide_by_zero():
    """测试异常"""
    with pytest.raises(ZeroDivisionError):
        divide(10, 0)
```

### 1.3 参数化测试

```python
import pytest
from src.my_package.calculator import multiply

@pytest.mark.parametrize("a,b,expected", [
    (2, 3, 6),
    (5, 5, 25),
    (0, 10, 0),
    (-2, 3, -6),
])
def test_multiply(a, b, expected):
    """参数化测试乘法"""
    assert multiply(a, b) == expected
```

## 第二部分 - Fixtures

### 2.1 基础 Fixture

```python
# tests/conftest.py
import pytest
from src.my_package.database import Database

@pytest.fixture
def db():
    """创建数据库连接"""
    db = Database(":memory:")
    yield db
    db.close()

@pytest.fixture
def sample_user():
    """示例用户数据"""
    return {"id": 1, "name": "Alice", "email": "alice@example.com"}
```

```python
# tests/test_database.py
import pytest
from src.my_package.user_service import UserService

def test_create_user(db, sample_user):
    """测试创建用户"""
    service = UserService(db)
    user = service.create(sample_user)
    assert user["name"] == "Alice"

def test_get_user(db, sample_user):
    """测试获取用户"""
    service = UserService(db)
    service.create(sample_user)
    retrieved = service.get_by_id(1)
    assert retrieved["email"] == "alice@example.com"
```

### 2.2 Fixture 作用域

```python
import pytest

@pytest.fixture(scope="function")
def func_fixture():
    """每个测试函数执行一次"""
    return "function"

@pytest.fixture(scope="module")
def module_fixture():
    """每个模块执行一次"""
    return "module"

@pytest.fixture(scope="session")
def session_fixture():
    """整个会话执行一次"""
    return "session"
```

## 第三部分 - Mock

### 3.1 使用 mock

```python
from unittest.mock import Mock, patch
import pytest
from src.my_package.api_client import APIClient

def test_api_client():
    """测试 API 客户端"""
    mock_response = Mock()
    mock_response.status_code = 200
    mock_response.json.return_value = {"temp": 25}
    
    with patch('requests.get', return_value=mock_response):
        client = APIClient()
        result = client.get_weather("Beijing")
        assert result["temp"] == 25
```

### 3.2 补丁装饰器

```python
import pytest
from unittest.mock import patch
from src.my_package.payment import PaymentProcessor

class TestPaymentProcessor:
    
    @patch('src.my_package.payment.stripe.Charge.create')
    def test_process_payment(self, mock_create):
        """测试支付处理"""
        mock_create.return_value = {"id": "ch_123", "status": "succeeded"}
        
        processor = PaymentProcessor()
        result = processor.process(100.00, "user_123")
        
        assert result["status"] == "succeeded"
        mock_create.assert_called_once()
```

## 第四部分 - 测试覆盖率

### 4.1 配置

```toml
# pyproject.toml
[tool.coverage.run]
source = ["src/my_package"]
omit = ["*/tests/*", "*/__init__.py"]

[tool.coverage.report]
exclude_lines = [
    "pragma: no cover",
    "def __repr__",
    "if TYPE_CHECKING:",
]
```

### 4.2 运行报告

```bash
# HTML 报告
pytest --cov=src/my_package --cov-report=html
open htmlcov/index.html

# 终端报告
pytest --cov=src/my_package --cov-report=term-missing

# XML 报告（CI 使用）
pytest --cov=src/my_package --cov-report=xml
```

## 第五部分 - 最佳实践

### 5.1 测试命名

```python
# ✅ 清晰的测试名
def test_create_user_with_valid_email():
    pass

def test_login_fails_with_invalid_password():
    pass

# ❌ 模糊的测试名
def test_user():
    pass

def test_stuff():
    pass
```

### 5.2 AAA 模式

```python
def test_withdraw_money():
    """Arrange-Act-Assert 模式"""
    # Arrange - 准备
    account = BankAccount(balance=100)
    
    # Act - 执行
    account.withdraw(50)
    
    # Assert - 断言
    assert account.balance == 50
```

### 5.3 测试隔离

```python
# ✅ 每个测试独立
def test_create():
    db = create_test_db()  # 每个测试创建自己的资源
    # 测试创建

def test_update():
    db = create_test_db()  # 不依赖其他测试
    # 测试更新

# ❌ 避免：测试间依赖
def test_create():
    global db
    db = create_test_db()

def test_update():
    update(db)  # 依赖 test_create 的结果
```

### 5.4 测试清单

| 实践 | 说明 |
|------|------|
| 为公共 API 编写测试 | 覆盖所有对外接口 |
| 保持测试独立 | 可并行运行，无依赖 |
| 使用有意义的名称 | 描述测试场景和预期 |
| 测试正常和异常路径 | 包括边界条件 |
| 保持测试简洁 | 每个测试 < 20 行 |
| 使用参数化 | 减少重复代码 |
